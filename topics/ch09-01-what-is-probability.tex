\section{What Is Probability?}

\begin{learningGoals}
\begin{itemize}
    \item Understand probability as a number between $0$ and $1$
    \item Interpret probability in terms of likelihood
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Probability Basics}
\begin{itemize}[leftmargin=*]
    \item Probability measures how likely an event is to happen
    \item $0$ means impossible, $1$ means certain, $0.5$ means equally likely or not
\end{itemize}
\end{conceptBox}

\begin{workedExample}{Probability Scale}
What is the probability of rolling a $7$ on a standard die?

Impossible: $P = 0$
\answerHighlight{$0$}
\end{workedExample}

\tipBox{Probability is always between $0$ and $1$.}

\begin{practiceBox}{Probability}
\resetProblems

\prob What is the probability of flipping heads on a fair coin?
\answerExplain{$0.5$}{There are $2$ outcomes, heads and tails. $1/2 = 0.5$.}

\prob What is the probability of picking a red marble from a bag with $3$ red and $7$ blue marbles?
\answerExplain{$0.3$}{There are $10$ marbles, $3$ are red. $3/10 = 0.3$.}

\prob What does a probability of $1$ mean?
\answerExplain{Certain event}{Probability $1$ means the event will always happen.}

\prob What does a probability of $0$ mean?
\answerExplain{Impossible event}{Probability $0$ means the event cannot happen.}

\prob What does a probability near $0.5$ mean?
\answerExplain{Equally likely or not}{The event is just as likely to happen as not.}

\end{practiceBox}
