% CCSS 7.RP.A.2.C — Writing Equations for Proportional Relationships
\section{Writing Equations for Proportional Relationships}

\begin{learningGoals}
\begin{itemize}
    \item Write equations of the form $y = kx$ for proportional relationships
    \item Use equations to find unknown values
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{The Proportional Equation}
Every proportional relationship can be written as:
$$y = kx$$
Find $k$ (the unit rate), then write the equation.
\begin{center}
\begin{tabular}{l l}
    \textbf{Situation} & \textbf{Equation} \\
    \hline
    Oranges cost $\$0.75$ each & $c = 0.75n$ \\
    A car goes $60$ mph & $d = 60t$ \\
    Earn $\$12$ per hour & $p = 12h$ \\
\end{tabular}
\end{center}
\keyIdea{Once you know $k$, you can predict $y$ for ANY value of $x$.}
\end{conceptBox}

\begin{workedExample}{From a Table to an Equation}
A farmer sells strawberries at a constant price.
\begin{center}
\begin{tabular}{|c|c|}
\hline \textbf{Pounds} ($x$) & \textbf{Cost} ($y$) \\ \hline
$3$ & $\$13.50$ \\ $5$ & $\$22.50$ \\ \hline
\end{tabular}
\end{center}
$k = 13.50 \div 3 = 4.50$, so the equation is $c = 4.50p$.
For $8$ pounds: $c = 4.50 \times 8 = \$36$.
\answerHighlight{$c = 4.50p$}
\end{workedExample}

\mascotSays{An equation is a prediction machine. Plug in any $x$, and out comes the matching $y$!}

\begin{practiceBox}{Writing Equations}
\resetProblems

\prob A bike rental costs $\$8$ per hour. Write an equation for total cost $c$ in terms of hours $h$.
\answerExplain{$c = 8h$}{The rate is $\$8$ per hour, so $k = 8$. Plug into $y = kx$: cost $= 8 \times$ hours, giving $c = 8h$.}

\prob Water flows at $3.5$ gallons per minute. Write an equation for gallons $g$ in terms of minutes $m$.
\answerExplain{$g = 3.5m$}{The unit rate is $3.5$ gallons per minute, so $k = 3.5$. The equation is $g = 3.5m$.}

\prob Use the table to write an equation.
\begin{center}
\begin{tabular}{|c|c|}
\hline $x$ & $y$ \\ \hline $4$ & $10$ \\ $6$ & $15$ \\ $10$ & $25$ \\ \hline
\end{tabular}
\end{center}
\answerExplain{$y = 2.5x$}{Find $k$ by dividing any $y$ by its $x$: $10 \div 4 = 2.5$. Check: $15 \div 6 = 2.5$ \checkmark. Write the equation: $y = 2.5x$.}

\prob A recipe uses $\frac{3}{4}$ cup of flour per serving. How much flour for $12$ servings?
\answerExplain{$9$ cups}{The equation is $f = \frac{3}{4}s$ where $s$ is servings. Plug in $12$: $f = \frac{3}{4} \times 12 = \frac{36}{4} = 9$ cups of flour.}

\prob A cell phone plan charges $\$0.04$ per text. You have $\$10$. How many texts can you send?
\answerExplain{$250$ texts}{Write the equation: $c = 0.04t$. Set $c = 10$ and solve: $10 = 0.04t$, so $t = 10 \div 0.04 = 250$ texts.}

\end{practiceBox}
