% CCSS 7.G.B.6 — Area of a Circle
\section{Area of a Circle}

\begin{learningGoals}
\begin{itemize}
    \item Use $A = \pi r^2$ to find the area of a circle
    \item Understand why the formula works
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Area Formula}
$A = \pi r^2$
\end{conceptBox}

\begin{workedExample}{Finding Area}
Find the area of a circle with radius $5$ cm.

$A = \pi r^2 = 3.14 \times 5^2 = 3.14 \times 25 = 78.5$ cm$^2$
\answerHighlight{$78.5$ cm$^2$}
\end{workedExample}

\tipBox{The area formula comes from rearranging the circle into a shape like a parallelogram.}

\begin{practiceBox}{Area of a Circle}
\resetProblems

\prob Find the area if the radius is $8$ m ($\pi \approx 3.14$).
\answerExplain{$201.06$ m$^2$}{$A = \pi r^2 = 3.14 \times 8^2 = 3.14 \times 64 = 201.06$ m$^2$.}

\prob Find the area if the diameter is $10$ in ($\pi \approx 3.14$).
\answerExplain{$78.5$ in$^2$}{Radius is $5$ in. $A = 3.14 \times 5^2 = 78.5$ in$^2$.}

\prob A circle has area $50.24$ cm$^2$. What is its radius?
\answerExplain{$4$ cm}{$A = \pi r^2$. $r^2 = 50.24/3.14 = 16$, so $r = 4$ cm.}

\prob If the radius is $3$ ft, what is the area?
\answerExplain{$28.26$ ft$^2$}{$A = 3.14 \times 3^2 = 3.14 \times 9 = 28.26$ ft$^2$.}

\prob A pizza has diameter $12$ in. What is its area?
\answerExplain{$113.04$ in$^2$}{Radius is $6$ in. $A = 3.14 \times 6^2 = 113.04$ in$^2$.}

\end{practiceBox}
