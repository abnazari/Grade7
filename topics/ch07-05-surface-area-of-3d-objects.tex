% CCSS 7.G.B.6 — Surface Area of Three-Dimensional Objects
\section{Surface Area of Three-Dimensional Objects}

\begin{learningGoals}
\begin{itemize}
    \item Find the surface area of cubes, prisms, and other 3D objects
    \item Use nets and formulas to calculate total surface area
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Surface Area}
Surface area is the total area covering the outside of a 3D object.
\end{conceptBox}

\begin{workedExample}{Surface Area of a Rectangular Prism}
Find the surface area of a box $4 \times 3 \times 2$ cm.

$2(4 \times 3) + 2(4 \times 2) + 2(3 \times 2) = 2(12) + 2(8) + 2(6) = 24 + 16 + 12 = 52$ cm$^2$
\answerHighlight{$52$ cm$^2$}
\end{workedExample}

\tipBox{Draw a net to see all faces. Add the area of each face for total surface area.}

\begin{practiceBox}{Surface Area}
\resetProblems

\prob Find the surface area of a cube with side $5$ m.
\answerExplain{$150$ m$^2$}{Cube has $6$ faces: $6 \times 5^2 = 6 \times 25 = 150$ m$^2$.}

\prob Find the surface area of a rectangular prism $7 \times 2 \times 3$ in.
\answerExplain{$74$ in$^2$}{$2(7 \times 2) + 2(7 \times 3) + 2(2 \times 3) = 28 + 42 + 12 = 82$ in$^2$.}

\prob A box is $6 \times 4 \times 2$ ft. Find its surface area.
\answerExplain{$88$ ft$^2$}{$2(6 \times 4) + 2(6 \times 2) + 2(4 \times 2) = 48 + 24 + 16 = 88$ ft$^2$.}

\prob Find the surface area of a prism with base $8 \times 3$ cm and height $5$ cm.
\answerExplain{$134$ cm$^2$}{Base: $8 \times 3 = 24$ cm$^2$. Sides: $2 \times 8 \times 5 = 80$ cm$^2$, $2 \times 3 \times 5 = 30$ cm$^2$. Total: $24 + 80 + 30 = 134$ cm$^2$.}

\prob A triangular prism has base $6$ m, height $4$ m, length $10$ m. Find surface area.
\answerExplain{$192$ m$^2$}{Base: $0.5 \times 6 \times 4 = 12$ m$^2$. Sides: $6 \times 10 = 60$ m$^2$, $4 \times 10 = 40$ m$^2$, $5 \times 10 = 50$ m$^2$. Total: $12 + 60 + 40 + 50 = 162$ m$^2$.}

\end{practiceBox}
