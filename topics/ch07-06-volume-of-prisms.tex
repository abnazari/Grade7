% CCSS 7.G.B.6 — Volume of Prisms
\section{Volume of Prisms}

\begin{learningGoals}
\begin{itemize}
    \item Find the volume of cubes and right prisms using $V = Bh$
    \item Apply volume formulas to real-world problems
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Volume Formula}
$V = Bh$ (area of base $\times$ height)
\end{conceptBox}

\begin{workedExample}{Volume of a Rectangular Prism}
Find the volume of a box $5 \times 3 \times 2$ cm.

Base: $5 \times 3 = 15$ cm$^2$
Height: $2$ cm
$V = 15 \times 2 = 30$ cm$^3$
\answerHighlight{$30$ cm$^3$}
\end{workedExample}

\tipBox{Volume measures how much space is inside a 3D object.}

\begin{practiceBox}{Volume}
\resetProblems

\prob Find the volume of a cube with side $4$ m.
\answerExplain{$64$ m$^3$}{Cube: $4^3 = 64$ m$^3$.}

\prob Find the volume of a rectangular prism $6 \times 2 \times 5$ in.
\answerExplain{$60$ in$^3$}{Base: $6 \times 2 = 12$ in$^2$. Height: $5$ in. $V = 12 \times 5 = 60$ in$^3$.}

\prob A box is $8 \times 3 \times 2$ ft. Find its volume.
\answerExplain{$48$ ft$^3$}{Base: $8 \times 3 = 24$ ft$^2$. Height: $2$ ft. $V = 24 \times 2 = 48$ ft$^3$.}

\prob Find the volume of a prism with base $7 \times 2$ cm and height $6$ cm.
\answerExplain{$84$ cm$^3$}{Base: $7 \times 2 = 14$ cm$^2$. Height: $6$ cm. $V = 14 \times 6 = 84$ cm$^3$.}

\prob A triangular prism has base $5$ m, height $3$ m, length $12$ m. Find volume.
\answerExplain{$180$ m$^3$}{Base: $0.5 \times 5 \times 3 = 7.5$ m$^2$. Length: $12$ m. $V = 7.5 \times 12 = 90$ m$^3$.}

\end{practiceBox}
