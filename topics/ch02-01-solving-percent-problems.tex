% CCSS 7.RP.A.3 — Solving Percent Problems
\section{Solving Percent Problems}

\begin{learningGoals}
\begin{itemize}
    \item Find a percent of a number, the whole, or the percent
    \item Solve percent problems using equations and proportions
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Three Types of Percent Problems}
Every percent problem has three parts: the \textbf{part}, the \textbf{whole}, and the \textbf{percent}.
$$\text{part} = \text{percent} \times \text{whole}$$
\begin{itemize}[leftmargin=*]
    \item \textbf{Find the part:} What is $30\%$ of $80$? → $0.30 \times 80 = 24$
    \item \textbf{Find the whole:} $15$ is $25\%$ of what? → $15 \div 0.25 = 60$
    \item \textbf{Find the percent:} $12$ is what percent of $48$? → $12 \div 48 = 0.25 = 25\%$
\end{itemize}
\end{conceptBox}

\begin{workedExample}{Using a Proportion}
$18$ is $40\%$ of what number?

Set up a proportion: $\dfrac{18}{w} = \dfrac{40}{100}$

Cross-multiply: $18 \times 100 = 40 \times w$ → $1800 = 40w$ → $w = 45$
\answerHighlight{The whole is $45$.}
\end{workedExample}

\mascotSays{``Is'' means equals. ``Of'' means multiply. So ``$20\%$ of $50$'' becomes $0.20 \times 50$.}

\begin{practiceBox}{Solving Percent Problems}
\resetProblems

\prob What is $15\%$ of $200$?
\answerExplain{$30$}{Convert the percent to a decimal: $15\% = 0.15$. Multiply: $0.15 \times 200 = 30$.}

\prob $36$ is what percent of $90$?
\answerExplain{$40\%$}{Divide the part by the whole: $36 \div 90 = 0.4$. Convert to a percent by multiplying by $100$: $0.4 \times 100 = 40\%$.}

\prob $27$ is $75\%$ of what number?
\answerExplain{$36$}{You know the part ($27$) and the percent ($75\% = 0.75$). Divide the part by the percent: $27 \div 0.75 = 36$. Check: $0.75 \times 36 = 27$ \checkmark.}

\prob A student got $42$ out of $60$ questions right. What percent is that?
\answerExplain{$70\%$}{Divide the part by the whole: $42 \div 60 = 0.7$. Multiply by $100$ to get the percent: $0.7 \times 100 = 70\%$.}

\prob A store has $240$ items. If $85\%$ are in stock, how many are in stock?
\answerExplain{$204$}{Convert $85\%$ to a decimal: $0.85$. Multiply by the total: $0.85 \times 240 = 204$ items in stock.}

\end{practiceBox}
