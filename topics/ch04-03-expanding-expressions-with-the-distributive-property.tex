% CCSS 7.EE.A.1 — Expanding Expressions with the Distributive Property
\section{Expanding with the Distributive Property}

\begin{learningGoals}
\begin{itemize}
    \item Use the distributive property to expand expressions
    \item Expand with negative and rational coefficients
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{The Distributive Property}
Multiply the number outside the parentheses by \textbf{every term} inside:
$$a(b + c) = ab + ac$$
\begin{itemize}[leftmargin=*]
    \item $3(2x + 5) = 6x + 15$
    \item $-2(4x - 3) = -8x + 6$ \quad (watch the signs!)
\end{itemize}
\end{conceptBox}

\begin{sideBySideExample}{Expanding with a Negative}
Expand $-4(3x - 7)$.
\tcblower
Multiply $-4$ by each term:

$-4 \times 3x = -12x$

$-4 \times (-7) = +28$

Result: $-12x + 28$
\end{sideBySideExample}

\mascotSays{A negative times a negative gives a positive. So $-4(3x - 7)$ gives $-12x + 28$, not $-12x - 28$!}

\begin{practiceBox}{Expanding with the Distributive Property}
\resetProblems

\prob Expand $5(2x + 3)$.
\answerExplain{$10x + 15$}{Multiply $5$ by each term inside: $5 \times 2x = 10x$ and $5 \times 3 = 15$. Result: $10x + 15$.}

\prob Expand $-3(4a - 2)$.
\answerExplain{$-12a + 6$}{Multiply $-3$ by each term: $-3 \times 4a = -12a$ and $-3 \times (-2) = +6$. A negative times a negative is positive.}

\prob Expand $\frac{1}{2}(6x - 10)$.
\answerExplain{$3x - 5$}{Multiply $\frac{1}{2}$ by each term: $\frac{1}{2} \times 6x = 3x$ and $\frac{1}{2} \times (-10) = -5$. Result: $3x - 5$.}

\prob Expand and simplify $2(3x + 1) + 4x$.
\answerExplain{$10x + 2$}{First expand: $2(3x + 1) = 6x + 2$. Then combine like terms: $6x + 2 + 4x = 10x + 2$.}

\prob Expand and simplify $-2(x - 5) + 3(2x + 1)$.
\answerExplain{$4x + 13$}{Expand each part: $-2x + 10 + 6x + 3$. Combine like terms: $(-2x + 6x) + (10 + 3) = 4x + 13$.}

\end{practiceBox}
