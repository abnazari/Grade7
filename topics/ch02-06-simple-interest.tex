% CCSS 7.RP.A.3 — Simple Interest
\section{Simple Interest: Earning and Paying Interest}

\begin{learningGoals}
\begin{itemize}
    \item Use the formula $I = Prt$ to calculate simple interest
    \item Find total amounts for savings and loans
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{The Simple Interest Formula}
\textbf{Simple interest} is money earned on savings or owed on loans.
$$I = Prt$$
\begin{itemize}[leftmargin=*]
    \item $I$ = interest (dollars earned or owed)
    \item $P$ = principal (starting amount)
    \item $r$ = annual rate (as a decimal)
    \item $t$ = time (in years)
\end{itemize}
\textbf{Total amount} = $P + I$
\end{conceptBox}

\begin{workedExample}{Saving Money}
You deposit $\$500$ in a bank that pays $4\%$ simple interest per year. How much interest do you earn in $3$ years? What is the total?

$I = Prt = 500 \times 0.04 \times 3 = \$60$

Total $= 500 + 60 = \$560$
\answerHighlight{You earn $\$60$ in interest. Total: $\$560$.}
\end{workedExample}

\mascotSays{Always convert the percent rate to a decimal first! $4\% = 0.04$, not $4$.}

\begin{practiceBox}{Simple Interest}
\resetProblems

\prob Find the interest: $P = \$800$, $r = 5\%$, $t = 2$ years.
\answerExplain{$\$80$}{Use the formula $I = Prt$. Convert the rate: $5\% = 0.05$. Multiply: $I = 800 \times 0.05 \times 2 = \$80$.}

\prob You borrow $\$1{,}200$ at $6\%$ for $4$ years. How much interest do you owe?
\answerExplain{$\$288$}{Use $I = Prt$. Convert: $6\% = 0.06$. Multiply: $I = 1{,}200 \times 0.06 \times 4 = \$288$. This is the extra amount you pay for borrowing.}

\prob You deposit $\$350$ at $3\%$ for $5$ years. What is the total amount?
\answerExplain{$\$402.50$}{First find the interest: $I = 350 \times 0.03 \times 5 = \$52.50$. Then add to the principal: $350 + 52.50 = \$402.50$.}

\prob A loan of $\$2{,}000$ charges $\$240$ interest over $3$ years. What is the rate?
\answerExplain{$4\%$}{Start with $I = Prt$ and solve for $r$: $240 = 2{,}000 \times r \times 3$. Simplify: $240 = 6{,}000r$. Divide: $r = 240 \div 6{,}000 = 0.04 = 4\%$.}

\prob You earned $\$75$ interest on $\$1{,}500$ at $2.5\%$. How many years?
\answerExplain{$2$ years}{Start with $I = Prt$ and solve for $t$: $75 = 1{,}500 \times 0.025 \times t$. Simplify: $75 = 37.5t$. Divide: $t = 75 \div 37.5 = 2$ years.}

\end{practiceBox}
