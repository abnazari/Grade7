% CCSS 7.RP.A.3 — Percent Increase and Decrease
\section{Percent Increase and Decrease}

\begin{learningGoals}
\begin{itemize}
    \item Calculate percent increase and percent decrease
    \item Use multipliers like $1.05$ and $0.80$ for quick calculations
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Percent Change}
\textbf{Percent increase} and \textbf{percent decrease} describe how much a value changes compared to the original.
$$\text{percent change} = \frac{|\text{new} - \text{original}|}{\text{original}} \times 100$$
\textbf{Shortcut with multipliers:}
\begin{itemize}[leftmargin=*]
    \item Increase by $5\%$: multiply by $1 + 0.05 = 1.05$
    \item Decrease by $20\%$: multiply by $1 - 0.20 = 0.80$
\end{itemize}
\end{conceptBox}

\begin{workedExample}{Using the Multiplier}
A jacket was $\$60$. The price increased by $15\%$. What is the new price?

Multiplier: $1 + 0.15 = 1.15$

New price: $60 \times 1.15 = 69$
\answerHighlight{The new price is $\$69$.}
\end{workedExample}

\begin{errorBox}[Common Mistake]
\wrongAnswer{A $\$50$ item is on sale for $30\%$ off. New price: $50 \times 0.30 = \$15$.}
$0.30$ gives the \textbf{discount amount}, not the sale price.
Correct: $50 \times 0.70 = \$35$ or $50 - 15 = \$35$.
\end{errorBox}

\begin{practiceBox}{Percent Increase and Decrease}
\resetProblems

\prob A town's population grew from $8{,}000$ to $9{,}200$. Find the percent increase.
\answerExplain{$15\%$}{First find the change: $9{,}200 - 8{,}000 = 1{,}200$. Divide by the original: $1{,}200 \div 8{,}000 = 0.15$. Multiply by $100$: the population increased by $15\%$.}

\prob A bike was $\$180$. It is now $25\%$ off. What is the sale price?
\answerExplain{$\$135$}{A $25\%$ discount means you pay $100\% - 25\% = 75\%$ of the price. Multiply: $180 \times 0.75 = \$135$.}

\prob A plant was $12$ cm tall. It grew to $15$ cm. Find the percent increase.
\answerExplain{$25\%$}{Find the change: $15 - 12 = 3$ cm. Divide by the original height: $3 \div 12 = 0.25$. Multiply by $100$: that is a $25\%$ increase.}

\prob What multiplier gives a $40\%$ decrease?
\answerExplain{$0.60$}{A $40\%$ decrease means you keep $100\% - 40\% = 60\%$. Convert to a decimal: $60\% = 0.60$. Multiply the original by $0.60$ to get the result.}

\prob A video game's price dropped from $\$40$ to $\$32$. Find the percent decrease.
\answerExplain{$20\%$}{Find the change: $40 - 32 = 8$. Divide by the original price: $8 \div 40 = 0.2$. Multiply by $100$: the price decreased by $20\%$.}

\end{practiceBox}
