% CCSS 7.RP.A.3 — Markups, Discounts, and Sales Tax
\section{Markups, Discounts, and Sales Tax}

\begin{learningGoals}
\begin{itemize}
    \item Calculate markups, discounts, and sales tax
    \item Find the final price after multiple percent changes
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Markups and Discounts}
\begin{itemize}[leftmargin=*]
    \item A \textbf{markup} is the amount a store adds to the cost. If a store pays $\$20$ and marks up $50\%$: selling price $= 20 \times 1.50 = \$30$.
    \item A \textbf{discount} is the amount subtracted from the price. A $\$30$ item at $20\%$ off: sale price $= 30 \times 0.80 = \$24$.
    \item \textbf{Sales tax} is added to the price at checkout. If tax is $8\%$: total $= \text{price} \times 1.08$.
\end{itemize}
\end{conceptBox}

\begin{workedExample}{Finding a Final Price}
A store buys a shirt for $\$15$ and marks it up $60\%$. A customer buys it during a $25\%$ off sale. Sales tax is $7\%$. What is the final price?

\textbf{Step 1:} Markup: $15 \times 1.60 = \$24$

\textbf{Step 2:} Discount: $24 \times 0.75 = \$18$

\textbf{Step 3:} Tax: $18 \times 1.07 = \$19.26$
\answerHighlight{The final price is $\$19.26$.}
\end{workedExample}

\mascotSays{Apply markup first, then discount, then tax. Tax is always applied \textbf{last}, on the price the customer actually pays.}

\begin{practiceBox}{Markups, Discounts, and Sales Tax}
\resetProblems

\prob A store buys headphones for $\$12$ and marks them up $75\%$. What is the selling price?
\answerExplain{$\$21$}{A $75\%$ markup means the price is $100\% + 75\% = 175\%$ of the cost. Multiply: $12 \times 1.75 = \$21$.}

\prob A $\$65$ jacket is $30\%$ off. What is the sale price?
\answerExplain{$\$45.50$}{A $30\%$ discount means you pay $100\% - 30\% = 70\%$. Multiply: $65 \times 0.70 = \$45.50$.}

\prob A meal costs $\$22$. Sales tax is $6\%$. What is the total?
\answerExplain{$\$23.32$}{Tax is added on top, so multiply by $1 + 0.06 = 1.06$. Total: $22 \times 1.06 = \$23.32$.}

\prob Shoes cost $\$80$, discount $15\%$, tax $8\%$. Find the final price.
\answerExplain{$\$73.44$}{First apply the discount: $80 \times 0.85 = \$68$. Then add tax to the sale price: $68 \times 1.08 = \$73.44$. Always apply tax last.}

\prob A store buys a game for $\$25$ and sells it for $\$40$. What is the percent markup?
\answerExplain{$60\%$}{Find the markup amount: $40 - 25 = \$15$. Divide by the original cost: $15 \div 25 = 0.60$. Multiply by $100$: that is a $60\%$ markup.}

\end{practiceBox}
