\section{Multiplying Integers and Rational Numbers}

\begin{learningGoals}
\begin{itemize}
    \item Apply the sign rules for multiplication
    \item Multiply signed fractions and decimals
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Sign Rules for Multiplication}
\begin{center}
\begin{tabular}{c c c}
    \textbf{Signs} & \textbf{Result} & \textbf{Example} \\
    \hline
    $(+) \times (+)$ & Positive & $3 \times 4 = 12$ \\
    $(-) \times (-)$ & Positive & $(-3)(-4) = 12$ \\
    $(+) \times (-)$ & Negative & $3 \times (-4) = -12$ \\
    $(-) \times (+)$ & Negative & $(-3) \times 4 = -12$ \\
\end{tabular}
\end{center}
\textbf{Same signs → positive. Different signs → negative.}

These rules apply to all rational numbers: integers, fractions, and decimals.
\end{conceptBox}

\begin{workedExample}{Multiplying Signed Fractions}
Find $\left(-\dfrac{2}{3}\right) \times \dfrac{3}{5}$.

Multiply numerators: $2 \times 3 = 6$. Multiply denominators: $3 \times 5 = 15$.

Different signs → negative: $-\dfrac{6}{15} = -\dfrac{2}{5}$
\answerHighlight{$-\dfrac{2}{5}$}
\end{workedExample}

\mascotSays{Count the negative signs. An even number of negatives gives a positive product. An odd number gives a negative product.}

\begin{practiceBox}{Multiplying Integers and Rational Numbers}
\resetProblems

\prob $(-6) \times (-9)$
\answerExplain{$54$}{Multiply the absolute values: $6 \times 9 = 54$. Same signs (both negative) give a positive result: $54$.}

\prob $(-4) \times 7$
\answerExplain{$-28$}{Multiply the absolute values: $4 \times 7 = 28$. Different signs (one negative, one positive) give a negative result: $-28$.}

\prob $\left(-\dfrac{3}{4}\right) \times \left(-\dfrac{2}{5}\right)$
\answerExplain{$\dfrac{3}{10}$}{Multiply numerators: $3 \times 2 = 6$. Multiply denominators: $4 \times 5 = 20$. Both negative → positive: $\frac{6}{20} = \frac{3}{10}$.}

\prob $(-1.5) \times 0.4$
\answerExplain{$-0.6$}{Multiply the absolute values: $1.5 \times 0.4 = 0.6$. Different signs → negative: $-0.6$.}

\prob $(-2) \times (-3) \times (-5)$
\answerExplain{$-30$}{Multiply in steps: $(-2) \times (-3) = 6$ (same signs → positive). Then $6 \times (-5) = -30$ (different signs → negative). Three negatives (odd count) give a negative product.}

\end{practiceBox}
