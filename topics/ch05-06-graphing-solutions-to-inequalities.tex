% CCSS 7.EE.B.4.B — Graphing Solutions to Inequalities
\section{Graphing Inequalities on a Number Line}

\begin{learningGoals}
\begin{itemize}
    \item Graph inequality solutions using open and closed circles
    \item Interpret graphs in real-world contexts
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Graphing Inequality Solutions}
Draw the solution on a number line:
\begin{itemize}[leftmargin=*]
    \item \textbf{Open circle} $\circ$: the endpoint is \textbf{not} included ($<$ or $>$).
    \item \textbf{Closed circle} $\bullet$: the endpoint \textbf{is} included ($\leq$ or $\geq$).
    \item \textbf{Shade} the direction of all solutions.
\end{itemize}
\begin{center}
\begin{tabular}{l l}
    $x > 3$ & open circle at $3$, shade right \\
    $x \leq -1$ & closed circle at $-1$, shade left \\
\end{tabular}
\end{center}
\end{conceptBox}

\begin{workedExample}{Solve and Graph}
Solve $2x - 1 < 5$ and graph the solution.

Add $1$: $2x < 6$. \quad Divide by $2$: $x < 3$.

\numberLine[-1]{-1}{5}

Open circle at $3$, shade everything to the left.
\answerHighlight{$x < 3$}
\end{workedExample}

\mascotSays{Think of the circle as a gate. Open = ``you can't come in.'' Closed = ``you're included!''}

\begin{practiceBox}{Graphing Solutions to Inequalities}
\resetProblems

\prob Solve and describe the graph: $x + 4 \geq 7$.
\answerExplain{$x \geq 3$; closed circle at $3$, shade right}{Subtract $4$: $x \geq 3$. Since it includes $3$ (``or equal to''), use a closed circle at $3$ and shade to the right.}

\prob Solve and describe the graph: $3n - 6 > 0$.
\answerExplain{$n > 2$; open circle at $2$, shade right}{Add $6$: $3n > 6$. Divide by $3$: $n > 2$. Strictly greater than $2$, so use an open circle at $2$ and shade right.}

\prob Solve and describe the graph: $-2x \leq 8$.
\answerExplain{$x \geq -4$; closed circle at $-4$, shade right}{Divide by $-2$ and flip the sign: $x \geq -4$. Closed circle at $-4$, shade right.}

\trueOrFalse{The graph of $x < 5$ uses a closed circle at $5$.}
\answerTF{False}

\prob A parking lot charges $\$3$ per hour. You have $\$20$. Write and solve an inequality for how many hours $h$ you can park. Describe the graph.
\answerExplain{$h \leq 6\frac{2}{3}$; closed circle at $6\frac{2}{3}$, shade left}{Inequality: $3h \leq 20$. Divide by $3$: $h \leq \frac{20}{3} = 6\frac{2}{3}$. You can park at most $6\frac{2}{3}$ hours. Closed circle at $6\frac{2}{3}$, shade left toward $0$.}

\end{practiceBox}
