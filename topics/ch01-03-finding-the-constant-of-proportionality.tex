\section{Finding the Constant of Proportionality}

\begin{learningGoals}
\begin{itemize}
    \item Find the constant of proportionality ($k$) from tables, graphs, and descriptions
    \item Understand that $k$ equals the unit rate
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{The Constant of Proportionality}
In a proportional relationship, every ratio $\dfrac{y}{x}$ equals the same value called $k$:
$$k = \frac{y}{x}$$
$k$ is the \textbf{unit rate} --- how much $y$ changes for every $1$ unit of $x$.
\begin{itemize}[leftmargin=*]
    \item \textbf{From a table:} divide any $y$ by its $x$.
    \item \textbf{From a graph:} read the $y$-value at $x = 1$, or pick any point and compute $\frac{y}{x}$.
    \item \textbf{From words:} divide the two quantities.
\end{itemize}
\end{conceptBox}

\begin{workedExample}{Finding $k$ from a Table}
Apples cost a fixed price per pound.
\begin{center}
\begin{tabular}{|c|c|}
\hline \textbf{Pounds} ($x$) & \textbf{Cost} ($y$) \\ \hline
$2$ & $\$3.50$ \\ $4$ & $\$7.00$ \\ $6$ & $\$10.50$ \\ \hline
\end{tabular}
\end{center}
$k = \frac{3.50}{2} = 1.75$. Check: $\frac{7.00}{4} = 1.75$ \checkmark
\answerHighlight{$k = 1.75$ → apples cost $\$1.75$ per pound.}
\end{workedExample}

\mascotSays{$k$, unit rate, rate of change --- they all mean the same thing. Always include \textbf{units} when you state $k$!}

\begin{practiceBox}{Finding the Constant of Proportionality}
\resetProblems

\prob Find $k$.
\begin{center}
\begin{tabular}{|c|c|}
\hline $x$ & $y$ \\ \hline $3$ & $12$ \\ $5$ & $20$ \\ $8$ & $32$ \\ \hline
\end{tabular}
\end{center}
\answerExplain{$k = 4$}{Pick any row and divide $y$ by $x$: $12 \div 3 = 4$. Check another row: $20 \div 5 = 4$ and $32 \div 8 = 4$. All ratios match, so $k = 4$.}

\prob Find $k$ and explain what it means.
\begin{center}
\begin{tabular}{|c|c|}
\hline \textbf{Hours} & \textbf{Miles} \\ \hline $\frac{1}{2}$ & $30$ \\ $1$ & $60$ \\ $2$ & $120$ \\ \hline
\end{tabular}
\end{center}
\answerExplain{$k = 60$ miles per hour}{Divide miles by hours: $30 \div \frac{1}{2} = 30 \times 2 = 60$. This means the car travels $60$ miles for every $1$ hour. Check: $60 \div 1 = 60$ and $120 \div 2 = 60$ \checkmark.}

\prob A proportional graph passes through $(6, 15)$. Find $k$.
\answerExplain{$k = 2.5$}{On a proportional graph, $k = y \div x$. Use the given point: $k = 15 \div 6 = 2.5$. This means $y$ increases by $2.5$ for every $1$ unit of $x$.}

\prob A printer prints $80$ pages in $5$ minutes at a constant rate. What is $k$?
\answerExplain{$k = 16$ pages/min}{Divide to find the unit rate: $80 \div 5 = 16$. The printer produces $16$ pages every minute.}

\prob The table shows a proportional relationship. Find the missing value.
\begin{center}
\begin{tabular}{|c|c|}
\hline $x$ & $y$ \\ \hline $4$ & $14$ \\ $6$ & $?$ \\ $10$ & $35$ \\ \hline
\end{tabular}
\end{center}
\answerExplain{$y = 21$}{First find $k$: $14 \div 4 = 3.5$. Then use $y = kx$ to find the missing value: $y = 3.5 \times 6 = 21$. Check: $35 \div 10 = 3.5$ \checkmark.}

\end{practiceBox}
