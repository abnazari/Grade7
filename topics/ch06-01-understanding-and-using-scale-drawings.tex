% CCSS 7.G.A.1 — Understanding and Using Scale Drawings
\section{Understanding and Using Scale Drawings}

\begin{learningGoals}
\begin{itemize}
    \item Use scale factors to find actual lengths from a scale drawing
    \item Understand how scale affects area
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Scale Drawings}
A \textbf{scale drawing} is a proportional copy of a real object — every length is multiplied by the same \textbf{scale factor}.
$$\text{actual length} = \text{drawing length} \times \text{scale factor}$$
If the scale is ``$1$ cm $= 5$ m,'' then the scale factor is $5$. A $3$ cm wall on the drawing is $3 \times 5 = 15$ m in real life.

\textbf{Area scales differently:} if lengths scale by $k$, areas scale by $k^2$.
\end{conceptBox}

\begin{workedExample}{Using a Map Scale}
A map uses the scale $1$ cm $= 20$ km. Two cities are $4.5$ cm apart on the map. What is the actual distance?

$4.5 \times 20 = 90$ km
\answerHighlight{The actual distance is $90$ km.}
\end{workedExample}

\mascotSays{Scale drawings keep all shapes proportional. Angles stay the same — only lengths change!}

\begin{practiceBox}{Scale Drawings}
\resetProblems

\prob A blueprint uses the scale $1$ in $= 8$ ft. A room is $2.5$ in wide on the blueprint. What is the actual width?
\answerExplain{$20$ ft}{Multiply drawing length by scale factor: $2.5 \times 8 = 20$ ft.}

\prob On a map, $3$ cm $= 45$ km. What is the scale factor?
\answerExplain{$15$ km per cm}{Divide to find the unit rate: $45 \div 3 = 15$. Each centimeter on the map represents $15$ km in real life.}

\prob A model car is $6$ inches long. The scale is $1:24$. How long is the real car?
\answerExplain{$144$ inches or $12$ ft}{Multiply by the scale factor: $6 \times 24 = 144$ inches. Convert: $144 \div 12 = 12$ feet.}

\prob A rectangle on a blueprint is $2$ cm by $3$ cm. The scale is $1$ cm $= 4$ m. Find the actual area.
\answerExplain{$96$ m$^2$}{Actual dimensions: $2 \times 4 = 8$ m and $3 \times 4 = 12$ m. Area: $8 \times 12 = 96$ m$^2$.}

\trueOrFalse{If a scale factor doubles all lengths, the area also doubles.}
\answerTF{False}

\end{practiceBox}
