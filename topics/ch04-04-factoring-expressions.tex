% CCSS 7.EE.A.1 — Factoring Expressions
\section{Factoring Expressions}

\begin{learningGoals}
\begin{itemize}
    \item Factor linear expressions by finding the greatest common factor
    \item Connect factoring to the distributive property in reverse
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Factoring = Distributive Property in Reverse}
\textbf{Factoring} means rewriting an expression as a product.
Find the \textbf{greatest common factor (GCF)} of all terms, then divide each term by it:
$$6x + 15 = 3(2x + 5)$$
\begin{itemize}[leftmargin=*]
    \item GCF of $6x$ and $15$ is $3$.
    \item Divide: $6x \div 3 = 2x$ and $15 \div 3 = 5$.
    \item Write: $3(2x + 5)$.
\end{itemize}
\textbf{Check by expanding:} $3(2x + 5) = 6x + 15$ \checkmark
\end{conceptBox}

\begin{workedExample}{Factoring with a Negative}
Factor $-8a + 12$.

GCF of $8$ and $12$ is $4$. Factor out $-4$:

$-8a \div (-4) = 2a$ \qquad $12 \div (-4) = -3$

$-4(2a - 3)$

Check: $-4(2a - 3) = -8a + 12$ \checkmark
\answerHighlight{$-4(2a - 3)$}
\end{workedExample}

\tipBox{Always check your factoring by expanding back out. If you get the original expression, you factored correctly!}

\begin{practiceBox}{Factoring Expressions}
\resetProblems

\prob Factor $10x + 25$.
\answerExplain{$5(2x + 5)$}{The GCF of $10$ and $25$ is $5$. Divide each term: $10x \div 5 = 2x$ and $25 \div 5 = 5$. Write: $5(2x + 5)$.}

\prob Factor $12y - 18$.
\answerExplain{$6(2y - 3)$}{The GCF of $12$ and $18$ is $6$. Divide: $12y \div 6 = 2y$ and $-18 \div 6 = -3$. Write: $6(2y - 3)$.}

\prob Factor $14a + 21b$.
\answerExplain{$7(2a + 3b)$}{The GCF of $14$ and $21$ is $7$. Divide: $14a \div 7 = 2a$ and $21b \div 7 = 3b$. Write: $7(2a + 3b)$.}

\prob Factor $-9x - 6$.
\answerExplain{$-3(3x + 2)$}{The GCF is $3$. Factor out $-3$: $-9x \div (-3) = 3x$ and $-6 \div (-3) = 2$. Write: $-3(3x + 2)$. Check: $-3(3x + 2) = -9x - 6$ \checkmark.}

\trueOrFalse{$4x + 9$ can be factored.}
\answerTF{False}

\end{practiceBox}
