% CCSS 7.NS.A.1 — Integers and Their Opposites
\section{Integers and Their Opposites}

\begin{learningGoals}
\begin{itemize}
    \item Identify integers and their opposites on a number line
    \item Find absolute values and understand what they measure
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Integers and Opposites}
\textbf{Integers} are whole numbers and their negatives: $\ldots, -3, -2, -1, 0, 1, 2, 3, \ldots$
Every integer has an \textbf{opposite} — the number that is the same distance from $0$ on the other side.
\begin{itemize}[leftmargin=*]
    \item The opposite of $5$ is $-5$. The opposite of $-3$ is $3$.
    \item A number plus its opposite always equals $0$: $\;5 + (-5) = 0$.
    \item The \textbf{absolute value} $|n|$ is the distance from $0$. It is always positive or zero.
\end{itemize}
\end{conceptBox}

\numberLine[-1]{-5}{5}

\begin{workedExample}{Real-World Opposites}
A diver descends $40$ feet below sea level, then rises $40$ feet.

Descent: $-40$ ft. \quad Rise: $+40$ ft.

$-40 + 40 = 0$ → back at sea level.

\answerHighlight{Opposite actions cancel to $0$.}
\end{workedExample}

\mascotSays{Think of opposites as ``undo'' buttons. Earning $\$20$ and spending $\$20$ puts you right back where you started!}

\begin{practiceBox}{Integers and Their Opposites}
\resetProblems

\prob What is the opposite of $-8$?
\answerExplain{$8$}{The opposite of a negative number is positive. $-8$ is $8$ units left of $0$, so its opposite is $8$ units right of $0$.}

\prob Find $|{-12}|$ and $|7|$.
\answerExplain{$12$ and $7$}{Absolute value measures distance from $0$. $-12$ is $12$ units from $0$, so $|-12| = 12$. $7$ is $7$ units from $0$, so $|7| = 7$.}

\prob A football team gains $6$ yards, then loses $6$ yards. Write this with integers and find the result.
\answerExplain{$0$ yards}{Gain $= +6$, loss $= -6$. Add: $6 + (-6) = 0$. The team is back where it started because the gain and loss are opposites.}

\trueOrFalse{The absolute value of a number can be negative.}
\answerTF{False}

\prob Order from least to greatest: $-4, \; 2, \; -7, \; 0, \; 5$.
\answerExplain{$-7, \; -4, \; 0, \; 2, \; 5$}{On a number line, numbers increase from left to right. $-7$ is farthest left, then $-4$, then $0$, then $2$, then $5$.}

\end{practiceBox}
