\section{Simulating Compound Events}

\begin{learningGoals}
\begin{itemize}
    \item Design and use simulations to model compound events
    \item Use random digits to generate frequencies
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Simulation}
A simulation uses random numbers or physical models to imitate real events.
\end{conceptBox}

\begin{workedExample}{Simulating Dice Rolls}
Simulate rolling two dice $100$ times. How many times do you expect a sum of $7$?

Probability: $1/6$; Expected: $100 \times 1/6 \approx 17$
\answerHighlight{$17$ times}
\end{workedExample}

\tipBox{Simulations help estimate probabilities when calculations are hard.}

\begin{practiceBox}{Simulating Events}
\resetProblems

\prob Simulate flipping a coin $50$ times. How many heads expected?
\answerExplain{$25$}{Probability: $0.5$; $50 \times 0.5 = 25$.}

\prob Simulate rolling a die $60$ times. How many times expect a $3$?
\answerExplain{$10$}{Probability: $1/6$; $60 \times 1/6 = 10$.}

\prob Simulate drawing a red marble from a bag ($3$ red, $7$ blue) $40$ times. How many reds expected?
\answerExplain{$12$}{Probability: $3/10$; $40 \times 0.3 = 12$.}

\prob Simulate picking two cards from a deck $100$ times. How many times expect both to be hearts?
\answerExplain{$5$}{Probability: $13/52 \times 12/51 \approx 0.059$; $100 \times 0.059 = 5.9 \approx 5$.}

\prob Simulate rolling two dice $36$ times. How many times expect a sum of $2$?
\answerExplain{$1$}{Probability: $1/36$; $36 \times 1/36 = 1$.}

\end{practiceBox}
