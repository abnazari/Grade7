% CCSS 7.G.B.5 — Angle Relationships
\section{Angle Relationships}

\begin{learningGoals}
\begin{itemize}
    \item Identify supplementary, complementary, and vertical angle pairs
    \item Write and solve equations to find unknown angles
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Angle Pairs}
\begin{itemize}[leftmargin=*]
    \item \textbf{Complementary angles} add to $90°$. \quad $\angle A + \angle B = 90°$
    \item \textbf{Supplementary angles} add to $180°$. \quad $\angle A + \angle B = 180°$
    \item \textbf{Vertical angles} are formed by two intersecting lines. They are always \textbf{equal}.
    \item \textbf{Adjacent angles} share a side and a vertex.
\end{itemize}
\end{conceptBox}

\begin{workedExample}{Finding an Unknown Angle}
Two angles are supplementary. One angle is $3x + 10$ and the other is $2x$. Find both angles.

$3x + 10 + 2x = 180$

$5x + 10 = 180$ → $5x = 170$ → $x = 34$

Angles: $3(34) + 10 = 112°$ and $2(34) = 68°$.

Check: $112 + 68 = 180°$ \checkmark
\answerHighlight{$112°$ and $68°$}
\end{workedExample}

\tipBox{Vertical angles are always equal. If you spot an X made by two lines, the opposite angles match!}

\begin{practiceBox}{Angle Relationships}
\resetProblems

\prob Two angles are complementary. One is $35°$. Find the other.
\answerExplain{$55°$}{Complementary angles sum to $90°$. Subtract: $90° - 35° = 55°$.}

\prob Two angles are supplementary. One is $128°$. Find the other.
\answerExplain{$52°$}{Supplementary angles sum to $180°$. Subtract: $180° - 128° = 52°$.}

\prob Two vertical angles are $4x + 5$ and $6x - 15$. Find $x$ and the angle measure.
\answerExplain{$x = 10$; angles are $45°$}{Vertical angles are equal: $4x + 5 = 6x - 15$. Subtract $4x$: $5 = 2x - 15$. Add $15$: $20 = 2x$. Divide: $x = 10$. Angle: $4(10) + 5 = 45°$.}

\prob An angle and its complement are in the ratio $2:3$. Find both angles.
\answerExplain{$36°$ and $54°$}{Let the angles be $2x$ and $3x$. They sum to $90°$: $2x + 3x = 90$, so $5x = 90$ and $x = 18$. Angles: $2(18) = 36°$ and $3(18) = 54°$.}

\prob Two supplementary angles are $(x + 30)°$ and $(2x)°$. Find both.
\answerExplain{$80°$ and $100°$}{$x + 30 + 2x = 180$. Combine: $3x + 30 = 180$. Subtract $30$: $3x = 150$. Divide: $x = 50$. Angles: $50 + 30 = 80°$ and $2(50) = 100°$.}

\end{practiceBox}
