% CCSS 7.NS.A.1.D — Adding and Subtracting Rational Numbers
\section{Adding and Subtracting Rational Numbers}

\begin{learningGoals}
\begin{itemize}
    \item Add and subtract positive and negative fractions and decimals
    \item Apply integer rules to all rational numbers
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Same Rules, New Numbers}
The same sign rules from integers work for \textbf{all rational numbers} — fractions and decimals too.
\begin{itemize}[leftmargin=*]
    \item \textbf{Same signs:} Add absolute values, keep the sign.
    \item \textbf{Different signs:} Subtract absolute values, keep the sign of the larger.
    \item For fractions, find a \textbf{common denominator} first, then apply the rules.
\end{itemize}
\end{conceptBox}

\begin{workedExample}{Adding Signed Fractions}
Find $-\dfrac{2}{3} + \dfrac{1}{4}$.

Common denominator: $12$.

$-\dfrac{8}{12} + \dfrac{3}{12}$

Different signs → subtract: $8 - 3 = 5$. Larger absolute value is negative.

\answerHighlight{$-\dfrac{5}{12}$}
\end{workedExample}

\mascotSays{Convert to a common denominator first, then treat the numerators like integers. The sign rules are exactly the same!}

\begin{practiceBox}{Adding and Subtracting Rational Numbers}
\resetProblems

\prob $-3.5 + 1.8$
\answerExplain{$-1.7$}{Different signs, so subtract the absolute values: $3.5 - 1.8 = 1.7$. The larger absolute value ($3.5$) is negative, so the answer is $-1.7$.}

\prob $-\dfrac{1}{2} - \dfrac{3}{4}$
\answerExplain{$-\dfrac{5}{4}$ or $-1\dfrac{1}{4}$}{Rewrite subtraction as addition: $-\frac{1}{2} + (-\frac{3}{4})$. Common denominator $4$: $-\frac{2}{4} + (-\frac{3}{4})$. Same signs, add: $\frac{2+3}{4} = \frac{5}{4}$. Both negative, so $-\frac{5}{4}$.}

\prob $4.2 + (-6.7)$
\answerExplain{$-2.5$}{Different signs, so subtract: $6.7 - 4.2 = 2.5$. The larger absolute value ($6.7$) is negative, so the answer is $-2.5$.}

\prob $\dfrac{5}{6} - \left(-\dfrac{1}{3}\right)$
\answerExplain{$\dfrac{7}{6}$ or $1\dfrac{1}{6}$}{Subtracting a negative means add: $\frac{5}{6} + \frac{1}{3}$. Common denominator $6$: $\frac{5}{6} + \frac{2}{6} = \frac{7}{6} = 1\frac{1}{6}$.}

\prob A diver is at $-12.5$ m. She goes up $4.8$ m, then down $3.2$ m. What is her position?
\answerExplain{$-10.9$ m}{Start at $-12.5$. Go up: $-12.5 + 4.8 = -7.7$. Go down: $-7.7 + (-3.2) = -10.9$ m.}

\end{practiceBox}
