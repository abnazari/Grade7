% CCSS 7.NS.A.2.B — Dividing Integers and Rational Numbers
\section{Dividing Integers and Rational Numbers}

\begin{learningGoals}
\begin{itemize}
    \item Apply the sign rules for division
    \item Divide signed fractions and decimals
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Sign Rules for Division}
Division follows the \textbf{same sign rules} as multiplication:
\begin{itemize}[leftmargin=*]
    \item \textbf{Same signs → positive:} $(-12) \div (-3) = 4$
    \item \textbf{Different signs → negative:} $12 \div (-3) = -4$
\end{itemize}
A negative sign can go in front of the fraction, on top, or on bottom — they are all equal:
$$-\frac{p}{q} = \frac{-p}{q} = \frac{p}{-q}$$
\textbf{You can never divide by zero.} $\;5 \div 0$ is undefined.
\end{conceptBox}

\begin{workedExample}{Dividing Signed Fractions}
Find $-\dfrac{3}{4} \div \dfrac{2}{3}$.

Flip the second fraction and multiply: $-\dfrac{3}{4} \times \dfrac{3}{2} = -\dfrac{9}{8}$

Different signs → negative.
\answerHighlight{$-\dfrac{9}{8}$ or $-1\dfrac{1}{8}$}
\end{workedExample}

\tipBox{Dividing by a fraction is the same as multiplying by its reciprocal. Then use the sign rules you already know!}

\begin{practiceBox}{Dividing Integers and Rational Numbers}
\resetProblems

\prob $(-48) \div (-6)$
\answerExplain{$8$}{Divide the absolute values: $48 \div 6 = 8$. Same signs (both negative) give a positive result: $8$.}

\prob $35 \div (-7)$
\answerExplain{$-5$}{Divide the absolute values: $35 \div 7 = 5$. Different signs → negative: $-5$.}

\prob $\dfrac{5}{6} \div \left(-\dfrac{1}{3}\right)$
\answerExplain{$-\dfrac{5}{2}$ or $-2\dfrac{1}{2}$}{Flip and multiply: $\frac{5}{6} \times \frac{3}{1} = \frac{15}{6} = \frac{5}{2}$. Different signs → negative: $-\frac{5}{2}$.}

\prob $(-4.8) \div 1.2$
\answerExplain{$-4$}{Divide the absolute values: $4.8 \div 1.2 = 4$. Different signs → negative: $-4$.}

\prob A submarine descends $240$ m in $8$ minutes at a constant rate. Write the rate as a signed number.
\answerExplain{$-30$ m/min}{Descending means negative depth: $-240 \div 8 = -30$. The submarine goes down $30$ meters each minute.}

\end{practiceBox}
