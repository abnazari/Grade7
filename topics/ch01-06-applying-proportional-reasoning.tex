\section{Applying Proportional Reasoning}

\begin{learningGoals}
\begin{itemize}
    \item Solve real-world proportion problems using the best strategy
    \item Choose between unit rate, proportion, or equation approaches
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Your Proportional Reasoning Toolbox}
\begin{center}
\begin{tabular}{c l}
    \textbf{1.} & \textbf{Unit rate} --- find the rate for $1$ unit, then multiply \\
    \textbf{2.} & \textbf{Proportion} --- set up $\frac{a}{b} = \frac{c}{d}$ and cross-multiply \\
    \textbf{3.} & \textbf{Equation} --- use $y = kx$ \\
    \textbf{4.} & \textbf{Scale factor} --- multiply or divide to scale up/down \\
\end{tabular}
\end{center}
Pick whichever feels most natural. They all give the same answer.
\end{conceptBox}

\begin{workedExample}{Unit Rate Strategy}
$5$ T-shirts cost $\$42.50$. How much for $8$?
\solutionLabel
$\$42.50 \div 5 = \$8.50$ per shirt. $\$8.50 \times 8 = \$68$.
\answerHighlight{$\$68.00$}
\end{workedExample}

\begin{workedExample}{Proportion Strategy}
A map scale: $2$ cm $= 15$ mi. Two cities are $7$ cm apart. Actual distance?
\solutionLabel
$\frac{2}{15} = \frac{7}{d}$ → $2d = 105$ → $d = 52.5$ miles.
\answerHighlight{$52.5$ miles}
\end{workedExample}

\mascotSays{Not sure which strategy? Try the unit rate first --- it works for almost everything!}

\tipBox{After solving, ask: does the answer make sense? Buying MORE should cost MORE. Always check your units.}

\begin{practiceBox}{Applying Proportional Reasoning}
\resetProblems

\prob A car travels $210$ miles on $6$ gallons. How far on $10$ gallons?
\answerExplain{$350$ miles}{Find the unit rate: $210 \div 6 = 35$ miles per gallon. Multiply by $10$: $35 \times 10 = 350$ miles.}

\prob A recipe for $4$ people uses $\frac{3}{2}$ cups of rice. How much for $10$ people?
\answerExplain{$3\frac{3}{4}$ cups}{Find rice per person: $\frac{3}{2} \div 4 = \frac{3}{8}$ cup. Multiply by $10$: $\frac{3}{8} \times 10 = \frac{30}{8} = 3\frac{3}{4}$ cups.}

\prob On a map, $3$ cm $= 24$ km. How many km is $8$ cm?
\answerExplain{$64$ km}{Set up a proportion: $\frac{3}{24} = \frac{8}{d}$. Cross-multiply: $3d = 24 \times 8 = 192$. Divide: $d = 192 \div 3 = 64$ km.}

\prob A printer prints $120$ pages in $8$ min. How long for $450$ pages?
\answerExplain{$30$ min}{Find the unit rate: $120 \div 8 = 15$ pages per minute. Divide: $450 \div 15 = 30$ minutes.}

\prob Small paint: $\frac{3}{4}$ gal for $\$18$. Large: $1\frac{1}{2}$ gal for $\$33$. Which is the better deal?
\answerExplain{Large by $\$2$/gal}{Find each unit price. Small: $18 \div 0.75 = \$24$ per gallon. Large: $33 \div 1.5 = \$22$ per gallon. The large can costs $\$2$ less per gallon, so it is the better deal.}

\end{practiceBox}
