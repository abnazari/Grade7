\section{Solving Multi-Step Problems with Rational Numbers}

\begin{learningGoals}
\begin{itemize}
    \item Solve multi-step equations with fractions and decimals
    \item Use estimation to check that answers are reasonable
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Handling Fractions and Decimals}
Some equations have fraction or decimal coefficients. Two strategies:
\begin{itemize}[leftmargin=*]
    \item \textbf{Multiply to clear fractions:} Multiply every term by the LCD.
    
    $\frac{x}{3} + \frac{1}{2} = 5$ → multiply by $6$: $2x + 3 = 30$
    \item \textbf{Work with decimals directly:} Use standard operations.
    
    $0.25x + 1.5 = 4$ → subtract $1.5$: $0.25x = 2.5$ → $x = 10$
\end{itemize}
\end{conceptBox}

\begin{workedExample}{Clearing Fractions}
Solve $\dfrac{2x}{5} - 3 = 7$.

Add $3$: $\dfrac{2x}{5} = 10$

Multiply by $5$: $2x = 50$

Divide by $2$: $x = 25$

Check: $\frac{2(25)}{5} - 3 = 10 - 3 = 7$ \checkmark
\answerHighlight{$x = 25$}
\end{workedExample}

\tipBox{Before solving, estimate the answer. If the equation says $0.5x + 3 = 10$, then $x$ should be around $14$. If you get $140$, you probably made a decimal mistake!}

\begin{practiceBox}{Multi-Step Problems with Rational Numbers}
\resetProblems

\prob Solve $0.3x + 2.1 = 5.1$.
\answerExplain{$x = 10$}{Subtract $2.1$: $0.3x = 3.0$. Divide by $0.3$: $x = 10$. Check: $0.3(10) + 2.1 = 3 + 2.1 = 5.1$ \checkmark.}

\prob Solve $\dfrac{x}{4} + \dfrac{3}{4} = 5$.
\answerExplain{$x = 17$}{Multiply every term by $4$: $x + 3 = 20$. Subtract $3$: $x = 17$. Check: $\frac{17}{4} + \frac{3}{4} = \frac{20}{4} = 5$ \checkmark.}

\prob Solve $-1.5y + 4 = -8$.
\answerExplain{$y = 8$}{Subtract $4$: $-1.5y = -12$. Divide by $-1.5$: $y = 8$. Check: $-1.5(8) + 4 = -12 + 4 = -8$ \checkmark.}

\prob Solve $\dfrac{3n}{2} - 5 = 10$.
\answerExplain{$n = 10$}{Add $5$: $\frac{3n}{2} = 15$. Multiply by $2$: $3n = 30$. Divide by $3$: $n = 10$. Check: $\frac{3(10)}{2} - 5 = 15 - 5 = 10$ \checkmark.}

\prob A store marks down shirts by $\frac{1}{3}$ of the price, then takes off another $\$4$. The final price is $\$12$. What was the original price?
\answerExplain{$\$24$}{Let $p$ = original. After markdown: $p - \frac{p}{3} = \frac{2p}{3}$. After $\$4$ off: $\frac{2p}{3} - 4 = 12$. Add $4$: $\frac{2p}{3} = 16$. Multiply by $\frac{3}{2}$: $p = 24$.}

\end{practiceBox}
