% CCSS 7.RP.A.3 — Percent Error
\section{Percent Error: How Close Are Your Estimates?}

\begin{learningGoals}
\begin{itemize}
    \item Calculate percent error using the formula
    \item Apply percent error to measurement and estimation problems
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Percent Error}
\textbf{Percent error} measures how far off an estimate or measurement is from the actual value.
$$\text{percent error} = \frac{|\text{estimated} - \text{actual}|}{\text{actual}} \times 100$$
The absolute value $|\,|$ means the answer is always positive --- it doesn't matter if you guessed too high or too low.
\end{conceptBox}

\begin{workedExample}{Finding Percent Error}
You estimated a room is $12$ feet long. The actual length is $14$ feet.

$\text{percent error} = \dfrac{|12 - 14|}{14} \times 100 = \dfrac{2}{14} \times 100 \approx 14.3\%$
\answerHighlight{The percent error is about $14.3\%$.}
\end{workedExample}

\begin{realWorld}[Why Percent Error Matters]
Scientists always report percent error in experiments. A chemist who predicts a reaction will produce $50$ g but gets $47$ g has a percent error of $6\%$. The smaller the percent error, the better the estimate.
\end{realWorld}

\begin{practiceBox}{Percent Error}
\resetProblems

\prob You guessed $30$ jelly beans in a jar. The actual count is $25$. Find the percent error.
\answerExplain{$20\%$}{Find the difference: $|30 - 25| = 5$. Divide by the actual value: $5 \div 25 = 0.2$. Multiply by $100$: the percent error is $20\%$.}

\prob A student estimated a book weighs $2$ pounds. It actually weighs $2.5$ pounds. Find the percent error.
\answerExplain{$20\%$}{Find the difference: $|2 - 2.5| = 0.5$. Divide by the actual: $0.5 \div 2.5 = 0.2$. Multiply by $100$: $20\%$ error. The absolute value makes the answer positive even though the estimate was too low.}

\prob A recipe calls for $3$ cups of flour. You used $3.2$ cups. What is your percent error?
\answerExplain{$\approx 6.7\%$}{Find the difference: $|3.2 - 3| = 0.2$. Divide by the actual: $0.2 \div 3 = 0.0\overline{6}$. Multiply by $100$: about $6.7\%$ error.}

\prob A builder estimated a project would cost $\$5{,}000$. The actual cost was $\$4{,}600$. Find the percent error.
\answerExplain{$\approx 8.7\%$}{Find the difference: $|5{,}000 - 4{,}600| = 400$. Divide by the actual value: $400 \div 4{,}600 \approx 0.087$. Multiply by $100$: about $8.7\%$ error.}

\trueOrFalse{If your estimate is exactly correct, the percent error is $0\%$.}
\answerTF{True}

\end{practiceBox}
