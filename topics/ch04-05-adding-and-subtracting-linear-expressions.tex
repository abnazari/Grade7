% CCSS 7.EE.A.1 — Adding and Subtracting Linear Expressions
\section{Adding and Subtracting Linear Expressions}

\begin{learningGoals}
\begin{itemize}
    \item Add linear expressions by combining like terms
    \item Subtract linear expressions by distributing the negative sign
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Adding and Subtracting Expressions}
\textbf{Adding:} Remove parentheses and combine like terms.
$$(3x + 5) + (2x - 4) = 5x + 1$$
\textbf{Subtracting:} Distribute the negative to \textbf{every term} in the second expression, then combine.
$$(3x + 5) - (2x - 4) = 3x + 5 - 2x + 4 = x + 9$$
\keyIdea{Subtracting flips every sign in the second expression.}
\end{conceptBox}

\begin{workedExample}{Subtracting Expressions}
Simplify $(7a - 3) - (4a + 5)$.

Distribute the negative: $7a - 3 - 4a - 5$

Combine like terms: $(7a - 4a) + (-3 - 5) = 3a - 8$
\answerHighlight{$3a - 8$}
\end{workedExample}

\begin{errorBox}[Don't Forget the Second Term]
\wrongAnswer{$(5x + 2) - (3x - 1) = 5x + 2 - 3x - 1 = 2x + 1$}
The negative must flip \textbf{both} signs: $-3x$ and $+1$.
Correct: $5x + 2 - 3x + 1 = 2x + 3$.
\end{errorBox}

\begin{practiceBox}{Adding and Subtracting Linear Expressions}
\resetProblems

\prob $(4x + 6) + (3x - 2)$
\answerExplain{$7x + 4$}{Remove parentheses: $4x + 6 + 3x - 2$. Combine like terms: $(4x + 3x) + (6 - 2) = 7x + 4$.}

\prob $(8n - 5) - (3n + 7)$
\answerExplain{$5n - 12$}{Distribute the negative: $8n - 5 - 3n - 7$. Combine: $(8n - 3n) + (-5 - 7) = 5n - 12$.}

\prob $(2y + 9) - (2y - 1)$
\answerExplain{$10$}{Distribute the negative: $2y + 9 - 2y + 1$. The variable terms cancel: $0y + 10 = 10$. The answer is just a constant.}

\prob $(-x + 4) + (6x - 8)$
\answerExplain{$5x - 4$}{Combine: $(-x + 6x) + (4 - 8) = 5x - 4$.}

\prob Perimeter of a triangle with sides $2x + 1$, $3x - 4$, and $x + 5$.
\answerExplain{$6x + 2$}{Add all three sides: $(2x + 1) + (3x - 4) + (x + 5)$. Combine: $(2x + 3x + x) + (1 - 4 + 5) = 6x + 2$.}

\end{practiceBox}
