\section{Tips, Commissions, and Fees}

\begin{learningGoals}
\begin{itemize}
    \item Calculate tips, commissions, and fees using percents
    \item Estimate tips mentally using benchmarks
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Percent in Everyday Money}
\begin{itemize}[leftmargin=*]
    \item A \textbf{tip} (gratuity) is extra money you pay for service. Common tips: $15\%$--$20\%$.
    \item A \textbf{commission} is a percent of the total sales a worker earns. A realtor earning $3\%$ commission on a $\$200{,}000$ sale gets $\$6{,}000$.
    \item A \textbf{fee} is a flat or percent charge for a service, like a $2.5\%$ transaction fee.
\end{itemize}
\end{conceptBox}

\begin{workedExample}{Quick Tip Estimation}
Your restaurant bill is $\$47.80$. Estimate a $20\%$ tip.

Round to $\$48$. Find $10\%$: $\$4.80$. Double it: $\$9.60$.
\answerHighlight{A $20\%$ tip is about $\$9.60$.}
\end{workedExample}

\tipBox{To estimate any tip: find $10\%$ by moving the decimal one place left. For $15\%$, add half of $10\%$. For $20\%$, double the $10\%$.}

\begin{practiceBox}{Tips, Commissions, and Fees}
\resetProblems

\prob A dinner bill is $\$35$. You leave a $18\%$ tip. How much is the tip?
\answerExplain{$\$6.30$}{Convert the percent to a decimal: $18\% = 0.18$. Multiply by the bill: $35 \times 0.18 = \$6.30$.}

\prob A salesperson earns $6\%$ commission. She sells $\$4{,}500$ worth of electronics. What is her commission?
\answerExplain{$\$270$}{Commission means she earns $6\%$ of total sales. Convert: $6\% = 0.06$. Multiply: $4{,}500 \times 0.06 = \$270$.}

\prob A ticket website charges a $12\%$ service fee. Tickets cost $\$75$. What is the total?
\answerExplain{$\$84$}{The fee is added on top, so you pay $100\% + 12\% = 112\%$. Multiply: $75 \times 1.12 = \$84$.}

\prob Estimate a $15\%$ tip on a $\$62$ bill using the $10\%$ method.
\answerExplain{About $\$9.30$}{Find $10\%$ by moving the decimal: $\$6.20$. Find $5\%$ by taking half of that: $\$3.10$. Add them: $6.20 + 3.10 = \$9.30$. That is $15\%$.}

\prob A real estate agent earns $4\%$ commission on a $\$350{,}000$ house. How much does she earn?
\answerExplain{$\$14{,}000$}{Convert the percent: $4\% = 0.04$. Multiply by the sale price: $350{,}000 \times 0.04 = \$14{,}000$.}

\end{practiceBox}
