% CCSS 7.EE.B.4.A — Writing Two-Step Equations from Word Problems
\section{Writing Two-Step Equations}

\begin{learningGoals}
\begin{itemize}
    \item Translate real-world problems into equations of the form $px + q = r$
    \item Identify the variable, rate, and starting value in a situation
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{From Words to Equations}
A \textbf{two-step equation} has a variable that is multiplied \textbf{and} added to (or subtracted from):
$$px + q = r$$
\begin{center}
\begin{tabular}{l l}
    \textbf{Situation} & \textbf{Equation} \\
    \hline
    Earn $\$8$/hr plus $\$20$ bonus, total $\$100$ & $8h + 20 = 100$ \\
    $3$ pizzas plus $\$5$ delivery $= \$32$ & $3p + 5 = 32$ \\
    Spend $\$4$/day from $\$50$, left with $\$18$ & $50 - 4d = 18$ \\
\end{tabular}
\end{center}
\end{conceptBox}

\begin{workedExample}{Setting Up the Equation}
A gym charges $\$25$ to join plus $\$10$ per month. You have spent $\$85$ total. How many months have you been a member?

Let $m =$ months. Total: $10m + 25 = 85$.
\answerHighlight{$10m + 25 = 85$}
\end{workedExample}

\mascotSays{Look for the rate (per something) — that goes with the variable. Look for the fixed amount — that is the constant.}

\begin{practiceBox}{Writing Two-Step Equations}
\resetProblems

\prob You buy $n$ notebooks at $\$3$ each and a $\$7$ binder. The total is $\$22$. Write an equation.
\answerExplain{$3n + 7 = 22$}{Each notebook costs $\$3$, so $n$ notebooks cost $3n$. Add the binder: $3n + 7$. Set equal to the total: $3n + 7 = 22$.}

\prob A taxi charges $\$2.50$ per mile plus a $\$3$ base fare. The ride costs $\$18$. Write an equation for the miles $m$.
\answerExplain{$2.50m + 3 = 18$}{The per-mile charge is $2.50m$. Add the base fare: $2.50m + 3$. Set equal to the total cost: $2.50m + 3 = 18$.}

\prob You have $\$60$. You spend $\$8$ per week. After how many weeks will you have $\$12$ left? Write an equation.
\answerExplain{$60 - 8w = 12$}{Start with $\$60$ and subtract $\$8$ each week: $60 - 8w$. You want $\$12$ left: $60 - 8w = 12$.}

\prob A phone plan charges $\$0.05$ per text plus $\$15$ monthly. The bill is $\$21.50$. Write an equation for texts $t$.
\answerExplain{$0.05t + 15 = 21.50$}{Text cost: $0.05t$. Add monthly fee: $0.05t + 15$. Set equal to bill: $0.05t + 15 = 21.50$.}

\prob Write an equation: ``Five times a number minus $9$ equals $31$.''
\answerExplain{$5n - 9 = 31$}{``Five times a number'' is $5n$. ``Minus $9$'' gives $5n - 9$. ``Equals $31$'' gives $5n - 9 = 31$.}

\end{practiceBox}
