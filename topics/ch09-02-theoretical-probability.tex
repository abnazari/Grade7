\section{Theoretical Probability}

\begin{learningGoals}
\begin{itemize}
    \item Calculate theoretical probability using favorable/total outcomes
    \item Develop uniform probability models
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Theoretical Probability}
$P = \frac{\text{favorable outcomes}}{\text{total outcomes}}$
\end{conceptBox}

\begin{workedExample}{Rolling a Die}
What is the probability of rolling a $4$ on a standard die?

$P = 1/6 \approx 0.167$
\answerHighlight{$1/6$ or $0.167$}
\end{workedExample}

\tipBox{All outcomes must be equally likely for theoretical probability.}

\begin{practiceBox}{Theoretical Probability}
\resetProblems

\prob What is the probability of drawing a heart from a standard deck of $52$ cards?
\answerExplain{$1/4$}{There are $13$ hearts out of $52$ cards. $13/52 = 1/4$.}

\prob What is the probability of rolling an even number on a die?
\answerExplain{$0.5$}{Even numbers: $2, 4, 6$ ($3$ out of $6$). $3/6 = 0.5$.}

\prob What is the probability of picking a green marble from $5$ green and $15$ yellow marbles?
\answerExplain{$0.25$}{$5$ out of $20$ marbles. $5/20 = 0.25$.}

\prob What is the probability of drawing a $2$ or $3$ from a deck of cards?
\answerExplain{$2/13$}{Each suit has $1$ of each, so $2$ out of $13$ per suit.}

\prob What is the probability of rolling a number greater than $4$ on a die?
\answerExplain{$1/3$}{Numbers $5$ and $6$ ($2$ out of $6$). $2/6 = 1/3$.}

\end{practiceBox}
