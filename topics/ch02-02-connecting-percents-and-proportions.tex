% CCSS 7.RP.A.3 — Connecting Percents and Proportions
\section{Connecting Percents and Proportions}

\begin{learningGoals}
\begin{itemize}
    \item Set up proportions to solve percent problems
    \item Understand that percent is a ratio with denominator $100$
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Percent as a Proportion}
``Percent'' means ``per hundred.'' Every percent problem is a proportion:
$$\frac{\text{part}}{\text{whole}} = \frac{\text{percent}}{100}$$
To solve, \textbf{cross-multiply} and divide.
\end{conceptBox}

\begin{workedExample}{Setting Up the Proportion}
A survey found that $33$ out of $150$ students ride the bus. What percent is that?

$\dfrac{33}{150} = \dfrac{p}{100}$ → $33 \times 100 = 150 \times p$ → $3300 = 150p$ → $p = 22$
\answerHighlight{$22\%$ of students ride the bus.}
\end{workedExample}

\begin{sideBySideExample}{Choosing the Method}
$65\%$ of a class of $40$ earned an A. How many students earned an A?
\tcblower
\textbf{Proportion:} $\dfrac{n}{40} = \dfrac{65}{100}$

$100n = 2600$ → $n = 26$

\textbf{Equation:} $n = 0.65 \times 40 = 26$ \checkmark

Both methods give the same answer.
\end{sideBySideExample}

\tipBox{The proportion method works every time: put the part over the whole on the left, and the percent over $100$ on the right. Cross-multiply to solve.}

\begin{practiceBox}{Connecting Percents and Proportions}
\resetProblems

\prob Set up a proportion and solve: What is $45\%$ of $80$?
\answerExplain{$36$}{Set up the proportion: $\frac{n}{80} = \frac{45}{100}$. Cross-multiply: $100n = 80 \times 45 = 3600$. Divide both sides by $100$: $n = 36$.}

\prob $54$ is what percent of $120$? Use a proportion.
\answerExplain{$45\%$}{Set up the proportion with the percent unknown: $\frac{54}{120} = \frac{p}{100}$. Cross-multiply: $120p = 54 \times 100 = 5400$. Divide: $p = 5400 \div 120 = 45$, so $45\%$.}

\prob $78$ is $60\%$ of what number?
\answerExplain{$130$}{Set up the proportion with the whole unknown: $\frac{78}{w} = \frac{60}{100}$. Cross-multiply: $60w = 78 \times 100 = 7800$. Divide: $w = 7800 \div 60 = 130$.}

\prob In a bag of $250$ marbles, $35$ are red. What percent are red?
\answerExplain{$14\%$}{Set up the proportion: $\frac{35}{250} = \frac{p}{100}$. Cross-multiply: $250p = 3500$. Divide: $p = 3500 \div 250 = 14$, so $14\%$ of the marbles are red.}

\prob A school has $480$ students. If $72\%$ brought lunch, how many brought lunch?
\answerExplain{$345.6 \approx 346$ students}{Set up: $\frac{n}{480} = \frac{72}{100}$. Cross-multiply: $100n = 480 \times 72 = 34{,}560$. Divide: $n = 345.6$. Round to $346$ since you can't have a partial student.}

\end{practiceBox}
