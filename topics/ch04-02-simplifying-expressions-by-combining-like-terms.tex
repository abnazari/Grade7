% CCSS 7.EE.A.1 — Simplifying Expressions by Combining Like Terms
\section{Combining Like Terms}

\begin{learningGoals}
\begin{itemize}
    \item Identify like terms in an expression
    \item Simplify expressions by combining like terms
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Like Terms}
\textbf{Like terms} have the same variable raised to the same power.
\begin{itemize}[leftmargin=*]
    \item $3x$ and $7x$ are like terms → combine to $10x$
    \item $5y$ and $5y^2$ are \textbf{not} like terms (different powers)
    \item Constants like $4$ and $-9$ are always like terms → combine to $-5$
\end{itemize}
The number in front of a variable is the \textbf{coefficient}. In $7x$, the coefficient is $7$.
\end{conceptBox}

\begin{workedExample}{Simplifying Step by Step}
Simplify $4x + 3 - 2x + 8$.

Group like terms: $(4x - 2x) + (3 + 8)$

Combine: $2x + 11$
\answerHighlight{$2x + 11$}
\end{workedExample}

\begin{errorBox}[Can't Combine Unlike Terms]
\wrongAnswer{$3x + 5 = 8x$}
$3x$ is a variable term and $5$ is a constant. They are not like terms.
Correct: $3x + 5$ is already simplified.
\end{errorBox}

\begin{practiceBox}{Combining Like Terms}
\resetProblems

\prob Simplify $6a + 2a - 3$.
\answerExplain{$8a - 3$}{$6a$ and $2a$ are like terms: $6a + 2a = 8a$. The $-3$ has no like term, so the result is $8a - 3$.}

\prob Simplify $5x - 9 + 3x + 4$.
\answerExplain{$8x - 5$}{Group like terms: $(5x + 3x) + (-9 + 4) = 8x + (-5) = 8x - 5$.}

\prob Simplify $-2y + 7 + 6y - 10$.
\answerExplain{$4y - 3$}{Combine variable terms: $-2y + 6y = 4y$. Combine constants: $7 + (-10) = -3$. Result: $4y - 3$.}

\prob Simplify $\frac{1}{2}n + \frac{3}{4}n - 2$.
\answerExplain{$\frac{5}{4}n - 2$ or $1\frac{1}{4}n - 2$}{Find a common denominator for the coefficients: $\frac{2}{4}n + \frac{3}{4}n = \frac{5}{4}n$. The constant stays: $\frac{5}{4}n - 2$.}

\trueOrFalse{$4x + 4y$ can be simplified to $8xy$.}
\answerTF{False}

\end{practiceBox}
