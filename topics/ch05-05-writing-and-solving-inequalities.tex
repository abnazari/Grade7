% CCSS 7.EE.B.4.B — Writing and Solving Inequalities
\section{Writing and Solving Inequalities}

\begin{learningGoals}
\begin{itemize}
    \item Write inequalities from word problems
    \item Solve two-step inequalities, including when to flip the sign
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Inequalities}
An \textbf{inequality} compares two expressions using $<$, $>$, $\leq$, or $\geq$.

Solve just like equations, with \textbf{one key rule}:
\keyIdea{When you multiply or divide by a \textbf{negative} number, \textbf{flip} the inequality sign.}
$$-2x > 6 \;\;\xrightarrow{\div (-2)}\;\; x < -3$$
\begin{center}
\begin{tabular}{l l}
    \textbf{Words} & \textbf{Symbol} \\
    \hline
    at least, no less than & $\geq$ \\
    at most, no more than & $\leq$ \\
    more than & $>$ \\
    fewer than, less than & $<$ \\
\end{tabular}
\end{center}
\end{conceptBox}

\begin{workedExample}{Solving a Two-Step Inequality}
Solve $3x - 7 \leq 14$.

Add $7$: $3x \leq 21$

Divide by $3$: $x \leq 7$
\answerHighlight{$x \leq 7$}
\end{workedExample}

\begin{errorBox}[Forgetting to Flip]
\wrongAnswer{$-5x > 20$ → $x > -4$}
When dividing by a negative, flip the sign.
Correct: $-5x > 20$ → $x < -4$.
\end{errorBox}

\begin{practiceBox}{Writing and Solving Inequalities}
\resetProblems

\prob Solve $2x + 5 > 17$.
\answerExplain{$x > 6$}{Subtract $5$: $2x > 12$. Divide by $2$: $x > 6$. Any number greater than $6$ is a solution.}

\prob Solve $-3n + 9 \leq 0$.
\answerExplain{$n \geq 3$}{Subtract $9$: $-3n \leq -9$. Divide by $-3$ and flip the sign: $n \geq 3$.}

\prob You have $\$50$ and spend $\$6$ per day. After how many days will you have less than $\$14$?
\answerExplain{More than $6$ days}{Equation: $50 - 6d < 14$. Subtract $50$: $-6d < -36$. Divide by $-6$ and flip: $d > 6$. After more than $6$ days.}

\prob Solve $\dfrac{x}{4} - 2 \geq 3$.
\answerExplain{$x \geq 20$}{Add $2$: $\frac{x}{4} \geq 5$. Multiply by $4$: $x \geq 20$.}

\prob A student needs at least $80$ points on a test. She earns $5$ points per question and starts with $15$ bonus points. How many questions must she answer correctly?
\answerExplain{At least $13$ questions}{Inequality: $5q + 15 \geq 80$. Subtract $15$: $5q \geq 65$. Divide by $5$: $q \geq 13$.}

\end{practiceBox}
