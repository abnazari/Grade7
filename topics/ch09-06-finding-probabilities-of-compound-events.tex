\section{Finding Probabilities of Compound Events}

\begin{learningGoals}
\begin{itemize}
    \item Calculate probabilities by counting favorable outcomes in a sample space
    \item Apply to everyday situations (dice, cards, etc.)
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Compound Probability}
$P = \frac{\text{favorable outcomes}}{\text{total outcomes}}$
\end{conceptBox}

\begin{workedExample}{Rolling Two Dice}
What is the probability of rolling a sum of $7$?

Favorable: $(1,6), (2,5), (3,4), (4,3), (5,2), (6,1)$ ($6$ out of $36$)
$P = 6/36 = 1/6$
\answerHighlight{$1/6$}
\end{workedExample}

\tipBox{Count all possible outcomes, then count those that fit the event.}

\begin{practiceBox}{Compound Probability}
\resetProblems

\prob What is the probability of flipping two heads in two coin tosses?
\answerExplain{$0.25$}{Outcomes: HH, HT, TH, TT. Only HH is favorable: $1/4 = 0.25$.}

\prob What is the probability of rolling a $2$ and a $3$ on two dice?
\answerExplain{$1/36$}{Only one outcome: $(2,3)$. $1/36$.}

\prob What is the probability of drawing two red marbles from a bag of $5$ red and $5$ blue (without replacement)?
\answerExplain{$2/9$}{First: $5/10$, second: $4/9$. $5/10 \times 4/9 = 20/90 = 2/9$.}

\prob What is the probability of picking a heart and then a spade from a deck (without replacement)?
\answerExplain{$1/17$}{First: $13/52$, second: $13/51$. $13/52 \times 13/51 = 169/2652 \approx 0.064$.}

\prob What is the probability of rolling an even number on both dice?
\answerExplain{$1/9$}{Each die: $3/6$ even. $3/6 \times 3/6 = 9/36 = 1/4$.}

\end{practiceBox}
