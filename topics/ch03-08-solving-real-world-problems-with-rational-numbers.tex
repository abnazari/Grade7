\section{Solving Real-World Problems with Rational Numbers}

\begin{learningGoals}
\begin{itemize}
    \item Solve multi-step problems using all four operations with rational numbers
    \item Choose the right operation and check that answers make sense
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Putting It All Together}
Real-world problems often mix addition, subtraction, multiplication, and division — sometimes with fractions \textbf{and} decimals in the same problem.
\begin{itemize}[leftmargin=*]
    \item \textbf{Read carefully.} Decide which operation each part of the problem needs.
    \item \textbf{Watch the signs.} Gains, deposits, and increases are positive. Losses, withdrawals, and decreases are negative.
    \item \textbf{Check your answer.} Does the sign make sense? Is the size reasonable?
\end{itemize}
\end{conceptBox}

\begin{workedExample}{Multi-Step Bank Problem}
You have $\$125.50$ in your account. You withdraw $\$40.75$, then deposit $\$18.20$, then pay a $\$3.50$ fee. What is your balance?

$125.50 - 40.75 + 18.20 - 3.50$

$= 84.75 + 18.20 - 3.50 = 102.95 - 3.50 = 99.45$
\answerHighlight{Balance: $\$99.45$}
\end{workedExample}

\begin{realWorld}[Temperature and Elevation]
Scientists use negative numbers every day. A weather station at $-50$ m elevation records $-12.5°$C at midnight. By noon the temperature rises $\frac{3}{4}$ of the way to $0°$C. Noon temperature: $-12.5 + \frac{3}{4}(12.5) = -12.5 + 9.375 = -3.125°$C.
\end{realWorld}

\begin{practiceBox}{Solving Real-World Problems}
\resetProblems

\prob A stock opens at $\$42.30$. It goes up $\$1.75$, down $\$3.20$, and up $\$0.85$. What is the closing price?
\answerExplain{$\$41.70$}{Add each change in order: $42.30 + 1.75 = 44.05$. Then $44.05 - 3.20 = 40.85$. Then $40.85 + 0.85 = 41.70$.}

\prob A recipe needs $2\frac{1}{4}$ cups of liquid. You use $\frac{2}{3}$ cup of milk and the rest is water. How much water?
\answerExplain{$1\frac{7}{12}$ cups}{Subtract: $2\frac{1}{4} - \frac{2}{3} = \frac{9}{4} - \frac{2}{3}$. Common denominator $12$: $\frac{27}{12} - \frac{8}{12} = \frac{19}{12} = 1\frac{7}{12}$ cups of water.}

\prob A football team gains $3$ yards, loses $7$ yards, gains $12$ yards, and loses $2$ yards. What is the net yardage?
\answerExplain{$6$ yards}{Add all changes: $3 + (-7) + 12 + (-2)$. Group positives: $3 + 12 = 15$. Group negatives: $(-7) + (-2) = -9$. Combine: $15 + (-9) = 6$ yards gained.}

\prob You split a $\$48.60$ bill equally among $4$ friends. Each friend also pays a $\$2.50$ tip. How much does each person pay total?
\answerExplain{$\$14.65$}{Divide the bill: $48.60 \div 4 = \$12.15$ per person. Add the tip: $12.15 + 2.50 = \$14.65$.}

\prob Evaluate: $-\dfrac{1}{2} \times 8 + \dfrac{3}{4} \div \left(-\dfrac{1}{2}\right)$
\answerExplain{$-5\frac{1}{2}$}{Multiply first: $-\frac{1}{2} \times 8 = -4$. Divide: $\frac{3}{4} \div (-\frac{1}{2}) = \frac{3}{4} \times (-2) = -\frac{3}{2}$. Add: $-4 + (-\frac{3}{2}) = -\frac{8}{2} + (-\frac{3}{2}) = -\frac{11}{2} = -5\frac{1}{2}$.}

\end{practiceBox}
