% CCSS 7.EE.A.1 — Writing and Evaluating Expressions
\section{Writing and Evaluating Expressions}

\begin{learningGoals}
\begin{itemize}
    \item Translate word phrases into algebraic expressions
    \item Evaluate expressions by substituting values for variables
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{From Words to Algebra}
An \textbf{algebraic expression} uses numbers, variables, and operations.
\begin{center}
\begin{tabular}{l l}
    \textbf{Words} & \textbf{Expression} \\
    \hline
    $5$ more than a number & $n + 5$ \\
    $3$ times a number minus $7$ & $3n - 7$ \\
    half of a number plus $4$ & $\frac{n}{2} + 4$ \\
\end{tabular}
\end{center}
To \textbf{evaluate}, replace each variable with a number and compute.
\end{conceptBox}

\begin{workedExample}{Evaluating with Rational Numbers}
Evaluate $2x - \frac{3}{4}$ when $x = -1.5$.

Substitute: $2(-1.5) - \frac{3}{4} = -3 - 0.75 = -3.75$
\answerHighlight{$-3.75$}
\end{workedExample}

\mascotSays{Always use parentheses when you substitute a negative number. Writing $2(-1.5)$ keeps you from losing the negative sign!}

\begin{practiceBox}{Writing and Evaluating Expressions}
\resetProblems

\prob Write an expression: ``$8$ less than $3$ times a number $n$.''
\answerExplain{$3n - 8$}{``$3$ times a number'' is $3n$. ``$8$ less than'' means subtract $8$ from that result: $3n - 8$.}

\prob Evaluate $4a + 7$ when $a = -2$.
\answerExplain{$-1$}{Substitute: $4(-2) + 7 = -8 + 7 = -1$.}

\prob Evaluate $\frac{x}{3} - 5$ when $x = 12$.
\answerExplain{$-1$}{Substitute: $\frac{12}{3} - 5 = 4 - 5 = -1$.}

\prob A phone plan charges $\$0.10$ per text plus a $\$15$ monthly fee. Write an expression for the total cost if you send $t$ texts.
\answerExplain{$0.10t + 15$}{Each text costs $\$0.10$, so $t$ texts cost $0.10t$. Add the $\$15$ fee: $0.10t + 15$.}

\prob Using the expression above, find the cost for $85$ texts.
\answerExplain{$\$23.50$}{Substitute $t = 85$: $0.10(85) + 15 = 8.50 + 15 = \$23.50$.}

\end{practiceBox}
