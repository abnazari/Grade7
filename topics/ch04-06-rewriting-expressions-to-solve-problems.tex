\section{Rewriting Expressions to Solve Problems}

\begin{learningGoals}
\begin{itemize}
    \item Rewrite expressions in different forms to reveal useful information
    \item Connect equivalent expressions to real-world meaning
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Different Forms, Same Value}
The same situation can be written in different ways, and each form tells you something different.
\begin{center}
\begin{tabular}{l l}
    \textbf{Form} & \textbf{What it tells you} \\
    \hline
    $a + 0.05a$ & Start with $a$, add $5\%$ of $a$ \\
    $1.05a$ & Total is $105\%$ of $a$ (simpler to compute) \\
\end{tabular}
\end{center}
Both are equivalent: $a + 0.05a = 1a + 0.05a = 1.05a$.
Rewriting helps you compute faster or see the problem differently.
\end{conceptBox}

\begin{workedExample}{Seeing the Big Picture}
A gym charges $\$30$ per month. A coupon gives $20\%$ off.

\textbf{Form 1:} $30 - 0.20(30) = 30 - 6 = 24$ (compute the discount, then subtract)

\textbf{Form 2:} $0.80(30) = 24$ (multiply by what you keep: $80\%$)

Both give $\$24$, but Form 2 is one step instead of two.
\answerHighlight{Rewriting $p - 0.20p$ as $0.80p$ is faster.}
\end{workedExample}

\begin{realWorld}[Is It Really a Better Deal?]
A store says ``Buy 2, get 1 free!'' That sounds like $33\%$ off. But if each shirt is $\$s$, you pay $2s$ for $3$ shirts. Price per shirt: $\frac{2s}{3} = \frac{2}{3}s \approx 0.67s$. That is about $33\%$ off each shirt. Rewriting the expression confirmed it!
\end{realWorld}

\begin{practiceBox}{Rewriting Expressions}
\resetProblems

\prob Rewrite $x + 0.08x$ as a single term. What does it represent?
\answerExplain{$1.08x$; an $8\%$ increase}{Factor out $x$: $x(1 + 0.08) = 1.08x$. This means the total is $108\%$ of the original — like adding $8\%$ sales tax.}

\prob A price $p$ is discounted $15\%$. Write two equivalent expressions for the sale price.
\answerExplain{$p - 0.15p$ and $0.85p$}{The discount is $0.15p$. Subtract: $p - 0.15p$. Factor: $p(1 - 0.15) = 0.85p$. Both give the same sale price.}

\prob Rewrite $3(x + 4) - 2x$ in simplest form.
\answerExplain{$x + 12$}{Expand: $3x + 12 - 2x$. Combine like terms: $(3x - 2x) + 12 = x + 12$.}

\prob A worker earns $\$h$ per hour plus $\$50$ bonus, working $40$ hours a week. Write an expression for weekly pay and simplify.
\answerExplain{$40h + 50$}{Hourly pay for $40$ hours is $40h$. Add the bonus: $40h + 50$. This is already simplified since $40h$ and $50$ are not like terms.}

\prob Show that ``increase by $25\%$ then decrease by $20\%$'' gives back the original price.
\answerExplain{$1.25p \times 0.80 = 1.00p = p$}{Increase by $25\%$: multiply by $1.25$. Then decrease by $20\%$: multiply by $0.80$. Combined: $1.25 \times 0.80 = 1.00$, so the final price equals the original $p$.}

\end{practiceBox}
