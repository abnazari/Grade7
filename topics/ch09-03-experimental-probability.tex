% CCSS 7.SP.C.7 — Experimental Probability
\section{Experimental Probability}

\begin{learningGoals}
\begin{itemize}
    \item Approximate probability by collecting data
    \item Understand long-run relative frequency
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Experimental Probability}
$P = \frac{\text{number of times event occurs}}{\text{total trials}}$
\end{conceptBox}

\begin{workedExample}{Coin Toss}
A coin is tossed $100$ times. Heads comes up $48$ times. What is the experimental probability of heads?

$P = 48/100 = 0.48$
\answerHighlight{$0.48$}
\end{workedExample}

\tipBox{Experimental probability may differ from theoretical probability in small samples.}

\begin{practiceBox}{Experimental Probability}
\resetProblems

\prob A die is rolled $60$ times. $10$ times a $6$ is rolled. What is the experimental probability of rolling a $6$?
\answerExplain{$0.167$}{$10/60 = 0.167$.}

\prob A spinner lands on red $25$ times out of $100$ spins. What is the probability?
\answerExplain{$0.25$}{$25/100 = 0.25$.}

\prob A coin is flipped $50$ times, heads comes up $28$ times. Probability?
\answerExplain{$0.56$}{$28/50 = 0.56$.}

\prob A card is drawn $40$ times, $12$ times it's a heart. Probability?
\answerExplain{$0.3$}{$12/40 = 0.3$.}

\prob A die is rolled $90$ times, $15$ times a $2$ is rolled. Probability?
\answerExplain{$0.167$}{$15/90 = 0.167$.}

\end{practiceBox}
