% CCSS 7.G.B.6 — Area of Composite Shapes
\section{Area of Composite Shapes}

\begin{learningGoals}
\begin{itemize}
    \item Find the area of figures made of triangles, quadrilaterals, and polygons
    \item Add and subtract areas to solve real-world problems
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Composite Shapes}
A composite shape is made by combining simpler shapes.
\end{conceptBox}

\begin{workedExample}{Finding Area of a Composite Shape}
A figure is made of a rectangle $6 \times 4$ cm and a semicircle with radius $2$ cm attached to one side. Find the total area ($\pi \approx 3.14$).

Rectangle: $6 \times 4 = 24$ cm$^2$
Semicircle: $A = \frac{1}{2} \pi r^2 = 0.5 \times 3.14 \times 4 = 6.28$ cm$^2$
Total: $24 + 6.28 = 30.28$ cm$^2$
\answerHighlight{$30.28$ cm$^2$}
\end{workedExample}

\tipBox{Break composite shapes into familiar pieces, find each area, then add or subtract.}

\begin{practiceBox}{Composite Area}
\resetProblems

\prob Find the area of a shape made of a rectangle $8 \times 3$ m and a triangle with base $8$ m, height $2$ m.
\answerExplain{$32$ m$^2$}{Rectangle: $8 \times 3 = 24$ m$^2$. Triangle: $0.5 \times 8 \times 2 = 8$ m$^2$. Total: $24 + 8 = 32$ m$^2$.}

\prob Find the area of a figure made of a square $5 \times 5$ in and a quarter circle with radius $5$ in ($\pi \approx 3.14$).
\answerExplain{$34.825$ in$^2$}{Square: $25$ in$^2$. Quarter circle: $0.25 \times 3.14 \times 25 = 19.625$ in$^2$. Total: $25 + 19.625 = 44.625$ in$^2$.}

\prob A composite shape is a rectangle $10 \times 4$ ft with a triangle $10$ ft base, $3$ ft height attached. Find the total area.
\answerExplain{$47$ ft$^2$}{Rectangle: $10 \times 4 = 40$ ft$^2$. Triangle: $0.5 \times 10 \times 3 = 15$ ft$^2$. Total: $40 + 15 = 55$ ft$^2$.}

\prob Find the area of a figure made of a rectangle $7 \times 2$ cm and a semicircle with radius $1$ cm ($\pi \approx 3.14$).
\answerExplain{$15.57$ cm$^2$}{Rectangle: $7 \times 2 = 14$ cm$^2$. Semicircle: $0.5 \times 3.14 \times 1^2 = 1.57$ cm$^2$. Total: $14 + 1.57 = 15.57$ cm$^2$.}

\prob A composite shape is made of two rectangles $5 \times 3$ m and $2 \times 3$ m. Find the total area.
\answerExplain{$21$ m$^2$}{First: $5 \times 3 = 15$ m$^2$. Second: $2 \times 3 = 6$ m$^2$. Total: $15 + 6 = 21$ m$^2$.}

\end{practiceBox}
