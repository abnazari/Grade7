% CCSS 7.G.A.1 — Reproducing Scale Drawings at a Different Scale
\section{Reproducing Scale Drawings at a Different Scale}

\begin{learningGoals}
\begin{itemize}
    \item Convert a scale drawing to a new scale
    \item Calculate the new scale factor from two given scales
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Changing the Scale}
To redraw at a new scale, find how the two scales relate:
$$\text{new drawing length} = \text{old drawing length} \times \frac{\text{new scale factor}}{\text{old scale factor}}$$
\textbf{Example:} Old scale: $1$ cm $= 10$ m. New scale: $1$ cm $= 5$ m.

Ratio: $\frac{10}{5} = 2$. Every old measurement is multiplied by $2$ on the new drawing.

A $3$ cm line becomes $3 \times 2 = 6$ cm.
\end{conceptBox}

\begin{workedExample}{Rescaling a Floor Plan}
A floor plan uses $1$ cm $= 8$ ft. You want to redraw it at $1$ cm $= 4$ ft. A wall is $5$ cm on the original. How long is it on the new drawing?

Scale ratio: $\frac{8}{4} = 2$. New length: $5 \times 2 = 10$ cm.
\answerHighlight{$10$ cm on the new drawing.}
\end{workedExample}

\tipBox{Going to a larger scale (fewer real units per cm) makes the drawing bigger. Going to a smaller scale makes it smaller.}

\begin{practiceBox}{Reproducing Scale Drawings}
\resetProblems

\prob A map uses $1$ cm $= 50$ km. You redraw it at $1$ cm $= 25$ km. A $4$ cm road becomes how long?
\answerExplain{$8$ cm}{Scale ratio: $\frac{50}{25} = 2$. New length: $4 \times 2 = 8$ cm. The new scale is more detailed, so the drawing gets bigger.}

\prob A model uses scale $1:100$. You need scale $1:50$. A $3$ cm wall becomes how long?
\answerExplain{$6$ cm}{Scale ratio: $\frac{100}{50} = 2$. New length: $3 \times 2 = 6$ cm.}

\prob A drawing uses $1$ cm $= 2$ m. Redraw at $1$ cm $= 6$ m. A $9$ cm room becomes how long?
\answerExplain{$3$ cm}{Scale ratio: $\frac{2}{6} = \frac{1}{3}$. New length: $9 \times \frac{1}{3} = 3$ cm. A smaller scale means a smaller drawing.}

\prob A rectangle is $4$ cm by $6$ cm on the original map ($1$ cm $= 10$ m). What is its area on a new map at $1$ cm $= 5$ m?
\answerExplain{$96$ cm$^2$}{Scale ratio: $\frac{10}{5} = 2$. New dimensions: $4 \times 2 = 8$ cm by $6 \times 2 = 12$ cm. New area: $8 \times 12 = 96$ cm$^2$.}

\trueOrFalse{Changing the scale of a drawing changes the angles in the shapes.}
\answerTF{False}

\end{practiceBox}
