\section{Solving Equations with the Distributive Property}

\begin{learningGoals}
\begin{itemize}
    \item Solve equations of the form $p(x + q) = r$
    \item Choose between distributing first or dividing first
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Two Approaches}
For equations like $3(x + 4) = 27$, you can solve two ways:
\begin{itemize}[leftmargin=*]
    \item \textbf{Distribute first:} $3x + 12 = 27$ → solve as a two-step equation.
    \item \textbf{Divide first:} $x + 4 = 9$ → solve in one step.
\end{itemize}
Both give the same answer. Pick whichever feels easier.
\end{conceptBox}

\begin{sideBySideExample}{Comparing Methods}
Solve $4(x - 2) = 20$.
\tcblower
\textbf{Method 1 — Distribute:}

$4x - 8 = 20$

$4x = 28 \to x = 7$

\textbf{Method 2 — Divide by $4$:}

$x - 2 = 5$

$x = 7$ \checkmark
\end{sideBySideExample}

\mascotSays{If the number outside the parentheses divides evenly into the other side, dividing first is faster!}

\begin{practiceBox}{Solving with the Distributive Property}
\resetProblems

\prob Solve $2(x + 5) = 18$.
\answerExplain{$x = 4$}{Divide both sides by $2$: $x + 5 = 9$. Subtract $5$: $x = 4$. Check: $2(4 + 5) = 2(9) = 18$ \checkmark.}

\prob Solve $5(n - 3) = -10$.
\answerExplain{$n = 1$}{Divide both sides by $5$: $n - 3 = -2$. Add $3$: $n = 1$. Check: $5(1 - 3) = 5(-2) = -10$ \checkmark.}

\prob Solve $-3(y + 4) = 15$.
\answerExplain{$y = -9$}{Divide by $-3$: $y + 4 = -5$. Subtract $4$: $y = -9$. Check: $-3(-9 + 4) = -3(-5) = 15$ \checkmark.}

\prob Solve $6(2x - 1) = 42$.
\answerExplain{$x = 4$}{Distribute: $12x - 6 = 42$. Add $6$: $12x = 48$. Divide by $12$: $x = 4$. Check: $6(2 \cdot 4 - 1) = 6(7) = 42$ \checkmark.}

\prob Each side of a square is $(x + 3)$ cm. The perimeter is $52$ cm. Find $x$.
\answerExplain{$x = 10$}{Perimeter $= 4(x + 3) = 52$. Divide by $4$: $x + 3 = 13$. Subtract $3$: $x = 10$. Each side is $13$ cm.}

\end{practiceBox}
