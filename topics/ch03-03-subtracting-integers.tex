% CCSS 7.NS.A.1.C — Subtracting Integers
\section{Subtracting Integers}

\begin{learningGoals}
\begin{itemize}
    \item Rewrite subtraction as adding the opposite
    \item Find the distance between two integers on a number line
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Subtraction = Adding the Opposite}
To subtract an integer, \textbf{add its opposite}:
$$p - q = p + (-q)$$
\begin{itemize}[leftmargin=*]
    \item $8 - 3 = 8 + (-3) = 5$
    \item $4 - 9 = 4 + (-9) = -5$
    \item $(-2) - 5 = (-2) + (-5) = -7$
    \item $(-6) - (-4) = (-6) + 4 = -2$
\end{itemize}
The \textbf{distance} between two numbers is $|a - b|$, always positive.
\end{conceptBox}

\begin{workedExample}{Keep--Change--Flip}
Find $(-3) - (-8)$.

\textbf{Keep} the first number: $-3$.

\textbf{Change} subtraction to addition.

\textbf{Flip} the sign of the second number: $-8 \to +8$.

$(-3) + 8 = 5$
\answerHighlight{$(-3) - (-8) = 5$}
\end{workedExample}

\begin{errorBox}[Watch the Double Negative]
\wrongAnswer{$(-5) - (-3) = -5 - 3 = -8$}
Subtracting a negative flips it to positive: $(-5) - (-3) = (-5) + 3 = -2$.
\end{errorBox}

\begin{practiceBox}{Subtracting Integers}
\resetProblems

\prob $10 - 16$
\answerExplain{$-6$}{Rewrite as adding the opposite: $10 + (-16)$. Different signs, so subtract: $16 - 10 = 6$. The larger absolute value is negative, so the answer is $-6$.}

\prob $(-7) - (-12)$
\answerExplain{$5$}{Rewrite: $(-7) + 12$. Different signs, so $12 - 7 = 5$. The larger absolute value is positive, so the answer is $5$.}

\prob The temperature dropped from $5°$C to $-8°$C. What was the change?
\answerExplain{$-13°$C}{Subtract: $-8 - 5 = -8 + (-5) = -13$. The temperature dropped $13$ degrees.}

\prob Find the distance between $-9$ and $4$ on a number line.
\answerExplain{$13$ units}{Distance $= |{-9} - 4| = |{-13}| = 13$. You can also count: from $-9$ to $0$ is $9$ units, then $0$ to $4$ is $4$ units, total $13$.}

\prob $(-3) - 5 - (-10)$
\answerExplain{$2$}{Rewrite both subtractions: $(-3) + (-5) + 10$. Add the negatives first: $-3 + (-5) = -8$. Then add $10$: $-8 + 10 = 2$.}

\end{practiceBox}
