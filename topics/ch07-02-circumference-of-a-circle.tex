% CCSS 7.G.B.4 — Circumference of a Circle
\section{Circumference of a Circle}

\begin{learningGoals}
\begin{itemize}
    \item Use $C = \pi d$ and $C = 2\pi r$ to find circumference
    \item Understand the relationship between circumference and diameter
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Circumference Formulas}
\begin{itemize}[leftmargin=*]
    \item $C = \pi d$ (diameter)
    \item $C = 2\pi r$ (radius)
\end{itemize}
\end{conceptBox}

\begin{workedExample}{Finding Circumference}
Find the circumference of a circle with diameter $12$ cm.

$C = \pi d = 3.14 \times 12 = 37.68$ cm
\answerHighlight{$37.68$ cm}
\end{workedExample}

\tipBox{The ratio $C/d$ is always $\pi$ for any circle.}

\begin{practiceBox}{Circumference}
\resetProblems

\prob Find the circumference if the radius is $9$ m ($\pi \approx 3.14$).
\answerExplain{$56.52$ m}{$C = 2\pi r = 2 \times 3.14 \times 9 = 56.52$ m.}

\prob Find the circumference if the diameter is $15$ in ($\pi \approx 3.14$).
\answerExplain{$47.1$ in}{$C = \pi d = 3.14 \times 15 = 47.1$ in.}

\prob A circle has circumference $31.4$ cm. What is its diameter?
\answerExplain{$10$ cm}{$d = C/\pi = 31.4/3.14 = 10$ cm.}

\prob If the radius is $6$ ft, what is the circumference?
\answerExplain{$37.68$ ft}{$C = 2\pi r = 2 \times 3.14 \times 6 = 37.68$ ft.}

\prob A bike wheel has diameter $0.7$ m. How far does it roll in one turn?
\answerExplain{$2.2$ m}{$C = \pi d = 3.14 \times 0.7 = 2.198$ m, rounded to $2.2$ m.}

\end{practiceBox}
