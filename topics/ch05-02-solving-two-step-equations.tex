\section{Solving Two-Step Equations}

\begin{learningGoals}
\begin{itemize}
    \item Solve equations of the form $px + q = r$ by undoing operations
    \item Check answers by substituting back into the original equation
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Undo in Reverse Order}
To solve a two-step equation, undo the operations \textbf{backward}:
\begin{itemize}[leftmargin=*]
    \item \textbf{Step 1:} Undo addition or subtraction (move the constant).
    \item \textbf{Step 2:} Undo multiplication or division (isolate the variable).
\end{itemize}
$$3x + 7 = 22 \;\;\xrightarrow{-7}\;\; 3x = 15 \;\;\xrightarrow{\div 3}\;\; x = 5$$
\textbf{Always check:} $3(5) + 7 = 15 + 7 = 22$ \checkmark
\end{conceptBox}

\begin{workedExample}{Solving Step by Step}
Solve $\dfrac{x}{4} - 3 = 5$.

Add $3$ to both sides: $\dfrac{x}{4} = 8$

Multiply both sides by $4$: $x = 32$

Check: $\frac{32}{4} - 3 = 8 - 3 = 5$ \checkmark
\answerHighlight{$x = 32$}
\end{workedExample}

\tipBox{Whatever you do to one side, do the exact same thing to the other side. That keeps the equation balanced!}

\begin{practiceBox}{Solving Two-Step Equations}
\resetProblems

\prob Solve $2x + 9 = 25$.
\answerExplain{$x = 8$}{Subtract $9$ from both sides: $2x = 16$. Divide both sides by $2$: $x = 8$. Check: $2(8) + 9 = 25$ \checkmark.}

\prob Solve $5n - 12 = 33$.
\answerExplain{$n = 9$}{Add $12$ to both sides: $5n = 45$. Divide by $5$: $n = 9$. Check: $5(9) - 12 = 45 - 12 = 33$ \checkmark.}

\prob Solve $\dfrac{y}{3} + 7 = 15$.
\answerExplain{$y = 24$}{Subtract $7$: $\frac{y}{3} = 8$. Multiply both sides by $3$: $y = 24$. Check: $\frac{24}{3} + 7 = 8 + 7 = 15$ \checkmark.}

\prob Solve $-4x + 10 = -6$.
\answerExplain{$x = 4$}{Subtract $10$: $-4x = -16$. Divide by $-4$: $x = 4$. Check: $-4(4) + 10 = -16 + 10 = -6$ \checkmark.}

\prob A gym charges $\$10$ per month plus $\$25$ to join. You spent $\$85$. How many months?
\answerExplain{$6$ months}{Equation: $10m + 25 = 85$. Subtract $25$: $10m = 60$. Divide by $10$: $m = 6$. Check: $10(6) + 25 = 85$ \checkmark.}

\end{practiceBox}
