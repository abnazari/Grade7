\section{Parts of a Circle}

\begin{learningGoals}
\begin{itemize}
    \item Identify and describe the center, radius, diameter, and circumference of a circle
    \item Relate diameter and radius; prepare for circumference and area formulas
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Circle Vocabulary}
\begin{itemize}[leftmargin=*]
    \item \textbf{Center}: The middle point of the circle
    \item \textbf{Radius}: Distance from center to edge
    \item \textbf{Diameter}: Distance across the circle through the center ($2 \times$ radius)
    \item \textbf{Circumference}: The distance around the circle
\end{itemize}
\end{conceptBox}

\begin{workedExample}{Labeling Circle Parts}
Draw a circle. Mark the center $O$, a radius $OA$, and a diameter $AB$ passing through $O$. Label the circumference as the edge.
\answerHighlight{Center $O$, radius $OA$, diameter $AB$, circumference}
\end{workedExample}

\tipBox{The diameter is always twice the radius: $d = 2r$.}

\begin{practiceBox}{Circle Parts}
\resetProblems

\prob What is the radius if the diameter is $10$ cm?
\answerExplain{$5$ cm}{Radius is half the diameter: $10 \div 2 = 5$ cm.}

\prob What is the diameter if the radius is $7$ in?
\answerExplain{$14$ in}{Diameter is twice the radius: $7 \times 2 = 14$ in.}

\prob Name all parts of a circle shown in a diagram with center $O$, radius $OA$, diameter $AB$, circumference $C$.
\answerExplain{Center, radius, diameter, circumference}{Identify each part by its definition and location.}

\prob If the circumference is $31.4$ cm and the diameter is $10$ cm, what is $\pi$ rounded to the nearest tenth?
\answerExplain{$3.1$}{$\pi = \text{circumference} \div \text{diameter} = 31.4 \div 10 = 3.14$. Rounded: $3.1$.}

\prob If the radius is $4$ m, what is the diameter and circumference (use $\pi \approx 3.14$)?
\answerExplain{Diameter $8$ m, circumference $25.12$ m}{Diameter: $4 \times 2 = 8$ m. Circumference: $2 \times 3.14 \times 4 = 25.12$ m.}

\end{practiceBox}
