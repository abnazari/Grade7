% CCSS 7.NS.A.2.D — Converting Rational Numbers to Decimals
\section{Converting Rational Numbers to Decimals}

\begin{learningGoals}
\begin{itemize}
    \item Convert fractions to decimals using long division
    \item Recognize terminating and repeating decimals
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Terminating vs. Repeating Decimals}
Every rational number becomes a decimal that either \textbf{terminates} (ends) or \textbf{repeats} (a pattern goes on forever).
\begin{itemize}[leftmargin=*]
    \item \textbf{Terminating:} $\frac{3}{8} = 0.375$ — the division comes out even.
    \item \textbf{Repeating:} $\frac{1}{3} = 0.333\ldots = 0.\overline{3}$ — the digit $3$ repeats forever.
\end{itemize}
Use a \textbf{bar} over the repeating digits: $\frac{2}{11} = 0.\overline{18}$ means $0.181818\ldots$
\end{conceptBox}

\begin{workedExample}{Long Division to Find the Decimal}
Convert $\frac{5}{6}$ to a decimal.

$5 \div 6 = 0.8333\ldots$

The digit $3$ starts repeating, so $\frac{5}{6} = 0.8\overline{3}$.
\answerHighlight{$0.8\overline{3}$}
\end{workedExample}

\mascotSays{If the only prime factors of the denominator are $2$ and $5$, the decimal terminates. Any other prime factor means it repeats!}

\begin{practiceBox}{Converting Rational Numbers to Decimals}
\resetProblems

\prob Convert $\frac{7}{8}$ to a decimal.
\answerExplain{$0.875$}{Divide $7 \div 8$: $8$ goes into $70$ eight times ($64$), remainder $6$. $8$ into $60$ seven times ($56$), remainder $4$. $8$ into $40$ five times ($40$), remainder $0$. So $\frac{7}{8} = 0.875$ (terminates).}

\prob Convert $\frac{4}{9}$ to a decimal.
\answerExplain{$0.\overline{4}$}{Divide $4 \div 9 = 0.444\ldots$ The remainder keeps being $4$, so the digit $4$ repeats forever. Write $0.\overline{4}$.}

\prob Convert $\frac{5}{11}$ to a decimal.
\answerExplain{$0.\overline{45}$}{Divide $5 \div 11 = 0.454545\ldots$ The pattern $45$ repeats. Write $0.\overline{45}$.}

\trueOrFalse{$\frac{3}{5}$ is a repeating decimal.}
\answerTF{False}

\prob Convert $-\frac{7}{12}$ to a decimal.
\answerExplain{$-0.58\overline{3}$}{Divide $7 \div 12 = 0.5833\ldots$ The digit $3$ repeats after $58$. Add the negative sign: $-0.58\overline{3}$.}

\end{practiceBox}
