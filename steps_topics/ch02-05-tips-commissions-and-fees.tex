\section{Tips, Commissions, and Fees}

\begin{stepGoal}
\begin{itemize}
    \item Calculate tips, commissions, and fees using percents
    \item Estimate tips mentally using the $10\%$ method
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Tip (Gratuity)}{Extra money you pay for service, usually $15\%$--$20\%$ of the bill.}
    \vocabItem{Commission}{A percent of total sales a worker earns as pay.}
    \vocabItem{Fee}{A flat or percent charge for a service.}
\end{stepVocabBox}

\begin{stepsCard}[How to Calculate a Tip, Commission, or Fee]
    \stepItem{Identify the \textbf{base amount} (bill, total sales, or service price).}
    \stepItem{Convert the percent to a \textbf{decimal} and \textbf{multiply} by the base.}
    \stepItem{If needed, \textbf{add} the result to the base to find the \textbf{total cost}.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{Your restaurant bill is $\$47.80$. Estimate a $20\%$ tip.}
    \stepShow{1} Base amount: $\$47.80$. Round to $\$48$ for easier mental math.

    \stepShow{2} Find $10\%$: $\$4.80$. Double it for $20\%$: $\$9.60$.

    \stepShow{3} Total: $47.80 + 9.60 = \$57.40$.
    \stepResult{A $20\%$ tip is about $\$9.60$. Total is $\$57.40$.}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{A salesperson earns $6\%$ commission. She sells $\$4{,}500$ worth of electronics. What is her commission?}
    \stepShow{1} Base amount: $\$4{,}500$ in total sales.

    \stepShow{2} Convert: $6\% = 0.06$. Multiply: $4{,}500 \times 0.06 = \$270$.

    \stepShow{3} The commission is the pay, so no need to add to the base.
    \stepResult{Her commission is $\$270$.}
\end{stepExample}

\mascotStepTip{Quick tip trick: find $10\%$ by moving the decimal one place left. For $15\%$, add half of $10\%$. For $20\%$, double the $10\%$.}

\begin{stepPractice}{Tips, Commissions, and Fees Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Find the Base Amount}
    \prob A dinner bill is $\$35$. You leave an $18\%$ tip. How much is the tip? \answerBlank[2cm]
    \answer{$\$6.30$}

    \stepPracticeHeader[step2Color]{Multiply to Find the Amount}
    \begin{multicols}{2}
    \prob A ticket website charges a $12\%$ service fee on $\$75$ tickets. What is the total? \answerBlank[2cm]
    \answer{$\$84$}
    \prob Estimate a $15\%$ tip on a $\$62$ bill. \answerBlank[2cm]
    \answer{About $\$9.30$}
    \end{multicols}

    \stepPracticeHeader[step3Color]{Real-World Applications}
    \prob A real estate agent earns $4\%$ commission on a $\$350{,}000$ house. How much does she earn? \answerBlank[2.5cm]
    \answer{$\$14{,}000$}

    \wordProblem{You and a friend split a $\$54$ dinner bill equally and each leave a $20\%$ tip on your share. How much does each person pay in total?}{\$}
    \answer{$\$32.40$}
\end{stepPractice}
