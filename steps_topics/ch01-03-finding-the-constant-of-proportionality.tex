\section{Finding the Constant of Proportionality}

\begin{stepGoal}
\begin{itemize}
    \item Find the constant of proportionality ($k$) from tables, graphs, and descriptions
    \item Understand that $k$ equals the unit rate
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Constant of Proportionality ($k$)}{The number every ratio $\frac{y}{x}$ equals. It is the \textbf{unit rate}.}
    \vocabItem{Unit Rate}{How much $y$ changes for every $1$ unit of $x$.}
\end{stepVocabBox}

\begin{stepsCard}[How to Find the Constant of Proportionality]
    \stepItem{Pick any pair $(x, y)$ from the table, graph, or description.}
    \stepItem{Divide: $k = \dfrac{y}{x}$.}
    \stepItem{Check with another pair — the answer should be the same.}
    \stepItem{State $k$ \textbf{with units} (e.g., ``$\$1.75$ per pound'').}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{Find $k$ from the table.
\begin{center}
\begin{tabular}{|c|c|}
\hline \textbf{Pounds} ($x$) & \textbf{Cost} ($y$) \\ \hline
$2$ & $\$3.50$ \\ $4$ & $\$7.00$ \\ $6$ & $\$10.50$ \\ \hline
\end{tabular}
\end{center}}
    \stepShow{1} Pick the first pair: $(2, \; 3.50)$.

    \stepShow{2} Divide: $k = 3.50 \div 2 = 1.75$.

    \stepShow{3} Check: $7.00 \div 4 = 1.75$ \checkmark \quad $10.50 \div 6 = 1.75$ \checkmark

    \stepShow{4} $k = 1.75$ dollars per pound.
    \stepResult{$k = 1.75$ — apples cost $\$1.75$ per pound.}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{A proportional graph passes through $(6, 15)$. Find $k$.}
    \stepShow{1} Use the point $(6, 15)$.

    \stepShow{2} Divide: $k = 15 \div 6 = 2.5$.

    \stepShow{3} Only one point given, but the line passes through $(0, 0)$, so $k$ is confirmed.

    \stepShow{4} $k = 2.5$ (per unit).
    \stepResult{$k = 2.5$}
\end{stepExample}

\mascotStepTip{$k$, unit rate, rate of change — they all mean the same thing. Always include \textbf{units} when you state $k$!}

\newpage
\begin{stepPractice}{Finding the Constant of Proportionality Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Pick a Pair and Divide}
    \prob Find $k$.
    \begin{center}
    \begin{tabular}{|c|c|}
    \hline $x$ & $y$ \\ \hline $3$ & $12$ \\ $5$ & $20$ \\ $8$ & $32$ \\ \hline
    \end{tabular}
    \end{center}
    \answer{$k = 4$}

    \stepPracticeHeader[step2Color]{Find $k$ with Units}
    \prob Find $k$ and explain what it means.
    \begin{center}
    \begin{tabular}{|c|c|}
    \hline \textbf{Hours} & \textbf{Miles} \\ \hline $\frac{1}{2}$ & $30$ \\ $1$ & $60$ \\ $2$ & $120$ \\ \hline
    \end{tabular}
    \end{center}
    \answer{$k = 60$ miles per hour}

    \prob A printer prints $80$ pages in $5$ minutes at a constant rate. What is $k$? \answerBlank[2.5cm]
    \answer{$k = 16$ pages per minute}

    \stepPracticeHeader[step3Color]{Use $k$ to Find a Missing Value}
    \prob The table shows a proportional relationship. Find the missing value.
    \begin{center}
    \begin{tabular}{|c|c|}
    \hline $x$ & $y$ \\ \hline $4$ & $14$ \\ $6$ & $?$ \\ $10$ & $35$ \\ \hline
    \end{tabular}
    \end{center}
    \answer{$y = 21$ \; ($k = 3.5$, so $3.5 \times 6 = 21$)}

    \wordProblem{A graph of a proportional relationship passes through $(4, 22)$. What is $k$? What is $y$ when $x = 10$?}{~}
    \answer{$k = 5.5$; when $x = 10$, $y = 5.5 \times 10 = 55$}
\end{stepPractice}
