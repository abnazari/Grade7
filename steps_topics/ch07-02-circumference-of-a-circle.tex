\section{Circumference of a Circle}

\begin{stepGoal}
\begin{itemize}
    \item Use $C = \pi d$ and $C = 2\pi r$ to find circumference
    \item Find the diameter or radius when given the circumference
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Circumference}{The distance around a circle.}
    \vocabItem{Pi ($\pi$)}{The ratio of circumference to diameter; $\pi \approx 3.14$.}
\end{stepVocabBox}

\begin{stepsCard}[How to Find the Circumference of a Circle]
    \stepItem{Decide whether you know the \textbf{diameter} or the \textbf{radius}.}
    \stepItem{Pick the formula: $C = \pi d$ (if diameter) or $C = 2\pi r$ (if radius).}
    \stepItem{Substitute the value and \textbf{multiply} to get the circumference.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{Find the circumference of a circle with diameter $12$ cm. Use $\pi \approx 3.14$.}
    \stepShow{1} You know the diameter: $d = 12$ cm.

    \stepShow{2} Use $C = \pi d$.

    \stepShow{3} $C = 3.14 \times 12 = 37.68$ cm.
    \stepResult{$C \approx 37.68$ cm}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{A circular garden has radius $7$ m. How much fencing is needed to go around it?}
    \stepShow{1} You know the radius: $r = 7$ m.

    \stepShow{2} Use $C = 2\pi r$.

    \stepShow{3} $C = 2 \times 3.14 \times 7 = 43.96$ m.
    \stepResult{About $43.96$ m of fencing is needed.}
\end{stepExample}

\watchOut{If you are given the diameter, do NOT double it. Only double the radius: $C = 2\pi r = \pi d$.}

\begin{stepPractice}{Circumference Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Identify Diameter or Radius}
    \begin{multicols}{2}
    \prob A circle has $r = 5$ in. Is this a radius or diameter? \answerBlank[2cm]
    \answer{radius}
    \prob A circle has $d = 20$ cm. Is this a radius or diameter? \answerBlank[2cm]
    \answer{diameter}
    \end{multicols}

    \stepPracticeHeader[step2Color]{Find the Circumference ($\pi \approx 3.14$)}
    \begin{multicols}{2}
    \prob $d = 10$ ft. $C =$ \answerBlank[2.5cm]
    \answer{$31.4$ ft}
    \prob $r = 9$ m. $C =$ \answerBlank[2.5cm]
    \answer{$56.52$ m}
    \end{multicols}

    \stepPracticeHeader[step3Color]{Work Backwards}
    \prob A circle has circumference $31.4$ cm. What is the diameter? \answerBlank[2.5cm]
    \answer{$10$ cm}

    \wordProblem{A bike wheel has diameter $0.7$ m. How far does it travel in one full turn?}{m}
    \answer{$\approx 2.198$ m}
\end{stepPractice}
