\section{Expanding with the Distributive Property}

\begin{stepGoal}
\begin{itemize}
    \item Use the distributive property to expand expressions
    \item Expand with negative and rational coefficients
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Distributive Property}{Multiply the outside number by \textbf{every} term inside the parentheses: $a(b + c) = ab + ac$.}
    \vocabItem{Expand}{Remove parentheses by distributing.}
\end{stepVocabBox}

\begin{stepsCard}[How to Expand Using the Distributive Property]
    \stepItem{Identify the number (or variable) \textbf{outside} the parentheses.}
    \stepItem{\textbf{Multiply} it by each term inside the parentheses, one at a time.}
    \stepItem{Write the results with the correct \textbf{signs} ($+$ or $-$) and \textbf{combine like terms} if needed.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{Expand $-4(3x - 7)$.}
    \stepShow{1} The number outside is $-4$.

    \stepShow{2} Multiply: $-4 \times 3x = -12x$ and $-4 \times (-7) = +28$.

    \stepShow{3} Write the result: $-12x + 28$. No like terms to combine.
    \stepResult{$-12x + 28$}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{Expand and simplify $2(3x + 1) + 4x$.}
    \stepShow{1} The number outside the parentheses is $2$.

    \stepShow{2} Multiply: $2 \times 3x = 6x$ and $2 \times 1 = 2$. So $2(3x + 1) = 6x + 2$.

    \stepShow{3} Combine with the remaining term: $6x + 2 + 4x = 10x + 2$.
    \stepResult{$10x + 2$}
\end{stepExample}

\mascotStepTip{A negative times a negative gives a positive! So $-4(3x - 7)$ gives $-12x + 28$, not $-12x - 28$.}

\newpage
\begin{stepPractice}{Distributive Property Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Identify the Outside Number}
    \prob In $5(2x + 3)$, what number gets multiplied by each term? \answerBlank[2cm]
    \answer{$5$}

    \stepPracticeHeader[step2Color]{Expand}
    \begin{multicols}{2}
    \prob Expand $5(2x + 3)$. \answerBlank[2.5cm]
    \answer{$10x + 15$}
    \prob Expand $-3(4a - 2)$. \answerBlank[2.5cm]
    \answer{$-12a + 6$}
    \end{multicols}

    \prob Expand $\frac{1}{2}(6x - 10)$. \answerBlank[2.5cm]
    \answer{$3x - 5$}

    \stepPracticeHeader[step3Color]{Expand and Simplify}
    \prob Expand and simplify $-2(x - 5) + 3(2x + 1)$. \answerBlank[3cm]
    \answer{$4x + 13$}

    \wordProblem{Three friends each buy $2$ notebooks at $\$x$ each and $1$ pen for $\$3$. Write and expand an expression for the total cost for all three friends.}{dollars}
    \answer{$3(2x + 3) = 6x + 9$}
\end{stepPractice}
