\section{Writing Two-Step Equations}

\begin{stepGoal}
\begin{itemize}
    \item Translate real-world problems into equations of the form $px + q = r$
    \item Identify the rate, starting value, and total in a situation
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Two-Step Equation}{An equation you solve in two steps, like $px + q = r$.}
    \vocabItem{Variable}{A letter that stands for the unknown number you want to find.}
    \vocabItem{Constant}{A fixed number that does not change, like a fee or starting amount.}
\end{stepVocabBox}

\begin{stepsCard}[How to Write a Two-Step Equation from a Word Problem]
    \stepItem{Read the problem and \textbf{identify the unknown} --- let a variable stand for it.}
    \stepItem{Find the \textbf{rate} (the ``per'' amount) and multiply it by the variable.}
    \stepItem{Find the \textbf{fixed amount} (starting cost, bonus, fee) and add or subtract it.}
    \stepItem{Set the expression \textbf{equal to the total} given in the problem.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{A gym charges $\$10$ per month plus a $\$25$ sign-up fee. You have spent $\$85$ in total. Write an equation for the number of months $m$.}
    \stepShow{1} Unknown: $m =$ number of months.

    \stepShow{2} Rate: $\$10$ per month $\to$ $10m$.

    \stepShow{3} Fixed amount: $\$25$ sign-up fee $\to$ add $25$.

    \stepShow{4} Total is $\$85$: $10m + 25 = 85$.
    \stepResult{$10m + 25 = 85$}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{You have $\$60$ and spend $\$8$ each week. After how many weeks will you have $\$12$ left? Write an equation for $w$.}
    \stepShow{1} Unknown: $w =$ number of weeks.

    \stepShow{2} Rate: spend $\$8$ per week $\to$ $8w$ (subtracted from starting amount).

    \stepShow{3} Fixed amount: start with $\$60$ $\to$ write $60 - 8w$.

    \stepShow{4} Amount left is $\$12$: $60 - 8w = 12$.
    \stepResult{$60 - 8w = 12$}
\end{stepExample}

\mascotStepTip{Look for the word ``per'' --- that tells you what gets multiplied by the variable. Everything else is a constant.}

\newpage
\begin{stepPractice}{Writing Two-Step Equations Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Identify the Unknown}
    \prob You buy $n$ notebooks at $\$3$ each and a $\$7$ binder. The total is $\$22$. What does $n$ stand for? \answerBlank[3cm]
    \answer{the number of notebooks}

    \stepPracticeHeader[step2Color]{Build the Expression}
    \prob A taxi charges $\$2.50$ per mile plus a $\$3$ base fare. Write an expression for the cost of $m$ miles. \answerBlank[3cm]
    \answer{$2.50m + 3$}

    \stepPracticeHeader[step3Color]{Write the Full Equation}
    \prob You buy $n$ notebooks at $\$3$ each and a $\$7$ binder. The total is $\$22$. Write an equation. \answerBlank[3cm]
    \answer{$3n + 7 = 22$}

    \prob A phone plan charges $\$0.05$ per text plus $\$15$ monthly. The bill is $\$21.50$. Write an equation for the number of texts $t$. \answerBlank[3cm]
    \answer{$0.05t + 15 = 21.50$}

    \prob Write an equation: ``Five times a number minus $9$ equals $31$.'' \answerBlank[3cm]
    \answer{$5n - 9 = 31$}

    \wordProblem{A school trip costs $\$4$ per student plus $\$50$ for the bus. The total bill is $\$130$. Write an equation for the number of students $s$.}{~}
    \answer{$4s + 50 = 130$}
\end{stepPractice}
