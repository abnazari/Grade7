\section{Solving Percent Problems}

\begin{stepGoal}
\begin{itemize}
    \item Find a percent of a number, the whole, or the percent
    \item Solve percent problems using equations and proportions
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Percent}{A rate per $100$. The symbol $\%$ means ``out of $100$.''}
    \vocabItem{Part}{The piece of the whole you are looking at.}
    \vocabItem{Whole}{The total amount (also called the base).}
\end{stepVocabBox}

\begin{stepsCard}[How to Solve Any Percent Problem]
    \stepItem{Identify which value is missing: the \textbf{part}, the \textbf{whole}, or the \textbf{percent}.}
    \stepItem{Write the equation: $\text{part} = \text{percent} \times \text{whole}$.}
    \stepItem{Convert the percent to a \textbf{decimal} (divide by $100$) and \textbf{solve} for the missing value.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{What is $30\%$ of $80$?}
    \stepShow{1} The \textbf{part} is missing. You know the percent ($30\%$) and the whole ($80$).

    \stepShow{2} Write the equation: $\text{part} = 30\% \times 80$.

    \stepShow{3} Convert: $30\% = 0.30$. Multiply: $0.30 \times 80 = 24$.
    \stepResult{$30\%$ of $80$ is $24$.}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{$27$ is $75\%$ of what number?}
    \stepShow{1} The \textbf{whole} is missing. You know the part ($27$) and the percent ($75\%$).

    \stepShow{2} Write: $27 = 75\% \times w$.

    \stepShow{3} Convert: $75\% = 0.75$. Solve: $w = 27 \div 0.75 = 36$.
    \stepResult{The whole is $36$.}
\end{stepExample}

\mascotStepTip{``Is'' means equals. ``Of'' means multiply. So ``$20\%$ of $50$'' becomes $0.20 \times 50$.}

    \answer{$27$}
    \end{multicols}

    \stepPracticeHeader[step2Color]{Find the Whole}
    \prob $36$ is $40\%$ of what number? \answerBlank[2cm]
    \answer{$90$}

    \stepPracticeHeader[step3Color]{Find the Percent}
    \prob $42$ out of $60$ questions are correct. What percent is that? \answerBlank[2cm]
    \answer{$70\%$}

    \wordProblem{A store has $240$ items. If $85\%$ are in stock, how many items are in stock?}{items}
