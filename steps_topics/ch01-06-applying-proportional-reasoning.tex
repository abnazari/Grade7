\section{Applying Proportional Reasoning}

\begin{stepGoal}
\begin{itemize}
    \item Solve real-world proportion problems using the best strategy
    \item Choose between unit rate, proportion, equation, or scale factor
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Proportion}{An equation that says two ratios are equal: $\frac{a}{b} = \frac{c}{d}$.}
    \vocabItem{Cross-Multiply}{Multiply diagonally and set the products equal to solve a proportion.}
\end{stepVocabBox}

\begin{stepsCard}[How to Solve a Proportional Reasoning Problem]
    \stepItem{Read the problem. Identify the two quantities and the rate.}
    \stepItem{Pick a strategy: \textbf{unit rate}, \textbf{proportion}, \textbf{equation}, or \textbf{scale factor}.}
    \stepItem{Solve and \textbf{label your answer} with units.}
    \stepItem{Check: does your answer make sense?}
\end{stepsCard}

% --- Worked Example 1: Unit Rate ---
\begin{stepExample}{$5$ T-shirts cost $\$42.50$. How much for $8$ T-shirts?}
    \stepShow{1} Quantities: shirts and cost. Rate: $\$42.50$ for $5$ shirts.

    \stepShow{2} Strategy: unit rate — find cost of $1$ shirt first.

    \stepShow{3} $\$42.50 \div 5 = \$8.50$ per shirt. \; $\$8.50 \times 8 = \$68.00$.

    \stepShow{4} Buying more costs more — $\$68 > \$42.50$ \checkmark
    \stepResult{$\$68.00$}
\end{stepExample}

% --- Worked Example 2: Proportion ---
\begin{stepExample}{A map scale: $2$ cm $= 15$ mi. Two cities are $7$ cm apart. Find the actual distance.}
    \stepShow{1} Quantities: cm on map and miles in real life.

    \stepShow{2} Strategy: proportion — $\dfrac{2}{15} = \dfrac{7}{d}$.

    \stepShow{3} Cross-multiply: $2d = 15 \times 7 = 105$, so $d = 52.5$ miles.

    \stepShow{4} $7$ cm is more than $2$ cm, so distance should be more than $15$ mi \checkmark
    \stepResult{$52.5$ miles}
\end{stepExample}

\mascotStepTip{Not sure which strategy to use? Try the unit rate first — it works for almost everything!}

\begin{stepPractice}{Applying Proportional Reasoning Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Identify the Strategy}
    \prob A car travels $210$ miles on $6$ gallons. How far on $10$ gallons? \answerBlank[2.5cm]
    \answer{$350$ miles}

    \stepPracticeHeader[step2Color]{Solve with Any Strategy}
    \prob A recipe for $4$ people uses $\frac{3}{2}$ cups of rice. How much for $10$ people? \answerBlank[2.5cm]
    \answer{$3\frac{3}{4}$ cups}

    \prob On a map, $3$ cm $= 24$ km. How many km is $8$ cm? \answerBlank[2.5cm]
    \answer{$64$ km}

    \stepPracticeHeader[step3Color]{Solve and Check}
    \prob A printer prints $120$ pages in $8$ min. How long for $450$ pages? \answerBlank[2.5cm]
    \answer{$30$ minutes}

    \wordProblem{Small paint: $\frac{3}{4}$ gal for $\$18$. Large: $1\frac{1}{2}$ gal for $\$33$. Which is the better deal and by how much per gallon?}{dollars per gallon}
    \answer{Large — $\$22$/gal vs.\ $\$24$/gal; saves $\$2$ per gallon.}
\end{stepPractice}
