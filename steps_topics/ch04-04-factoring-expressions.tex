\section{Factoring Expressions}

\begin{stepGoal}
\begin{itemize}
    \item Factor linear expressions by finding the greatest common factor
    \item Connect factoring to the distributive property in reverse
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Factor}{To rewrite an expression as a \textbf{product} (multiplication).}
    \vocabItem{Greatest Common Factor (GCF)}{The largest number that divides evenly into every term.}
\end{stepVocabBox}

\begin{stepsCard}[How to Factor a Linear Expression]
    \stepItem{Find the \textbf{GCF} of all the terms' coefficients.}
    \stepItem{\textbf{Divide} each term by the GCF.}
    \stepItem{Write the answer as \textbf{GCF $\times$ (quotients)}.}
    \stepItem{\textbf{Check} by expanding --- you should get the original expression.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{Factor $6x + 15$.}
    \stepShow{1} GCF of $6$ and $15$ is $3$.

    \stepShow{2} Divide: $6x \div 3 = 2x$ and $15 \div 3 = 5$.

    \stepShow{3} Write: $3(2x + 5)$.

    \stepShow{4} Check: $3(2x + 5) = 6x + 15$ \checkmark
    \stepResult{$3(2x + 5)$}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{Factor $-8a + 12$.}
    \stepShow{1} GCF of $8$ and $12$ is $4$. Factor out $-4$ (take the negative).

    \stepShow{2} Divide: $-8a \div (-4) = 2a$ and $12 \div (-4) = -3$.

    \stepShow{3} Write: $-4(2a - 3)$.

    \stepShow{4} Check: $-4(2a - 3) = -8a + 12$ \checkmark
    \stepResult{$-4(2a - 3)$}
\end{stepExample}

\mascotStepTip{Factoring is the reverse of the distributive property. Always check your work by expanding back out!}

\begin{stepPractice}{Factoring Expressions Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Find the GCF}
    \begin{multicols}{2}
    \prob GCF of $10$ and $25$: \answerBlank[2cm]
    \answer{$5$}
    \prob GCF of $12$ and $18$: \answerBlank[2cm]
    \answer{$6$}
    \end{multicols}

    \stepPracticeHeader[step2Color]{Factor the Expression}
    \prob Factor $10x + 25$. \answerBlank[3cm]
    \answer{$5(2x + 5)$}

    \prob Factor $12y - 18$. \answerBlank[3cm]
    \answer{$6(2y - 3)$}

    \prob Factor $14a + 21b$. \answerBlank[3cm]
    \answer{$7(2a + 3b)$}

    \stepPracticeHeader[step3Color]{Factor with Negatives}
    \prob Factor $-9x - 6$. \answerBlank[3cm]
    \answer{$-3(3x + 2)$}

    \wordProblem{A rectangular garden has an area of $8x + 20$ square feet. If the width is $4$ feet, write a factored expression for the length.}{feet}
    \answer{$4(2x + 5)$, so the length is $2x + 5$ feet}
\end{stepPractice}
