\section{Percent Increase and Decrease}

\begin{stepGoal}
\begin{itemize}
    \item Calculate percent increase and percent decrease
    \item Use multipliers for quick percent-change calculations
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Percent Increase}{The percent a value goes \textbf{up} from the original.}
    \vocabItem{Percent Decrease}{The percent a value goes \textbf{down} from the original.}
    \vocabItem{Multiplier}{A single decimal you multiply by: $1.15$ means a $15\%$ increase; $0.80$ means a $20\%$ decrease.}
\end{stepVocabBox}

\begin{stepsCard}[How to Find Percent Change]
    \stepItem{Find the \textbf{amount of change}: $|\text{new} - \text{original}|$.}
    \stepItem{\textbf{Divide} the change by the \textbf{original} value.}
    \stepItem{\textbf{Multiply by $100$} to convert the decimal to a percent.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{A town's population grew from $8{,}000$ to $9{,}200$. Find the percent increase.}
    \stepShow{1} Change: $9{,}200 - 8{,}000 = 1{,}200$.

    \stepShow{2} Divide by the original: $1{,}200 \div 8{,}000 = 0.15$.

    \stepShow{3} Convert: $0.15 \times 100 = 15\%$.
    \stepResult{The population increased by $15\%$.}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{A jacket was $\$60$. The price increased by $15\%$. What is the new price?}
    \stepShow{1} Find the increase: $0.15 \times 60 = \$9$.

    \stepShow{2} Add to the original: $60 + 9 = \$69$. (Or use the multiplier: $60 \times 1.15 = \$69$.)

    \stepShow{3} Check: $\frac{9}{60} \times 100 = 15\%$ \checkmark
    \stepResult{The new price is $\$69$.}
\end{stepExample}

\watchOut{$0.30$ gives the \textbf{discount amount}, not the sale price. A $30\%$ off sale means you pay $70\%$, so multiply by $0.70$.}

\newpage
\begin{stepPractice}{Percent Change Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Find the Change}
    \prob A bike was $\$180$. It is now $25\%$ off. What is the sale price? \answerBlank[2cm]
    \answer{$\$135$}

    \stepPracticeHeader[step2Color]{Calculate the Percent Change}
    \begin{multicols}{2}
    \prob A plant was $12$ cm tall. It grew to $15$ cm. Find the percent increase. \answerBlank[2cm]
    \answer{$25\%$}
    \prob A game's price dropped from $\$40$ to $\$32$. Find the percent decrease. \answerBlank[2cm]
    \answer{$20\%$}
    \end{multicols}

    \stepPracticeHeader[step3Color]{Use a Multiplier}
    \prob What multiplier gives a $40\%$ decrease? \answerBlank[2cm]
    \answer{$0.60$}

    \wordProblem{A pair of sneakers originally costs $\$75$. The price goes up by $20\%$. What is the new price?}{\$}
    \answer{$\$90$}
\end{stepPractice}
