\section{Connecting Percents and Proportions}

\begin{stepGoal}
\begin{itemize}
    \item Set up proportions to solve percent problems
    \item Understand that percent is a ratio with denominator $100$
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Proportion}{An equation showing two ratios are equal: $\frac{a}{b} = \frac{c}{d}$.}
    \vocabItem{Cross-Multiply}{Multiply diagonally across the equal sign to solve a proportion.}
\end{stepVocabBox}

\begin{stepsCard}[How to Solve Percent Problems with Proportions]
    \stepItem{Set up the proportion: $\dfrac{\text{part}}{\text{whole}} = \dfrac{\text{percent}}{100}$.}
    \stepItem{\textbf{Cross-multiply}: multiply the diagonal pairs.}
    \stepItem{\textbf{Divide} to solve for the unknown value.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{$33$ out of $150$ students ride the bus. What percent is that?}
    \stepShow{1} Set up: $\dfrac{33}{150} = \dfrac{p}{100}$.

    \stepShow{2} Cross-multiply: $33 \times 100 = 150 \times p$, so $3{,}300 = 150p$.

    \stepShow{3} Divide: $p = 3{,}300 \div 150 = 22$.
    \stepResult{$22\%$ of the students ride the bus.}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{$78$ is $60\%$ of what number?}
    \stepShow{1} Set up: $\dfrac{78}{w} = \dfrac{60}{100}$.

    \stepShow{2} Cross-multiply: $78 \times 100 = 60 \times w$, so $7{,}800 = 60w$.

    \stepShow{3} Divide: $w = 7{,}800 \div 60 = 130$.
    \stepResult{The whole is $130$.}
\end{stepExample}

\watchOut{Always put the \textbf{part} on top and the \textbf{whole} on the bottom of the left fraction. Switching them gives the wrong answer!}

\begin{stepPractice}{Percents and Proportions Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Set Up the Proportion}
    \prob What is $45\%$ of $80$? Set up a proportion and solve. \answerBlank[2cm]
    \answer{$36$}

    \stepPracticeHeader[step2Color]{Cross-Multiply and Solve}
    \begin{multicols}{2}
    \prob $54$ is what percent of $120$? \answerBlank[2cm]
    \answer{$45\%$}
    \prob In a bag of $250$ marbles, $35$ are red. What percent are red? \answerBlank[2cm]
    \answer{$14\%$}
    \end{multicols}

    \stepPracticeHeader[step3Color]{Find the Unknown Whole}
    \prob $78$ is $60\%$ of what number? \answerBlank[2cm]
    \answer{$130$}

    \wordProblem{A school has $480$ students. If $72\%$ brought lunch, how many brought lunch?}{students}
    \answer{$346$ students}
\end{stepPractice}
