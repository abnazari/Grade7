\section{Integers and Their Opposites}

\begin{stepGoal}
\begin{itemize}
    \item Identify integers and their opposites on a number line
    \item Find the absolute value of an integer
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Integer}{A whole number or its negative: $\ldots, -3, -2, -1, 0, 1, 2, 3, \ldots$}
    \vocabItem{Opposite}{The number the same distance from $0$ on the other side of the number line.}
    \vocabItem{Absolute Value}{The distance from $0$, written $|n|$. Always $\geq 0$.}
\end{stepVocabBox}

\begin{stepsCard}[How to Find Opposites and Absolute Values]
    \stepItem{Locate the number on a \textbf{number line}.}
    \stepItem{The \textbf{opposite} is the same distance from $0$ on the other side. A number plus its opposite equals $0$.}
    \stepItem{The \textbf{absolute value} is the distance from $0$ — drop the sign.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{Find the opposite and absolute value of $-7$.}
    \stepShow{1} Locate $-7$ on the number line — it is $7$ units to the left of $0$.

    \stepShow{2} The opposite is $7$ units to the \textbf{right} of $0$, which is $7$.
    Check: $-7 + 7 = 0$ $\checkmark$

    \stepShow{3} Distance from $0$ is $7$, so $|-7| = 7$.
    \stepResult{Opposite: $7$ \quad Absolute value: $7$}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{A diver descends $40$ ft below sea level, then rises $40$ ft. What is the final position?}
    \stepShow{1} Descent is $-40$ (below sea level). Rise is $+40$ (above starting point).

    \stepShow{2} These are opposites: $-40 + 40 = 0$.

    \stepShow{3} $|-40| = 40$ and $|40| = 40$ — same distance, opposite directions.
    \stepResult{Back at sea level ($0$ ft). Opposite actions cancel to $0$.}
\end{stepExample}

\mascotStepTip{Think of opposites as ''undo'' buttons. Earning \$20 and spending \$20 puts you right back where you started!}
