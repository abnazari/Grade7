\section{Dividing Integers and Rational Numbers}

\begin{stepGoal}
\begin{itemize}
    \item Apply the sign rules for dividing integers
    \item Divide signed fractions and decimals
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Quotient}{The result of division.}
    \vocabItem{Reciprocal}{The ``flipped'' fraction — the reciprocal of $\frac{a}{b}$ is $\frac{b}{a}$.}
\end{stepVocabBox}

\begin{stepsCard}[How to Divide Signed Numbers]
    \stepItem{Divide the \textbf{absolute values}. For fractions, multiply by the \textbf{reciprocal} (Keep--Change--Flip).}
    \stepItem{Apply the \textbf{sign rule}: same signs $\to$ \textbf{positive}; different signs $\to$ \textbf{negative}.}
    \stepItem{\textbf{Simplify} the result. Remember: you can never divide by zero.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{Find $(-48) \div (-6)$.}
    \stepShow{1} Divide absolute values: $48 \div 6 = 8$.

    \stepShow{2} Both negative $\to$ same signs $\to$ \textbf{positive}.

    \stepShow{3} Already simplified.
    \stepResult{$(-48) \div (-6) = 8$}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{Find $-\dfrac{3}{4} \div \dfrac{2}{3}$.}
    \stepShow{1} Multiply by the reciprocal: $-\dfrac{3}{4} \times \dfrac{3}{2} = \dfrac{9}{8}$.

    \stepShow{2} One negative, one positive $\to$ different signs $\to$ \textbf{negative}.

    \stepShow{3} Result: $-\dfrac{9}{8} = -1\dfrac{1}{8}$.
    \stepResult{$-\dfrac{9}{8}$ or $-1\dfrac{1}{8}$}
\end{stepExample}

\watchOut{You can NEVER divide by zero. $5 \div 0$ is undefined — there is no answer!}

\newpage
\begin{stepPractice}{Dividing Integers and Rational Numbers Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Divide the Absolute Values}
    \begin{multicols}{2}
    \prob $|{-48}| \div |{-6}| =$ \answerBlank[2cm]
    \answer{$8$}
    \prob $35 \div 7 =$ \answerBlank[2cm]
    \answer{$5$}
    \end{multicols}

    \stepPracticeHeader[step2Color]{Apply the Sign Rule}
    \begin{multicols}{2}
    \prob $35 \div (-7) =$ \answerBlank[2cm]
    \answer{$-5$}
    \prob $(-4.8) \div 1.2 =$ \answerBlank[2cm]
    \answer{$-4$}
    \end{multicols}

    \stepPracticeHeader[step3Color]{Divide Fractions and Simplify}
    \prob $\dfrac{5}{6} \div \left(-\dfrac{1}{3}\right) =$ \answerBlank[2.5cm]
    \answer{$-2\frac{1}{2}$}

    \wordProblem{A submarine descends $240$ m in $8$ minutes at a constant rate. Write the rate as a signed number.}{m/min}
    \answer{$-30$ m/min}
\end{stepPractice}
