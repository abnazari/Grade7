\section{Volume of Prisms}

\begin{stepGoal}
\begin{itemize}
    \item Use $V = Bh$ to find the volume of cubes and right prisms
    \item Solve real-world problems involving volume
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Volume}{The amount of space inside a 3D object, measured in cubic units.}
    \vocabItem{Base Area ($B$)}{The area of the bottom (or top) face of the prism.}
\end{stepVocabBox}

\begin{stepsCard}[How to Find the Volume of a Prism]
    \stepItem{Identify the \textbf{base} of the prism and find its \textbf{area} ($B$).}
    \stepItem{Identify the \textbf{height} ($h$) — the distance between the two bases.}
    \stepItem{Multiply: $V = B \times h$.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{Find the volume of a rectangular prism: $6 \times 4 \times 3$ cm.}
    \stepShow{1} Base is a rectangle: $B = 6 \times 4 = 24$ cm$^{2}$.

    \stepShow{2} Height: $h = 3$ cm.

    \stepShow{3} $V = 24 \times 3 = 72$ cm$^{3}$.
    \stepResult{$72$ cm$^{3}$}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{A triangular prism has a base triangle with base $8$ m and height $5$ m. The prism is $10$ m long. Find the volume.}
    \stepShow{1} Base is a triangle: $B = \frac{1}{2} \times 8 \times 5 = 20$ m$^{2}$.

    \stepShow{2} Height (length) of the prism: $h = 10$ m.

    \stepShow{3} $V = 20 \times 10 = 200$ m$^{3}$.
    \stepResult{$200$ m$^{3}$}
\end{stepExample}

\watchOut{Don't confuse the \textbf{height of the base shape} (used to find $B$) with the \textbf{height of the prism} (used in $V = Bh$).}

\begin{stepPractice}{Volume of Prisms Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Find the Base Area}
    \begin{multicols}{2}
    \prob Rectangle $5 \times 3$ in. $B =$ \answerBlank[2cm]
    \answer{$15$ in$^{2}$}
    \prob Triangle: base $6$ m, height $4$ m. $B =$ \answerBlank[2cm]
    \answer{$12$ m$^{2}$}
    \end{multicols}

    \stepPracticeHeader[step2Color]{Find the Volume}
    \begin{multicols}{2}
    \prob Cube, side $= 5$ cm. $V =$ \answerBlank[2cm]
    \answer{$125$ cm$^{3}$}
    \prob Prism: $7 \times 3 \times 4$ ft. $V =$ \answerBlank[2cm]
    \answer{$84$ ft$^{3}$}
    \end{multicols}

    \stepPracticeHeader[step3Color]{Real-World Volume}
    \prob A fish tank is $12 \times 8 \times 6$ inches. How many cubic inches of water does it hold? \answerBlank[2.5cm]
    \answer{$576$ in$^{3}$}

    \wordProblem{A sandbox is shaped like a triangular prism. The triangle base is $4$ ft and its height is $3$ ft. The sandbox is $6$ ft long. What is its volume?}{ft$^{3}$}
    \answer{$36$ ft$^{3}$}
\end{stepPractice}
