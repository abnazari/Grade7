\section{Simple Interest: Earning and Paying Interest}

\begin{stepGoal}
\begin{itemize}
    \item Use the formula $I = Prt$ to calculate simple interest
    \item Find total amounts for savings and loans
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Principal ($P$)}{The starting amount of money saved or borrowed.}
    \vocabItem{Interest Rate ($r$)}{The annual percent charged or earned, written as a decimal.}
    \vocabItem{Simple Interest ($I$)}{Money earned on savings or owed on a loan: $I = Prt$.}
\end{stepVocabBox}

\begin{stepsCard}[How to Calculate Simple Interest]
    \stepItem{Identify $P$ (principal), $r$ (rate as a \textbf{decimal}), and $t$ (time in \textbf{years}).}
    \stepItem{Plug into the formula: $I = P \times r \times t$.}
    \stepItem{Find the \textbf{total amount}: $A = P + I$.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{You deposit $\$500$ at $4\%$ simple interest for $3$ years. How much interest do you earn? What is the total?}
    \stepShow{1} $P = 500$, $r = 4\% = 0.04$, $t = 3$.

    \stepShow{2} $I = 500 \times 0.04 \times 3 = \$60$.

    \stepShow{3} Total: $A = 500 + 60 = \$560$.
    \stepResult{You earn $\$60$ in interest. Total: $\$560$.}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{A loan of $\$2{,}000$ charges $\$240$ interest over $3$ years. What is the rate?}
    \stepShow{1} $P = 2{,}000$, $I = 240$, $t = 3$. The rate $r$ is unknown.

    \stepShow{2} Plug in: $240 = 2{,}000 \times r \times 3$, so $240 = 6{,}000r$.

    \stepShow{3} Solve: $r = 240 \div 6{,}000 = 0.04 = 4\%$.
    \stepResult{The annual interest rate is $4\%$.}
\end{stepExample}

\watchOut{Always convert the percent rate to a decimal first! $4\% = 0.04$, not $4$. Forgetting this gives an answer $100$ times too large.}

\newpage
\begin{stepPractice}{Simple Interest Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Identify the Values}
    \prob Find the interest: $P = \$800$, $r = 5\%$, $t = 2$ years. \answerBlank[2cm]
    \answer{$\$80$}

    \stepPracticeHeader[step2Color]{Use the Formula}
    \begin{multicols}{2}
    \prob You borrow $\$1{,}200$ at $6\%$ for $4$ years. How much interest do you owe? \answerBlank[2cm]
    \answer{$\$288$}
    \prob You deposit $\$350$ at $3\%$ for $5$ years. What is the total amount? \answerBlank[2cm]
    \answer{$\$402.50$}
    \end{multicols}

    \stepPracticeHeader[step3Color]{Solve for a Missing Value}
    \prob You earned $\$75$ interest on $\$1{,}500$ at $2.5\%$. How many years? \answerBlank[2cm]
    \answer{$2$ years}

    \wordProblem{Maya deposits $\$600$ in a savings account that pays $3.5\%$ simple interest. How much money will she have after $4$ years?}{\$}
    \answer{$\$684$}
\end{stepPractice}
