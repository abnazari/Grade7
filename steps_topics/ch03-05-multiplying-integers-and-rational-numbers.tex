\section{Multiplying Integers and Rational Numbers}

\begin{stepGoal}
\begin{itemize}
    \item Apply the sign rules for multiplying integers
    \item Multiply signed fractions and decimals
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Product}{The result of multiplication.}
    \vocabItem{Sign Rule}{Same signs $\to$ positive product. Different signs $\to$ negative product.}
\end{stepVocabBox}

\begin{stepsCard}[How to Multiply Signed Numbers]
    \stepItem{Multiply the \textbf{absolute values} (ignore the signs for now).}
    \stepItem{Count the \textbf{negative signs}. Even count $\to$ \textbf{positive}. Odd count $\to$ \textbf{negative}.}
    \stepItem{For fractions: multiply numerators, multiply denominators, then \textbf{simplify}.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{Find $(-6) \times (-9)$.}
    \stepShow{1} Multiply absolute values: $6 \times 9 = 54$.

    \stepShow{2} Two negatives (even count) $\to$ \textbf{positive}.

    \stepShow{3} (No fractions to simplify.)
    \stepResult{$(-6) \times (-9) = 54$}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{Find $\left(-\dfrac{2}{3}\right) \times \dfrac{3}{5}$.}
    \stepShow{1} Multiply absolute values: numerators $2 \times 3 = 6$, denominators $3 \times 5 = 15$.

    \stepShow{2} One negative (odd count) $\to$ \textbf{negative}.

    \stepShow{3} Simplify: $-\dfrac{6}{15} = -\dfrac{2}{5}$.
    \stepResult{$-\dfrac{2}{5}$}
\end{stepExample}

\mascotStepTip{Count the negative signs before you multiply. Even = positive, odd = negative. It works for any number of factors!}

\begin{stepPractice}{Multiplying Integers and Rational Numbers Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Multiply the Absolute Values}
    \begin{multicols}{2}
    \prob $|-4| \times |7| =$ \answerBlank[2cm]
    \answer{$28$}
    \prob $\left|\!-\dfrac{3}{4}\right| \times \left|\!-\dfrac{2}{5}\right| =$ \answerBlank[2cm]
    \answer{$\frac{6}{20} = \frac{3}{10}$}
    \end{multicols}

    \stepPracticeHeader[step2Color]{Apply the Sign Rule}
    \begin{multicols}{2}
    \prob $(-4) \times 7 =$ \answerBlank[2cm]
    \answer{$-28$}
    \prob $\left(-\dfrac{3}{4}\right) \times \left(-\dfrac{2}{5}\right) =$ \answerBlank[2cm]
    \answer{$\frac{3}{10}$}
    \end{multicols}

    \stepPracticeHeader[step3Color]{Multiply and Simplify}
    \prob $(-1.5) \times 0.4 =$ \answerBlank[2cm]
    \answer{$-0.6$}

    \wordProblem{A submarine descends at $3$ ft per second for $8$ seconds. Write the depth change as a signed number and find the result.}{ft}
    \answer{$(-3) \times 8 = -24$ ft}
\end{stepPractice}
