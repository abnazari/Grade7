\section{Adding and Subtracting Rational Numbers}

\begin{stepGoal}
\begin{itemize}
    \item Add and subtract positive and negative fractions
    \item Add and subtract positive and negative decimals
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Rational Number}{Any number that can be written as a fraction $\frac{a}{b}$ where $b \neq 0$.}
    \vocabItem{Common Denominator}{A shared denominator needed before you can add or subtract fractions.}
\end{stepVocabBox}

\begin{stepsCard}[How to Add or Subtract Signed Fractions and Decimals]
    \stepItem{For fractions, find a \textbf{common denominator}. For decimals, line up the \textbf{decimal points}.}
    \stepItem{Use the \textbf{same sign rules} as integers — same signs: add absolute values and keep the sign; different signs: subtract absolute values and keep the sign of the larger.}
    \stepItem{For subtraction, \textbf{add the opposite} first (Keep--Change--Flip), then follow step 2.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{Find $-\dfrac{2}{3} + \dfrac{1}{4}$.}
    \stepShow{1} Common denominator: $12$. Rewrite: $-\dfrac{8}{12} + \dfrac{3}{12}$.

    \stepShow{2} Different signs $\to$ subtract: $8 - 3 = 5$. Larger absolute value is $8$ (negative).

    \stepShow{3} (No subtraction to convert — already addition.)
    \stepResult{$-\dfrac{5}{12}$}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{Find $4.2 + (-6.7)$.}
    \stepShow{1} Line up decimals: $4.2$ and $6.7$.

    \stepShow{2} Different signs $\to$ subtract: $6.7 - 4.2 = 2.5$. Larger absolute value is $6.7$ (negative).

    \stepShow{3} (Already addition.)
    \stepResult{$4.2 + (-6.7) = -2.5$}
\end{stepExample}

\mascotStepTip{Convert fractions to a common denominator first, then treat the numerators like integers. The sign rules are exactly the same!}

\begin{stepPractice}{Adding and Subtracting Rational Numbers Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Find Common Denominators / Line Up Decimals}
    \prob Rewrite $-\dfrac{1}{2} - \dfrac{3}{4}$ with a common denominator: $-\dfrac{\text{\answerBlank[1cm]}}{4} - \dfrac{3}{4}$
    \answer{$2$}

    \stepPracticeHeader[step2Color]{Add or Subtract}
    \begin{multicols}{2}
    \prob $-3.5 + 1.8 =$ \answerBlank[2cm]
    \answer{$-1.7$}
    \prob $-\dfrac{1}{2} - \dfrac{3}{4} =$ \answerBlank[2cm]
    \answer{$-1\frac{1}{4}$}
    \end{multicols}

    \stepPracticeHeader[step3Color]{Subtraction $\to$ Add the Opposite}
    \begin{multicols}{2}
    \prob $\dfrac{5}{6} - \left(-\dfrac{1}{3}\right) =$ \answerBlank[2cm]
    \answer{$1\frac{1}{6}$}
    \prob $-2.8 - 1.5 =$ \answerBlank[2cm]
    \answer{$-4.3$}
    \end{multicols}

    \wordProblem{A diver is at $-12.5$ m. She goes up $4.8$ m, then down $3.2$ m. What is her depth?}{m}
    \answer{$-10.9$ m}
\end{stepPractice}
