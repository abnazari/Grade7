\section{Subtracting Integers}

\begin{stepGoal}
\begin{itemize}
    \item Rewrite subtraction as adding the opposite
    \item Find the distance between two integers on a number line
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Additive Inverse}{The opposite of a number — used to turn subtraction into addition.}
    \vocabItem{Distance}{How far apart two numbers are: $|a - b|$, always positive.}
\end{stepVocabBox}

\begin{stepsCard}[How to Subtract Integers (Keep--Change--Flip)]
    \stepItem{\textbf{Keep} the first number exactly as it is.}
    \stepItem{\textbf{Change} the subtraction sign to addition.}
    \stepItem{\textbf{Flip} the sign of the second number (find its opposite).}
    \stepItem{Now \textbf{add} using the integer addition rules.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{Find $(-3) - (-8)$.}
    \stepShow{1} Keep $-3$.

    \stepShow{2} Change subtraction to addition: $(-3) + \;\ldots$

    \stepShow{3} Flip $-8$ to $+8$: $(-3) + 8$.

    \stepShow{4} Different signs, subtract: $8 - 3 = 5$. Larger absolute value is positive.
    \stepResult{$(-3) - (-8) = 5$}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{Find the distance between $-9$ and $4$ on a number line.}
    \stepShow{1} Keep $-9$.

    \stepShow{2} Change to addition: compute $-9 - 4$ or $4 - (-9)$.

    \stepShow{3} Flip: $4 + 9 = 13$.

    \stepShow{4} Distance $= |{-9} - 4| = |{-13}| = 13$.
    \stepResult{$13$ units apart}
\end{stepExample}

\watchOut{Subtracting a negative does NOT make it more negative. $(-5) - (-3) = (-5) + 3 = -2$, not $-8$!}

\newpage
\begin{stepPractice}{Subtracting Integers Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Rewrite as Addition}
    \begin{multicols}{2}
    \prob $10 - 16 =\; 10 +$ \answerBlank[2cm]
    \answer{$(-16)$}
    \prob $(-7) - (-12) =\; (-7) +$ \answerBlank[2cm]
    \answer{$12$}
    \end{multicols}

    \stepPracticeHeader[step2Color]{Subtract (Keep--Change--Flip)}
    \begin{multicols}{2}
    \prob $10 - 16 =$ \answerBlank[2cm]
    \answer{$-6$}
    \prob $(-7) - (-12) =$ \answerBlank[2cm]
    \answer{$5$}
    \end{multicols}

    \stepPracticeHeader[step3Color]{Find the Distance}
    \prob Find the distance between $-5$ and $8$ on a number line. \answerBlank[2cm]
    \answer{$13$ units}

    \wordProblem{The temperature dropped from $5°$C to $-8°$C. What was the change in temperature?}{°C}
    \answer{$-13°$C}
\end{stepPractice}
