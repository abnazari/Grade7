\section{Writing Equations for Proportional Relationships}

\begin{stepGoal}
\begin{itemize}
    \item Write equations of the form $y = kx$ for proportional relationships
    \item Use equations to find unknown values
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Proportional Equation}{An equation in the form $y = kx$, where $k$ is the constant of proportionality.}
    \vocabItem{Variable}{A letter that stands for a number you can change, like $x$ or $n$.}
\end{stepVocabBox}

\begin{stepsCard}[How to Write a Proportional Equation]
    \stepItem{Find $k$ — the unit rate (divide $y$ by $x$).}
    \stepItem{Write the equation: $y = kx$.}
    \stepItem{To find an unknown, \textbf{plug in} the known value and solve.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{A farmer sells strawberries at a constant price.
\begin{center}
\begin{tabular}{|c|c|}
\hline \textbf{Pounds} ($p$) & \textbf{Cost} ($c$) \\ \hline
$3$ & $\$13.50$ \\ $5$ & $\$22.50$ \\ \hline
\end{tabular}
\end{center}
Write an equation. Then find the cost of $8$ pounds.}
    \stepShow{1} $k = 13.50 \div 3 = 4.50$

    \stepShow{2} Equation: $c = 4.50p$

    \stepShow{3} For $8$ pounds: $c = 4.50 \times 8 = 36$
    \stepResult{$c = 4.50p$; \; $8$ pounds cost $\$36$.}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{A cell phone plan charges $\$0.04$ per text. You have $\$10$. How many texts can you send?}
    \stepShow{1} $k = 0.04$ (given directly — $\$0.04$ per text).

    \stepShow{2} Equation: $c = 0.04t$

    \stepShow{3} Set $c = 10$ and solve: $10 = 0.04t$, so $t = 10 \div 0.04 = 250$
    \stepResult{You can send $250$ texts.}
\end{stepExample}

\mascotStepTip{An equation is a prediction machine. Plug in any $x$, and out comes the matching $y$!}

\begin{stepPractice}{Writing Equations Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Find $k$}
    \prob Use the table to find $k$. \answerBlank[2cm]
    \begin{center}
    \begin{tabular}{|c|c|}
    \hline $x$ & $y$ \\ \hline $4$ & $10$ \\ $6$ & $15$ \\ $10$ & $25$ \\ \hline
    \end{tabular}
    \end{center}
    \answer{$k = 2.5$}

    \stepPracticeHeader[step2Color]{Write the Equation}
    \prob A bike rental costs $\$8$ per hour. Write an equation for total cost $c$ in terms of hours $h$. \answerBlank[3cm]
    \answer{$c = 8h$}

    \prob Water flows at $3.5$ gallons per minute. Write an equation for gallons $g$ in terms of minutes $m$. \answerBlank[3cm]
    \answer{$g = 3.5m$}

    \stepPracticeHeader[step3Color]{Use the Equation}
    \prob A recipe uses $\frac{3}{4}$ cup of flour per serving. How much flour for $12$ servings? \answerBlank[2.5cm]
    \answer{$9$ cups}

    \wordProblem{Movie tickets cost $\$12.50$ each. Write an equation and find how many tickets you can buy with $\$75$.}{tickets}
    \answer{$c = 12.50t$; \; $75 \div 12.50 = 6$ tickets}
\end{stepPractice}
