\section{Solving Real-World Problems with Rational Numbers}

\begin{stepGoal}
\begin{itemize}
    \item Solve multi-step problems using all four operations with rational numbers
    \item Choose the right operation and check that answers make sense
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Multi-Step Problem}{A problem that needs more than one operation to solve.}
    \vocabItem{Estimate}{A rough answer using rounded numbers — use it to check your work.}
\end{stepVocabBox}

\begin{stepsCard}[How to Solve Multi-Step Rational Number Problems]
    \stepItem{\textbf{Read} the problem. Decide which operations are needed and assign signs (gains/deposits = positive; losses/withdrawals = negative).}
    \stepItem{\textbf{Write} an expression or solve step by step, following order of operations.}
    \stepItem{\textbf{Check} — does the sign make sense? Is the size reasonable? Estimate to verify.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{You have $\$125.50$ in your account. You withdraw $\$40.75$, deposit $\$18.20$, then pay a $\$3.50$ fee. What is your balance?}
    \stepShow{1} Start: $+125.50$. Withdraw: $-40.75$. Deposit: $+18.20$. Fee: $-3.50$.

    \stepShow{2} $125.50 - 40.75 + 18.20 - 3.50 = 84.75 + 18.20 - 3.50 = 99.45$.

    \stepShow{3} Estimate: $126 - 41 + 18 - 4 = 99$. Close to $99.45$ $\checkmark$
    \stepResult{Balance: $\$99.45$}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{Evaluate: $-\dfrac{1}{2} \times 8 + \dfrac{3}{4} \div \left(-\dfrac{1}{2}\right)$}
    \stepShow{1} Two operations: multiplication and division (do these before addition).

    \stepShow{2} Multiply: $-\frac{1}{2} \times 8 = -4$. \quad Divide: $\frac{3}{4} \times (-2) = -\frac{3}{2}$.\par
    Add: $-4 + \left(-\frac{3}{2}\right) = -\frac{8}{2} + \left(-\frac{3}{2}\right) = -\frac{11}{2}$.

    \stepShow{3} Both terms are negative, so a negative sum makes sense. $-\frac{11}{2} = -5\frac{1}{2}$ $\checkmark$
    \stepResult{$-5\dfrac{1}{2}$}
\end{stepExample}

\mascotStepTip{When a problem has lots of steps, write down each step on its own line. It's much easier to catch sign errors!}

\begin{stepPractice}{Solving Real-World Problems Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Identify Operations and Signs}
    \prob A stock opens at $\$42.30$. It goes up $\$1.75$, down $\$3.20$, and up $\$0.85$. Write the expression. \answerBlank[4cm]
    \answer{$42.30 + 1.75 - 3.20 + 0.85$}

    \stepPracticeHeader[step2Color]{Solve Step by Step}
    \prob Find the closing price from the problem above. \answerBlank[2cm]
    \answer{$\$41.70$}

    \begin{multicols}{2}
    \prob $(-2) \times 5 + 3 =$ \answerBlank[2cm]
    \answer{$-7$}
    \prob $12 \div (-4) - (-3) =$ \answerBlank[2cm]
    \answer{$0$}
    \end{multicols}

    \stepPracticeHeader[step3Color]{Check Your Answer}
    \wordProblem{You split a $\$48.60$ bill equally among $4$ friends. Each friend also pays a $\$2.50$ tip. How much does each person pay?}{dollars}
    \answer{$\$14.65$}

    \wordProblem{A football team gains $3$ yd, loses $7$ yd, gains $12$ yd, and loses $2$ yd. What is the net yardage?}{yards}
    \answer{$6$ yards}
\end{stepPractice}
