\section{Percent Error: How Close Are Your Estimates?}

\begin{stepGoal}
\begin{itemize}
    \item Calculate percent error using the formula
    \item Apply percent error to measurement and estimation problems
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Percent Error}{How far off an estimate is from the actual value, expressed as a percent.}
    \vocabItem{Absolute Value}{The distance from zero — always positive. Written $|~|$.}
\end{stepVocabBox}

\begin{stepsCard}[How to Calculate Percent Error]
    \stepItem{Find the \textbf{difference}: $|\text{estimated} - \text{actual}|$ (use absolute value).}
    \stepItem{\textbf{Divide} the difference by the \textbf{actual} value.}
    \stepItem{\textbf{Multiply by $100$} to express the result as a percent.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{You estimated a room is $12$ feet long. The actual length is $14$ feet. Find the percent error.}
    \stepShow{1} Difference: $|12 - 14| = 2$.

    \stepShow{2} Divide by actual: $2 \div 14 \approx 0.1429$.

    \stepShow{3} Convert: $0.1429 \times 100 \approx 14.3\%$.
    \stepResult{The percent error is about $14.3\%$.}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{A recipe calls for $3$ cups of flour. You used $3.2$ cups. What is the percent error?}
    \stepShow{1} Difference: $|3.2 - 3| = 0.2$.

    \stepShow{2} Divide by actual: $0.2 \div 3 \approx 0.0667$.

    \stepShow{3} Convert: $0.0667 \times 100 \approx 6.7\%$.
    \stepResult{The percent error is about $6.7\%$.}
\end{stepExample}

\mascotStepTip{Always divide by the \textbf{actual} value, not the estimate. The actual value is your reference point!}

\begin{stepPractice}{Percent Error Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Find the Difference}
    \prob You guessed $30$ jelly beans in a jar. The actual count is $25$. What is the difference? \answerBlank[2cm]
    \answer{$5$}

    \stepPracticeHeader[step2Color]{Calculate the Percent Error}
    \begin{multicols}{2}
    \prob Estimated: $30$. Actual: $25$. Find the percent error. \answerBlank[2cm]
    \answer{$20\%$}
    \prob Estimated: $2$ lb. Actual: $2.5$ lb. Find the percent error. \answerBlank[2cm]
    \answer{$20\%$}
    \end{multicols}

    \stepPracticeHeader[step3Color]{Apply Percent Error}
    \prob A builder estimated a project costs $\$5{,}000$. The actual cost was $\$4{,}600$. Find the percent error. \answerBlank[2cm]
    \answer{$\approx 8.7\%$}

    \wordProblem{A scientist predicted a reaction would produce $50$ grams but got $47$ grams. What is the percent error?}{~}
    \answer{$6\%$}
\end{stepPractice}
