\section{Adding and Subtracting Linear Expressions}

\begin{stepGoal}
\begin{itemize}
    \item Add linear expressions by combining like terms
    \item Subtract linear expressions by distributing the negative sign
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Linear Expression}{An expression with a variable raised to the first power (e.g.\ $3x + 5$).}
    \vocabItem{Distribute the Negative}{When subtracting, flip the sign of \textbf{every} term in the second expression.}
\end{stepVocabBox}

\begin{stepsCard}[How to Add or Subtract Linear Expressions]
    \stepItem{Remove the parentheses. If \textbf{subtracting}, change the sign of every term in the second expression.}
    \stepItem{\textbf{Group} like terms together.}
    \stepItem{\textbf{Combine} the like terms to simplify.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{Simplify $(4x + 6) + (3x - 2)$.}
    \stepShow{1} Remove parentheses (addition --- signs stay the same): $4x + 6 + 3x - 2$.

    \stepShow{2} Group like terms: $(4x + 3x) + (6 - 2)$.

    \stepShow{3} Combine: $7x + 4$.
    \stepResult{$7x + 4$}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{Simplify $(7a - 3) - (4a + 5)$.}
    \stepShow{1} Distribute the negative to the second expression: $7a - 3 - 4a - 5$.

    \stepShow{2} Group like terms: $(7a - 4a) + (-3 - 5)$.

    \stepShow{3} Combine: $3a - 8$.
    \stepResult{$3a - 8$}
\end{stepExample}

\watchOut{When subtracting, flip \textbf{both} signs in the second expression. $(5x + 2) - (3x - 1) = 5x + 2 - 3x + 1 = 2x + 3$, not $2x + 1$.}

% PRACTICE

\newpage
\begin{stepPractice}{Adding and Subtracting Expressions Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Remove Parentheses}
    \prob Write $(8n - 5) - (3n + 7)$ without parentheses. \answerBlank[3cm]
    \answer{$8n - 5 - 3n - 7$}

    \stepPracticeHeader[step2Color]{Add Expressions}
    \begin{multicols}{2}
    \prob $(2y + 9) + (5y - 3)$ \answerBlank[2.5cm]
    \answer{$7y + 6$}
    \prob $(-x + 4) + (6x - 8)$ \answerBlank[2.5cm]
    \answer{$5x - 4$}
    \end{multicols}

    \stepPracticeHeader[step3Color]{Subtract Expressions}
    \prob $(8n - 5) - (3n + 7)$ \answerBlank[2.5cm]
    \answer{$5n - 12$}

    \prob $(2y + 9) - (2y - 1)$ \answerBlank[2.5cm]
    \answer{$10$}

    \wordProblem{A triangle has sides $2x + 1$, $3x - 4$, and $x + 5$. Write a simplified expression for the perimeter.}{units}
    \answer{$6x + 2$}
\end{stepPractice}
