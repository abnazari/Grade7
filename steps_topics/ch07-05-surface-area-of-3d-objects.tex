\section{Surface Area of Three-Dimensional Objects}

\begin{stepGoal}
\begin{itemize}
    \item Find the surface area of cubes and rectangular prisms
    \item Use nets to identify all faces of a 3D object
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Surface Area}{The total area of all faces (outside surfaces) of a 3D object.}
    \vocabItem{Net}{A flat pattern that folds into a 3D shape, showing every face.}
\end{stepVocabBox}

\begin{stepsCard}[How to Find the Surface Area of a Prism]
    \stepItem{List all \textbf{faces} of the object (use a net if it helps).}
    \stepItem{Find the \textbf{area of each face}.}
    \stepItem{\textbf{Add} all the face areas together for the total surface area.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{Find the surface area of a rectangular prism: $5 \times 3 \times 2$ cm.}
    \stepShow{1} A rectangular prism has $6$ faces: top/bottom ($5 \times 3$), front/back ($5 \times 2$), left/right ($3 \times 2$).

    \stepShow{2} Top/bottom: $2(5 \times 3) = 30$ cm$^{2}$. \quad Front/back: $2(5 \times 2) = 20$ cm$^{2}$. \quad Sides: $2(3 \times 2) = 12$ cm$^{2}$.

    \stepShow{3} Total: $30 + 20 + 12 = 62$ cm$^{2}$.
    \stepResult{$62$ cm$^{2}$}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{Find the surface area of a cube with side length $4$ in.}
    \stepShow{1} A cube has $6$ identical square faces.

    \stepShow{2} Each face: $4 \times 4 = 16$ in$^{2}$.

    \stepShow{3} Total: $6 \times 16 = 96$ in$^{2}$.
    \stepResult{$96$ in$^{2}$}
\end{stepExample}

\mascotStepTip{For a rectangular prism, there are always \textbf{3 pairs} of matching faces. Find the area of each pair and add!}
