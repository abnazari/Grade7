\section{Integers and Their Opposites}

\begin{learningGoals}
\begin{itemize}
    \item Identify integers, opposites, and absolute values on a number line
    \item Compare and order rational numbers using inequality symbols
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Integers, Opposites, and Absolute Value}
\textbf{Integers} are whole numbers and their negatives: $\ldots, -3, -2, -1, 0, 1, 2, 3, \ldots$

Every integer has an \textbf{opposite} — the same distance from $0$ on the other side.
\begin{itemize}[leftmargin=*]
    \item The opposite of $5$ is $-5$. The opposite of $-3$ is $3$.
    \item A number plus its opposite always equals $0$: $\;5 + (-5) = 0$.
    \item The \textbf{absolute value} $|n|$ is the distance from $0$. It is always $\geq 0$.
\end{itemize}
\end{conceptBox}

\numberLine[-1]{-5}{5}

\begin{workedExample}{Comparing Rational Numbers in Multiple Forms}
Order from least to greatest: $-1.5, \;\frac{3}{4}, \;-\frac{7}{4}, \;0.2, \;-1$

Convert to decimals: $-1.5, \;0.75, \;-1.75, \;0.2, \;-1$.

On the number line, left to right: $-1.75 < -1.5 < -1 < 0.2 < 0.75$.

\answerHighlight{$-\frac{7}{4}, \;-1.5, \;-1, \;0.2, \;\frac{3}{4}$}
\end{workedExample}

\mascotSays{To compare fractions and decimals, convert them all to the same form. Decimals are usually easiest!}

\tipBox{On a number line, numbers \textbf{increase} from left to right. So $-3 < -1$ because $-3$ is farther left.}

\begin{practiceBox}{Integers and Their Opposites}
\resetProblems

\prob What is the opposite of $-8$? Place both $-8$ and its opposite on a number line.
\answerExplain{$8$}{The opposite of $-8$ is $8$. Both are $8$ units from $0$ on opposite sides of the number line.}

\prob Find $|{-12}|$ and $|7|$.
\answerExplain{$12$ and $7$}{Absolute value is distance from $0$. $|-12| = 12$ and $|7| = 7$.}

\prob Use $<$, $>$, or $=$ to compare: $-\frac{2}{3}$ \answerBlank[1.5cm] $-0.5$
\answerExplain{$<$}{Convert: $-\frac{2}{3} \approx -0.667$. Since $-0.667$ is farther left than $-0.5$, we have $-\frac{2}{3} < -0.5$.}

\prob Order from least to greatest: $\frac{1}{2}, \;-0.75, \;-\frac{1}{4}, \;0.3, \;-1$
\answerExplain{$-1, \;-0.75, \;-\frac{1}{4}, \;0.3, \;\frac{1}{2}$}{Convert all to decimals: $0.5, -0.75, -0.25, 0.3, -1$. Order from most negative to most positive: $-1 < -0.75 < -0.25 < 0.3 < 0.5$.}

\trueOrFalse{$|-3| > |2|$}
\answerTF{True}

\end{practiceBox}
