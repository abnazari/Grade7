\section{Solving Percent Problems}

\begin{learningGoals}
\begin{itemize}
    \item Find a percent of a number, the whole, or the percent
    \item Apply percents to real-world money contexts: discounts, tax, tips, and markups
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Three Types of Percent Problems}
Every percent problem has three parts: the \textbf{part}, the \textbf{whole}, and the \textbf{percent}.
$$\text{part} = \text{percent} \times \text{whole}$$
\begin{itemize}[leftmargin=*]
    \item \textbf{Find the part:} What is $30\%$ of $80$? $\rightarrow$ $0.30 \times 80 = 24$
    \item \textbf{Find the whole:} $15$ is $25\%$ of what? $\rightarrow$ $15 \div 0.25 = 60$
    \item \textbf{Find the percent:} $12$ is what percent of $48$? $\rightarrow$ $12 \div 48 = 0.25 = 25\%$
\end{itemize}
\end{conceptBox}

\begin{workedExample}{Percent in a Money Context}
A jacket costs $\$60$. It is on sale for $25\%$ off. There is a $7\%$ sales tax on the sale price. How much do you pay?

\textbf{Step 1:} Discount $= 0.25 \times 60 = \$15$. Sale price $= 60 - 15 = \$45$.

\textbf{Step 2:} Tax $= 0.07 \times 45 = \$3.15$. Total $= 45 + 3.15 = \$48.15$.
\answerHighlight{You pay $\$48.15$.}
\end{workedExample}

\begin{realWorld}[Every Time You Shop]
Percents are everywhere when you spend money. A $20\%$ tip at a restaurant, a $15\%$ off coupon at the store, or $6.5\%$ sales tax on a video game — all use the same formula: \textbf{part $=$ percent $\times$ whole}.
\end{realWorld}

\mascotSays{``Is'' means equals. ``Of'' means multiply. So ``$20\%$ of $\$50$'' becomes $0.20 \times 50 = \$10$.}

\begin{practiceBox}{Solving Percent Problems}
\resetProblems

\prob A meal costs $\$35$. You leave a $20\%$ tip. How much is the tip?
\answerExplain{$\$7$}{Convert: $20\% = 0.20$. Multiply: $0.20 \times 35 = \$7$. That's the amount you add for the server.}

\prob A store marks up a shirt from $\$12$ to $\$18$. What is the markup percent?
\answerExplain{$50\%$}{The markup is $18 - 12 = \$6$. Divide by the original: $6 \div 12 = 0.5$. Multiply by $100$: $0.5 \times 100 = 50\%$.}

\prob A backpack costs $\$40$ after a $20\%$ discount. What was the original price?
\answerExplain{$\$50$}{After $20\%$ off, you pay $80\%$. So $\$40 = 0.80 \times \text{original}$. Divide: $40 \div 0.80 = \$50$.}

\prob A phone costs $\$250$ plus $8\%$ sales tax. How much is the total?
\answerExplain{$\$270$}{Tax $= 0.08 \times 250 = \$20$. Total $= 250 + 20 = \$270$.}

\prob You spent $\$42$ on a meal including a $\$7$ tip. What percent tip did you leave on the food?
\answerExplain{$20\%$}{Food cost $= 42 - 7 = \$35$. Percent $= 7 \div 35 = 0.20 = 20\%$.}

\end{practiceBox}
