\section{Angle Relationships}

\begin{learningGoals}
\begin{itemize}
    \item Identify supplementary, complementary, vertical, and adjacent angle pairs
    \item Solve multi-step algebraic equations to find unknown angles
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Angle Pairs}
\begin{itemize}[leftmargin=*]
    \item \textbf{Complementary angles} add to $90¬į$. \quad $\angle A + \angle B = 90¬į$
    \item \textbf{Supplementary angles} add to $180¬į$. \quad $\angle A + \angle B = 180¬į$
    \item \textbf{Vertical angles} are formed by two intersecting lines. They are always \textbf{equal}.
    \item \textbf{Adjacent angles} share a side and a vertex.
\end{itemize}
When two lines intersect, they form two pairs of vertical angles and four pairs of supplementary angles.
\end{conceptBox}

\begin{workedExample}{Multi-Step Angle Problem}
Two lines intersect. One pair of vertical angles is $(4x - 10)¬į$ and $(2x + 30)¬į$. A third angle is supplementary to one of them. Find all three angles.

\textbf{Step 1:} Vertical angles are equal: $4x - 10 = 2x + 30$. Solve: $2x = 40$, so $x = 20$.

\textbf{Step 2:} Vertical angles $= 4(20) - 10 = 70¬į$.

\textbf{Step 3:} The supplementary angle $= 180¬į - 70¬į = 110¬į$.
\answerHighlight{$70¬į$, $70¬į$, and $110¬į$}
\end{workedExample}

\tipBox{Vertical angles are always equal. If you spot an X made by two lines, opposite angles match!}

\mascotSays{When a problem gives you angles with expressions like $3x + 10$, set up an equation using the angle relationship ‚ÄĒ then solve for $x$.}

\begin{practiceBox}{Angle Relationships}
\resetProblems

\prob Two angles are complementary. One is $35¬į$. Find the other.
\answerExplain{$55¬į$}{Complementary angles sum to $90¬į$. Subtract: $90¬į - 35¬į = 55¬į$.}

\prob Two supplementary angles are $(3x + 15)¬į$ and $(2x + 5)¬į$. Find both angles.
\answerExplain{$111¬į$ and $69¬į$}{Sum to $180¬į$: $3x + 15 + 2x + 5 = 180$. Combine: $5x + 20 = 180$. Solve: $5x = 160$, $x = 32$. Angles: $3(32) + 15 = 111¬į$ and $2(32) + 5 = 69¬į$. Check: $111 + 69 = 180¬į$ \checkmark.}

\prob Two vertical angles are $4x + 5$ and $6x - 15$. Find $x$ and the angle measure.
\answerExplain{$x = 10$; angles are $45¬į$}{Vertical angles are equal: $4x + 5 = 6x - 15$. Subtract $4x$: $5 = 2x - 15$. Add $15$: $20 = 2x$. So $x = 10$. Angle: $4(10) + 5 = 45¬į$.}

\prob Three angles meet at a point on a line: $(x + 20)¬į$, $(2x)¬į$, and $40¬į$. Find $x$.
\answerExplain{$x = 40$}{Angles on a line sum to $180¬į$: $(x + 20) + 2x + 40 = 180$. Combine: $3x + 60 = 180$. Solve: $3x = 120$, $x = 40$. Check: $60 + 80 + 40 = 180¬į$ \checkmark.}

\prob An angle is $20¬į$ more than $3$ times its complement. Find both angles.
\answerExplain{$17.5¬į$ and $72.5¬į$}{Let the complement be $x$. The angle is $3x + 20$. Sum: $x + 3x + 20 = 90$. Combine: $4x = 70$, $x = 17.5¬į$. The angle: $3(17.5) + 20 = 72.5¬į$. Check: $17.5 + 72.5 = 90¬į$ \checkmark.}

\end{practiceBox}
