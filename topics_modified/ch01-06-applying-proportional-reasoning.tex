\section{Applying Proportional Reasoning}

\begin{learningGoals}
\begin{itemize}
    \item Solve proportion problems using the best strategy — including estimation
    \item Check answers for reasonableness before and after solving
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Your Proportional Reasoning Toolbox}
\begin{center}
\begin{tabular}{c l}
    \textbf{1.} & \textbf{Estimate first} --- round and predict before you calculate \\
    \textbf{2.} & \textbf{Unit rate} --- find the rate for $1$ unit, then multiply \\
    \textbf{3.} & \textbf{Proportion} --- set up $\frac{a}{b} = \frac{c}{d}$ and cross-multiply \\
    \textbf{4.} & \textbf{Equation} --- use $y = kx$ \\
    \textbf{5.} & \textbf{Scale factor} --- multiply or divide to scale up/down \\
\end{tabular}
\end{center}
Pick whichever strategy feels most natural. They all give the same answer. But \textbf{always estimate first} and \textbf{check at the end}.
\end{conceptBox}

\begin{workedExample}{Estimate-First Strategy}
$5$ T-shirts cost $\$42.50$. How much for $8$?

\textbf{Estimate:} Each shirt is about $\$42 \div 5 \approx \$8$. So $8$ shirts $\approx \$64$--$\$72$.

\textbf{Solve:} $\$42.50 \div 5 = \$8.50$ per shirt. $\$8.50 \times 8 = \$68$.

\textbf{Check:} $\$68$ falls in our estimate range. \checkmark
\answerHighlight{$\$68.00$}
\end{workedExample}

\begin{warningBox}[Always Check Your Answer!]
After solving any proportion problem, ask yourself:
\begin{itemize}[leftmargin=*]
    \item Does the answer fall near my estimate?
    \item If I'm buying \textbf{more}, the cost should be \textbf{higher}.
    \item If I'm going a \textbf{shorter} distance, it should take \textbf{less} time.
    \item Do the \textbf{units} make sense?
\end{itemize}
If the answer seems off, re-check your setup.
\end{warningBox}

\mascotSays{Estimating first is like using a map before driving. You may not know the exact distance, but you'll know if you're heading the wrong way!}

\begin{practiceBox}{Applying Proportional Reasoning}
\resetProblems

\prob A car travels $210$ miles on $6$ gallons. \textbf{Estimate} the distance on $10$ gallons, then solve exactly.
\answerExplain{$350$ miles}{Estimate: $210 \div 6 \approx 35$ mpg, so $35 \times 10 \approx 350$. Exact: $210 \div 6 = 35$ mpg. $35 \times 10 = 350$ miles. Check: estimate matches. \checkmark}

\prob A recipe for $4$ people uses $\frac{3}{2}$ cups of rice. \textbf{Estimate} the amount for $10$ people, then solve.
\answerExplain{$3\frac{3}{4}$ cups}{Estimate: $\frac{3}{2}$ cups serves $4$, so $10$ is $2.5$ times as many — about $3.5$--$4$ cups. Exact: $\frac{3}{2} \div 4 = \frac{3}{8}$ per person. $\frac{3}{8} \times 10 = \frac{30}{8} = 3\frac{3}{4}$ cups. Check: $3.75$ is near our estimate. \checkmark}

\prob On a map, $3$ cm $= 24$ km. A road measures $8$ cm. \textbf{Estimate}, then find the exact distance.
\answerExplain{$64$ km}{Estimate: $3$ cm is $24$ km, so $1$ cm $\approx 8$ km, and $8$ cm $\approx 64$ km. Exact: $\frac{3}{24} = \frac{8}{d}$, cross-multiply: $3d = 192$, $d = 64$ km. Check: estimate matches exactly. \checkmark}

\prob A printer prints $120$ pages in $8$ min. How long for $450$ pages? Is your answer reasonable?
\answerExplain{$30$ min}{Estimate: $120$ pages in $8$ min is about $15$ per min. $450 \div 15 = 30$ min. Exact: same calculation. Check: $450$ is nearly $4 \times 120$, so time should be nearly $4 \times 8 = 32$ min — $30$ is close. \checkmark}

\prob Small paint: $\frac{3}{4}$ gal for $\$18$. Large: $1\frac{1}{2}$ gal for $\$33$. Estimate which is the better deal, then verify.
\answerExplain{Large by $\$2$/gal}{Estimate: Small is about $\$18$ for $\frac{3}{4}$ gal $= \$24$/gal. Large is about $\$33$ for $1.5$ gal $\approx \$22$/gal. Exact: Small: $18 \div 0.75 = \$24$/gal. Large: $33 \div 1.5 = \$22$/gal. Large saves $\$2$ per gallon. \checkmark}

\end{practiceBox}
