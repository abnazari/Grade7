\worksheetHeader{Find the Value of $k$}

\begin{speedDrill}{Constant of Proportionality Sprint}
\timerBadge{4}

Find $k = \frac{y}{x}$ for each relationship.

\bigskip
\renewcommand{\arraystretch}{1.9}
\begin{center}
\begin{tabular}{|C{6.5cm}|C{3.5cm}|}
\hline
\textbf{Relationship} & \textbf{$k$} \\
\hline
$(3, 18)$ & \answerBlank[2.5cm] \\
\hline
$\$7.50$ for $3$ notebooks & \answerBlank[2.5cm] \\
\hline
$90$ miles in $1.5$ hours & \answerBlank[2.5cm] \\
\hline
$(8, 10)$ & \answerBlank[2.5cm] \\
\hline
$\frac{3}{4}$ cup per $2$ servings & \answerBlank[2.5cm] \\
\hline
Table row $x = 5,\; y = 22.5$ & \answerBlank[2.5cm] \\
\hline
\end{tabular}
\end{center}
\end{speedDrill}
\worksheetAnswer{(1)~$k = 18 \div 3 = 6$. (2)~$k = 7.50 \div 3 = 2.5$, so the cost is $\$2.50$ per notebook. (3)~$k = 90 \div 1.5 = 60$, so the speed is $60$ miles per hour. (4)~$k = 10 \div 8 = 1.25 = \frac{5}{4}$. (5)~$k = \frac{3}{4} \div 2 = \frac{3}{8}$ cup per serving. (6)~$k = 22.5 \div 5 = 4.5$. The constant of proportionality is always the output divided by the input.}

\worksheetHeader{Unit Rate or Constant? Same Idea!}

\begin{matchIt}{Match Each Relationship to Its Constant of Proportionality}

\matchRow{$4$ tickets cost $\$26$}{$k = 3.5$ dollars per pound}
\matchRow{$2$ pounds of apples cost $\$7$}{$k = 12$ points per round}
\matchRow{$(6, 48)$}{$k = 6.5$ dollars per ticket}
\matchRow{A game score rises from $(1, 12)$ to $(4, 48)$ in a proportional pattern}{$k = 8$}
\matchRow{$\frac{3}{5}$ gallon paints $\frac{3}{2}$ fence panels}{$k = \frac{2}{5}$ gallon per panel}
\end{matchIt}
\worksheetAnswer{$4$ tickets cost $\$26$ matches $k = 6.5$ dollars per ticket because $26 \div 4 = 6.5$. $2$ pounds of apples cost $\$7$ matches $k = 3.5$ dollars per pound because $7 \div 2 = 3.5$. The point $(6, 48)$ matches $k = 8$ because $48 \div 6 = 8$. The game score pattern matches $k = 12$ points per round because $12 \div 1 = 12$ and $48 \div 4 = 12$. The paint situation matches $k = \frac{2}{5}$ gallon per panel because $\frac{3}{5} \div \frac{3}{2} = \frac{3}{5} \times \frac{2}{3} = \frac{2}{5}$.}

\worksheetHeader{Spot the Wrong Constant}

\begin{errorDetective}{Which Constant Does Not Belong?}

Priya found $k$ for each relationship.

\bigskip

Relationship: $(4, 30)$
\errorWork{$k = 4 \div 30 = \frac{2}{15}$}
Correct or wrong? \answerBlank[2cm] \quad Correct $k$: \answerBlank[4cm]

\bigskip

Relationship: $12$ pages in $\frac{1}{2}$ minute
\errorWork{$k = 12 \div \frac{1}{2} = 6$ pages per minute}
Correct or wrong? \answerBlank[2cm] \quad Correct $k$: \answerBlank[4cm]

\bigskip

Relationship: $(2, 9)$ and $(6, 27)$
\errorWork{$k = \frac{9}{2}$ from the first point and $k = \frac{27}{6} = \frac{9}{2}$ from the second point.}
Correct or wrong? \answerBlank[2cm] \quad Why? \answerBlank[6cm]
\end{errorDetective}
\worksheetAnswer{(1)~Wrong. Priya divided in the wrong order. The constant is $k = y \div x = 30 \div 4 = 7.5$, not $4 \div 30$. (2)~Wrong. Dividing by $\frac{1}{2}$ should double the value: $12 \div \frac{1}{2} = 24$, so $k = 24$ pages per minute. (3)~Correct. From the first point, $9 \div 2 = \frac{9}{2}$. From the second point, $27 \div 6 = \frac{9}{2}$ after simplifying. Since both ratios match, the constant of proportionality really is $\frac{9}{2}$, or $4.5$.}

\worksheetHeader{School Store Rates}

\begin{mathScenario}{Use $k$ to Predict New Values}

The student store sells folders at a proportional rate.

\bigskip
\scenarioItem{3 folders}{$\$4.50$}
\scenarioItem{8 folders}{unknown cost}
\scenarioItem{12 folders}{unknown cost}
\scenarioItem{Store manager's note}{``The value of $k$ stays the same for every purchase.''}

\bigskip

What is the constant of proportionality? \answerBlank[4cm]

\bigskip

How much would $8$ folders cost? \answerBlank[4cm]

\bigskip

How many folders could a student buy with $\$18$? \answerBlank[4cm]
\end{mathScenario}
\worksheetAnswer{(1)~The constant of proportionality is $k = 4.50 \div 3 = 1.50$, so each folder costs $\$1.50$. (2)~For $8$ folders, multiply by the unit rate: $8 \times 1.50 = \$12$. (3)~To find the number of folders for $\$18$, divide by the unit rate: $18 \div 1.50 = 12$. A student could buy $12$ folders.}

\worksheetHeader{Why Do Units Matter?}

\begin{explainIt}{State the Constant Clearly}

A student writes, ``The constant of proportionality is $6$.'' Their table compares pounds of almonds to dollars.
Why is that answer incomplete? Rewrite it the right way and explain what the $6$ means.

\writeLines{5}

\bigskip

Another student says the constant for a graph through $(1, 2.5)$ and $(4, 10)$ is $4$ because the points go up by $4$ in $x$ from $1$ to $5$. Explain the mistake and find the real constant.

\writeLines{6}
\end{explainIt}
\worksheetAnswer{(1)~The answer is incomplete because a constant of proportionality should include units and meaning. If the table compares pounds of almonds to dollars, then $k = 6$ means $\$6$ per pound, not just the number $6$. A clear statement is: ``The constant of proportionality is $6$ dollars per pound, so each pound of almonds costs $\$6$.'' (2)~The student is mixing up the horizontal change with the ratio $\frac{y}{x}$. The constant of proportionality comes from dividing output by input: $2.5 \div 1 = 2.5$ and $10 \div 4 = 2.5$. So the real constant is $k = 2.5$.}
