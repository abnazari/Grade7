\section{Solving Real-World Problems with Rational Numbers}

\begin{stepGoal}
\begin{itemize}
    \item Solve multi-step problems using all four operations with rational numbers
    \item Estimate first, solve, and check that answers make sense
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Estimate}{A rough answer using rounded numbers — use it to predict and verify.}
    \vocabItem{Net Change}{The overall result after combining all gains and losses.}
\end{stepVocabBox}

\begin{stepsCard}[How to Solve Multi-Step Rational Number Problems]
    \stepItem{\textbf{Estimate first} — round the numbers and predict a reasonable answer.}
    \stepItem{\textbf{Solve} step by step. Assign signs (gains = positive, losses = negative) and follow order of operations.}
    \stepItem{\textbf{Check} — compare your exact answer to the estimate. Does the sign make sense in context?}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{At midnight in Fairbanks, the temperature is $-28.5°$F. By noon, it has risen $\frac{3}{4}$ of the way toward $0°$F. What is the noon temperature?}
    \stepShow{1} Estimate: $\frac{3}{4}$ of $28.5 \approx 21$. So about $-28.5 + 21 \approx -7°$F.

    \stepShow{2} Rise $= \frac{3}{4} \times 28.5 = \frac{3}{4} \times \frac{57}{2} = \frac{171}{8} = 21.375$.\par
    Noon temp $= -28.5 + 21.375 = -7.125°$F.

    \stepShow{3} Estimate was $\approx -7°$F. Actual is $-7.125°$F. $\checkmark$ Negative makes sense — still below zero.
    \stepResult{$-7.125°$F}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{A hiker descends $1{,}250$ ft from a summit at $4{,}380$ ft, rests, then climbs $\frac{1}{2}$ of the distance back up. Final elevation?}
    \stepShow{1} Estimate: $4{,}380 - 1{,}250 \approx 3{,}100$. Climb back $\approx 625$. Final $\approx 3{,}725$ ft.

    \stepShow{2} After descent: $4{,}380 - 1{,}250 = 3{,}130$ ft.\par
    Climb back: $\frac{1}{2} \times 1{,}250 = 625$ ft. Final: $3{,}130 + 625 = 3{,}755$ ft.

    \stepShow{3} Estimate was $\approx 3{,}725$. Actual is $3{,}755$. $\checkmark$ Positive elevation makes sense.
    \stepResult{$3{,}755$ ft}
\end{stepExample}

\mascotStepTip{When a problem has lots of steps, write each step on its own line. It's much easier to catch sign errors!}

\begin{stepPractice}{Solving Real-World Problems Practice (AK)}
    \resetProblems

    \stepPracticeHeader[step1Color]{Estimate First}
    \prob A fishing boat catches $345.5$ lb on Monday and $278.75$ lb on Tuesday. Estimate the total. \answerBlank[2.5cm]
    \answer{$\approx 624$ lb}

    \stepPracticeHeader[step2Color]{Solve Step by Step}
    \prob The crew keeps $\frac{2}{3}$ of the total $624.25$ lb and sells the rest at $\$4.50$/lb. How much do they earn? \answerBlank[2.5cm]
    \answer{$\approx \$937.88$}

    \prob The temperature in Barrow is $-18°$F at $6$ AM. It drops $5.5°$F, then rises $12.25°$F. Noon temperature? \answerBlank[2.5cm]
    \answer{$-11.25°$F}

    \stepPracticeHeader[step3Color]{Check and Interpret}
    \prob A family budgets $\$1{,}200$/month: $\frac{2}{5}$ on housing, $\frac{1}{4}$ on food, and $\$180$ on fuel. How much is left? \answerBlank[2.5cm]
    \answer{$\$240$}

    \wordProblem{Evaluate: $-\dfrac{1}{2} \times 8 + \dfrac{3}{4} \div \left(-\dfrac{1}{2}\right)$. Explain your steps.}{}
    \answer{$-5\frac{1}{2}$}
\end{stepPractice}
