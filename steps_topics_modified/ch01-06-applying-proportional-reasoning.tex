\section{Applying Proportional Reasoning}

\begin{stepGoal}
\begin{itemize}
    \item Solve proportion problems using the best strategy — including estimation
    \item Check answers for reasonableness before and after solving
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Estimate}{A quick, rounded answer that helps you check if your exact answer makes sense.}
    \vocabItem{Reasonableness}{Asking ``Does my answer make sense?'' by comparing it to your estimate.}
\end{stepVocabBox}

\begin{stepsCard}[How to Solve Proportional Problems with Estimation]
    \stepItem{\textbf{Estimate first} — round the numbers and predict an approximate answer.}
    \stepItem{Pick a strategy: \textbf{unit rate}, \textbf{proportion}, \textbf{equation}, or \textbf{scale factor}.}
    \stepItem{Solve and \textbf{label your answer} with units.}
    \stepItem{\textbf{Check}: does your exact answer fall near your estimate?}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{$5$ T-shirts cost $\$42.50$. How much for $8$ T-shirts?}
    \stepShow{1} Estimate: $\$42 \div 5 \approx \$8$ each, so $8$ shirts $\approx \$64$--$\$72$.

    \stepShow{2} Strategy: unit rate.

    \stepShow{3} $\$42.50 \div 5 = \$8.50$ per shirt. \; $\$8.50 \times 8 = \$68.00$.

    \stepShow{4} $\$68$ is in the $\$64$--$\$72$ range \checkmark
    \stepResult{$\$68.00$}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{A recipe for $4$ people uses $\frac{3}{2}$ cups of rice. How much for $10$ people?}
    \stepShow{1} Estimate: $10$ is $2.5$ times $4$, so about $1.5 \times 2.5 \approx 3.5$--$4$ cups.

    \stepShow{2} Strategy: unit rate.

    \stepShow{3} $\frac{3}{2} \div 4 = \frac{3}{8}$ cup per person. \; $\frac{3}{8} \times 10 = \frac{30}{8} = 3\frac{3}{4}$ cups.

    \stepShow{4} $3.75$ cups is near our estimate of $3.5$--$4$ \checkmark
    \stepResult{$3\frac{3}{4}$ cups}
\end{stepExample}

\watchOut{After solving, always ask: ``Does my answer make sense?'' If you're buying MORE, the cost should be HIGHER. If you're going a SHORTER distance, it should take LESS time.}

\newpage
\begin{stepPractice}{Applying Proportional Reasoning Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Estimate First}
    \prob A car travels $210$ miles on $6$ gallons. Estimate the distance on $10$ gallons, then solve exactly. \answerBlank[2.5cm]
    \answer{Estimate $\approx 350$ mi; exact $= 350$ miles}

    \stepPracticeHeader[step2Color]{Solve with Any Strategy}
    \prob On a map, $3$ cm $= 24$ km. A road is $8$ cm. Estimate, then find the exact distance. \answerBlank[2.5cm]
    \answer{Estimate $\approx 64$ km; exact $= 64$ km}

    \prob A printer prints $120$ pages in $8$ min. How long for $450$ pages? Is your answer reasonable? \answerBlank[2.5cm]
    \answer{$30$ min ($450$ is $\approx 4 \times 120$, so $\approx 4 \times 8 = 32$ — close \checkmark)}

    \stepPracticeHeader[step3Color]{Solve and Check Reasonableness}
    \prob Small paint: $\frac{3}{4}$ gal for $\$18$. Large: $1\frac{1}{2}$ gal for $\$33$. Estimate both unit prices, then solve. \answerBlank[3cm]
    \answer{Small $\approx \$24$/gal; Large $\approx \$22$/gal. Large saves $\$2$/gal.}

    \wordProblem{A factory makes $540$ bolts in $3$ hours. Estimate how many bolts in $7$ hours, then solve exactly.}{bolts}
    \answer{Estimate: $\approx 180 \times 7 = 1{,}260$. Exact: $540 \div 3 = 180$/hr; $180 \times 7 = 1{,}260$ bolts.}
\end{stepPractice}
