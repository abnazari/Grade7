\section{Solving Percent Problems}

\begin{stepGoal}
\begin{itemize}
    \item Find a percent of a number, the whole, or the percent
    \item Apply percents to real-world money contexts: discounts, tax, tips, and markups
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Percent}{A rate per $100$. The symbol $\%$ means ``out of $100$.''}
    \vocabItem{Part}{The piece of the whole you are looking at.}
    \vocabItem{Whole}{The total amount (the base).}
\end{stepVocabBox}

\begin{stepsCard}[How to Solve Percent Problems in Money Contexts]
    \stepItem{Identify which value is missing: the \textbf{part}, the \textbf{whole}, or the \textbf{percent}.}
    \stepItem{Write the equation: $\text{part} = \text{percent} \times \text{whole}$.}
    \stepItem{Convert the percent to a \textbf{decimal} and \textbf{solve}. For multi-step money problems, apply each percent change in order.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{A jacket costs $\$60$. It is $25\%$ off with $7\%$ sales tax on the sale price. How much do you pay?}
    \stepShow{1} Two steps: first find the discount (part missing), then find the tax (part missing again).

    \stepShow{2} Discount: $\text{part} = 0.25 \times 60 = \$15$. Sale price: $60 - 15 = \$45$.

    \stepShow{3} Tax: $0.07 \times 45 = \$3.15$. Total: $45 + 3.15 = \$48.15$.
    \stepResult{You pay $\$48.15$.}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{A backpack costs $\$40$ after a $20\%$ discount. What was the original price?}
    \stepShow{1} The \textbf{whole} (original price) is missing. After $20\%$ off you pay $80\%$.

    \stepShow{2} Write: $40 = 0.80 \times w$.

    \stepShow{3} Solve: $w = 40 \div 0.80 = \$50$.
    \stepResult{The original price was $\$50$.}
\end{stepExample}

\mascotStepTip{``Is'' means equals. ``Of'' means multiply. So ``$20\%$ of $\$50$'' becomes $0.20 \times 50 = \$10$.}

\newpage
\begin{stepPractice}{Percent Problems in Money Contexts}
    \resetProblems

    \stepPracticeHeader[step1Color]{Find the Part}
    \prob A meal costs $\$35$. You leave a $20\%$ tip. How much is the tip? \answerBlank[2cm]
    \answer{$\$7$}

    \stepPracticeHeader[step2Color]{Find the Whole or Percent}
    \begin{multicols}{2}
    \prob A store marks up a shirt from $\$12$ to $\$18$. What is the markup percent? \answerBlank[2cm]
    \answer{$50\%$}
    \prob A phone costs $\$250$ plus $8\%$ sales tax. What is the total? \answerBlank[2cm]
    \answer{$\$270$}
    \end{multicols}

    \stepPracticeHeader[step3Color]{Multi-Step Money Problems}
    \prob You spent $\$42$ on a meal including a $\$7$ tip. What percent tip did you leave on the food? \answerBlank[2cm]
    \answer{$20\%$}

    \wordProblem{Sneakers cost $\$90$. They are $30\%$ off, and sales tax is $5\%$. What is the final price?}{\$}
    \answer{$\$66.15$}
\end{stepPractice}
