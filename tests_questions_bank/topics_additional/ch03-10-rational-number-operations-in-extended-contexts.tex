\begin{practiceQuestion}{3-10-q01}{mc}
\begin{questionText}
A submarine is at $-250$ feet. It rises $80$ feet, then dives $120$ feet. What is its depth?
\end{questionText}
\choiceA{$-150$ ft}
\choiceB{$-290$ ft}
\choiceC{$-350$ ft}
\choiceD{$-170$ ft}
\correctAnswer{B}
\explanation{$-250 + 80 = -170$. Then $-170 + (-120) = -290$ ft.}
\end{practiceQuestion}

\begin{practiceQuestion}{3-10-q03}{mc}
\begin{questionText}
The temperature was $-4\dfrac{1}{2}°$F. It dropped another $6\dfrac{3}{4}°$F. What is the new temperature?
\end{questionText}
\choiceA{$-2\dfrac{1}{4}°$F}
\choiceB{$-10\dfrac{1}{4}°$F}

\end{practiceQuestion}

The bar graph shows temperature changes over $4$ days.

  \node[below] at (5.5,-4.7) {\footnotesize Thu};
