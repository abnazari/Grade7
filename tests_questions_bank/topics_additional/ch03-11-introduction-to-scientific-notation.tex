\begin{practiceQuestion}{3-11-q30}{gsa}
\begin{questionText}
The bar chart compares populations of four countries.

\begin{center}
\begin{tikzpicture}[yscale=0.6,xscale=1.2]
  \draw[thick,->] (0,0) -- (0,7) node[above] {\footnotesize Population ($\times 10^7$)};
  \draw[thick,->] (0,0) -- (5.5,0);
  \foreach \y in {1,...,6} \draw (-0.15,\y) -- (0.15,\y) node[left=4pt] {\footnotesize $\y$};
  \fill[funBlue!60] (0.5,0) rectangle (1.5,3.3);
  \fill[funRed!60] (2,0) rectangle (3,1.4);
  \fill[funGreen!60] (3.5,0) rectangle (4.5,5.1);
  \node[below] at (1,0) {\footnotesize X};
  \node[below] at (2.5,0) {\footnotesize Y};
  \node[below] at (4,0) {\footnotesize Z};
\end{tikzpicture}
\end{center}

If Country Z has about $5.1 \times 10^7$ people, write this population in standard form. Then estimate how many more people Country Z has than Country Y ($1.4 \times 10^7$).
\end{questionText}
\correctAnswer{$51{,}000{,}000$ people. Country Z has about $3.7 \times 10^7$ ($37{,}000{,}000$) more people than Country Y.}
\explanation{$5.1 \times 10^7 = 51{,}000{,}000$. Difference: $5.1 \times 10^7 - 1.4 \times 10^7 = 3.7 \times 10^7 = 37{,}000{,}000$.}
\end{practiceQuestion}
