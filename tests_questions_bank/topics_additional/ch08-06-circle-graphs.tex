\begin{practiceQuestion}{8-6-q30}{gsa}
\begin{questionText}
A survey of $120$ students asked about their favorite after-school activity. The results are:

\begin{center}
\begin{tabular}{|l|c|}
\hline
\textbf{Activity} & \textbf{Students} \\
\hline
Sports & $48$ \\
\hline
Music & $30$ \\
\hline
Art & $18$ \\
\hline
Reading & $24$ \\
\hline
\end{tabular}
\end{center}

Find the percent and angle (in degrees) for each category.
\end{questionText}
\correctAnswer{Sports: $40\%$, $144°$; Music: $25\%$, $90°$; Art: $15\%$, $54°$; Reading: $20\%$, $72°$}
\explanation{Sports: $\frac{48}{120} = 40\%$, $0.40 \times 360° = 144°$. Music: $\frac{30}{120} = 25\%$, $90°$. Art: $\frac{18}{120} = 15\%$, $54°$. Reading: $\frac{24}{120} = 20\%$, $72°$. Check: $144+90+54+72 = 360°$.}
\end{practiceQuestion}
