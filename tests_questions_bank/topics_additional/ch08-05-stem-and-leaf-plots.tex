\begin{practiceQuestion}{8-5-q01}{mc}
\begin{questionText}
In a stem-and-leaf plot, $5 \mid 3$ means:
\end{questionText}
\choiceA{$5.3$}
\choiceB{$35$}
\choiceC{$53$}
\choiceD{$5 + 3 = 8$}
\correctAnswer{C}
\explanation{The stem is the tens digit and the leaf is the ones digit, so $5 \mid 3$ means $53$.}
\end{practiceQuestion}
