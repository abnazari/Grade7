\begin{practiceQuestion}{m-3-8-q01}{mc}
\begin{questionText}
In Fairbanks, the temperature at $6$:$00$ a.m. was $-28.4°$F. By noon the temperature had risen by $15.7°$F. What was the temperature at noon?
\end{questionText}
\choiceA{$-44.1°$F}
\choiceB{$-12.7°$F}
\choiceC{$12.7°$F}
\choiceD{$-13.7°$F}
\correctAnswer{B}
\explanation{$-28.4 + 15.7 = -12.7$. The temperature at noon was $-12.7°$F.}
\end{practiceQuestion}

\begin{practiceQuestion}{m-3-8-q04}{mc}
\begin{questionText}
A hiking group on the Chilkoot Trail descended $1{,}450$ feet over $3.5$ hours. What was their average rate of elevation change per hour?
\end{questionText}
\choiceA{$-414\frac{2}{7}$ ft/hr}
\choiceB{$414\frac{2}{7}$ ft/hr}
\choiceC{$-5{,}075$ ft/hr}
\choiceD{$-400$ ft/hr}
\correctAnswer{A}
\explanation{Descent is negative. $\frac{-1{,}450}{3.5} = \frac{-1{,}450}{\frac{7}{2}} = -1{,}450 \times \frac{2}{7} = -\frac{2{,}900}{7} = -414\frac{2}{7}$ ft/hr.}
\end{practiceQuestion}

\begin{practiceQuestion}{m-3-8-q06}{mc}
\begin{questionText}
A bush pilot's fuel tank holds $87.5$ gallons. The plane uses $12\frac{1}{4}$ gallons per hour. After $5$ hours of flying, how much fuel remains?
\end{questionText}
\choiceA{$25\frac{1}{4}$ gallons}
\choiceB{$26\frac{1}{4}$ gallons}
\choiceC{$26.25$ gallons}
\choiceD{Both B and C}
\correctAnswer{D}
\explanation{Fuel used: $12\frac{1}{4} \times 5 = \frac{49}{4} \times 5 = \frac{245}{4} = 61\frac{1}{4}$ gal. Remaining: $87.5 - 61.25 = 26.25 = 26\frac{1}{4}$ gal.}
\end{practiceQuestion}

\begin{practiceQuestion}{m-3-8-q12}{mc}
\begin{questionText}
The tide at Juneau falls $2\frac{1}{3}$ feet per hour. If the current level is $14\frac{1}{2}$ feet above low tide, how many hours until it reaches $2\frac{1}{6}$ feet above low tide?
\end{questionText}
\choiceA{$5\frac{2}{7}$ hr}
\choiceB{$\frac{37}{7}$ hr}
\choiceC{Both A and B}
\choiceD{$6$ hr}
\correctAnswer{C}
\explanation{Drop needed: $14\frac{1}{2} - 2\frac{1}{6} = \frac{29}{2} - \frac{13}{6} = \frac{87}{6} - \frac{13}{6} = \frac{74}{6} = \frac{37}{3}$ ft. Time: $\frac{37/3}{7/3} = \frac{37}{3} \times \frac{3}{7} = \frac{37}{7} = 5\frac{2}{7}$ hr.}
\end{practiceQuestion}

\begin{practiceQuestion}{m-3-8-q19}{sa}
\begin{questionText}
A dog sled team covers $9\frac{3}{4}$ miles in the morning and $7\frac{2}{3}$ miles in the afternoon. If the total race distance is $25$ miles, how far do they still need to travel? Show your work.
\end{questionText}
\correctAnswer{$7\frac{7}{12}$ miles}
\explanation{Distance covered: $9\frac{3}{4} + 7\frac{2}{3} = \frac{39}{4} + \frac{23}{3} = \frac{117}{12} + \frac{92}{12} = \frac{209}{12} = 17\frac{5}{12}$. Remaining: $25 - 17\frac{5}{12} = \frac{300}{12} - \frac{209}{12} = \frac{91}{12} = 7\frac{7}{12}$ mi.}
\end{practiceQuestion}
