\begin{practiceQuestion}{m-2-6-q01}{mc}
\begin{questionText}
What is the simple interest earned on $\$500$ at $4\%$ for $3$ years?
\end{questionText}
\choiceA{$\$6$}
\choiceB{$\$20$}
\choiceC{$\$60$}
\choiceD{$\$600$}
\correctAnswer{C}
\explanation{$I = Prt = 500 \times 0.04 \times 3 = \$60$.}
\end{practiceQuestion}
