\begin{practiceQuestion}{7-2-q30}{gsa}
\begin{questionText}
The figure shows a running track made of two straight sides and two semicircles. Each straight side is $100$ m. Each semicircle has a diameter of $50$ m. What is the total distance around the track? Use $\pi \approx 3.14$.

\begin{center}
\begin{tikzpicture}[scale=0.04]
  \draw[thick,funBlue] (0,25) -- (100,25);
  \draw[thick,funBlue] (0,-25) -- (100,-25);
  \draw[thick,funBlue] (100,25) arc (90:-90:25);
  \draw[thick,funBlue] (0,-25) arc (270:90:25);
  \draw[thick,funRed,<->] (0,-35) -- (100,-35);
  \node[below,funRed] at (50,-35) {$100$ m};
  \draw[thick,funRed,<->] (110,25) -- (110,-25);
  \node[right,funRed] at (110,0) {$50$ m};
\end{tikzpicture}
\end{center}
\end{questionText}
\correctAnswer{$357$ m}
\explanation{Two straight sides: $2 \times 100 = 200$ m. Two semicircles form one full circle with $d = 50$: $C = \pi d = 3.14 \times 50 = 157$ m. Total: $200 + 157 = 357$ m.}
\end{practiceQuestion}
