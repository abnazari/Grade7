\begin{practiceQuestion}{3-4-q01}{mc}
\begin{questionText}
What is $-3.5 + 1.8$?
\end{questionText}
\choiceA{$-5.3$}
\choiceB{$-1.7$}
\choiceC{$1.7$}
\choiceD{$5.3$}
\correctAnswer{B}
\explanation{Different signs: $3.5 - 1.8 = 1.7$. The larger absolute value ($3.5$) is negative, so the answer is $-1.7$.}
\end{practiceQuestion}

\begin{practiceQuestion}{3-4-q18}{mc}
\begin{questionText}
What must you add to $-\dfrac{3}{8}$ to get $\dfrac{1}{2}$?
\end{questionText}
\choiceA{$\dfrac{1}{8}$}
\choiceB{$\dfrac{7}{8}$}
\choiceC{$-\dfrac{1}{8}$}
\choiceD{$-\dfrac{7}{8}$}
\correctAnswer{B}
\explanation{$-\frac{3}{8} + n = \frac{1}{2} = \frac{4}{8}$. So $n = \frac{4}{8} - (-\frac{3}{8}) = \frac{4}{8} + \frac{3}{8} = \frac{7}{8}$.}
\end{practiceQuestion}
