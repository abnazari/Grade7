\begin{practiceQuestion}{3-5-q29}{gsa}

Use the area model below to find the product $\left(-\dfrac{2}{3}\right) \times \dfrac{3}{4}$.

\begin{center}
\begin{tikzpicture}[scale=1.5]
  \draw[thick] (0,0) rectangle (3,4);
  % Vertical divisions (thirds)
  \draw[dashed] (1,0) -- (1,4);
  \draw[dashed] (2,0) -- (2,4);
  % Horizontal divisions (fourths)
  \draw[dashed] (0,1) -- (3,1);
  \draw[dashed] (0,2) -- (3,2);
  \draw[dashed] (0,3) -- (3,3);
  % Shade 2/3 columns and 3/4 rows overlap
  \fill[funBlue!30] (0,0) rectangle (2,3);
  \node at (1,1.5) {$6$ of $12$};
  \node[below] at (0.5,-0.1) {\footnotesize $\frac{1}{3}$};
  \node[below] at (1.5,-0.1) {\footnotesize $\frac{1}{3}$};
  \node[below] at (2.5,-0.1) {\footnotesize $\frac{1}{3}$};
  \node[left] at (-0.1,0.5) {\footnotesize $\frac{1}{4}$};
  \node[left] at (-0.1,1.5) {\footnotesize $\frac{1}{4}$};
  \node[left] at (-0.1,2.5) {\footnotesize $\frac{1}{4}$};
