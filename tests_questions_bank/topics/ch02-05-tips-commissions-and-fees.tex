\begin{practiceQuestion}{2-5-q01}{mc}
\begin{questionText}
A restaurant bill is $\$40$. You leave a $15\%$ tip. How much is the tip?
\end{questionText}
\choiceA{$\$4.00$}
\choiceB{$\$5.00$}
\choiceC{$\$6.00$}
\choiceD{$\$8.00$}
\correctAnswer{C}
\explanation{$15\% = 0.15$. Tip: $40 \times 0.15 = \$6.00$.}
