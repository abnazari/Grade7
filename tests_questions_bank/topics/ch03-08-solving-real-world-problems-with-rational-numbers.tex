\begin{practiceQuestion}{3-8-q01}{mc}
\begin{questionText}
You have $\$125.50$ in your bank account. You withdraw $\$40.75$ and then deposit $\$18.20$. What is your new balance?
\end{questionText}
\choiceA{$\$184.45$}
\choiceB{$\$102.95$}
\choiceC{$\$66.55$}
\choiceD{$\$147.30$}
\correctAnswer{B}
\explanation{$125.50 - 40.75 + 18.20 = 84.75 + 18.20 = \$102.95$.}
\end{practiceQuestion}

\begin{practiceQuestion}{3-8-q06}{mc}
\begin{questionText}
Evaluate: $-\dfrac{1}{2} \times 8 + \dfrac{3}{4} \div \left(-\dfrac{1}{2}\right)$
\end{questionText}
\choiceA{$-2\dfrac{1}{2}$}
\choiceB{$-5\dfrac{1}{2}$}
\choiceC{$5\dfrac{1}{2}$}
\choiceD{$-2$}
\correctAnswer{B}
\explanation{$-\frac{1}{2} \times 8 = -4$. $\frac{3}{4} \div (-\frac{1}{2}) = \frac{3}{4} \times (-2) = -\frac{3}{2}$. $-4 + (-\frac{3}{2}) = -\frac{11}{2} = -5\frac{1}{2}$.}
\end{practiceQuestion}

\begin{practiceQuestion}{3-8-q18}{mc}
\begin{questionText}
A pool is being filled and drained simultaneously. Water enters at $3\dfrac{1}{2}$ gallons per minute and drains at $1\dfrac{3}{4}$ gallons per minute. What is the net change after $8$ minutes?
\end{questionText}
\choiceA{$14$ gallons}
\choiceB{$42$ gallons}
\choiceC{$28$ gallons}
\choiceD{$10$ gallons}
\correctAnswer{A}
\explanation{Net rate: $3\frac{1}{2} - 1\frac{3}{4} = \frac{7}{2} - \frac{7}{4} = \frac{14}{4} - \frac{7}{4} = \frac{7}{4}$ gal/min. After $8$ min: $\frac{7}{4} \times 8 = 14$ gallons.}
\end{practiceQuestion}
