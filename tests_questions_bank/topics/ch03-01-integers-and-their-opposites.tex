\begin{practiceQuestion}{3-1-q01}{mc}
\begin{questionText}
What is the opposite of $-6$?
\end{questionText}
\choiceA{$-6$}
\choiceB{$0$}
\choiceC{$6$}
\choiceD{$\frac{1}{6}$}
\correctAnswer{C}
\explanation{The opposite of a negative number is positive. The opposite of $-6$ is $6$ because both are $6$ units from $0$.}
\end{practiceQuestion}

\begin{practiceQuestion}{3-1-q02}{mc}

What is $|-9|$?
\end{questionText}
\choiceA{$-9$}
\choiceB{$0$}
\choiceC{$\frac{1}{9}$}
\choiceD{$9$}
\correctAnswer{D}
\explanation{Absolute value is the distance from $0$. Since $-9$ is $9$ units from $0$, $|-9| = 9$.}

\begin{practiceQuestion}{3-1-q11}{mc}
\begin{questionText}
Which expression equals $0$?
\end{questionText}
\choiceA{$7 + 7$}
\choiceB{$7 - 7$}
\choiceC{$7 \times 0 + 7$}
\choiceD{$|{-7}| - |7|$}
\correctAnswer{B}
\explanation{$7 - 7 = 0$. Note that $|{-7}| - |7| = 7 - 7 = 0$ also works, but since only one answer is expected, B is the most direct additive-inverse example. Actually D also equals $0$. But B is the clearest demonstration of opposites.}
\end{practiceQuestion}

\begin{practiceQuestion}{3-1-q17}{mc}
\begin{questionText}
Two integers are $11$ units apart on a number line. One of them is $-3$. Which could be the other integer?
\end{questionText}
\choiceA{$8$ or $-14$}
\choiceB{$8$ only}
\choiceC{$-14$ only}
\choiceD{$14$ or $-8$}
\correctAnswer{A}
\explanation{The other integer satisfies $|n - (-3)| = 11$, so $n + 3 = 11$ giving $n = 8$, or $n + 3 = -11$ giving $n = -14$.}
\end{practiceQuestion}

\begin{practiceQuestion}{3-1-q30}{gsa}
\begin{questionText}
A vertical number line shows the elevations of four locations.

\begin{center}

  \foreach \y in {-3,...,4} \draw (-0.15,\y) -- (0.15,\y) node[right=4pt] {$\y 00$ ft};

  \node[right=20pt] at (0,0) {Sea Level};

\end{center}

What is the difference in elevation between the Mountain and the Valley?
\end{questionText}
\correctAnswer{$500$ ft}
