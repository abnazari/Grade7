\begin{practiceQuestion}{8-2-q01}{mc}
\begin{questionText}
A random sample of $25$ fish from a lake has a mean length of $14$ inches. The best prediction for the mean length of all fish in the lake is:
\end{questionText}
\choiceA{Exactly $14$ inches}
\choiceB{About $14$ inches}
\choiceC{More than $14$ inches}
\choiceD{Less than $14$ inches}
\correctAnswer{B}
\explanation{The sample mean is a good estimate of the population mean, but may not be exact due to random variation.}
\end{practiceQuestion}

\begin{practiceQuestion}{8-2-q02}{mc}
\begin{questionText}
A teacher collects three random samples of $20$ students from a school. The sample means for test scores are $78$, $81$, and $80$. What is the best estimate for the population mean?
\end{questionText}
\choiceA{$78$}
\choiceB{$81$}
\choiceC{$79.67$}
\choiceD{$80$}
\correctAnswer{C}
\explanation{Average the three sample means: $\frac{78+81+80}{3} = \frac{239}{3} \approx 79.67$.}
\end{practiceQuestion}
