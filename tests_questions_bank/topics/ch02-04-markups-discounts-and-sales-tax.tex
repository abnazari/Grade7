\begin{practiceQuestion}{2-4-q13}{mc}
\begin{questionText}
Which is the correct order to calculate a final price?
\end{questionText}
\choiceA{Tax → Discount → Markup}
\choiceB{Markup → Tax → Discount}
\choiceC{Discount → Tax → Markup}
