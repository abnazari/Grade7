\begin{practiceQuestion}{7-3-q01}{mc}
\begin{questionText}
What is the area of a circle with a radius of $3$ cm? Use $\pi \approx 3.14$.
\end{questionText}
\choiceA{$9.42$ cm$^2$}
\choiceB{$18.84$ cm$^2$}
\choiceC{$28.26$ cm$^2$}
\choiceD{$113.04$ cm$^2$}
