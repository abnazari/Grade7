\begin{practiceQuestion}{2-3-q01}{mc}
\begin{questionText}
A shirt originally costs $\$40$. The price increases by $10\%$. What is the new price?
\end{questionText}
\choiceA{$\$4$}
\choiceB{$\$36$}
\choiceC{$\$44$}
\choiceD{$\$50$}
\correctAnswer{C}
\explanation{Multiply by the increase multiplier: $40 \times 1.10 = \$44$.}
\end{practiceQuestion}

\begin{practiceQuestion}{2-3-q02}{mc}
\begin{questionText}
A population grew from $5{,}000$ to $5{,}750$. What is the percent increase?
\end{questionText}
\choiceA{$12\%$}
\choiceB{$13\%$}
\choiceC{$15\%$}
\choiceD{$17.5\%$}
\correctAnswer{C}
\explanation{Change: $5{,}750 - 5{,}000 = 750$. Percent change: $750 \div 5{,}000 = 0.15 = 15\%$.}

\begin{practiceQuestion}{2-3-q26}{gmc}
\begin{questionText}
The bar graph shows a store's monthly sales. What is the percent increase from January to March?

\begin{center}
\begin{tikzpicture}[scale=0.55]
  \draw[thick,->] (0,0) -- (0,7) node[above]{\small Sales (\$)};
  \draw[thick,->] (0,0) -- (7,0);

  \fill[funBlue!60] (0.5,0) rectangle (1.5,4);
  \fill[funBlue!60] (2.5,0) rectangle (3.5,3);

  \node[below,font=\small] at (3,0) {Feb};
  \node[below,font=\small] at (5,0) {Mar};
  % Y-axis
  \foreach \y/\lab in {2/2000,4/4000,6/6000} {
    \draw (-0.1,\y) -- (0.1,\y) node[left,font=\small,xshift=-2pt] {\lab};
  }
  \node[above,font=\small] at (1,4) {$4{,}000$};
  \node[above,font=\small] at (3,3) {$3{,}000$};
