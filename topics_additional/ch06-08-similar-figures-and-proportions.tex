% State-specific — Similar Figures and Proportions
\section{Similar Figures and Proportions}

\begin{learningGoals}
\begin{itemize}
    \item Identify similar figures and find missing side lengths
    \item Use scale factors to solve real-world problems
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Similar Figures}
Two figures are \textbf{similar} if they have the same shape but not necessarily the same size.
\begin{itemize}[leftmargin=*]
    \item Corresponding angles are \textbf{equal}.
    \item Corresponding sides are \textbf{proportional} (same scale factor).
\end{itemize}
$$\frac{\text{side of figure A}}{\text{corresponding side of figure B}} = \text{scale factor}$$
\end{conceptBox}

\begin{workedExample}{Finding a Missing Side}
Triangle $ABC \sim$ Triangle $DEF$. $AB = 6$, $DE = 9$, $BC = 8$. Find $EF$.

Scale factor: $\frac{9}{6} = 1.5$

$EF = 8 \times 1.5 = 12$
\answerHighlight{$EF = 12$}
\end{workedExample}

\tipBox{Set up a proportion with corresponding sides: $\frac{AB}{DE} = \frac{BC}{EF}$. Cross-multiply to solve!}

\begin{practiceBox}{Similar Figures}
\resetProblems

\prob Rectangle $ABCD \sim$ Rectangle $EFGH$. $AB = 4$, $EF = 10$, $BC = 6$. Find $FG$.
\answerExplain{$15$}{Scale factor: $\frac{10}{4} = 2.5$. $FG = 6 \times 2.5 = 15$.}

\prob Two similar triangles have sides $5$ and $15$. What is the scale factor?
\answerExplain{$3$}{$15 \div 5 = 3$. The larger triangle is $3$ times the smaller.}

\prob A tree casts a $12$-ft shadow. A $5$-ft person casts a $3$-ft shadow. How tall is the tree?
\answerExplain{$20$ ft}{Set up: $\frac{5}{3} = \frac{h}{12}$. Cross-multiply: $3h = 60$. Divide: $h = 20$ ft.}

\prob Triangle $PQR \sim$ Triangle $STU$. $PQ = 10$, $ST = 4$, $QR = 15$, $PR = 20$. Find $TU$ and $SU$.
\answerExplain{$TU = 6$, $SU = 8$}{Scale: $\frac{4}{10} = 0.4$. $TU = 15 \times 0.4 = 6$. $SU = 20 \times 0.4 = 8$.}

\prob A photo is $4$ in.\ by $6$ in. It is enlarged so the short side is $10$ in. What is the long side?
\answerExplain{$15$ in.}{Scale: $\frac{10}{4} = 2.5$. Long side: $6 \times 2.5 = 15$ in.}

\end{practiceBox}
