\section{Data Displays Extended}

\begin{learningGoals}
\begin{itemize}
    \item Read and create frequency tables, double bar graphs, and line graphs
    \item Solve problems using information from data displays
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Types of Data Displays}
\begin{itemize}[leftmargin=*]
    \item \textbf{Frequency table:} lists categories and how often each occurs.
    \item \textbf{Double bar graph:} compares two data sets side by side using bars.
    \item \textbf{Line graph:} shows how data changes over time. Points are connected.
\end{itemize}
Choose the display that best fits the question. Tables are great for exact values. Graphs help you spot trends and compare.
\end{conceptBox}

\begin{workedExample}{Reading a Double Bar Graph}
A double bar graph shows test scores for Class A and Class B.

Class A: Mon $70$, Tue $75$, Wed $80$. \quad Class B: Mon $65$, Tue $80$, Wed $85$.

Which class improved more from Monday to Wednesday?

Class A: $80 - 70 = 10$ points. \quad Class B: $85 - 65 = 20$ points.
\answerHighlight{Class B improved more ($20$ points vs.\ $10$).}
\end{workedExample}

\tipBox{When comparing two data sets on the same graph, look at both the heights AND the trends!}

\begin{practiceBox}{Data Displays}
\resetProblems

\prob A frequency table shows: A-grade: $8$, B-grade: $12$, C-grade: $5$, D-grade: $3$. How many students total?
\answerExplain{$28$}{Add all frequencies: $8 + 12 + 5 + 3 = 28$.}

\prob From the table in Problem 1, what percent earned a B?
\answerExplain{$\approx 42.9\%$}{$\frac{12}{28} \approx 0.4286 \approx 42.9\%$.}

\prob A line graph shows temperatures: Mon $60°$, Tue $55°$, Wed $58°$, Thu $62°$. Between which two days was the biggest change?
\answerExplain{Mon to Tue ($5°$ drop)}{Mon to Tue: $|60 - 55| = 5$. Tue to Wed: $3$. Wed to Thu: $4$. The biggest change is $5°$.}

\prob In a double bar graph, Team X scored $45, 50, 55$ over three games. Team Y scored $40, 55, 50$. Which team had the higher average?
\answerExplain{Team X}{Team X average: $(45+50+55) \div 3 = 50$. Team Y: $(40+55+50) \div 3 = 48.3$. Team X is higher.}

\trueOrFalse{A line graph is the best choice for showing change over time.}
\answerTF{True}

\end{practiceBox}
