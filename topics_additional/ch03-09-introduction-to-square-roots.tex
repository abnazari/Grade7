% State-specific — Introduction to Square Roots
\section{Introduction to Square Roots}

\begin{learningGoals}
\begin{itemize}
    \item Find square roots of perfect squares
    \item Estimate square roots of non-perfect squares
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Perfect Squares and Square Roots}
A \textbf{perfect square} is a number you get by multiplying an integer by itself.

$1, 4, 9, 16, 25, 36, 49, 64, 81, 100, 121, 144, \ldots$

The \textbf{square root} undoes squaring: $\sqrt{25} = 5$ because $5 \times 5 = 25$.

To \textbf{estimate} $\sqrt{n}$ when $n$ is not a perfect square, find the two perfect squares $n$ sits between.
\end{conceptBox}

\begin{workedExample}{Estimating a Square Root}
Estimate $\sqrt{40}$.

$36 < 40 < 49$, so $6 < \sqrt{40} < 7$.

$40$ is closer to $36$, so $\sqrt{40} \approx 6.3$.
\answerHighlight{$\sqrt{40} \approx 6.3$}
\end{workedExample}

\tipBox{Memorize perfect squares up to $15^2 = 225$. It makes estimating much faster!}

\begin{practiceBox}{Square Roots}
\resetProblems

\prob Find $\sqrt{81}$.
\answerExplain{$9$}{$9 \times 9 = 81$, so $\sqrt{81} = 9$.}

\prob Find $\sqrt{144}$.
\answerExplain{$12$}{$12 \times 12 = 144$, so $\sqrt{144} = 12$.}

\prob Between which two whole numbers is $\sqrt{50}$?
\answerExplain{$7$ and $8$}{$49 < 50 < 64$, so $7 < \sqrt{50} < 8$.}

\prob Estimate $\sqrt{20}$ to one decimal place.
\answerExplain{$\approx 4.5$}{$16 < 20 < 25$, so $4 < \sqrt{20} < 5$. Since $20$ is close to the middle, $\sqrt{20} \approx 4.5$.}

\prob A square garden has area $196$ ft$^2$. Find the side length.
\answerExplain{$14$ ft}{Side $= \sqrt{196} = 14$ ft because $14 \times 14 = 196$.}

\end{practiceBox}
