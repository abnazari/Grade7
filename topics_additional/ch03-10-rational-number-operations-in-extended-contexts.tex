\section{Rational Number Operations in Extended Contexts}

\begin{learningGoals}
\begin{itemize}
    \item Apply all four operations with rational numbers in real-world situations
    \item Interpret positive and negative answers in context
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Rational Numbers in the Real World}
Rational numbers appear everywhere:
\begin{itemize}[leftmargin=*]
    \item \textbf{Temperature:} $-15°$F means $15$ degrees below zero
    \item \textbf{Elevation:} $-200$ ft means $200$ ft below sea level
    \item \textbf{Money:} $-\$50$ means $\$50$ in debt
    \item \textbf{Stocks:} $+2.75$ means the price went up $\$2.75$
\end{itemize}
Choose the right operation, watch the signs, and check that your answer makes sense.
\end{conceptBox}

\begin{workedExample}{Multi-Step Temperature Problem}
At $6$ AM the temperature is $-8.5°$C. It rises $3.2°$ each hour for $4$ hours, then drops $5.6°$. Final temperature?

Rise: $3.2 \times 4 = 12.8°$. After rise: $-8.5 + 12.8 = 4.3°$C.

After drop: $4.3 - 5.6 = -1.3°$C.
\answerHighlight{$-1.3°$C}
\end{workedExample}

\mascotSays{Negative means below (zero, sea level, or the starting point). Positive means above!}

\begin{practiceBox}{Extended Contexts}
\resetProblems

\prob A submarine is at $-250$ ft. It rises $80$ ft, then dives $120$ ft. What is its depth?
\answerExplain{$-290$ ft}{$-250 + 80 = -170$. Then $-170 - 120 = -290$ ft.}

\prob A stock opens at $\$34.50$. It changes by $-\$1.25$, $+\$0.80$, and $-\$2.10$. Closing price?
\answerExplain{$\$31.95$}{$34.50 - 1.25 + 0.80 - 2.10 = 31.95$.}

\prob The temperature was $-4\frac{1}{2}°$F. It dropped another $6\frac{3}{4}°$. New temperature?
\answerExplain{$-11\frac{1}{4}°$F}{$-4\frac{1}{2} - 6\frac{3}{4} = -\frac{9}{2} - \frac{27}{4} = -\frac{18}{4} - \frac{27}{4} = -\frac{45}{4} = -11\frac{1}{4}°$F.}

\prob You owe $\$85.50$. You pay $\frac{2}{3}$ of it. How much do you still owe?
\answerExplain{$\$28.50$}{Paid: $\frac{2}{3} \times 85.50 = \$57$. Remaining: $85.50 - 57 = \$28.50$.}

\prob Death Valley is $-282$ ft. Mount Whitney is $14{,}505$ ft. What is the difference in elevation?
\answerExplain{$14{,}787$ ft}{$14{,}505 - (-282) = 14{,}505 + 282 = 14{,}787$ ft.}

\end{practiceBox}
