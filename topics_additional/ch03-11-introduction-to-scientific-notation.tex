% State-specific — Introduction to Scientific Notation
\section{Introduction to Scientific Notation}

\begin{learningGoals}
\begin{itemize}
    \item Write large and small numbers in scientific notation
    \item Compare and order numbers in scientific notation
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Scientific Notation}
Scientific notation writes a number as:
$$a \times 10^n$$
\begin{itemize}[leftmargin=*]
    \item $a$ is a number from $1$ to less than $10$ (including $1$, not including $10$)
    \item $n$ is an integer that tells you how many places to move the decimal
    \item \textbf{Positive} $n$: large number (move decimal right)
    \item \textbf{Negative} $n$: small number (move decimal left)
\end{itemize}
Example: $45{,}000 = 4.5 \times 10^4$ \quad and \quad $0.003 = 3 \times 10^{-3}$
\end{conceptBox}

\begin{workedExample}{Converting to Scientific Notation}
Write $7{,}200{,}000$ in scientific notation.

Move the decimal $6$ places left: $7.2$

$7{,}200{,}000 = 7.2 \times 10^6$
\answerHighlight{$7.2 \times 10^6$}
\end{workedExample}

\mascotSays{Count the hops! The exponent equals how many places you moved the decimal.}

\begin{practiceBox}{Scientific Notation}
\resetProblems

\prob Write $93{,}000{,}000$ in scientific notation.
\answerExplain{$9.3 \times 10^7$}{Move the decimal $7$ places left to get $9.3$. So $93{,}000{,}000 = 9.3 \times 10^7$.}

\prob Write $0.00056$ in scientific notation.
\answerExplain{$5.6 \times 10^{-4}$}{Move the decimal $4$ places right to get $5.6$. Since the number is small, the exponent is negative: $5.6 \times 10^{-4}$.}

\prob Convert $3.07 \times 10^5$ to standard form.
\answerExplain{$307{,}000$}{Move the decimal $5$ places right: $3.07 \to 307{,}000$.}

\prob Which is larger: $6.1 \times 10^8$ or $9.2 \times 10^7$?
\answerExplain{$6.1 \times 10^8$}{Compare exponents first. $10^8 > 10^7$, so $6.1 \times 10^8$ is larger.}

\prob The Sun is about $1.5 \times 10^8$ km from Earth. Write this in standard form.
\answerExplain{$150{,}000{,}000$ km}{Move the decimal $8$ places right: $1.5 \to 150{,}000{,}000$.}

\end{practiceBox}
