% State-specific — Stem-and-Leaf Plots
\section{Stem-and-Leaf Plots}

\begin{learningGoals}
\begin{itemize}
    \item Organize data using a stem-and-leaf plot
    \item Read and interpret stem-and-leaf plots to find measures of center
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{How Stem-and-Leaf Plots Work}
A \textbf{stem-and-leaf plot} organizes numbers by splitting each value into a \textbf{stem} (all digits except the last) and a \textbf{leaf} (the last digit).

Example data: $23, 25, 31, 34, 34, 38, 42$

\begin{center}
\begin{tabular}{r|l}
    \textbf{Stem} & \textbf{Leaf} \\
    \hline
    $2$ & $3\;\;5$ \\
    $3$ & $1\;\;4\;\;4\;\;8$ \\
    $4$ & $2$
\end{tabular}
\end{center}
Key: $3 \mid 4$ means $34$. Leaves are always written in order.
\end{conceptBox}

\begin{workedExample}{Finding the Median from a Stem-and-Leaf Plot}
Use the plot above. There are $7$ values: $23, 25, 31, 34, 34, 38, 42$.

The middle value (4th) is $34$.
\answerHighlight{Median $= 34$}
\end{workedExample}

\tipBox{Count the total number of leaves to find how many data values there are!}

\begin{practiceBox}{Stem-and-Leaf Plots}
\resetProblems

\prob Make a stem-and-leaf plot for: $15, 18, 22, 25, 27, 31, 33, 38$.
\answerExplain{Stems $1, 2, 3$; Leaves: $5\;8$, $2\;5\;7$, $1\;3\;8$}{Split each number: stem $1$: leaves $5, 8$; stem $2$: leaves $2, 5, 7$; stem $3$: leaves $1, 3, 8$.}

\prob From this plot, find the median. Stem $4$: $1\;3\;5$ \; Stem $5$: $0\;2\;7\;8$ \; Stem $6$: $4$
\answerExplain{$51$}{Data: $41, 43, 45, 50, 52, 57, 58, 64$. Eight values. Median $= \frac{50+52}{2} = 51$.}

\prob What is the range of the data set in Problem 2?
\answerExplain{$23$}{Range $= 64 - 41 = 23$.}

\prob How many data values are shown in: Stem $7$: $2\;2\;5$ \; Stem $8$: $0\;3\;6\;9$ \; Stem $9$: $1\;4$?
\answerExplain{$9$ values}{Count the leaves: $3 + 4 + 2 = 9$.}

\trueOrFalse{In a stem-and-leaf plot, $6 \mid 3$ means $63$.}
\answerTF{True}

\end{practiceBox}
