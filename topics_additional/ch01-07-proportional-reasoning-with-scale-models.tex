\section{Proportional Reasoning with Scale Models}

\begin{learningGoals}
\begin{itemize}
    \item Use scale factors to find actual dimensions from a model
    \item Apply proportional reasoning to scale-model problems
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Scale Models}
A \textbf{scale model} is a smaller (or larger) copy of a real object. The \textbf{scale factor} tells you how model measurements compare to real ones.

Scale $1 : 50$ means every $1$ cm on the model equals $50$ cm in real life.

\textbf{Actual length} $=$ model length $\times$ scale factor

\textbf{Model length} $=$ actual length $\div$ scale factor
\end{conceptBox}

\begin{workedExample}{Finding Actual Dimensions}
A model car is $8$ cm long. The scale is $1 : 25$. What is the actual length?

Actual $= 8 \times 25 = 200$ cm $= 2$ m
\answerHighlight{The real car is $2$ m long.}
\end{workedExample}

\mascotSays{Scale factor works like a unit rate --- multiply to go big, divide to go small!}

\begin{practiceBox}{Scale Models}
\resetProblems

\prob A model building is $12$ cm tall. Scale: $1 : 200$. Find the actual height.
\answerExplain{$24$ m}{Multiply: $12 \times 200 = 2{,}400$ cm. Convert: $2{,}400 \div 100 = 24$ m.}

\prob A bridge is $450$ m long. Scale: $1 : 500$. How long is the model?
\answerExplain{$90$ cm}{Divide: $450 \div 500 = 0.9$ m $= 90$ cm.}

\prob A model airplane wing is $15$ cm. The real wing is $12$ m. What is the scale factor?
\answerExplain{$1 : 80$}{Convert: $12$ m $= 1{,}200$ cm. Divide: $1{,}200 \div 15 = 80$. Scale is $1 : 80$.}

\prob A room is $6$ m by $4$ m. Scale: $1 : 50$. Find the model's dimensions.
\answerExplain{$12$ cm by $8$ cm}{Length: $600 \div 50 = 12$ cm. Width: $400 \div 50 = 8$ cm.}

\prob A $1 : 100$ model house is $9$ cm tall and $14$ cm wide. Find the actual dimensions.
\answerExplain{$9$ m tall, $14$ m wide}{Multiply each by $100$: $9 \times 100 = 900$ cm $= 9$ m; $14 \times 100 = 1{,}400$ cm $= 14$ m.}

\end{practiceBox}
