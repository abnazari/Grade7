% State-specific — Cylinder Surface Area and Volume
\section{Cylinder Surface Area and Volume}

\begin{learningGoals}
\begin{itemize}
    \item Find the surface area and volume of cylinders
    \item Apply cylinder formulas to real-world problems
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Cylinder Formulas}
A \textbf{cylinder} has two circular bases and a curved surface.

\formulaCard{Volume}{$V = \pi r^2 h$}

\formulaCard{Surface Area}{$SA = 2\pi r^2 + 2\pi r h$}

\begin{itemize}[leftmargin=*]
    \item $r$ = radius of the base
    \item $h$ = height of the cylinder
    \item $2\pi r^2$ = area of both circular bases
    \item $2\pi rh$ = area of the curved side (like a label wrapped around)
\end{itemize}
\end{conceptBox}

\begin{workedExample}{Finding Volume and Surface Area}
A can has radius $3$ cm and height $10$ cm. Use $\pi \approx 3.14$.

$V = \pi r^2 h = 3.14 \times 9 \times 10 = 282.6$ cm$^3$

$SA = 2\pi r^2 + 2\pi rh = 2(3.14)(9) + 2(3.14)(3)(10) = 56.52 + 188.4 = 244.92$ cm$^2$
\answerHighlight{$V \approx 282.6$ cm$^3$; $SA \approx 244.92$ cm$^2$}
\end{workedExample}

\mascotSays{Think of surface area like wrapping a can --- two circles for the top and bottom, plus a rectangle that wraps around!}

\begin{practiceBox}{Cylinders}
\resetProblems

\prob Find the volume of a cylinder with $r = 5$ cm and $h = 8$ cm. Use $\pi \approx 3.14$.
\answerExplain{$628$ cm$^3$}{$V = 3.14 \times 25 \times 8 = 628$ cm$^3$.}

\prob Find the surface area of a cylinder with $r = 4$ in.\ and $h = 6$ in.
\answerExplain{$\approx 251.2$ in$^2$}{$SA = 2(3.14)(16) + 2(3.14)(4)(6) = 100.48 + 150.72 = 251.2$ in$^2$.}

\prob A water tank is a cylinder with $r = 2$ m and $h = 5$ m. How many cubic meters of water does it hold?
\answerExplain{$\approx 62.8$ m$^3$}{$V = 3.14 \times 4 \times 5 = 62.8$ m$^3$.}

\prob A soup can has diameter $8$ cm and height $12$ cm. Find its volume.
\answerExplain{$\approx 602.88$ cm$^3$}{Radius $= 4$ cm. $V = 3.14 \times 16 \times 12 = 602.88$ cm$^3$.}

\prob A cylinder has volume $314$ cm$^3$ and $r = 5$ cm. Find the height.
\answerExplain{$4$ cm}{$314 = 3.14 \times 25 \times h$. So $314 = 78.5h$. Divide: $h = 314 \div 78.5 = 4$ cm.}

\end{practiceBox}
