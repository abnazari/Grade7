% State-specific — Transformations on the Coordinate Plane
\section{Transformations on the Coordinate Plane}

\begin{learningGoals}
\begin{itemize}
    \item Identify and perform translations, reflections, and rotations
    \item Describe how transformations change coordinates
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Three Types of Transformations}
\begin{itemize}[leftmargin=*]
    \item \textbf{Translation} (slide): Move every point the same distance and direction.
    $(x, y) \to (x + a,\; y + b)$
    \item \textbf{Reflection} (flip): Mirror a figure across a line.
    Over the $x$-axis: $(x, y) \to (x, -y)$.
    Over the $y$-axis: $(x, y) \to (-x, y)$.
    \item \textbf{Rotation} (turn): Spin a figure around a point.
    $90°$ clockwise about the origin: $(x, y) \to (y, -x)$.
\end{itemize}
All three keep the \textbf{size and shape} the same.
\end{conceptBox}

\begin{workedExample}{Translating a Point}
Translate $A(2, 5)$ right $3$ and down $4$.

$(2 + 3,\; 5 - 4) = (5, 1)$
\answerHighlight{$A'(5, 1)$}
\end{workedExample}

\mascotSays{After a reflection over the $x$-axis, the $x$-coordinate stays the same but the $y$ flips sign!}

\begin{practiceBox}{Transformations}
\resetProblems

\prob Translate $B(1, -3)$ left $4$ and up $6$. Find $B'$.
\answerExplain{$B'(-3, 3)$}{Left $4$: $1 - 4 = -3$. Up $6$: $-3 + 6 = 3$. So $B'(-3, 3)$.}

\prob Reflect $C(4, -2)$ over the $x$-axis. Find $C'$.
\answerExplain{$C'(4, 2)$}{Over the $x$-axis: keep $x$, flip $y$. $(4, -2) \to (4, 2)$.}

\prob Reflect $D(-5, 3)$ over the $y$-axis. Find $D'$.
\answerExplain{$D'(5, 3)$}{Over the $y$-axis: flip $x$, keep $y$. $(-5, 3) \to (5, 3)$.}

\prob Rotate $E(3, 1)$ by $90°$ clockwise about the origin. Find $E'$.
\answerExplain{$E'(1, -3)$}{$90°$ clockwise: $(x, y) \to (y, -x)$. $(3, 1) \to (1, -3)$.}

\prob A triangle has vertices $P(1, 2)$, $Q(4, 2)$, $R(4, 5)$. Translate it right $2$, down $3$. List the new vertices.
\answerExplain{$P'(3,-1)$, $Q'(6,-1)$, $R'(6, 2)$}{Add $2$ to each $x$ and subtract $3$ from each $y$: $(1+2, 2-3)=(3,-1)$; $(4+2, 2-3)=(6,-1)$; $(4+2, 5-3)=(6,2)$.}

\end{practiceBox}
