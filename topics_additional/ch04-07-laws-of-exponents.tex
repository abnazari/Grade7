% State-specific — Laws of Exponents
\section{Laws of Exponents}

\begin{learningGoals}
\begin{itemize}
    \item Apply the product, quotient, and power-of-a-power rules
    \item Evaluate expressions using exponent laws
\end{itemize}
\end{learningGoals}

\begin{conceptBox}{Exponent Rules}
\begin{center}
\begin{tabular}{l l l}
    \textbf{Rule} & \textbf{Formula} & \textbf{Example} \\
    \hline
    Product Rule & $a^m \cdot a^n = a^{m+n}$ & $2^3 \cdot 2^4 = 2^7 = 128$ \\
    Quotient Rule & $\dfrac{a^m}{a^n} = a^{m-n}$ & $\dfrac{5^6}{5^2} = 5^4 = 625$ \\
    Power of a Power & $(a^m)^n = a^{mn}$ & $(3^2)^3 = 3^6 = 729$ \\
    Zero Exponent & $a^0 = 1$ \;(if $a \neq 0$) & $7^0 = 1$ \\
\end{tabular}
\end{center}
\end{conceptBox}

\begin{workedExample}{Using the Product Rule}
Simplify $4^3 \cdot 4^2$.

Same base, so add exponents: $4^{3+2} = 4^5 = 1{,}024$.
\answerHighlight{$4^5 = 1{,}024$}
\end{workedExample}

\tipBox{Exponent rules only work when the bases are the same. $2^3 \cdot 3^2$ cannot be simplified with these rules.}

\begin{practiceBox}{Laws of Exponents}
\resetProblems

\prob Simplify $6^2 \cdot 6^3$.
\answerExplain{$6^5 = 7{,}776$}{Add exponents: $6^{2+3} = 6^5$. Calculate: $6^5 = 7{,}776$.}

\prob Simplify $\dfrac{10^7}{10^4}$.
\answerExplain{$10^3 = 1{,}000$}{Subtract exponents: $10^{7-4} = 10^3 = 1{,}000$.}

\prob Simplify $(2^3)^4$.
\answerExplain{$2^{12} = 4{,}096$}{Multiply exponents: $2^{3 \times 4} = 2^{12}$. Calculate: $2^{12} = 4{,}096$.}

\prob What is $(-3)^0$?
\answerExplain{$1$}{Any nonzero number to the zero power equals $1$.}

\prob Simplify $\left(\frac{1}{2}\right)^2 \cdot \left(\frac{1}{2}\right)^3$.
\answerExplain{$\left(\frac{1}{2}\right)^5 = \frac{1}{32}$}{Add exponents: $\left(\frac{1}{2}\right)^{2+3} = \left(\frac{1}{2}\right)^5 = \frac{1}{32}$.}

\end{practiceBox}
