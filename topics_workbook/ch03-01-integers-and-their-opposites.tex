\section{Integers and Their Opposites}
\topicTitle{Integers and Their Opposites}

\begin{quickReview}{Integers and Opposites}
\textbf{Integers} are whole numbers and their negatives: $\ldots, -3, -2, -1, 0, 1, 2, 3, \ldots$

\begin{itemize}[leftmargin=*]
    \item Every integer has an \textbf{opposite} — the number the same distance from $0$ on the other side.
    \item The opposite of $7$ is $-7$. The opposite of $-4$ is $4$. The opposite of $0$ is $0$.
    \item A number plus its opposite always equals $0$: $\;a + (-a) = 0$.
    \item The \textbf{absolute value} $|n|$ is the distance from $0$. It is always $\geq 0$.
\end{itemize}

\medskip
\textbf{Example:} $|{-9}| = 9$ because $-9$ is $9$ units from $0$. \quad $|5| = 5$ because $5$ is $5$ units from $0$.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Find the Opposite}

\begin{multicols}{2}
\prob What is the opposite of $6$? \answerBlank[2cm]
\answerExplain{$-6$}{The opposite of a positive number is negative. Since $6$ is $6$ units right of $0$, its opposite is $6$ units left of $0$, which is $-6$.}

\prob What is the opposite of $-10$? \answerBlank[2cm]
\answerExplain{$10$}{The opposite of a negative number is positive. $-10$ is $10$ units left of $0$, so its opposite is $10$ units right of $0$, which is $10$.}

\prob What is the opposite of $0$? \answerBlank[2cm]
\answerExplain{$0$}{Zero is its own opposite. It sits at the center of the number line and is the same distance ($0$ units) from itself on both sides.}

\prob Find $|{-14}|$. \answerBlank[2cm]
\answerExplain{$14$}{Absolute value measures distance from $0$. Since $-14$ is $14$ units from $0$, $|{-14}| = 14$.}

\prob Find $|23|$. \answerBlank[2cm]
\answerExplain{$23$}{Absolute value measures distance from $0$. Since $23$ is $23$ units from $0$, $|23| = 23$.}

\prob What is $8 + (-8)$? \answerBlank[2cm]
\answerExplain{$0$}{A number plus its opposite always equals $0$. Since $8$ and $-8$ are opposites, $8 + (-8) = 0$.}
\end{multicols}
\end{practiceBox}

\begin{practiceBox}{Opposites and Absolute Value}

\practiceHeader{Write the opposite of each integer.}

\begin{multicols}{3}
\prob $-25$ \answerBlank[2cm]
\answerExplain{$25$}{The opposite of $-25$ is the number the same distance from $0$ on the positive side, which is $25$.}

\prob $17$ \answerBlank[2cm]
\answerExplain{$-17$}{The opposite of $17$ is the number the same distance from $0$ on the negative side, which is $-17$.}

\prob $-100$ \answerBlank[2cm]
\answerExplain{$100$}{The opposite of $-100$ is $100$. Both are $100$ units from $0$ on opposite sides of the number line.}
\end{multicols}

\practiceHeader{Find each absolute value.}

\begin{multicols}{3}
\prob $|{-33}|$ \answerBlank[2cm]
\answerExplain{$33$}{Absolute value is distance from $0$. Since $-33$ is $33$ units from $0$, $|{-33}| = 33$.}

\prob $|0|$ \answerBlank[2cm]
\answerExplain{$0$}{Zero is $0$ units from itself on the number line, so $|0| = 0$.}

\prob $|{-71}|$ \answerBlank[2cm]
\answerExplain{$71$}{Since $-71$ is $71$ units from $0$ on the number line, $|{-71}| = 71$.}
\end{multicols}

\practiceHeader{Compare using $<$, $>$, or $=$.}

\prob $-5$ \answerBlank[1cm] $3$
\answerExplain{$<$}{On the number line, $-5$ is to the left of $3$, so $-5 < 3$. Negative numbers are always less than positive numbers.}

\prob $|{-8}|$ \answerBlank[1cm] $|6|$
\answerExplain{$>$}{$|{-8}| = 8$ and $|6| = 6$. Since $8 > 6$, we have $|{-8}| > |6|$.}

\prob $-12$ \answerBlank[1cm] $-4$
\answerExplain{$<$}{On the number line, $-12$ is farther left than $-4$, so $-12 < -4$. A more negative number is always less.}

\end{practiceBox}

% ── Practice B: Ordering Integers ────────────────────────────────────────────
\begin{practiceBox}[funTeal]{Ordering Integers}

\practiceHeader[funTeal]{Order each set from least to greatest.}

\prob $3, \; -7, \; 1, \; -2, \; 0$ \answerBlank[5cm]
\answerExplain{$-7, \; -2, \; 0, \; 1, \; 3$}{Place each number on a number line. From left to right: $-7$ is farthest left, then $-2$, then $0$, then $1$, then $3$.}

\prob $-15, \; 8, \; -3, \; 12, \; -8$ \answerBlank[5cm]
\answerExplain{$-15, \; -8, \; -3, \; 8, \; 12$}{Order negatives first (more negative = smaller): $-15 < -8 < -3$. Then the positives: $8 < 12$.}

\prob $-1, \; -10, \; 10, \; 0, \; -5, \; 5$ \answerBlank[5cm]
\answerExplain{$-10, \; -5, \; -1, \; 0, \; 5, \; 10$}{Negatives from most negative to least: $-10, -5, -1$. Then $0$, and positives in order: $5, 10$.}

\end{practiceBox}

% ── Practice C: True or False? ───────────────────────────────────────────────
\begin{practiceBox}[funPurple]{True or False?}

\trueOrFalse{The absolute value of a number can be negative.}
\answerExplain{False}{Absolute value measures distance, which is always zero or positive. It can never be negative.}

\trueOrFalse{$-3$ is farther from $0$ than $2$ is.}
\answerExplain{True}{$|{-3}| = 3$ and $|2| = 2$. Since $3 > 2$, the number $-3$ is farther from $0$.}

\trueOrFalse{Every positive integer has a negative opposite.}
\answerExplain{True}{By definition, the opposite of a positive integer $n$ is $-n$, which is negative.}

\trueOrFalse{$|{-5}| = |5|$}
\answerExplain{True}{Both $-5$ and $5$ are $5$ units from $0$, so their absolute values are equal: $|{-5}| = 5 = |5|$.}

\end{practiceBox}

% ── Practice D: Word Problems ────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{A submarine is at $-200$ feet (below sea level). It rises $200$ feet. What is its new position? Write the answer using integers.}{feet}
\answerExplain{$0$ feet (sea level)}{Rising $200$ feet from $-200$: $-200 + 200 = 0$. The submarine is at sea level because $-200$ and $200$ are opposites.}

\wordProblem{The temperature at midnight was $-8°$C. During the day, the temperature changed by the opposite amount. What was the daytime temperature change, and what was the resulting temperature?}{°C}
\answerExplain{Change of $8°$C; resulting temperature $0°$C}{The opposite of $-8$ is $8$, so the temperature rose $8°$C. New temperature: $-8 + 8 = 0°$C.}

\wordProblem{Mount Everest's peak is about $8{,}849$ meters above sea level. The Mariana Trench is about $10{,}994$ meters below sea level. Write each elevation as an integer and find which has a greater absolute value.}{meters}
\answerExplain{$8{,}849$ m and $-10{,}994$ m; Mariana Trench has greater absolute value}{Everest: $+8{,}849$. Mariana Trench: $-10{,}994$. Compare: $|{-10{,}994}| = 10{,}994 > 8{,}849 = |8{,}849|$. The trench is farther from sea level.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}

\prob If $|x| = 7$, what are the possible values of $x$? \answerBlank[3cm]
\answerExplain{$x = 7$ or $x = -7$}{Absolute value equals $7$ means the number is $7$ units from $0$. Two numbers are $7$ units from $0$: $7$ (to the right) and $-7$ (to the left).}

\prob If $|a| = |b|$ and $a \neq b$, what must be true about $a$ and $b$? \answerBlank[5cm]
\answerExplain{$a$ and $b$ are opposites ($a = -b$)}{If two different numbers have the same absolute value, they are the same distance from $0$ but on opposite sides. That means one is the opposite of the other.}

\end{challengeBox}
