\section{Solving Two-Step Equations}
\topicTitle{Solving Two-Step Equations}

\begin{quickReview}{Solving Two-Step Equations}
To solve $px + q = r$, undo operations in \textbf{reverse order}:
\begin{itemize}[leftmargin=*]
    \item \textbf{Step 1:} Undo addition or subtraction (move the constant).
    \item \textbf{Step 2:} Undo multiplication or division (isolate the variable).
\end{itemize}

\medskip
\textbf{Example:} Solve $3x + 7 = 22$.

$3x + 7 = 22 \;\xrightarrow{-\,7}\; 3x = 15 \;\xrightarrow{\div\,3}\; x = 5$.

\smallskip
\textbf{Always check:} $3(5) + 7 = 15 + 7 = 22$ \checkmark
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{One-Step Equations}
Solve each equation.

\begin{multicols}{2}
\prob $x + 9 = 14$ \answerBlank[2.5cm]
\answerExplain{$x = 5$}{Subtract $9$ from both sides: $x = 14 - 9 = 5$.}

\prob $n - 6 = 11$ \answerBlank[2.5cm]
\answerExplain{$n = 17$}{Add $6$ to both sides: $n = 11 + 6 = 17$.}

\prob $4y = 36$ \answerBlank[2.5cm]
\answerExplain{$y = 9$}{Divide both sides by $4$: $y = 36 \div 4 = 9$.}

\prob $\dfrac{m}{5} = 8$ \answerBlank[2.5cm]
\answerExplain{$m = 40$}{Multiply both sides by $5$: $m = 8 \times 5 = 40$.}

\prob $-3a = 21$ \answerBlank[2.5cm]
\answerExplain{$a = -7$}{Divide both sides by $-3$: $a = 21 \div (-3) = -7$.}

\prob $w + 15 = 3$ \answerBlank[2.5cm]
\answerExplain{$w = -12$}{Subtract $15$ from both sides: $w = 3 - 15 = -12$.}
\end{multicols}
\end{practiceBox}

% ── Practice A: Solve Two-Step Equations ─────────────────────────────────────
\begin{practiceBox}{Solve Two-Step Equations}

\practiceHeader{Solve each equation. Show your work.}

\prob $2x + 5 = 19$ \answerBlank[3cm]
\answerExplain{$x = 7$}{Subtract $5$: $2x = 14$. Divide by $2$: $x = 7$. Check: $2(7) + 5 = 14 + 5 = 19$ \checkmark.}

\prob $4n - 3 = 25$ \answerBlank[3cm]
\answerExplain{$n = 7$}{Add $3$: $4n = 28$. Divide by $4$: $n = 7$. Check: $4(7) - 3 = 28 - 3 = 25$ \checkmark.}

\prob $\dfrac{y}{3} + 8 = 14$ \answerBlank[3cm]
\answerExplain{$y = 18$}{Subtract $8$: $\frac{y}{3} = 6$. Multiply by $3$: $y = 18$. Check: $\frac{18}{3} + 8 = 6 + 8 = 14$ \checkmark.}

\prob $-5x + 12 = -3$ \answerBlank[3cm]
\answerExplain{$x = 3$}{Subtract $12$: $-5x = -15$. Divide by $-5$: $x = 3$. Check: $-5(3) + 12 = -15 + 12 = -3$ \checkmark.}

\prob $7a + 10 = -4$ \answerBlank[3cm]
\answerExplain{$a = -2$}{Subtract $10$: $7a = -14$. Divide by $7$: $a = -2$. Check: $7(-2) + 10 = -14 + 10 = -4$ \checkmark.}

\prob $\dfrac{m}{6} - 5 = 3$ \answerBlank[3cm]
\answerExplain{$m = 48$}{Add $5$: $\frac{m}{6} = 8$. Multiply by $6$: $m = 48$. Check: $\frac{48}{6} - 5 = 8 - 5 = 3$ \checkmark.}

\end{practiceBox}

% ── Practice B: Equations with Negatives ─────────────────────────────────────
\begin{practiceBox}[funTeal]{Equations with Negative Numbers}

\practiceHeader[funTeal]{Solve. Watch the signs carefully!}

\prob $-3x - 4 = 11$ \answerBlank[3cm]
\answerExplain{$x = -5$}{Add $4$: $-3x = 15$. Divide by $-3$: $x = -5$. Check: $-3(-5) - 4 = 15 - 4 = 11$ \checkmark.}

\prob $6n + 7 = -5$ \answerBlank[3cm]
\answerExplain{$n = -2$}{Subtract $7$: $6n = -12$. Divide by $6$: $n = -2$. Check: $6(-2) + 7 = -12 + 7 = -5$ \checkmark.}

\prob $-2y + 9 = 1$ \answerBlank[3cm]
\answerExplain{$y = 4$}{Subtract $9$: $-2y = -8$. Divide by $-2$: $y = 4$. Check: $-2(4) + 9 = -8 + 9 = 1$ \checkmark.}

\prob $\dfrac{p}{-4} + 6 = 3$ \answerBlank[3cm]
\answerExplain{$p = 12$}{Subtract $6$: $\frac{p}{-4} = -3$. Multiply by $-4$: $p = 12$. Check: $\frac{12}{-4} + 6 = -3 + 6 = 3$ \checkmark.}

\prob $-8k - 5 = -37$ \answerBlank[3cm]
\answerExplain{$k = 4$}{Add $5$: $-8k = -32$. Divide by $-8$: $k = 4$. Check: $-8(4) - 5 = -32 - 5 = -37$ \checkmark.}

\end{practiceBox}

% ── Spot the Error ───────────────────────────────────────────────────────────
\begin{errorBox}
A student solved $5x - 8 = 22$ this way:

$5x - 8 = 22$

$5x = 14$ \quad (subtracted $8$)

$x = \frac{14}{5}$

\medskip
\prob What mistake did the student make? What is the correct answer? \answerBlank[3cm]
\answerExplain{$x = 6$}{The student subtracted $8$ instead of adding $8$. The correct first step is: $5x = 22 + 8 = 30$. Then $x = 30 \div 5 = 6$. To undo ``$-8$,'' you must add $8$, not subtract.}
\end{errorBox}

% ── Practice C: Word Problems ────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{A gym charges $\$15$ per month plus a $\$30$ registration fee. You have paid a total of $\$135$. How many months have you been a member?}{months}
\answerExplain{$7$ months}{Equation: $15m + 30 = 135$. Subtract $30$: $15m = 105$. Divide by $15$: $m = 7$ months.}

\wordProblem{You start with $\$80$ and spend $\$6$ per day. After how many days will you have $\$20$ left?}{days}
\answerExplain{$10$ days}{Equation: $80 - 6d = 20$. Subtract $80$: $-6d = -60$. Divide by $-6$: $d = 10$ days.}

\wordProblem{A catering company charges $\$7$ per guest plus a $\$50$ set-up fee. The total bill was $\$169$. How many guests were at the party?}{guests}
\answerExplain{$17$ guests}{Equation: $7g + 50 = 169$. Subtract $50$: $7g = 119$. Divide by $7$: $g = 17$ guests.}

\wordProblem{After depositing $\$25$ a week into an account that started with $\$40$, Maya had $\$240$ in total. How many weeks did she deposit money?}{weeks}
\answerExplain{$8$ weeks}{Equation: $25w + 40 = 240$. Subtract $40$: $25w = 200$. Divide by $25$: $w = 8$ weeks.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}

\prob The perimeter of a rectangle is $50$ cm. The length is $3$ cm more than the width. Write and solve an equation to find the width $w$.
\answerExplain{$w = 11$ cm}{Length $= w + 3$. Perimeter: $2w + 2(w + 3) = 50$. Simplify: $4w + 6 = 50$. Subtract $6$: $4w = 44$. Divide by $4$: $w = 11$ cm. Length $= 14$ cm.}

\prob Two siblings share their earnings. One earns $\$5$ more than twice what the other earns. Together they earned $\$53$. Write and solve an equation for the lesser amount $x$.
\answerExplain{$x = 16$}{The other earns $2x + 5$. Together: $x + (2x + 5) = 53$. Simplify: $3x + 5 = 53$. Subtract $5$: $3x = 48$. Divide by $3$: $x = 16$. The other earns $\$37$.}

\end{challengeBox}
