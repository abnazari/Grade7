\section{Solving Percent Problems}
\topicTitle{Solving Percent Problems}

\begin{quickReview}{Solving Percent Problems}
Every percent problem has three parts: the \textbf{part}, the \textbf{whole}, and the \textbf{percent}.
$$\text{part} = \text{percent} \times \text{whole}$$

\begin{itemize}[leftmargin=*]
    \item \textbf{Find the part:} What is $30\%$ of $80$? $\to$ $0.30 \times 80 = 24$
    \item \textbf{Find the whole:} $15$ is $25\%$ of what? $\to$ $15 \div 0.25 = 60$
    \item \textbf{Find the percent:} $12$ is what percent of $48$? $\to$ $12 \div 48 = 0.25 = 25\%$
\end{itemize}

\medskip
\textbf{Tip:} ``Is'' means equals. ``Of'' means multiply. Convert percents to decimals by dividing by $100$.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Find the Part}
Convert the percent to a decimal and multiply.

\begin{multicols}{2}
\prob What is $50\%$ of $80$? \answerBlank[3cm]
\answerExplain{$40$}{Convert $50\%$ to a decimal: $0.50$. Multiply: $0.50 \times 80 = 40$. Half of $80$ is $40$.}

\prob What is $10\%$ of $150$? \answerBlank[3cm]
\answerExplain{$15$}{Convert $10\%$ to a decimal: $0.10$. Multiply: $0.10 \times 150 = 15$. Moving the decimal one place left gives $15$.}

\prob What is $25\%$ of $60$? \answerBlank[3cm]
\answerExplain{$15$}{Convert $25\%$ to a decimal: $0.25$. Multiply: $0.25 \times 60 = 15$. One-quarter of $60$ is $15$.}

\prob What is $20\%$ of $45$? \answerBlank[3cm]
\answerExplain{$9$}{Convert $20\%$ to a decimal: $0.20$. Multiply: $0.20 \times 45 = 9$. One-fifth of $45$ is $9$.}

\prob What is $75\%$ of $120$? \answerBlank[3cm]
\answerExplain{$90$}{Convert $75\%$ to a decimal: $0.75$. Multiply: $0.75 \times 120 = 90$. Three-quarters of $120$ is $90$.}

\prob What is $100\%$ of $37$? \answerBlank[3cm]
\answerExplain{$37$}{$100\%$ means the entire amount. $1.00 \times 37 = 37$. Any number times $1$ is itself.}
\end{multicols}
\end{practiceBox}

% ── Practice D: Word Problems ────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{A student scored $42$ out of $60$ on a test. What percent did the student score?}{}
\answerExplain{$70\%$}{Divide the part by the whole: $42 \div 60 = 0.70$. Multiply by $100$: the student scored $70\%$.}

\wordProblem{A store has $300$ items. If $85\%$ are in stock, how many items are in stock?}{items}
\answerExplain{$255$ items}{Convert $85\%$ to a decimal: $0.85$. Multiply: $0.85 \times 300 = 255$ items are in stock.}

\wordProblem{A class raised $\$180$ for a field trip. This is $72\%$ of the goal. What is the full goal?}{dollars}
\answerExplain{$\$250$}{The part is $\$180$ and the percent is $72\%$. Divide: $180 \div 0.72 = 250$. The goal is $\$250$.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob A school has $640$ students. $55\%$ are girls. Of the girls, $40\%$ play a sport. How many girls play a sport? \answerBlank[3cm]
\answerExplain{$141$ girls (rounding $140.8$)}{First find the number of girls: $0.55 \times 640 = 352$. Then find how many play a sport: $0.40 \times 352 = 140.8$. Round to $141$ since you can't have a partial student.}

\prob In a survey, $36$ people said yes. This was $45\%$ of those surveyed. Of all people surveyed, $30\%$ were under age $18$. How many were under $18$? \answerBlank[3cm]
\answerExplain{$24$ people}{First find the total surveyed: $36 \div 0.45 = 80$. Then find $30\%$ of $80$: $0.30 \times 80 = 24$ people under $18$.}
\end{challengeBox}
