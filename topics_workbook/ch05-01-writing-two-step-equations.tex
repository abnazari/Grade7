\section{Writing Two-Step Equations}
\topicTitle{Writing Two-Step Equations from Word Problems}

\begin{quickReview}{Writing Two-Step Equations}
A \textbf{two-step equation} has the form $px + q = r$, where:
\begin{itemize}[leftmargin=*]
    \item $p$ = the \textbf{rate} (per item, per hour, per month, etc.)
    \item $x$ = the \textbf{unknown} quantity you are solving for
    \item $q$ = a \textbf{fixed amount} (fee, bonus, starting value, etc.)
    \item $r$ = the \textbf{total}
\end{itemize}

\medskip
\textbf{Example:} A gym charges $\$10$ per month plus a $\$25$ sign-up fee. You have spent $\$85$ total.

Rate $= 10$, variable $= m$ (months), fixed $= 25$, total $= 85$ → $10m + 25 = 85$.

\smallskip
\textbf{Tip:} Look for ``per'' (rate) and a one-time amount (constant). The rate goes with the variable.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Identify the Parts}
For each situation, identify the \textbf{rate}, the \textbf{fixed amount}, and the \textbf{total}. You do NOT need to write the equation yet.

\prob A streaming service costs $\$12$ per month plus a $\$5$ activation fee. The total so far is $\$41$.
\answerExplain{Rate $= \$12$/month, Fixed $= \$5$, Total $= \$41$}{The per-month charge ($\$12$) is the rate. The one-time activation fee ($\$5$) is the fixed amount. The total spent is $\$41$.}

\prob You earn $\$9$ per hour and receive a $\$15$ bonus. Your paycheck is $\$78$.
\answerExplain{Rate $= \$9$/hour, Fixed $= \$15$, Total $= \$78$}{The hourly wage ($\$9$) is the rate. The bonus ($\$15$) is the fixed amount. The paycheck ($\$78$) is the total.}

\prob A phone plan costs $\$0.10$ per text plus $\$20$ per month. The bill is $\$35$.
\answerExplain{Rate $= \$0.10$/text, Fixed $= \$20$, Total $= \$35$}{The per-text charge ($\$0.10$) is the rate. The monthly base ($\$20$) is the fixed amount. The bill ($\$35$) is the total.}

\prob You start with $\$100$ and spend $\$7$ per day. You have $\$37$ left.
\answerExplain{Rate $= \$7$/day, Fixed (start) $= \$100$, Remaining $= \$37$}{The daily spending ($\$7$) is the rate. The starting amount ($\$100$) is the fixed value. The remaining balance ($\$37$) is what's left.}

\prob A taxi charges $\$3.50$ per mile plus a $\$2$ base fare. The total fare is $\$19.50$.
\answerExplain{Rate $= \$3.50$/mile, Fixed $= \$2$, Total $= \$19.50$}{The per-mile charge ($\$3.50$) is the rate. The base fare ($\$2$) is the fixed amount. The total fare ($\$19.50$) is the total.}

\end{practiceBox}

% ── Practice A: Translate to Equations ───────────────────────────────────────
\begin{practiceBox}{Translate Words into Equations}

\practiceHeader{Write an equation for each situation. Use the variable given.}

\prob You buy $n$ notebooks at $\$4$ each and a $\$6$ pen set. The total is $\$30$. \answerBlank[3cm]
\answerExplain{$4n + 6 = 30$}{Each notebook costs $\$4$, so $n$ notebooks cost $4n$. Add the $\$6$ pen set: $4n + 6$. Set equal to the total: $4n + 6 = 30$.}

\prob A plumber charges $\$50$ per hour plus a $\$75$ service call fee. The bill is $\$275$. Let $h$ be hours. \answerBlank[3cm]
\answerExplain{$50h + 75 = 275$}{The hourly charge is $50h$. Add the service fee: $50h + 75$. Set equal to the bill: $50h + 75 = 275$.}

\prob You have $\$200$ and spend $\$15$ per week. After $w$ weeks you have $\$50$ left. \answerBlank[3cm]
\answerExplain{$200 - 15w = 50$}{Start with $\$200$ and subtract $\$15$ each week: $200 - 15w$. Set equal to what remains: $200 - 15w = 50$.}

\prob A concert ticket costs $\$12$ each. There is an $\$8$ processing fee. You paid $\$56$ total for $t$ tickets. \answerBlank[3cm]
\answerExplain{$12t + 8 = 56$}{Ticket cost: $12t$. Add the fee: $12t + 8$. Set equal to the total: $12t + 8 = 56$.}

\prob Three times a number increased by $7$ equals $28$. Let the number be $x$. \answerBlank[3cm]
\answerExplain{$3x + 7 = 28$}{``Three times a number'' means $3x$. ``Increased by $7$'' means add $7$: $3x + 7$. ``Equals $28$'' gives $3x + 7 = 28$.}

\prob A dog walker earns $\$6$ per dog walked. She pays $\$10$ for supplies. Her profit is $\$38$. Let $d$ be dogs walked. \answerBlank[3cm]
\answerExplain{$6d - 10 = 38$}{Earnings: $6d$. Subtract supply cost: $6d - 10$. Set equal to profit: $6d - 10 = 38$.}

\end{practiceBox}

% ── Practice B: Translate Verbal Expressions ─────────────────────────────────
\begin{practiceBox}[funTeal]{From Words to Algebra}

\practiceHeader[funTeal]{Write an equation. Use $n$ for the unknown number.}

\prob A number divided by $5$, plus $3$, equals $11$. \answerBlank[3cm]
\answerExplain{$\frac{n}{5} + 3 = 11$}{``A number divided by $5$'' is $\frac{n}{5}$. ``Plus $3$'' gives $\frac{n}{5} + 3$. ``Equals $11$'' gives $\frac{n}{5} + 3 = 11$.}

\prob Seven less than twice a number is $19$. \answerBlank[3cm]
\answerExplain{$2n - 7 = 19$}{``Twice a number'' is $2n$. ``Seven less than'' means subtract $7$: $2n - 7$. ``Is $19$'' gives $2n - 7 = 19$.}

\prob Four times a number decreased by $10$ equals $-2$. \answerBlank[3cm]
\answerExplain{$4n - 10 = -2$}{``Four times a number'' is $4n$. ``Decreased by $10$'' gives $4n - 10$. ``Equals $-2$'' gives $4n - 10 = -2$.}

\prob Half of a number plus $8$ is $20$. \answerBlank[3cm]
\answerExplain{$\frac{n}{2} + 8 = 20$}{``Half of a number'' is $\frac{n}{2}$. ``Plus $8$'' gives $\frac{n}{2} + 8$. ``Is $20$'' gives $\frac{n}{2} + 8 = 20$.}

\prob Nine more than a number divided by $3$ equals $12$. \answerBlank[3cm]
\answerExplain{$\frac{n}{3} + 9 = 12$}{``A number divided by $3$'' is $\frac{n}{3}$. ``Nine more than'' means add $9$: $\frac{n}{3} + 9$. ``Equals $12$'' gives $\frac{n}{3} + 9 = 12$.}

\end{practiceBox}

% ── Practice C: True or False ────────────────────────────────────────────────
\begin{practiceBox}[funPurple]{True or False?}

\trueOrFalse{The equation for ``$\$5$ per book plus $\$3$ shipping equals $\$23$'' is $5b + 3 = 23$.}
\answerExplain{True}{The rate is $\$5$ per book, so $b$ books cost $5b$. Adding the $\$3$ shipping gives $5b + 3$. Setting equal to $\$23$ gives $5b + 3 = 23$.}

\trueOrFalse{``Eight less than three times a number is $13$'' translates to $8 - 3n = 13$.}
\answerExplain{False}{``Three times a number'' is $3n$. ``Eight less than'' means subtract $8$ from $3n$: $3n - 8 = 13$, not $8 - 3n = 13$. The order matters.}

\trueOrFalse{If you start with $\$75$ and spend $\$5$ per day for $d$ days, the amount left is $75 - 5d$.}
\answerExplain{True}{You begin with $\$75$ and lose $\$5$ each day. After $d$ days, you've spent $5d$, leaving $75 - 5d$.}

\trueOrFalse{The equation $2x + 10 = 30$ and $10 + 2x = 30$ represent the same situation.}
\answerExplain{True}{Addition is commutative, so $2x + 10$ and $10 + 2x$ are equivalent expressions. Both equations have the same solution.}

\end{practiceBox}

% ── Practice D: Word Problems ────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{Maria is saving for a $\$120$ bicycle. She already has $\$30$ and saves $\$15$ per week. Write an equation for the number of weeks $w$ until she has enough.}{weeks}
\answerExplain{$15w + 30 = 120$}{Maria saves $\$15$ per week, so after $w$ weeks she saves $15w$. Add her starting $\$30$: $15w + 30$. Set equal to $\$120$: $15w + 30 = 120$.}

\wordProblem{A pool is being drained at $20$ gallons per minute. It currently holds $500$ gallons. Write an equation for the time $t$ (in minutes) until the pool has $100$ gallons left.}{minutes}
\answerExplain{$500 - 20t = 100$}{Start with $500$ gallons and remove $20$ each minute: $500 - 20t$. Set equal to $100$ gallons remaining: $500 - 20t = 100$.}

\wordProblem{An online store charges $\$8$ per item plus $\$5.50$ for shipping. Your order total is $\$37.50$. Write an equation for the number of items $n$ you ordered.}{items}
\answerExplain{$8n + 5.50 = 37.50$}{Each item costs $\$8$, so $n$ items cost $8n$. Add shipping: $8n + 5.50$. Set equal to the total: $8n + 5.50 = 37.50$.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}

\prob Two friends go to a fair. Admission is $\$10$ per person. Ride tickets cost $\$3$ each. Together they spend $\$62$. If each person buys the same number of tickets, write an equation for the number of tickets $t$ each person buys.
\answerExplain{$2(10) + 2(3t) = 62$, or $6t + 20 = 62$}{Two admissions cost $2 \times 10 = 20$. Each person buys $t$ tickets at $\$3$ each, so $2 \times 3t = 6t$ for both. Total: $6t + 20 = 62$.}

\prob Write a real-world situation that could be modeled by the equation $4x - 12 = 36$. Then explain what $x$ represents.
\answerExplain{Answers vary}{Sample: ``You earn $\$4$ per chore. After you pay back the $\$12$ you owe your sister, you have $\$36$.'' Here $x$ is the number of chores. The equation models: $4x - 12 = 36$.}

\end{challengeBox}
