\section{Writing and Solving Inequalities}
\topicTitle{Writing and Solving Inequalities}

\begin{quickReview}{Inequalities}
An \textbf{inequality} compares two expressions using $<$, $>$, $\leq$, or $\geq$.

\begin{itemize}[leftmargin=*]
    \item Solve the same way as equations — but with \textbf{one key rule}:
    \item \keyIdea{When you multiply or divide by a \textbf{negative} number, \textbf{flip} the inequality sign.}
\end{itemize}

\medskip
\begin{center}
\begin{tabular}{l l}
    \textbf{Words} & \textbf{Symbol} \\
    \hline
    at least, no less than & $\geq$ \\
    at most, no more than & $\leq$ \\
    more than, greater than & $>$ \\
    fewer than, less than & $<$ \\
\end{tabular}
\end{center}

\textbf{Example:} Solve $-2x > 6$. Divide by $-2$ and flip: $x < -3$.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Translate Words to Symbols}
Write the correct inequality symbol ($<$, $>$, $\leq$, $\geq$).

\begin{multicols}{2}
\prob ``at most $50$'' means $x$ \answerBlank[1.5cm] $50$
\answerExplain{$\leq$}{``At most'' means the value can equal $50$ or be less. The symbol is $\leq$.}

\prob ``more than $12$'' means $x$ \answerBlank[1.5cm] $12$
\answerExplain{$>$}{``More than'' means strictly greater, not equal. The symbol is $>$.}

\prob ``no less than $7$'' means $x$ \answerBlank[1.5cm] $7$
\answerExplain{$\geq$}{``No less than'' means the value is $7$ or greater. The symbol is $\geq$.}

\prob ``fewer than $20$'' means $x$ \answerBlank[1.5cm] $20$
\answerExplain{$<$}{``Fewer than'' means strictly less, not equal. The symbol is $<$.}

\prob ``at least $100$'' means $x$ \answerBlank[1.5cm] $100$
\answerExplain{$\geq$}{``At least'' means the value can be $100$ or more. The symbol is $\geq$.}
\end{multicols}
\end{practiceBox}

% ── Practice A: One-Step Inequalities ────────────────────────────────────────
\begin{practiceBox}{Solve One-Step Inequalities}

\practiceHeader{Solve each inequality.}

\begin{multicols}{2}
\prob $x + 5 > 12$ \answerBlank[2.5cm]
\answerExplain{$x > 7$}{Subtract $5$ from both sides: $x > 12 - 5 = 7$.}

\prob $n - 3 \leq 10$ \answerBlank[2.5cm]
\answerExplain{$n \leq 13$}{Add $3$ to both sides: $n \leq 10 + 3 = 13$.}

\prob $4y > 28$ \answerBlank[2.5cm]
\answerExplain{$y > 7$}{Divide both sides by $4$: $y > 28 \div 4 = 7$. No sign flip since $4$ is positive.}

\prob $-6a \geq 18$ \answerBlank[2.5cm]
\answerExplain{$a \leq -3$}{Divide both sides by $-6$ and flip the sign: $a \leq 18 \div (-6) = -3$.}

\prob $\dfrac{m}{3} < 5$ \answerBlank[2.5cm]
\answerExplain{$m < 15$}{Multiply both sides by $3$: $m < 5 \times 3 = 15$.}

\prob $-5k < 35$ \answerBlank[2.5cm]
\answerExplain{$k > -7$}{Divide by $-5$ and flip: $k > 35 \div (-5) = -7$.}
\end{multicols}
\end{practiceBox}

% ── Practice B: Two-Step Inequalities ────────────────────────────────────────
\begin{practiceBox}[funTeal]{Solve Two-Step Inequalities}

\practiceHeader[funTeal]{Solve. Remember to flip the sign when dividing by a negative!}

\prob $2x + 7 > 19$ \answerBlank[3cm]
\answerExplain{$x > 6$}{Subtract $7$: $2x > 12$. Divide by $2$: $x > 6$.}

\prob $3n - 8 \leq 13$ \answerBlank[3cm]
\answerExplain{$n \leq 7$}{Add $8$: $3n \leq 21$. Divide by $3$: $n \leq 7$.}

\prob $-4y + 5 > 21$ \answerBlank[3cm]
\answerExplain{$y < -4$}{Subtract $5$: $-4y > 16$. Divide by $-4$ and flip: $y < -4$.}

\prob $\dfrac{x}{2} - 6 \geq 1$ \answerBlank[3cm]
\answerExplain{$x \geq 14$}{Add $6$: $\frac{x}{2} \geq 7$. Multiply by $2$: $x \geq 14$.}

\prob $-5a - 3 < 22$ \answerBlank[3cm]
\answerExplain{$a > -5$}{Add $3$: $-5a < 25$. Divide by $-5$ and flip: $a > -5$.}

\prob $9 - 2k \geq 15$ \answerBlank[3cm]
\answerExplain{$k \leq -3$}{Subtract $9$: $-2k \geq 6$. Divide by $-2$ and flip: $k \leq -3$.}

\end{practiceBox}

% ── Spot the Error ───────────────────────────────────────────────────────────
\begin{errorBox}
A student solved $-3x + 6 > 15$ this way:

$-3x + 6 > 15$

$-3x > 9$ \quad (subtracted $6$)

$x > -3$ \quad (divided by $-3$)

\medskip
\prob What mistake did the student make? What is the correct answer? \answerBlank[3cm]
\answerExplain{$x < -3$}{The student forgot to flip the inequality sign when dividing by $-3$. When you divide (or multiply) both sides by a negative number, the inequality direction reverses. The correct answer is $x < -3$.}
\end{errorBox}

% ── Practice C: Write and Solve ──────────────────────────────────────────────
\begin{practiceBox}[funOrange]{Write and Solve Inequalities from Words}

\prob ``Five times a number plus $3$ is at most $43$.'' Write and solve. \answerBlank[3cm]
\answerExplain{$n \leq 8$}{Inequality: $5n + 3 \leq 43$. Subtract $3$: $5n \leq 40$. Divide by $5$: $n \leq 8$.}

\prob ``A number divided by $4$ minus $2$ is greater than $3$.'' Write and solve. \answerBlank[3cm]
\answerExplain{$n > 20$}{Inequality: $\frac{n}{4} - 2 > 3$. Add $2$: $\frac{n}{4} > 5$. Multiply by $4$: $n > 20$.}

\prob ``Twice a number decreased by $11$ is no less than $5$.'' Write and solve. \answerBlank[3cm]
\answerExplain{$n \geq 8$}{Inequality: $2n - 11 \geq 5$. Add $11$: $2n \geq 16$. Divide by $2$: $n \geq 8$.}

\end{practiceBox}

% ── Practice D: Word Problems ────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{You have $\$75$ and earn $\$8$ per hour babysitting. How many hours must you work to have at least $\$131$?}{hours}
\answerExplain{At least $7$ hours}{Inequality: $8h + 75 \geq 131$. Subtract $75$: $8h \geq 56$. Divide by $8$: $h \geq 7$. You must work at least $7$ hours.}

\wordProblem{A parking garage charges $\$5$ per hour. You have $\$30$. For how many hours can you park?}{hours}
\answerExplain{At most $6$ hours}{Inequality: $5h \leq 30$. Divide by $5$: $h \leq 6$. You can park for at most $6$ hours.}

\wordProblem{A student needs at least $90$ points to earn an A. She has $62$ points and earns $7$ points per assignment. How many more assignments must she complete?}{assignments}
\answerExplain{At least $4$ assignments}{Inequality: $7a + 62 \geq 90$. Subtract $62$: $7a \geq 28$. Divide by $7$: $a \geq 4$.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}

\prob A gym offers two plans. Plan A: $\$30$ per month. Plan B: $\$12$ per month plus $\$3$ per visit. For how many visits per month is Plan A cheaper?
\answerExplain{More than $6$ visits}{Plan A $<$ Plan B: $30 < 12 + 3v$. Subtract $12$: $18 < 3v$. Divide by $3$: $6 < v$, or $v > 6$. Plan A is cheaper when you visit more than $6$ times per month.}

\prob The sum of three consecutive integers is greater than $60$. What is the smallest possible value of the first integer?
\answerExplain{$20$}{Let the integers be $n$, $n+1$, $n+2$. Sum: $n + (n+1) + (n+2) > 60$. Simplify: $3n + 3 > 60$. Subtract $3$: $3n > 57$. Divide by $3$: $n > 19$. The smallest integer greater than $19$ is $20$.}

\end{challengeBox}
