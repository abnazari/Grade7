\topicTitle{Dividing Integers and Rational Numbers}

\begin{quickReview}{Dividing Signed Numbers}
Division follows the \textbf{same sign rules} as multiplication:

\begin{itemize}[leftmargin=*]
    \item \textbf{Same signs $\to$ positive:} $(-20) \div (-4) = 5$
    \item \textbf{Different signs $\to$ negative:} $20 \div (-4) = -5$
    \item A negative sign can go in front, on top, or on bottom: $-\dfrac{p}{q} = \dfrac{-p}{q} = \dfrac{p}{-q}$
    \item To \textbf{divide fractions}, multiply by the reciprocal (Keep--Change--Flip).
    \item \textbf{Division by zero is undefined.}
\end{itemize}

\medskip
\textbf{Example:} $-\dfrac{3}{4} \div \dfrac{2}{3} = -\dfrac{3}{4} \times \dfrac{3}{2} = -\dfrac{9}{8} = -1\dfrac{1}{8}$
\end{quickReview}

% ── Practice C: True or False? ───────────────────────────────────────────────
\begin{practiceBox}[funPurple]{True or False?}

\trueOrFalse{$(-5) \div 0 = 0$}
\answerExplain{False}{Division by zero is undefined. You can never divide any number by $0$.}

\trueOrFalse{$-\dfrac{3}{4} = \dfrac{-3}{4} = \dfrac{3}{-4}$}
\answerExplain{True}{A negative sign can be placed in front of the fraction, on the numerator, or on the denominator — all three are equivalent.}

\trueOrFalse{Dividing two negative numbers always gives a negative result.}
\answerExplain{False}{Same signs (both negative) give a \emph{positive} result. For example, $(-12) \div (-3) = 4$.}

\end{practiceBox}

% ── Practice D: Word Problems ────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{A submarine descends $240$ meters in $8$ minutes at a constant rate. Write the rate as a signed number.}{meters per minute}
\answerExplain{$-30$ meters per minute}{Descending means negative: $-240 \div 8 = -30$. The submarine goes down $30$ meters each minute.}

\wordProblem{The temperature dropped $13.5°$C over $3$ hours. What was the average change per hour?}{°C per hour}
\answerExplain{$-4.5°$C per hour}{A drop is negative: $-13.5 \div 3 = -4.5$. The temperature fell an average of $4.5°$C each hour.}

\wordProblem{A baker uses $\frac{3}{4}$ of a bag of flour to make $6$ equal loaves. What fraction of the bag goes into each loaf?}{of a bag}
\answerExplain{$\dfrac{1}{8}$ of a bag}{Divide: $\frac{3}{4} \div 6 = \frac{3}{4} \times \frac{1}{6} = \frac{3}{24} = \frac{1}{8}$ of the bag per loaf.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}

\prob Evaluate: $\dfrac{(-8) \times (-3)}{(-6)}$ \answerBlank[2cm]
\answerExplain{$-4$}{Numerator: $(-8)(-3) = 24$ (same signs $\to$ positive). Divide: $24 \div (-6) = -4$ (different signs $\to$ negative).}

\prob If $\dfrac{x}{-5} = 12$, what is $x$? \answerBlank[2cm]
\answerExplain{$-60$}{Multiply both sides by $-5$: $x = 12 \times (-5) = -60$. Check: $\frac{-60}{-5} = 12$. \checkmark}

\end{challengeBox}
