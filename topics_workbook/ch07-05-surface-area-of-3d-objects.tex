\section{Surface Area of Three-Dimensional Objects}
\topicTitle{Surface Area of Three-Dimensional Objects}

\begin{quickReview}{Surface Area of 3D Objects}
\textbf{Surface area} (SA) is the total area of all the faces of a 3D object.

\begin{itemize}[leftmargin=*]
    \item \textbf{Cube:} $SA = 6s^2$ \quad (six identical square faces)
    \item \textbf{Rectangular prism:} $SA = 2lw + 2lh + 2wh$ \quad (three pairs of rectangular faces)
    \item \textbf{Triangular prism:} Add the area of the two triangular bases plus the three rectangular faces.
\end{itemize}

\medskip
\textbf{Tip:} Drawing or imagining the \textbf{net} (the unfolded shape) helps you see every face.

\smallskip
\textbf{Example:} A box $5 \times 3 \times 2$ cm: $SA = 2(5 \times 3) + 2(5 \times 2) + 2(3 \times 2) = 30 + 20 + 12 = 62$ cm$^2$.
\end{quickReview}

% ── Practice B: Triangular Prisms ────────────────────────────────────────────
\begin{practiceBox}[funTeal]{Surface Area of Triangular Prisms}

\practiceHeader[funTeal]{Find the total surface area. The triangular base has the given base $b$ and height $h_t$.}

\prob Triangle: $b = 6$ cm, $h_t = 4$ cm. Sides of triangle: $6$ cm, $5$ cm, $5$ cm. Prism length: $10$ cm. \answerBlank[3cm]
\answerExplain{$184$ cm$^2$}{Two triangles: $2 \times \frac{1}{2} \times 6 \times 4 = 24$ cm$^2$. Three rectangles: $(6 \times 10) + (5 \times 10) + (5 \times 10) = 60 + 50 + 50 = 160$ cm$^2$. Total: $24 + 160 = 184$ cm$^2$.}

\prob Triangle: $b = 8$ m, $h_t = 3$ m. Sides of triangle: $8$ m, $5$ m, $5$ m. Prism length: $12$ m. \answerBlank[3cm]
\answerExplain{$240$ m$^2$}{Two triangles: $2 \times \frac{1}{2} \times 8 \times 3 = 24$ m$^2$. Three rectangles: $(8 \times 12) + (5 \times 12) + (5 \times 12) = 96 + 60 + 60 = 216$ m$^2$. Total: $24 + 216 = 240$ m$^2$.}

\prob Triangle: $b = 10$ in, $h_t = 5$ in. Sides of triangle: $10$ in, $\sqrt{50} \approx 7.07$ in, $\sqrt{50} \approx 7.07$ in. Prism length: $8$ in. \answerBlank[3cm]
\answerExplain{$\approx 243.12$ in$^2$}{Two triangles: $2 \times \frac{1}{2} \times 10 \times 5 = 50$ in$^2$. Three rectangles: $(10 \times 8) + (7.07 \times 8) + (7.07 \times 8) = 80 + 56.56 + 56.56 = 193.12$ in$^2$. Total: $50 + 193.12 \approx 243.12$ in$^2$.}

\end{practiceBox}

% ── Practice C: True or False ────────────────────────────────────────────────
\begin{practiceBox}[funPurple]{True or False?}

\trueOrFalse{A cube has $6$ faces that are all the same size.}
\answerExplain{True}{Every face of a cube is a square of the same side length, so all $6$ faces are identical.}

\trueOrFalse{Surface area is measured in cubic units.}
\answerExplain{False}{Surface area measures the total area covering the outside, so it is measured in \emph{square} units (cm$^2$, m$^2$, etc.). Cubic units are for volume.}

\trueOrFalse{A rectangular prism with dimensions $4 \times 4 \times 4$ has the same surface area as a cube with side $4$.}
\answerExplain{True}{A rectangular prism with all equal dimensions \emph{is} a cube. Both give $SA = 6 \times 16 = 96$ square units.}

\end{practiceBox}

% ── Word Problems ────────────────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{A gift box is $30$ cm long, $20$ cm wide, and $10$ cm tall. How much wrapping paper is needed to cover the entire box (no overlap)?}{cm$^2$}
\answerExplain{$2{,}200$ cm$^2$}{$SA = 2(30 \times 20) + 2(30 \times 10) + 2(20 \times 10) = 1200 + 600 + 400 = 2{,}200$ cm$^2$.}

\wordProblem{Each face of a Rubik's cube is $5.7$ cm on a side. What is the total surface area of the cube?}{cm$^2$}
\answerExplain{$194.94$ cm$^2$}{$SA = 6s^2 = 6 \times 5.7^2 = 6 \times 32.49 = 194.94$ cm$^2$.}

\wordProblem{A camping tent has the shape of a triangular prism. The triangular face has base $2.4$ m and height $1.8$ m. The three side lengths of the triangle are $2.4$ m, $2.12$ m, and $2.12$ m. The tent is $3$ m long. What is the total surface area of the tent (including the floor)?}{m$^2$}
\answerExplain{$24.24$ m$^2$}{Two triangles: $2 \times \frac{1}{2} \times 2.4 \times 1.8 = 4.32$ m$^2$. Three rectangles: $(2.4 \times 3) + (2.12 \times 3) + (2.12 \times 3) = 7.2 + 6.36 + 6.36 = 19.92$ m$^2$. Total: $4.32 + 19.92 = 24.24$ m$^2$.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob A room is $5$ m long, $4$ m wide, and $3$ m tall. You need to paint all four walls and the ceiling (not the floor). What is the total area to paint? \answerBlank[3cm]
\answerExplain{$74$ m$^2$}{Two long walls: $2 \times (5 \times 3) = 30$ m$^2$. Two short walls: $2 \times (4 \times 3) = 24$ m$^2$. Ceiling: $5 \times 4 = 20$ m$^2$. Total: $30 + 24 + 20 = 74$ m$^2$.}

\prob A cube has a surface area of $150$ cm$^2$. What is the length of one side? \answerBlank[3cm]
\answerExplain{$5$ cm}{$SA = 6s^2$. So $s^2 = 150 \div 6 = 25$. Take the square root: $s = 5$ cm.}
\end{challengeBox}
