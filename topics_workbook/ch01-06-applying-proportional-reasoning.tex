\section{Applying Proportional Reasoning}
\topicTitle{Applying Proportional Reasoning to Real-World Problems}

\begin{quickReview}{Your Proportional Reasoning Toolbox}
You have four strategies --- pick whichever fits the problem best:

\begin{enumerate}[leftmargin=*]
    \item \textbf{Unit rate} --- find the rate for $1$ unit, then multiply or divide.
    \item \textbf{Proportion} --- set up $\dfrac{a}{b} = \dfrac{c}{d}$ and cross-multiply.
    \item \textbf{Equation} --- use $y = kx$.
    \item \textbf{Scale factor} --- multiply or divide to scale up/down.
\end{enumerate}

\medskip
\textbf{Example:} $5$ T-shirts cost $\$42.50$. How much for $8$?

Unit rate: $42.50 \div 5 = \$8.50$ each. Then $8.50 \times 8 = \$68.00$.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Solve using any strategy.}

\prob $4$ notebooks cost $\$10$. How much for $1$ notebook? \answerBlank[3cm]
\answerExplain{$\$2.50$}{Divide total by quantity: $10 \div 4 = \$2.50$ per notebook.}

\prob A car goes $180$ miles on $6$ gallons. How far on $1$ gallon? \answerBlank[3cm]
\answerExplain{$30$ miles}{Divide distance by gallons: $180 \div 6 = 30$ miles per gallon.}

\prob A recipe for $2$ servings uses $3$ eggs. How many eggs for $6$ servings? \answerBlank[3cm]
\answerExplain{$9$ eggs}{Scale factor: $6 \div 2 = 3$. Multiply eggs: $3 \times 3 = 9$ eggs.}

\prob A map scale says $1$ cm $= 5$ km. Two towns are $8$ cm apart. Actual distance? \answerBlank[3cm]
\answerExplain{$40$ km}{Multiply by the scale: $8 \times 5 = 40$ km.}

\prob $12$ muffins cost $\$9$. What is the cost per muffin? \answerBlank[3cm]
\answerExplain{$\$0.75$}{Divide to find the unit price: $9 \div 12 = \$0.75$ per muffin.}
\end{practiceBox}

% ── Practice A: Unit Rate Strategy ───────────────────────────────────────────
\begin{practiceBox}{Using the Unit Rate Strategy}

\practiceHeader{Find the unit rate first, then answer the question.}

\prob A factory produces $360$ bottles in $8$ hours. How many bottles in $15$ hours?
\answerBlank[3cm]
\answerExplain{$675$ bottles}{Unit rate: $360 \div 8 = 45$ bottles per hour. In $15$ hours: $45 \times 15 = 675$ bottles.}

\prob A cyclist rides $42$ km in $3$ hours. How far in $5$ hours at the same speed?
\answerBlank[3cm]
\answerExplain{$70$ km}{Unit rate: $42 \div 3 = 14$ km per hour. In $5$ hours: $14 \times 5 = 70$ km.}

\prob $6$ pounds of apples cost $\$8.40$. How much for $10$ pounds?
\answerBlank[3cm]
\answerExplain{$\$14$}{Unit rate: $8.40 \div 6 = \$1.40$ per pound. For $10$ pounds: $1.40 \times 10 = \$14$.}

\prob A printer prints $90$ pages in $6$ minutes. How long for $300$ pages?
\answerBlank[3cm]
\answerExplain{$20$ minutes}{Unit rate: $90 \div 6 = 15$ pages per minute. Time: $300 \div 15 = 20$ minutes.}

\end{practiceBox}

% ── Practice B: Proportion Strategy ──────────────────────────────────────────
\begin{practiceBox}[funTeal]{Setting Up and Solving Proportions}

\practiceHeader[funTeal]{Set up a proportion $\frac{a}{b} = \frac{c}{d}$ and cross-multiply.}

\prob A map scale: $3$ cm $= 21$ km. Two cities are $5$ cm apart on the map. What is the actual distance?
\answerBlank[3cm]
\answerExplain{$35$ km}{Proportion: $\frac{3}{21} = \frac{5}{d}$. Cross-multiply: $3d = 105$, so $d = 35$ km.}

\prob A shadow problem: A $6$-foot person casts a $4$-foot shadow. A tree casts a $20$-foot shadow. How tall is the tree?
\answerBlank[3cm]
\answerExplain{$30$ feet}{Proportion: $\frac{6}{4} = \frac{h}{20}$. Cross-multiply: $4h = 120$, so $h = 30$ feet.}

\prob If $7$ tickets cost $\$52.50$, how much for $12$ tickets?
\answerBlank[3cm]
\answerExplain{$\$90$}{Proportion: $\frac{7}{52.50} = \frac{12}{c}$. Cross-multiply: $7c = 630$, so $c = \$90$.}

\prob A recipe for $8$ people needs $5$ cups of flour. How much for $20$ people?
\answerBlank[3cm]
\answerExplain{$12.5$ cups}{Proportion: $\frac{8}{5} = \frac{20}{f}$. Cross-multiply: $8f = 100$, so $f = 12.5$ cups.}

\end{practiceBox}

% ── Practice C: Best Deal ────────────────────────────────────────────────────
\begin{practiceBox}[funOrange]{Which Is the Better Deal?}

\practiceHeader[funOrange]{Find the unit price for each option and circle the better deal.}

\prob Brand A: $5$ bars of soap for $\$4.50$. Brand B: $8$ bars of soap for $\$6.80$. \answerBlank[4cm]
\answerExplain{Brand B ($\$0.85$/bar $<$ $\$0.90$/bar)}{Brand A: $4.50 \div 5 = \$0.90$ per bar. Brand B: $6.80 \div 8 = \$0.85$ per bar. Brand B is cheaper per bar.}

\prob Small juice: $\frac{3}{4}$ liter for $\$1.80$. Large juice: $1\frac{1}{2}$ liters for $\$3.30$. \answerBlank[4cm]
\answerExplain{Large ($\$2.20$/L $<$ $\$2.40$/L)}{Small: $1.80 \div 0.75 = \$2.40$ per liter. Large: $3.30 \div 1.5 = \$2.20$ per liter. The large bottle is the better deal.}

\prob Pack A: $15$ pencils for $\$3.75$. Pack B: $24$ pencils for $\$5.52$. \answerBlank[4cm]
\answerExplain{Pack B ($\$0.23$/pencil $<$ $\$0.25$/pencil)}{Pack A: $3.75 \div 15 = \$0.25$ per pencil. Pack B: $5.52 \div 24 = \$0.23$ per pencil. Pack B costs less per pencil.}

\end{practiceBox}

% ── Practice D: Multi-Step Word Problems ─────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Multi-Step Word Problems}

\wordProblem{A recipe for $4$ servings calls for $\frac{3}{2}$ cups of rice and $2$ cups of beans. How much rice and how many cups of beans are needed for $14$ servings?}{cups of rice and cups of beans}
\answerExplain{$5\frac{1}{4}$ cups of rice and $7$ cups of beans}{Scale factor: $14 \div 4 = 3.5$. Rice: $\frac{3}{2} \times 3.5 = \frac{3}{2} \times \frac{7}{2} = \frac{21}{4} = 5\frac{1}{4}$ cups. Beans: $2 \times 3.5 = 7$ cups.}

\wordProblem{A car travels $280$ miles on $8$ gallons. Gas costs $\$3.50$ per gallon. What is the fuel cost for a $700$-mile trip?}{dollars}
\answerExplain{$\$70$}{Miles per gallon: $280 \div 8 = 35$. Gallons for $700$ miles: $700 \div 35 = 20$. Cost: $20 \times 3.50 = \$70$.}

\wordProblem{On a map, $2$ cm $= 15$ miles. A hiking trail measures $9$ cm on the map and a cyclist completes the trail at $12$ miles per hour. How long does the ride take?}{hours}
\answerExplain{$5\frac{5}{8}$ hours}{Actual distance: $\frac{9}{2} \times 15 = \frac{135}{2} = 67.5$ miles. Time: $67.5 \div 12 = 5.625 = 5\frac{5}{8}$ hours.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob A store sells paint in two sizes. Small: $\frac{3}{4}$ gallon for $\$18$. Large: $2\frac{1}{2}$ gallons for $\$55$. You need exactly $5$ gallons. What is the cheapest way to buy them, and what is the total cost? \answerBlank[5cm]
\answerExplain{$2$ large cans for $\$110$}{Unit prices: Small $= 18 \div 0.75 = \$24$/gal. Large $= 55 \div 2.5 = \$22$/gal. The large can is cheaper per gallon. Two large cans give $2 \times 2.5 = 5$ gallons exactly, costing $2 \times 55 = \$110$.}

\prob A train travels at a constant speed. It passes a $300$-meter bridge in $20$ seconds and a $450$-meter bridge in $25$ seconds. What is the length of the train and its speed? \answerBlank[5cm]
\answerExplain{Train is $300$ m; speed $= 30$ m/s}{The train covers its own length plus the bridge. Set up: $(L + 300)/20 = (L + 450)/25$. Cross-multiply: $25(L + 300) = 20(L + 450)$, giving $25L + 7500 = 20L + 9000$, so $5L = 1500$ and $L = 300$ m. Speed $= (300 + 300)/20 = 600/20 = 30$ m/s.}
\end{challengeBox}
