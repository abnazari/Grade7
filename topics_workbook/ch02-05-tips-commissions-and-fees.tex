\section{Tips, Commissions, and Fees}
\topicTitle{Tips, Commissions, and Fees}

\begin{quickReview}{Tips, Commissions, and Fees}
Percents appear in many everyday money situations:
\begin{itemize}[leftmargin=*]
    \item A \textbf{tip} (gratuity) is extra money for service. Common tips: $15\%$--$20\%$.
    \item A \textbf{commission} is a percent of total sales a worker earns as pay.
    \item A \textbf{fee} is a flat or percent-based charge for a service.
\end{itemize}

\medskip
\textbf{Quick tip estimation:} Find $10\%$ by moving the decimal one place left. Double it for $20\%$; add half of $10\%$ for $15\%$.

\smallskip
\textbf{Example:} Bill is $\$52$. $10\% = \$5.20$. For $20\%$: $5.20 \times 2 = \$10.40$. For $15\%$: $5.20 + 2.60 = \$7.80$.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Estimate the Tip}
Use the $10\%$ method to estimate each tip mentally.

\begin{multicols}{2}
\prob $20\%$ tip on a $\$30$ bill \answerBlank[3cm]
\answerExplain{$\$6$}{$10\%$ of $\$30 = \$3$. Double it for $20\%$: $3 \times 2 = \$6$.}

\prob $15\%$ tip on a $\$40$ bill \answerBlank[3cm]
\answerExplain{$\$6$}{$10\%$ of $\$40 = \$4$. Half of that is $\$2$. Add: $4 + 2 = \$6$ for $15\%$.}

\prob $20\%$ tip on a $\$55$ bill \answerBlank[3cm]
\answerExplain{$\$11$}{$10\%$ of $\$55 = \$5.50$. Double: $5.50 \times 2 = \$11$.}

\prob $10\%$ tip on a $\$82$ bill \answerBlank[3cm]
\answerExplain{$\$8.20$}{Move the decimal one place left: $10\%$ of $\$82 = \$8.20$.}

\prob $15\%$ tip on a $\$60$ bill \answerBlank[3cm]
\answerExplain{$\$9$}{$10\%$ of $\$60 = \$6$. Half of that is $\$3$. Add: $6 + 3 = \$9$ for $15\%$.}

\prob $20\%$ tip on a $\$73$ bill \answerBlank[3cm]
\answerExplain{About $\$14.60$}{$10\%$ of $\$73 = \$7.30$. Double: $7.30 \times 2 = \$14.60$.}
\end{multicols}
\end{practiceBox}

% ── Practice A: Exact Tips ───────────────────────────────────────────────────
\begin{practiceBox}{Calculating Exact Tips and Totals}

\practiceHeader{Find the tip amount and the total bill.}

\prob A dinner bill is $\$48$. You leave an $18\%$ tip. How much is the tip? What is the total?
\answerBlank[3cm]
\answerExplain{Tip: $\$8.64$; Total: $\$56.64$}{Tip: $48 \times 0.18 = \$8.64$. Total: $48 + 8.64 = \$56.64$.}

\prob A lunch bill is $\$27.50$. You leave a $15\%$ tip. How much is the tip? What is the total?
\answerBlank[3cm]
\answerExplain{Tip: $\$4.13$ (rounded); Total: $\$31.63$}{Tip: $27.50 \times 0.15 = \$4.125 \approx \$4.13$. Total: $27.50 + 4.13 = \$31.63$.}

\prob A haircut costs $\$35$. You leave a $20\%$ tip. What is the total you pay?
\answerBlank[3cm]
\answerExplain{$\$42$}{Tip: $35 \times 0.20 = \$7$. Total: $35 + 7 = \$42$. Or use the multiplier: $35 \times 1.20 = \$42$.}

\end{practiceBox}

% ── Practice B: Commissions ──────────────────────────────────────────────────
\begin{practiceBox}[funTeal]{Calculating Commissions}

\practiceHeader[funTeal]{Find each commission amount.}

\prob A salesperson earns $5\%$ commission on $\$3{,}200$ in sales. \answerBlank[3cm]
\answerExplain{$\$160$}{Commission: $3{,}200 \times 0.05 = \$160$. The salesperson earns $5$ cents for every dollar sold.}

\prob A real estate agent earns $3\%$ on a $\$250{,}000$ home sale. \answerBlank[3cm]
\answerExplain{$\$7{,}500$}{Commission: $250{,}000 \times 0.03 = \$7{,}500$.}

\prob A car salesman earns $4\%$ commission. He sold a car for $\$28{,}000$. How much did he earn? \answerBlank[3cm]
\answerExplain{$\$1{,}120$}{Commission: $28{,}000 \times 0.04 = \$1{,}120$.}

\prob An art dealer earned $\$900$ commission on a painting. The commission rate is $6\%$. What was the sale price of the painting? \answerBlank[3cm]
\answerExplain{$\$15{,}000$}{Set up: $900 = 0.06 \times p$. Divide: $p = 900 \div 0.06 = \$15{,}000$.}

\end{practiceBox}

% ── Practice C: Fees ─────────────────────────────────────────────────────────
\begin{practiceBox}[funPurple]{Service Fees}

\practiceHeader[funPurple]{Calculate the fee and total cost.}

\prob A ticket website charges a $12\%$ service fee. Tickets cost $\$85$. What is the total? \answerBlank[3cm]
\answerExplain{$\$95.20$}{Fee multiplier: $1.12$. Total: $85 \times 1.12 = \$95.20$.}

\prob A money transfer app charges a $2.5\%$ fee. You send $\$200$. How much is the fee? \answerBlank[3cm]
\answerExplain{$\$5$}{Fee: $200 \times 0.025 = \$5$. The recipient gets $\$200$ and you pay $\$5$ extra.}

\prob A shipping company charges $\$10$ plus $4\%$ of the item value. You ship an item worth $\$250$. Total shipping cost? \answerBlank[3cm]
\answerExplain{$\$20$}{Percent fee: $250 \times 0.04 = \$10$. Total: $10 + 10 = \$20$.}

\end{practiceBox}

% ── Practice D: Word Problems ────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Mixed Word Problems}

\wordProblem{Your family's restaurant bill is $\$86$. You want to leave a $20\%$ tip and the sales tax is $7\%$. What is the total amount you pay? (Tip and tax are both calculated on the bill.)}{dollars}
\answerExplain{$\$109.22$}{Tip: $86 \times 0.20 = \$17.20$. Tax: $86 \times 0.07 = \$6.02$. Total: $86 + 17.20 + 6.02 = \$109.22$.}

\wordProblem{A salesperson earns a base salary of $\$400$ per week plus $8\%$ commission on sales. This week she sold $\$5{,}500$ worth of goods. What is her total earnings?}{dollars}
\answerExplain{$\$840$}{Commission: $5{,}500 \times 0.08 = \$440$. Total earnings: $400 + 440 = \$840$.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob Two friends dine separately. Alex's bill is $\$42$ with a $20\%$ tip. Jordan's bill is $\$50$ with a $15\%$ tip. Who pays more in total, and by how much? \answerBlank[3cm]
\answerExplain{Jordan pays more by $\$7.10$}{Alex total: $42 \times 1.20 = \$50.40$. Jordan total: $50 \times 1.15 = \$57.50$. Difference: $57.50 - 50.40 = \$7.10$. Jordan pays more.}

\prob A salesperson earns $6\%$ commission on the first $\$10{,}000$ in sales and $9\%$ on anything above $\$10{,}000$. She sold $\$18{,}000$. What is her total commission? \answerBlank[3cm]
\answerExplain{$\$1{,}320$}{First $\$10{,}000$: $10{,}000 \times 0.06 = \$600$. Above $\$10{,}000$: $(18{,}000 - 10{,}000) \times 0.09 = 8{,}000 \times 0.09 = \$720$. Total: $600 + 720 = \$1{,}320$.}
\end{challengeBox}
