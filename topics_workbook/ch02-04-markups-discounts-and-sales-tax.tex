\section{Markups, Discounts, and Sales Tax}
\topicTitle{Markups, Discounts, and Sales Tax}

\begin{quickReview}{Markups, Discounts, and Sales Tax}
\begin{itemize}[leftmargin=*]
    \item A \textbf{markup} is the amount a store adds to cost to set the selling price.
          Selling price $= \text{cost} \times (1 + \text{markup rate})$.
    \item A \textbf{discount} (markdown) is the amount taken off the regular price.
          Sale price $= \text{price} \times (1 - \text{discount rate})$.
    \item \textbf{Sales tax} is added at checkout.
          Total $= \text{price} \times (1 + \text{tax rate})$.
\end{itemize}

\medskip
\textbf{Example:} A store buys a hat for $\$10$, marks it up $60\%$, then charges $5\%$ tax.

Selling price: $10 \times 1.60 = \$16$. \quad Total with tax: $16 \times 1.05 = \$16.80$.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Quick Calculations}

\begin{multicols}{2}
\prob A $\$20$ item with a $50\%$ markup costs \answerBlank[2cm]
\answerExplain{$\$30$}{Markup: $20 \times 1.50 = \$30$. A $50\%$ markup means the selling price is $150\%$ of the cost.}

\prob A $\$40$ item at $10\%$ off costs \answerBlank[2cm]
\answerExplain{$\$36$}{Discount: $40 \times 0.90 = \$36$. A $10\%$ discount means you pay $90\%$ of the price.}

\prob A $\$50$ item with $8\%$ sales tax totals \answerBlank[2cm]
\answerExplain{$\$54$}{Tax: $50 \times 1.08 = \$54$. Add $8\%$ on top of the price.}

\prob A $\$15$ item with a $100\%$ markup costs \answerBlank[2cm]
\answerExplain{$\$30$}{Markup: $15 \times 2.00 = \$30$. A $100\%$ markup means the price doubles.}

\prob A $\$80$ item at $25\%$ off costs \answerBlank[2cm]
\answerExplain{$\$60$}{Discount: $80 \times 0.75 = \$60$. You pay $75\%$ of the original price.}

\prob A $\$100$ item with $6\%$ tax totals \answerBlank[2cm]
\answerExplain{$\$106$}{Tax: $100 \times 1.06 = \$106$. Sales tax adds to the price.}
\end{multicols}
\end{practiceBox}

% ── Practice A: Markups ──────────────────────────────────────────────────────
\begin{practiceBox}{Calculating Markups}

\practiceHeader{Find the selling price after the markup.}

\prob A store buys headphones for $\$18$ and marks them up $75\%$. Selling price? \answerBlank[3cm]
\answerExplain{$\$31.50$}{Multiplier for $75\%$ markup: $1.75$. Selling price: $18 \times 1.75 = \$31.50$.}

\prob A shop buys a phone case for $\$8$ and marks it up $120\%$. Selling price? \answerBlank[3cm]
\answerExplain{$\$17.60$}{Multiplier for $120\%$ markup: $2.20$. Selling price: $8 \times 2.20 = \$17.60$.}

\prob A store sells a game for $\$45$. The cost was $\$30$. What is the percent markup? \answerBlank[3cm]
\answerExplain{$50\%$}{Markup amount: $45 - 30 = \$15$. Percent: $15 \div 30 = 0.50$, so $50\%$ markup.}

\prob A bakery buys ingredients for $\$6$ per cake and sells each cake for $\$15$. Find the percent markup. \answerBlank[3cm]
\answerExplain{$150\%$}{Markup amount: $15 - 6 = \$9$. Percent: $9 \div 6 = 1.50$, so $150\%$ markup.}

\end{practiceBox}

% ── Practice B: Discounts ────────────────────────────────────────────────────
\begin{practiceBox}[funTeal]{Calculating Discounts}

\practiceHeader[funTeal]{Find the sale price after the discount.}

\prob A $\$65$ jacket is $30\%$ off. What is the sale price? \answerBlank[3cm]
\answerExplain{$\$45.50$}{Multiplier for $30\%$ discount: $0.70$. Sale price: $65 \times 0.70 = \$45.50$.}

\prob A $\$120$ pair of shoes is $15\%$ off. Sale price? \answerBlank[3cm]
\answerExplain{$\$102$}{Multiplier for $15\%$ discount: $0.85$. Sale price: $120 \times 0.85 = \$102$.}

\prob A laptop that regularly costs $\$800$ is now $\$680$. What is the percent discount? \answerBlank[3cm]
\answerExplain{$15\%$}{Discount amount: $800 - 680 = \$120$. Percent: $120 \div 800 = 0.15$, so $15\%$ off.}

\prob A bike originally costs $\$350$. After a discount, it costs $\$280$. What percent off is it? \answerBlank[3cm]
\answerExplain{$20\%$}{Discount amount: $350 - 280 = \$70$. Percent: $70 \div 350 = 0.20$, so $20\%$ off.}

\end{practiceBox}

% ── Practice C: Sales Tax ────────────────────────────────────────────────────
\begin{practiceBox}[funPurple]{Adding Sales Tax}

\practiceHeader[funPurple]{Find the total cost after sales tax is added.}

\prob A meal costs $\$24$. Sales tax is $7\%$. What is the total? \answerBlank[3cm]
\answerExplain{$\$25.68$}{Tax multiplier: $1.07$. Total: $24 \times 1.07 = \$25.68$. Tax is always added to the price.}

\prob You buy a book for $\$16.50$. Tax is $5\%$. Total? \answerBlank[3cm]
\answerExplain{$\$17.33$ (rounded)}{Total: $16.50 \times 1.05 = \$17.325$. Round to $\$17.33$ (round up at the half cent).}

\prob A video game costs $\$59.99$. Tax is $9\%$. What is the total? \answerBlank[3cm]
\answerExplain{$\$65.39$ (rounded)}{Total: $59.99 \times 1.09 = \$65.3891$. Round to $\$65.39$.}

\end{practiceBox}

% ── Practice D: Multi-Step Problems ──────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Multi-Step Problems}

\wordProblem{A store buys a shirt for $\$12$ and marks it up $80\%$. A customer buys it during a $20\%$ off sale. Sales tax is $6\%$. What is the final price?}{dollars}
\answerExplain{$\$18.29$ (rounded)}{Markup: $12 \times 1.80 = \$21.60$. Discount: $21.60 \times 0.80 = \$17.28$. Tax: $17.28 \times 1.06 = \$18.3168 \approx \$18.32$.}

\wordProblem{A furniture store buys a table for $\$150$ and sells it for $\$240$. What is the percent markup?}{percent}
\answerExplain{$60\%$}{Markup amount: $240 - 150 = \$90$. Percent markup: $90 \div 150 = 0.60$, so $60\%$.}

\wordProblem{Sneakers regularly cost $\$95$. They are $40\%$ off, and sales tax is $8\%$. What is the total cost?}{dollars}
\answerExplain{$\$61.56$}{Discount: $95 \times 0.60 = \$57$. Tax: $57 \times 1.08 = \$61.56$. Apply the discount first, then add tax.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob A store buys an item for $\$50$. After a $70\%$ markup and a $30\%$ discount, the selling price is the same as after what single markup on the original cost? \answerBlank[3cm]
\answerExplain{$19\%$ markup}{Markup: $50 \times 1.70 = \$85$. Discount: $85 \times 0.70 = \$59.50$. Single markup: $(59.50 - 50) \div 50 = 9.50 \div 50 = 0.19$, so a $19\%$ markup gives the same result.}

\prob Two stores sell the same TV. Store A: original $\$500$, $20\%$ off, then $8\%$ tax. Store B: original $\$475$, $10\%$ off, then $8\%$ tax. Which store has the lower final price, and by how much? \answerBlank[3cm]
\answerExplain{Store A is cheaper by $\$30.78$}{Store A: $500 \times 0.80 = \$400$, then $400 \times 1.08 = \$432$. Store B: $475 \times 0.90 = \$427.50$, then $427.50 \times 1.08 = \$461.70$. Difference: $461.70 - 432 = \$29.70$.}
\end{challengeBox}
