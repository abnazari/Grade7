\section{Parts of a Circle}
\topicTitle{Parts of a Circle}

\begin{quickReview}{Parts of a Circle}
A \textbf{circle} is the set of all points that are the same distance from a fixed point called the \textbf{center}.

\begin{itemize}[leftmargin=*]
    \item \textbf{Radius ($r$)}: Distance from the center to any point on the circle.
    \item \textbf{Diameter ($d$)}: Distance across the circle through the center. Always $d = 2r$.
    \item \textbf{Circumference ($C$)}: The distance around the circle (its perimeter).
    \item \textbf{Chord}: A segment with both endpoints on the circle. The diameter is the longest chord.
\end{itemize}

\medskip
\textbf{Key relationship:} $d = 2r$ \quad or equivalently \quad $r = \dfrac{d}{2}$.

\smallskip
\textbf{Example:} If the radius is $6$ cm, the diameter is $2 \times 6 = 12$ cm.
\end{quickReview}

% ── Practice A: Identify Circle Parts ────────────────────────────────────────
\begin{practiceBox}{Name That Part}

\practiceHeader{Identify the circle part described.}

\begin{multicols}{2}
\prob The point at the exact middle of a circle. \answerBlank[3cm]
\answerExplain{Center}{The center is the fixed point from which every point on the circle is the same distance away.}

\prob A line segment from the center to a point on the circle. \answerBlank[3cm]
\answerExplain{Radius}{A radius connects the center to any point on the circle's edge.}

\prob A line segment that passes through the center with both endpoints on the circle. \answerBlank[3cm]
\answerExplain{Diameter}{The diameter is a chord that passes through the center. It is the longest possible chord.}

\prob The distance around the entire circle. \answerBlank[3cm]
\answerExplain{Circumference}{The circumference is the perimeter of a circle — the total distance around it.}

\prob A line segment with both endpoints on the circle that does \emph{not} pass through the center. \answerBlank[3cm]
\answerExplain{Chord}{A chord connects two points on the circle. When it passes through the center, it is called a diameter.}

\prob The longest chord in any circle. \answerBlank[3cm]
\answerExplain{Diameter}{The diameter is the longest chord because it passes through the center, giving it the maximum possible length.}
\end{multicols}
\end{practiceBox}

% ── Practice B: Compute Missing Measurements ─────────────────────────────────
\begin{practiceBox}[funTeal]{Find the Missing Measurement}

\practiceHeader[funTeal]{Use $d = 2r$ to find the unknown.}

\prob A clock face has a radius of $15$ cm. What is its diameter? \answerBlank[3cm]
\answerExplain{$30$ cm}{Diameter equals twice the radius: $d = 2 \times 15 = 30$ cm.}

\prob A dinner plate has a diameter of $26$ cm. What is its radius? \answerBlank[3cm]
\answerExplain{$13$ cm}{Radius is half the diameter: $r = 26 \div 2 = 13$ cm.}

\prob A bicycle wheel has a radius of $33$ cm. What is the diameter of the wheel? \answerBlank[3cm]
\answerExplain{$66$ cm}{Diameter is twice the radius: $d = 2 \times 33 = 66$ cm.}

\prob A circular garden has a diameter of $4.6$ m. What is the radius? \answerBlank[3cm]
\answerExplain{$2.3$ m}{Radius is half the diameter: $r = 4.6 \div 2 = 2.3$ m.}

\prob A manhole cover has a radius of $\frac{3}{4}$ m. What is its diameter? \answerBlank[3cm]
\answerExplain{$1\frac{1}{2}$ m}{Diameter is twice the radius: $d = 2 \times \frac{3}{4} = \frac{6}{4} = \frac{3}{2} = 1\frac{1}{2}$ m.}

\prob A quarter (coin) has a diameter of about $2.4$ cm. What is its radius? \answerBlank[3cm]
\answerExplain{$1.2$ cm}{Radius is half the diameter: $r = 2.4 \div 2 = 1.2$ cm.}

\end{practiceBox}

% ── Practice C: True or False ────────────────────────────────────────────────
\begin{practiceBox}[funPurple]{True or False?}

\trueOrFalse{The diameter of a circle is always twice its radius.}
\answerExplain{True}{By definition, $d = 2r$. This relationship holds for every circle.}

\trueOrFalse{A radius is a chord of a circle.}
\answerExplain{False}{A chord must have both endpoints on the circle. A radius has one endpoint at the center and one on the circle, so it is not a chord.}

\trueOrFalse{Every diameter is a chord.}
\answerExplain{True}{The diameter has both endpoints on the circle, so it meets the definition of a chord. It is the longest chord.}

\trueOrFalse{A circle with radius $10$ cm has a diameter of $5$ cm.}
\answerExplain{False}{The diameter is twice the radius: $d = 2 \times 10 = 20$ cm, not $5$ cm.}

\end{practiceBox}

% ── Word Problems ────────────────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{A circular pond has a diameter of $18$ m. A frog sits at the center. How far does the frog need to hop to reach the edge?}{m}
\answerExplain{$9$ m}{The distance from the center to the edge is the radius: $r = 18 \div 2 = 9$ m.}

\wordProblem{A pizza has a radius of $7$ in. What is the distance across the pizza through its center?}{in}
\answerExplain{$14$ in}{The distance across through the center is the diameter: $d = 2 \times 7 = 14$ in.}

\wordProblem{A circular table has a diameter of $1.2$ m. A centerpiece exactly covers the radius from the center to the edge. How long is the centerpiece?}{m}
\answerExplain{$0.6$ m}{The centerpiece covers the radius: $r = 1.2 \div 2 = 0.6$ m.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob Two circles share the same center. The smaller circle has a radius of $4$ cm and the larger circle has a diameter of $20$ cm. How much longer is the radius of the larger circle than the radius of the smaller circle? \answerBlank[3cm]
\answerExplain{$6$ cm}{The larger circle's radius is $20 \div 2 = 10$ cm. The difference is $10 - 4 = 6$ cm.}

\prob A circular running track has an outer diameter of $50$ m and an inner diameter of $42$ m. What is the width of the track (the difference between the outer and inner radii)? \answerBlank[3cm]
\answerExplain{$4$ m}{Outer radius: $50 \div 2 = 25$ m. Inner radius: $42 \div 2 = 21$ m. Width: $25 - 21 = 4$ m.}
\end{challengeBox}
