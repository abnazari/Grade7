\section{Finding the Constant of Proportionality}
\topicTitle{Finding the Constant of Proportionality}

\begin{quickReview}{The Constant of Proportionality}
In a proportional relationship, every ratio $\dfrac{y}{x}$ equals the same value called \textbf{$k$}:
$$k = \frac{y}{x}$$

\begin{itemize}[leftmargin=*]
    \item $k$ is the \textbf{unit rate} --- how much $y$ changes for every $1$ unit of $x$.
    \item \textbf{From a table:} divide any $y$ by its $x$.
    \item \textbf{From a graph:} read the $y$-value at $x = 1$, or pick any point and compute $y \div x$.
    \item \textbf{From words:} divide the two given quantities.
\end{itemize}

\medskip
\textbf{Example:} If $3$ pounds of grapes cost $\$7.50$, then $k = 7.50 \div 3 = 2.50$. The grapes cost $\$2.50$ per pound.
\end{quickReview}

\practiceHeader[funGreen]{Find $k$ for each pair.}

\begin{multicols}{2}
\prob $x = 2$, $y = 10$ \quad $k = $ \answerBlank[2cm]
\answerExplain{$k = 5$}{Divide $y$ by $x$: $10 \div 2 = 5$.}

\prob $x = 4$, $y = 28$ \quad $k = $ \answerBlank[2cm]
\answerExplain{$k = 7$}{Divide $y$ by $x$: $28 \div 4 = 7$.}

\prob $x = 5$, $y = 12.5$ \quad $k = $ \answerBlank[2cm]
\answerExplain{$k = 2.5$}{Divide $y$ by $x$: $12.5 \div 5 = 2.5$.}

\prob $x = 3$, $y = 4.5$ \quad $k = $ \answerBlank[2cm]
\answerExplain{$k = 1.5$}{Divide $y$ by $x$: $4.5 \div 3 = 1.5$.}

\prob $x = 8$, $y = 6$ \quad $k = $ \answerBlank[2cm]
\answerExplain{$k = 0.75$}{Divide $y$ by $x$: $6 \div 8 = 0.75$.}

% ── Practice A: Finding k from Tables ────────────────────────────────────────
\begin{practiceBox}{Finding $k$ from Tables}

\practiceHeader{Find $k$ and explain what it means in context.}

\prob A bakery sells cupcakes at a constant price.
\begin{center}
\begin{tabular}{|c|c|}
\hline \textbf{Cupcakes} ($x$) & \textbf{Cost} ($y$) \\ \hline
$3$ & $\$7.50$ \\ $5$ & $\$12.50$ \\ $8$ & $\$20.00$ \\ \hline
\end{tabular}
\end{center}
$k = $ \answerBlank[3cm] \quad Meaning: \answerBlank[5cm]
\answerExplain{$k = 2.50$; each cupcake costs $\$2.50$}{Pick any row: $7.50 \div 3 = 2.50$. Check: $12.50 \div 5 = 2.50$ and $20.00 \div 8 = 2.50$. The unit rate $k = 2.50$ means each cupcake costs $\$2.50$.}

\prob A factory produces widgets at a constant rate.
\begin{center}
\begin{tabular}{|c|c|}
\hline \textbf{Hours} ($x$) & \textbf{Widgets} ($y$) \\ \hline
$2$ & $90$ \\ $4$ & $180$ \\ $7$ & $315$ \\ \hline
\end{tabular}
\end{center}
$k = $ \answerBlank[3cm] \quad Meaning: \answerBlank[5cm]
\answerExplain{$k = 45$; the factory makes $45$ widgets per hour}{Divide: $90 \div 2 = 45$. Check: $180 \div 4 = 45$, $315 \div 7 = 45$. The constant of proportionality means the factory produces $45$ widgets every hour.}

\prob \begin{center}
\begin{tabular}{|c|c|}
\hline $x$ & $y$ \\ \hline
$6$ & $9$ \\ $10$ & $15$ \\ $14$ & $21$ \\ \hline
\end{tabular}
\end{center}
$k = $ \answerBlank[3cm]
\answerExplain{$k = 1.5$}{Divide: $9 \div 6 = 1.5$. Check: $15 \div 10 = 1.5$ and $21 \div 14 = 1.5$. All rows confirm $k = 1.5$.}

\prob \begin{center}
\begin{tabular}{|c|c|}
\hline $x$ & $y$ \\ \hline
$\frac{1}{3}$ & $2$ \\ $1$ & $6$ \\ $2$ & $12$ \\ \hline
\end{tabular}
\end{center}
$k = $ \answerBlank[3cm]
\answerExplain{$k = 6$}{Divide using the second row for simplicity: $6 \div 1 = 6$. Check: $2 \div \frac{1}{3} = 2 \times 3 = 6$ and $12 \div 2 = 6$. So $k = 6$.}

\end{practiceBox}

% ── Practice B: Finding k from Graphs & Descriptions ─────────────────────────
\begin{practiceBox}[funTeal]{Finding $k$ from Graphs and Descriptions}

\prob A proportional graph passes through the point $(4, 22)$. What is $k$?
\answerBlank[3cm]
\answerExplain{$k = 5.5$}{On a proportional graph, $k = y \div x$: $22 \div 4 = 5.5$.}

\prob A proportional graph passes through $(1, 8)$. What is $k$?
\answerBlank[3cm]
\answerExplain{$k = 8$}{When $x = 1$, the $y$-value IS the unit rate. So $k = 8$.}

\prob A car uses gas at a constant rate and travels $210$ miles on $6$ gallons. What is $k$?
\answerBlank[3cm]
\answerExplain{$k = 35$ miles per gallon}{Divide distance by gallons: $210 \div 6 = 35$. The car gets $35$ miles per gallon.}

\prob A printer produces $96$ pages in $8$ minutes at a constant rate. What is $k$?
\answerBlank[3cm]
\answerExplain{$k = 12$ pages per minute}{Divide pages by time: $96 \div 8 = 12$ pages per minute.}

\prob A proportional relationship passes through $(5, 3)$. What is the value of $y$ when $x = 15$?
\answerBlank[3cm]
\answerExplain{$y = 9$}{Find $k$: $3 \div 5 = 0.6$. Then $y = 0.6 \times 15 = 9$. Alternatively, $x$ tripled from $5$ to $15$, so $y$ also triples: $3 \times 3 = 9$.}

\end{practiceBox}

\end{center}
\answerExplain{$y = 16$}{Find $k$: $8 \div 5 = 1.6$. Check: $24 \div 15 = 1.6$ \checkmark. Missing value: $y = 1.6 \times 10 = 16$.}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob Two proportional relationships share the point $(6, 18)$. Relationship A also passes through $(2, 6)$. Relationship B passes through $(3, 12)$. Do A and B have the same constant of proportionality? Explain.
\answerExplain{No}{Relationship A: $k = 6 \div 2 = 3$ (check: $18 \div 6 = 3$ \checkmark). Relationship B: $k = 12 \div 3 = 4$ (but $18 \div 6 = 3 \neq 4$). B is not actually proportional through both given points or they belong to different lines. A has $k = 3$ and B has $k = 4$, so they are different.}

\prob A proportional table has $k = \frac{3}{4}$. If $x = 20$, find $y$. If $y = 24$, find $x$.
\answerExplain{$y = 15$; $x = 32$}{When $x = 20$: $y = \frac{3}{4} \times 20 = 15$. When $y = 24$: $x = 24 \div \frac{3}{4} = 24 \times \frac{4}{3} = 32$.}
\end{challengeBox}
