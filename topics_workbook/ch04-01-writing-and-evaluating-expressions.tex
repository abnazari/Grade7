\section{Writing and Evaluating Expressions}
\topicTitle{Writing and Evaluating Expressions}

\begin{quickReview}{Writing and Evaluating Expressions}
An \textbf{algebraic expression} uses numbers, variables, and operations to represent a quantity.

\begin{itemize}[leftmargin=*]
    \item \textbf{Translate words to algebra:} ``$5$ more than a number $n$'' $\to$ $n + 5$; \quad ``$3$ times a number minus $7$'' $\to$ $3n - 7$.
    \item \textbf{Evaluate} means substitute a value for the variable and compute.
    \item Use \textbf{parentheses} when substituting negatives: $2(-3)$, not $2 \cdot -3$.
\end{itemize}

\medskip
\textbf{Example:} Evaluate $4x - 1$ when $x = -2$. \quad Substitute: $4(-2) - 1 = -8 - 1 = -9$.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Translate each phrase into an algebraic expression.}

\begin{multicols}{2}
\prob $7$ more than a number $n$ \answerBlank[3cm]
\answerExplain{$n + 7$}{``More than'' means addition. Add $7$ to the number $n$: $n + 7$.}

\prob A number $x$ decreased by $4$ \answerBlank[3cm]
\answerExplain{$x - 4$}{``Decreased by'' means subtraction. Subtract $4$ from $x$: $x - 4$.}

\prob Twice a number $y$ \answerBlank[3cm]
\answerExplain{$2y$}{``Twice'' means two times. Multiply the number by $2$: $2y$.}

\prob A number $m$ divided by $5$ \answerBlank[3cm]
\answerExplain{$\frac{m}{5}$}{``Divided by $5$'' means $m \div 5$, which we write as the fraction $\frac{m}{5}$.}

\prob $10$ less than $6$ times a number $p$ \answerBlank[3cm]
\answerExplain{$6p - 10$}{``$6$ times a number'' is $6p$. ``$10$ less than'' means subtract $10$ from that: $6p - 10$.}

\prob Half of a number $k$ plus $3$ \answerBlank[3cm]
\answerExplain{$\frac{k}{2} + 3$}{``Half of $k$'' is $\frac{k}{2}$. ``Plus $3$'' means add $3$: $\frac{k}{2} + 3$.}
\end{multicols}
\end{practiceBox}

% ── Practice A: Evaluating with Integers ─────────────────────────────────────
\begin{practiceBox}{Evaluating Expressions with Integers}

\practiceHeader{Evaluate each expression for the given value.}

\begin{multicols}{2}
\prob $3x + 5$ when $x = 4$ \answerBlank[3cm]
\answerExplain{$17$}{Substitute $x = 4$: $3(4) + 5 = 12 + 5 = 17$.}

\prob $2a - 9$ when $a = -3$ \answerBlank[3cm]
\answerExplain{$-15$}{Substitute $a = -3$: $2(-3) - 9 = -6 - 9 = -15$.}

\prob $-4n + 1$ when $n = 5$ \answerBlank[3cm]
\answerExplain{$-19$}{Substitute $n = 5$: $-4(5) + 1 = -20 + 1 = -19$.}

\prob $\frac{x}{2} + 7$ when $x = -8$ \answerBlank[3cm]
\answerExplain{$3$}{Substitute $x = -8$: $\frac{-8}{2} + 7 = -4 + 7 = 3$.}

\prob $10 - 3y$ when $y = -2$ \answerBlank[3cm]
\answerExplain{$16$}{Substitute $y = -2$: $10 - 3(-2) = 10 + 6 = 16$. A negative times a negative gives a positive.}

\prob $5 + 2x$ when $x = -6$ \answerBlank[3cm]
\answerExplain{$-7$}{Substitute $x = -6$: $5 + 2(-6) = 5 + (-12) = -7$.}
\end{multicols}
\end{practiceBox}

% ── Practice B: Evaluating with Fractions & Decimals ─────────────────────────
\begin{practiceBox}[funTeal]{Evaluating with Fractions and Decimals}

\practiceHeader[funTeal]{Evaluate each expression for the given value.}

\prob $6x - 1.5$ when $x = 0.5$ \answerBlank[3cm]
\answerExplain{$1.5$}{Substitute $x = 0.5$: $6(0.5) - 1.5 = 3 - 1.5 = 1.5$.}

\prob $\frac{3}{4}n + 2$ when $n = 8$ \answerBlank[3cm]
\answerExplain{$8$}{Substitute $n = 8$: $\frac{3}{4}(8) + 2 = 6 + 2 = 8$.}

\prob $-2.5a + 4$ when $a = -2$ \answerBlank[3cm]
\answerExplain{$9$}{Substitute $a = -2$: $-2.5(-2) + 4 = 5 + 4 = 9$.}

\prob $\frac{x}{3} - \frac{1}{2}$ when $x = 9$ \answerBlank[3cm]
\answerExplain{$\frac{5}{2}$ or $2\frac{1}{2}$}{Substitute $x = 9$: $\frac{9}{3} - \frac{1}{2} = 3 - \frac{1}{2} = \frac{6}{2} - \frac{1}{2} = \frac{5}{2} = 2\frac{1}{2}$.}

\end{practiceBox}

% ── Practice C: Writing Expressions from Situations ──────────────────────────
\begin{practiceBox}[funPurple]{\faGlobe~Real-World Expressions}

\practiceHeader[funPurple]{Write an expression for each situation, then evaluate.}

\prob An online store charges $\$3$ per item plus $\$5$ for shipping. Write an expression for the total cost of $n$ items, then find the cost for $8$ items.
\answerBlank[4cm]
\answerExplain{$3n + 5$; \quad $\$29$}{Each item costs $\$3$, so $n$ items cost $3n$. Add shipping: $3n + 5$. For $n = 8$: $3(8) + 5 = 24 + 5 = 29$.}

\prob A swimming pool drains $12$ gallons per minute. It starts with $500$ gallons. Write an expression for the water remaining after $m$ minutes, then find how much is left after $15$ minutes.
\answerBlank[4cm]
\answerExplain{$500 - 12m$; \quad $320$ gallons}{The pool loses $12$ gallons each minute, so after $m$ minutes: $500 - 12m$. For $m = 15$: $500 - 12(15) = 500 - 180 = 320$.}

\prob A taxi ride costs $\$2.50$ plus $\$1.75$ per mile. Write an expression for a ride of $d$ miles, then find the cost for $6$ miles.
\answerBlank[4cm]
\answerExplain{$1.75d + 2.50$; \quad $\$13.00$}{The per-mile charge is $1.75d$. Add the base fare: $1.75d + 2.50$. For $d = 6$: $1.75(6) + 2.50 = 10.50 + 2.50 = 13.00$.}

\end{practiceBox}

% ── Practice D: True or False ────────────────────────────────────────────────
\begin{practiceBox}[funOrange]{True or False?}

\trueOrFalse{The expression ``$5$ less than $x$'' is written as $5 - x$.}
\answerExplain{False}{``$5$ less than $x$'' means start with $x$ and subtract $5$: $x - 5$. Writing $5 - x$ reverses the order and gives a different value.}

\trueOrFalse{If $x = -3$, then $x^{2} = 9$.}
\answerExplain{True}{Substitute: $(-3)^{2} = (-3) \times (-3) = 9$. Squaring a negative number gives a positive result.}

\trueOrFalse{The expression $2x + 3$ equals $10$ when $x = 3$.}
\answerExplain{False}{Substitute $x = 3$: $2(3) + 3 = 6 + 3 = 9$, not $10$.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob Write an expression for the total cost of a concert trip: tickets cost $\$t$ each for $4$ friends, plus $\$20$ for parking, plus each friend pays $\$8$ for food. Evaluate when $t = 35$.
\answerBlank[4cm]
\answerExplain{$4t + 20 + 4(8) = 4t + 52$; \quad $\$192$}{Ticket cost: $4t$. Parking: $20$. Food: $4 \times 8 = 32$. Total expression: $4t + 52$. For $t = 35$: $4(35) + 52 = 140 + 52 = 192$.}

\prob Evaluate $\frac{2x + 3}{x - 1}$ when $x = 5$. Then find another integer value of $x$ that makes this expression equal a whole number.
\answerBlank[4cm]
\answerExplain{$\frac{13}{4}$ when $x = 5$; \quad $x = 3$ gives $\frac{9}{2}$, $x = 2$ gives $7$}{For $x = 5$: $\frac{2(5) + 3}{5 - 1} = \frac{13}{4}$. For $x = 2$: $\frac{2(2) + 3}{2 - 1} = \frac{7}{1} = 7$, a whole number.}
\end{challengeBox}
