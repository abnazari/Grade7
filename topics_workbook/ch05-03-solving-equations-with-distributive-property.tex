\section{Solving Equations with the Distributive Property}
\topicTitle{Solving Equations with the Distributive Property}

\begin{quickReview}{Equations with Parentheses}
For equations like $p(x + q) = r$, you have \textbf{two approaches}:
\begin{itemize}[leftmargin=*]
    \item \textbf{Distribute first:} Expand the parentheses, then solve as a two-step equation.
    \item \textbf{Divide first:} If the number outside divides evenly into the other side, divide both sides first.
\end{itemize}

\medskip
\textbf{Example:} Solve $4(x - 2) = 20$.

\textbf{Method 1 (distribute):} $4x - 8 = 20 \;\to\; 4x = 28 \;\to\; x = 7$.

\textbf{Method 2 (divide by $4$):} $x - 2 = 5 \;\to\; x = 7$.

\smallskip
Both methods give the same answer. Pick whichever is easier!
\end{quickReview}

\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

Expand each expression. (No solving needed.)

\begin{multicols}{2}
\prob $3(x + 4)$ \answerBlank[3cm]
\answerExplain{$3x + 12$}{Multiply $3$ by each term inside: $3 \cdot x + 3 \cdot 4 = 3x + 12$.}

\prob $5(n - 2)$ \answerBlank[3cm]
\answerExplain{$5n - 10$}{Multiply $5$ by each term: $5 \cdot n - 5 \cdot 2 = 5n - 10$.}

\prob $-2(y + 6)$ \answerBlank[3cm]
\answerExplain{$-2y - 12$}{Multiply $-2$ by each term: $(-2)(y) + (-2)(6) = -2y - 12$.}

\prob $4(2a - 3)$ \answerBlank[3cm]
\answerExplain{$8a - 12$}{Multiply $4$ by each term: $4 \cdot 2a - 4 \cdot 3 = 8a - 12$.}

\prob $-3(5 - x)$ \answerBlank[3cm]
\answerExplain{$-15 + 3x$}{Multiply $-3$ by each term: $(-3)(5) + (-3)(-x) = -15 + 3x$.}

\prob $6(m + 1)$ \answerBlank[3cm]
\answerExplain{$6m + 6$}{Multiply $6$ by each term: $6 \cdot m + 6 \cdot 1 = 6m + 6$.}
\end{multicols}
\end{practiceBox}

% ── Practice A: Solve by Distributing First ──────────────────────────────────
\begin{practiceBox}{Solve by Distributing First}

\practiceHeader{Expand the parentheses, then solve.}

\prob $2(x + 3) = 14$ \answerBlank[3cm]
\answerExplain{$x = 4$}{Distribute: $2x + 6 = 14$. Subtract $6$: $2x = 8$. Divide by $2$: $x = 4$. Check: $2(4 + 3) = 2(7) = 14$ \checkmark.}

\prob $3(n - 5) = 12$ \answerBlank[3cm]
\answerExplain{$n = 9$}{Distribute: $3n - 15 = 12$. Add $15$: $3n = 27$. Divide by $3$: $n = 9$. Check: $3(9 - 5) = 3(4) = 12$ \checkmark.}

\prob $-4(y + 2) = 20$ \answerBlank[3cm]
\answerExplain{$y = -7$}{Distribute: $-4y - 8 = 20$. Add $8$: $-4y = 28$. Divide by $-4$: $y = -7$. Check: $-4(-7 + 2) = -4(-5) = 20$ \checkmark.}

\prob $5(2x + 1) = 35$ \answerBlank[3cm]
\answerExplain{$x = 3$}{Distribute: $10x + 5 = 35$. Subtract $5$: $10x = 30$. Divide by $10$: $x = 3$. Check: $5(2 \cdot 3 + 1) = 5(7) = 35$ \checkmark.}

\prob $-2(3a - 4) = 26$ \answerBlank[3cm]
\answerExplain{$a = -3$}{Distribute: $-6a + 8 = 26$. Subtract $8$: $-6a = 18$. Divide by $-6$: $a = -3$. Check: $-2(3(-3) - 4) = -2(-13) = 26$ \checkmark.}

\end{practiceBox}

% ── Practice B: Solve by Dividing First ──────────────────────────────────────
\begin{practiceBox}[funTeal]{Solve by Dividing First}

\practiceHeader[funTeal]{Divide both sides by the number outside the parentheses, then solve.}

\prob $6(x + 2) = 42$ \answerBlank[3cm]
\answerExplain{$x = 5$}{Divide by $6$: $x + 2 = 7$. Subtract $2$: $x = 5$. Check: $6(5 + 2) = 6(7) = 42$ \checkmark.}

\prob $4(n - 7) = -8$ \answerBlank[3cm]
\answerExplain{$n = 5$}{Divide by $4$: $n - 7 = -2$. Add $7$: $n = 5$. Check: $4(5 - 7) = 4(-2) = -8$ \checkmark.}

\prob $-5(y + 3) = -30$ \answerBlank[3cm]
\answerExplain{$y = 3$}{Divide by $-5$: $y + 3 = 6$. Subtract $3$: $y = 3$. Check: $-5(3 + 3) = -5(6) = -30$ \checkmark.}

\prob $8(k - 1) = 56$ \answerBlank[3cm]
\answerExplain{$k = 8$}{Divide by $8$: $k - 1 = 7$. Add $1$: $k = 8$. Check: $8(8 - 1) = 8(7) = 56$ \checkmark.}

\prob $-3(x + 10) = 12$ \answerBlank[3cm]
\answerExplain{$x = -14$}{Divide by $-3$: $x + 10 = -4$. Subtract $10$: $x = -14$. Check: $-3(-14 + 10) = -3(-4) = 12$ \checkmark.}

\end{practiceBox}

% ── Practice C: Choose Your Method ───────────────────────────────────────────
\begin{practiceBox}[funOrange]{Choose Your Method}

\practiceHeader[funOrange]{Solve using whichever method you prefer.}

\prob $3(2x - 5) = 9$ \answerBlank[3cm]
\answerExplain{$x = 4$}{Divide by $3$: $2x - 5 = 3$. Add $5$: $2x = 8$. Divide by $2$: $x = 4$. Check: $3(2 \cdot 4 - 5) = 3(3) = 9$ \checkmark.}

\prob $-6(n + 4) = -18$ \answerBlank[3cm]
\answerExplain{$n = -1$}{Divide by $-6$: $n + 4 = 3$. Subtract $4$: $n = -1$. Check: $-6(-1 + 4) = -6(3) = -18$ \checkmark.}

\prob $7(y - 3) = -14$ \answerBlank[3cm]
\answerExplain{$y = 1$}{Divide by $7$: $y - 3 = -2$. Add $3$: $y = 1$. Check: $7(1 - 3) = 7(-2) = -14$ \checkmark.}

\prob $2(4a + 3) = 30$ \answerBlank[3cm]
\answerExplain{$a = 3$}{Distribute: $8a + 6 = 30$. Subtract $6$: $8a = 24$. Divide by $8$: $a = 3$. Check: $2(4 \cdot 3 + 3) = 2(15) = 30$ \checkmark.}

\end{practiceBox}

% ── Practice D: Word Problems ────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{Each side of a square flower bed is $(x + 5)$ feet. The perimeter is $48$ feet. Find $x$.}{feet}
\answerExplain{$x = 7$}{Perimeter $= 4(x + 5) = 48$. Divide by $4$: $x + 5 = 12$. Subtract $5$: $x = 7$. Each side is $12$ feet.}

\wordProblem{Three friends each bought a meal and a $\$2$ drink. Together they spent $\$33$. What was the cost of each meal?}{dollars}
\answerExplain{$\$9$}{Equation: $3(m + 2) = 33$. Divide by $3$: $m + 2 = 11$. Subtract $2$: $m = 9$.}

\wordProblem{A teacher splits $42$ pencils equally among groups. Each group also receives $3$ extra erasers. If each group gets $(p + 3)$ items total and there are $6$ groups, find $p$ (pencils per group).}{pencils}
\answerExplain{$p = 4$}{Total items: $6(p + 3) = 42$. Divide by $6$: $p + 3 = 7$. Subtract $3$: $p = 4$ pencils per group.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}

\prob Solve $-2(3x - 7) + 4 = 10$.
\answerExplain{$x = \frac{4}{3}$}{Distribute: $-6x + 14 + 4 = 10$. Combine: $-6x + 18 = 10$. Subtract $18$: $-6x = -8$. Divide by $-6$: $x = \frac{8}{6} = \frac{4}{3}$. Check: $-2(3 \cdot \frac{4}{3} - 7) + 4 = -2(4 - 7) + 4 = -2(-3) + 4 = 6 + 4 = 10$ \checkmark.}

\prob The lengths of the three sides of a triangle are $(x + 4)$, $(x + 4)$, and $(2x - 1)$ cm. The perimeter is $31$ cm. Find the length of each side.
\answerExplain{$10$ cm, $10$ cm, $11$ cm}{Perimeter: $(x + 4) + (x + 4) + (2x - 1) = 31$. Simplify: $4x + 7 = 31$. Subtract $7$: $4x = 24$. Divide by $4$: $x = 6$. Sides: $6 + 4 = 10$ cm, $10$ cm, $2(6) - 1 = 11$ cm. Check: $10 + 10 + 11 = 31$ \checkmark.}

\end{challengeBox}
