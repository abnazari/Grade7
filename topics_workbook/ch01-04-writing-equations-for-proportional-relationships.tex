\section{Writing Equations for Proportional Relationships}
\topicTitle{Writing Equations for Proportional Relationships}

\begin{quickReview}{Writing Proportional Equations}
Every proportional relationship can be written as:
$$y = kx$$
Find $k$ (the unit rate), then plug it in.

\begin{itemize}[leftmargin=*]
    \item \textbf{From a table:} divide any $y$ by $x$ to get $k$, then write $y = kx$.
    \item \textbf{From words:} identify the rate (``per'', ``each'', ``every'') and use it as $k$.
    \item \textbf{Using the equation:} plug in $x$ to predict $y$, or plug in $y$ to find $x$.
\end{itemize}

\medskip
\textbf{Example:} Oranges cost $\$0.75$ each. Equation: $c = 0.75n$.

For $12$ oranges: $c = 0.75 \times 12 = \$9.00$.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Write the equation $y = kx$ for each situation.}

\prob Bananas cost $\$0.60$ each. \answerBlank[4cm]
\answerExplain{$c = 0.60n$}{The rate is $\$0.60$ per banana, so $k = 0.60$ and the equation is $c = 0.60n$.}

\prob A car drives $55$ miles per hour. \answerBlank[4cm]
\answerExplain{$d = 55t$}{The rate is $55$ miles per hour, so $k = 55$ and the equation is $d = 55t$, where $t$ is time in hours.}

\prob A painter paints $3$ walls per hour. \answerBlank[4cm]
\answerExplain{$w = 3h$}{The rate is $3$ walls per hour, so $k = 3$ and the equation is $w = 3h$.}

\prob A factory makes $120$ toys per day. \answerBlank[4cm]
\answerExplain{$t = 120d$}{The production rate is $120$ toys per day, so $k = 120$ and the equation is $t = 120d$.}

\prob A membership costs $\$15$ per month (no sign-up fee). \answerBlank[4cm]
\answerExplain{$c = 15m$}{The rate is $\$15$ per month with no extra fees, so $k = 15$ and the equation is $c = 15m$.}
\end{practiceBox}

% ── Practice A: From Tables ──────────────────────────────────────────────────
\begin{practiceBox}{From Tables to Equations}

\practiceHeader{Find $k$, write the equation, and use it to answer the question.}

\prob \begin{center}
\begin{tabular}{|c|c|}
\hline $x$ & $y$ \\ \hline $2$ & $9$ \\ $4$ & $18$ \\ $6$ & $27$ \\ \hline
\end{tabular}
\end{center}
Equation: \answerBlank[3cm] \quad Find $y$ when $x = 10$: \answerBlank[3cm]
\answerExplain{$y = 4.5x$; $y = 45$}{Find $k$: $9 \div 2 = 4.5$. Equation: $y = 4.5x$. When $x = 10$: $y = 4.5 \times 10 = 45$.}

\prob A farm stand:
\begin{center}
\begin{tabular}{|c|c|}
\hline \textbf{Pounds} ($p$) & \textbf{Cost} ($c$) \\ \hline $3$ & $\$10.50$ \\ $5$ & $\$17.50$ \\ \hline
\end{tabular}
\end{center}
Equation: \answerBlank[3cm] \quad Cost for $7$ pounds: \answerBlank[3cm]
\answerExplain{$c = 3.50p$; $\$24.50$}{Find $k$: $10.50 \div 3 = 3.50$. Equation: $c = 3.50p$. For $7$ pounds: $c = 3.50 \times 7 = \$24.50$.}

\prob \begin{center}
\begin{tabular}{|c|c|}
\hline $x$ & $y$ \\ \hline $4$ & $6$ \\ $8$ & $12$ \\ $12$ & $18$ \\ \hline
\end{tabular}
\end{center}
Equation: \answerBlank[3cm] \quad Find $x$ when $y = 30$: \answerBlank[3cm]
\answerExplain{$y = 1.5x$; $x = 20$}{Find $k$: $6 \div 4 = 1.5$. Equation: $y = 1.5x$. When $y = 30$: $30 = 1.5x$, so $x = 30 \div 1.5 = 20$.}

\prob \begin{center}
\begin{tabular}{|c|c|}
\hline \textbf{Minutes} ($m$) & \textbf{Pages} ($p$) \\ \hline $5$ & $40$ \\ $10$ & $80$ \\ $15$ & $120$ \\ \hline
\end{tabular}
\end{center}
Equation: \answerBlank[3cm] \quad How many minutes for $200$ pages? \answerBlank[3cm]
\answerExplain{$p = 8m$; $25$ minutes}{Find $k$: $40 \div 5 = 8$ pages per minute. Equation: $p = 8m$. For $200$ pages: $200 = 8m$, so $m = 200 \div 8 = 25$ minutes.}

\end{practiceBox}

% ── Practice B: Using Equations ──────────────────────────────────────────────
\begin{practiceBox}[funTeal]{Using Equations to Solve Problems}

\prob The equation $d = 4.5t$ gives distance in miles after $t$ hours of hiking. How far after $6$ hours?
\answerBlank[3cm]
\answerExplain{$27$ miles}{Substitute $t = 6$: $d = 4.5 \times 6 = 27$ miles.}

\prob The equation $c = 0.08w$ gives the cost (in dollars) of $w$ grams of candy. How many grams can you buy with $\$5$?
\answerBlank[3cm]
\answerExplain{$62.5$ grams}{Set $c = 5$: $5 = 0.08w$, so $w = 5 \div 0.08 = 62.5$ grams.}

\prob The equation $p = 12h$ gives the pay for $h$ hours of work. How many hours to earn $\$180$?
\answerBlank[3cm]
\answerExplain{$15$ hours}{Set $p = 180$: $180 = 12h$, so $h = 180 \div 12 = 15$ hours.}

\prob A recipe uses $\frac{2}{3}$ cup of milk per serving. Write an equation and find how many cups are needed for $9$ servings.
\answerBlank[4cm]
\answerExplain{$m = \frac{2}{3}s$; $6$ cups}{The rate is $\frac{2}{3}$ cup per serving, so $m = \frac{2}{3}s$. For $9$ servings: $m = \frac{2}{3} \times 9 = 6$ cups.}

\prob Water flows from a tap at $2.5$ liters per minute. Write an equation and find how long to fill an $80$-liter tank.
\answerBlank[4cm]
\answerExplain{$w = 2.5t$; $32$ minutes}{The equation is $w = 2.5t$. Set $w = 80$: $80 = 2.5t$, so $t = 80 \div 2.5 = 32$ minutes.}

\end{practiceBox}

% ── Practice C: Multiple Choice ──────────────────────────────────────────────
\begin{practiceBox}[funPurple]{Pick the Correct Equation}

\circleAnswer{A store sells trail mix for $\$6$ per pound. Which equation gives the total cost $c$ for $p$ pounds?}{$c = 6p$}{$c = p + 6$}{$c = \frac{p}{6}$}{$p = 6c$}
\answerExplain{A}{The price is $\$6$ per pound with no extra fee. A proportional equation is $y = kx$, so $c = 6p$. Choice B adds $6$ instead of multiplying. Choice C divides incorrectly. Choice D reverses the variables.}

\circleAnswer{A graph passes through $(0, 0)$ and $(5, 20)$. Which equation matches?}{$y = 5x$}{$y = 20x$}{$y = 4x$}{$y = x + 15$}
\answerExplain{C}{Find $k$: $20 \div 5 = 4$, so $y = 4x$. Choice A uses $k = 5$ (wrong). Choice B uses $k = 20$ (wrong). Choice D is not proportional because of the $+15$.}

\circleAnswer{Which equation is NOT proportional?}{$y = 3x$}{$y = 0.5x$}{$y = 2x + 1$}{$y = 7x$}
\answerExplain{C}{A proportional equation has the form $y = kx$ with no added constant. Equation $y = 2x + 1$ has a $+1$, so it is not proportional. The other three are all $y = kx$.}

\end{practiceBox}

% ── Word Problems ────────────────────────────────────────────────────────────
\begin{practiceBox}[funOrange]{\faGlobe~Word Problems}

\wordProblem{A bike rental charges $\$9$ per hour. Write an equation and find the cost of renting a bike for $3\frac{1}{2}$ hours.}{dollars}
\answerExplain{$c = 9h$; $\$31.50$}{The rate is $\$9$ per hour, so $c = 9h$. For $3.5$ hours: $c = 9 \times 3.5 = \$31.50$.}

\wordProblem{A bus travels at a constant speed. It covers $195$ miles in $3$ hours. Write an equation for the distance $d$ in terms of hours $h$ and find how far the bus travels in $5$ hours.}{miles}
\answerExplain{$d = 65h$; $325$ miles}{Find $k$: $195 \div 3 = 65$ mph. Equation: $d = 65h$. For $5$ hours: $d = 65 \times 5 = 325$ miles.}

\wordProblem{Tomatoes cost $\$2.80$ per pound. You have $\$14$. How many pounds can you buy?}{pounds}
\answerExplain{$5$ pounds}{Equation: $c = 2.80p$. Set $c = 14$: $14 = 2.80p$, so $p = 14 \div 2.80 = 5$ pounds.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob Two equations: $y = 3.5x$ and $y = 4x$. For what value of $x$ does the second equation give a $y$ that is $6$ more than the first? \answerBlank[3cm]
\answerExplain{$x = 12$}{Set up the equation: $4x = 3.5x + 6$. Subtract $3.5x$: $0.5x = 6$. Divide: $x = 12$.}

\prob An equation $y = kx$ passes through $\left(\frac{2}{3},\; 5\right)$. Find $k$ and use it to find $y$ when $x = 4$.
\answerBlank[4cm]
\answerExplain{$k = 7.5$; $y = 30$}{Find $k$: $5 \div \frac{2}{3} = 5 \times \frac{3}{2} = \frac{15}{2} = 7.5$. When $x = 4$: $y = 7.5 \times 4 = 30$.}
\end{challengeBox}

\encouragement{You can turn any proportional situation into a powerful equation --- that is real math power!}
