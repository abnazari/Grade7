\section{Factoring Expressions}
\topicTitle{Factoring Expressions}

\begin{quickReview}{Factoring = Distributive Property in Reverse}
\textbf{Factoring} means rewriting a sum or difference as a \textbf{product}.

\begin{itemize}[leftmargin=*]
    \item Find the \textbf{greatest common factor (GCF)} of all terms.
    \item Divide each term by the GCF.
    \item Write the GCF outside and the quotients inside parentheses.
\end{itemize}

\medskip
\textbf{Example:} Factor $6x + 15$. \quad GCF $= 3$. \quad $6x \div 3 = 2x$, \quad $15 \div 3 = 5$. \quad Answer: $3(2x + 5)$.

\smallskip
\textbf{Check:} Expand $3(2x + 5) = 6x + 15$ \checkmark
\end{quickReview}

% ── Practice A: Basic Factoring ──────────────────────────────────────────────
\begin{practiceBox}{Factor Each Expression}

\practiceHeader{Find the GCF, then write the factored form.}

\begin{multicols}{2}
\prob $10x + 20$ \answerBlank[3cm]
\answerExplain{$10(x + 2)$}{GCF of $10$ and $20$ is $10$. Divide: $10x \div 10 = x$, $20 \div 10 = 2$. Write: $10(x + 2)$.}

\prob $8a - 12$ \answerBlank[3cm]
\answerExplain{$4(2a - 3)$}{GCF of $8$ and $12$ is $4$. Divide: $8a \div 4 = 2a$, $-12 \div 4 = -3$. Write: $4(2a - 3)$.}

\prob $15n + 25$ \answerBlank[3cm]
\answerExplain{$5(3n + 5)$}{GCF of $15$ and $25$ is $5$. Divide: $15n \div 5 = 3n$, $25 \div 5 = 5$. Write: $5(3n + 5)$.}

\prob $6y - 9$ \answerBlank[3cm]
\answerExplain{$3(2y - 3)$}{GCF of $6$ and $9$ is $3$. Divide: $6y \div 3 = 2y$, $-9 \div 3 = -3$. Write: $3(2y - 3)$.}

\prob $18m + 24$ \answerBlank[3cm]
\answerExplain{$6(3m + 4)$}{GCF of $18$ and $24$ is $6$. Divide: $18m \div 6 = 3m$, $24 \div 6 = 4$. Write: $6(3m + 4)$.}

\prob $21x - 14$ \answerBlank[3cm]
\answerExplain{$7(3x - 2)$}{GCF of $21$ and $14$ is $7$. Divide: $21x \div 7 = 3x$, $-14 \div 7 = -2$. Write: $7(3x - 2)$.}
\end{multicols}
\end{practiceBox}

% ── Practice B: Factoring with Two Variables ─────────────────────────────────
\begin{practiceBox}[funTeal]{Factoring with Two Variable Terms}

\prob $12a + 18b$ \answerBlank[3cm]
\answerExplain{$6(2a + 3b)$}{GCF of $12$ and $18$ is $6$. Divide: $12a \div 6 = 2a$, $18b \div 6 = 3b$. Write: $6(2a + 3b)$.}

\prob $20x - 30y$ \answerBlank[3cm]
\answerExplain{$10(2x - 3y)$}{GCF of $20$ and $30$ is $10$. Divide: $20x \div 10 = 2x$, $-30y \div 10 = -3y$. Write: $10(2x - 3y)$.}

\prob $9p + 15q$ \answerBlank[3cm]
\answerExplain{$3(3p + 5q)$}{GCF of $9$ and $15$ is $3$. Divide: $9p \div 3 = 3p$, $15q \div 3 = 5q$. Write: $3(3p + 5q)$.}

\prob $16m - 24n$ \answerBlank[3cm]
\answerExplain{$8(2m - 3n)$}{GCF of $16$ and $24$ is $8$. Divide: $16m \div 8 = 2m$, $-24n \div 8 = -3n$. Write: $8(2m - 3n)$.}

\end{practiceBox}

% ── Practice C: Factoring with Negatives ─────────────────────────────────────
\begin{practiceBox}[funPurple]{Factoring Out a Negative}

\practiceHeader[funPurple]{Factor out the negative GCF.}

\prob $-6x - 10$ \answerBlank[3cm]
\answerExplain{$-2(3x + 5)$}{GCF is $2$; factor out $-2$: $-6x \div (-2) = 3x$ and $-10 \div (-2) = 5$. Write: $-2(3x + 5)$. Check: $-2(3x + 5) = -6x - 10$ \checkmark.}

\prob $-12a + 8$ \answerBlank[3cm]
\answerExplain{$-4(3a - 2)$}{GCF is $4$; factor out $-4$: $-12a \div (-4) = 3a$ and $8 \div (-4) = -2$. Write: $-4(3a - 2)$. Check: $-4(3a - 2) = -12a + 8$ \checkmark.}

\prob $-15n - 20$ \answerBlank[3cm]
\answerExplain{$-5(3n + 4)$}{GCF is $5$; factor out $-5$: $-15n \div (-5) = 3n$ and $-20 \div (-5) = 4$. Write: $-5(3n + 4)$. Check: $-5(3n + 4) = -15n - 20$ \checkmark.}

\end{practiceBox}

% ── Practice D: True or False ────────────────────────────────────────────────
\begin{practiceBox}[funOrange]{True or False?}

\trueOrFalse{$4x + 9$ can be factored.}
\answerExplain{False}{The GCF of $4$ and $9$ is $1$. Factoring out $1$ does not simplify the expression, so $4x + 9$ cannot be factored further.}

\trueOrFalse{$3(5x - 2) = 15x - 6$.}
\answerExplain{True}{Expand to check: $3 \times 5x = 15x$ and $3 \times (-2) = -6$. So $3(5x - 2) = 15x - 6$.}

\trueOrFalse{The factored form of $8y + 6$ is $2(4y + 3)$.}
\answerExplain{True}{GCF of $8$ and $6$ is $2$. Divide: $8y \div 2 = 4y$, $6 \div 2 = 3$. Write: $2(4y + 3)$. This is correct.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob Factor $36x + 48y - 60$. \answerBlank[4cm]
\answerExplain{$12(3x + 4y - 5)$}{GCF of $36$, $48$, and $60$ is $12$. Divide each: $36x \div 12 = 3x$, $48y \div 12 = 4y$, $-60 \div 12 = -5$. Write: $12(3x + 4y - 5)$.}

\prob A rectangle has area $24x + 16$. One side has length $8$. Write an expression for the other side.
\answerBlank[4cm]
\answerExplain{$3x + 2$}{Area $=$ length $\times$ width. Factor: $24x + 16 = 8(3x + 2)$. Since one side is $8$, the other side is $3x + 2$.}
\end{challengeBox}
