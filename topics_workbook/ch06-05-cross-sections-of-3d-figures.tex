\section{Cross-Sections of 3D Figures}
\topicTitle{Cross-Sections of 3D Figures}

\begin{quickReview}{What Is a Cross-Section?}
A \textbf{cross-section} is the flat (2D) shape you see when you slice through a 3D figure with a plane — like cutting through a loaf of bread.

\begin{itemize}[leftmargin=*]
    \item \textbf{Rectangular prism} (box): cuts can produce rectangles, squares, or triangles.
    \item \textbf{Rectangular pyramid}: cuts can produce triangles, trapezoids, or rectangles.
    \item \textbf{Cylinder}: horizontal cuts give circles; vertical cuts give rectangles.
    \item \textbf{Cone}: horizontal cuts give circles; vertical cuts through the apex give triangles.
    \item \textbf{Sphere}: every cross-section is a circle.
\end{itemize}

\medskip
\textbf{Key idea:} The shape depends on \textbf{where} and \textbf{at what angle} you cut.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Name the Cross-Section}
Identify the 2D shape formed by each slice.

\begin{multicols}{2}
\prob Slice a rectangular prism parallel to its base. \answerBlank[3cm]
\answerExplain{Rectangle}{A horizontal cut parallel to the base of a rectangular prism always produces a rectangle identical to the base.}

\prob Slice a sphere with any plane. \answerBlank[3cm]
\answerExplain{Circle}{Every cross-section of a sphere is a circle. The size depends on how close the cut is to the center.}

\prob Slice a cone parallel to its base. \answerBlank[3cm]
\answerExplain{Circle}{A horizontal cut parallel to the circular base of a cone produces a smaller circle.}

\prob Slice a cylinder parallel to its base. \answerBlank[3cm]
\answerExplain{Circle}{A horizontal cut through a cylinder parallel to its circular base creates a circle with the same radius.}

\prob Slice a cube with a cut through two opposite edges. \answerBlank[3cm]
\answerExplain{Rectangle}{A cut through two opposite edges of a cube creates a rectangular cross-section.}
\end{multicols}
\end{practiceBox}

% ── Practice A: Rectangular Prisms ───────────────────────────────────────────
\begin{practiceBox}{Slicing Rectangular Prisms}

\practiceHeader{Describe the cross-section for each type of cut through a rectangular prism (box).}

\prob A cut parallel to the top face, halfway down. \answerBlank[3cm]
\answerExplain{Rectangle}{A horizontal cut parallel to the base gives a rectangle with the same length and width as the base.}

\prob A cut parallel to one of the side faces. \answerBlank[3cm]
\answerExplain{Rectangle}{A vertical cut parallel to a side face produces a rectangle with the same height and depth as that face.}

\prob A diagonal cut from one top edge to the opposite bottom edge. \answerBlank[3cm]
\answerExplain{Rectangle}{A diagonal cut connecting opposite edges of a rectangular prism typically produces a rectangle (larger than any face).}

\prob A cut from one corner of the top face to the opposite corner of the top face and from one corner of the bottom face to the opposite corner of the bottom face. \answerBlank[3cm]
\answerExplain{Rectangle}{A vertical diagonal cut through a rectangular prism creates a rectangular cross-section whose width is the diagonal of the base.}

\prob A cut that slices off one corner of the prism (through three adjacent edges). \answerBlank[3cm]
\answerExplain{Triangle}{Cutting off a corner by slicing through three edges that meet at one vertex creates a triangular cross-section.}
\end{practiceBox}

% ── Practice B: Pyramids ─────────────────────────────────────────────────────
\begin{practiceBox}[funTeal]{Slicing Pyramids}

\practiceHeader[funTeal]{Describe the cross-section for each cut through a right rectangular pyramid.}

\prob A horizontal cut parallel to the base, one-third from the top. \answerBlank[3cm]
\answerExplain{A smaller rectangle}{A cut parallel to the rectangular base of a pyramid gives a smaller rectangle. The higher the cut, the smaller the rectangle.}

\prob A vertical cut through the apex and the midpoint of one base edge. \answerBlank[3cm]
\answerExplain{Triangle}{A vertical cut through the apex of a pyramid and down through the base creates a triangular cross-section.}

\prob A vertical cut parallel to one side of the base, not through the apex. \answerBlank[3cm]
\answerExplain{Trapezoid}{A vertical cut parallel to a base edge but not passing through the apex slices through two slanted faces, forming a trapezoid.}

\prob A horizontal cut at the very base of the pyramid. \answerBlank[3cm]
\answerExplain{Rectangle}{At the base, the cross-section is the base itself — a rectangle.}
\end{practiceBox}

% ── Practice C: Multiple Choice ──────────────────────────────────────────────
\begin{practiceBox}[funPurple]{Choose the Best Answer}

\circleAnswer{Which cross-section is NOT possible when slicing a rectangular prism?}{Rectangle}{Triangle}{Circle}{Square}
\answerExplain{C}{A rectangular prism has flat faces and straight edges. You can get rectangles, squares, and triangles, but never a circle, because a circle requires a curved surface.}

\circleAnswer{A cone is sliced vertically through its apex. The cross-section is a:}{Circle}{Rectangle}{Triangle}{Trapezoid}
\answerExplain{C}{A vertical cut through the apex of a cone passes through the tip and the base diameter, creating a triangular cross-section (an isosceles triangle).}

\circleAnswer{A horizontal slice of a cylinder always produces a:}{Rectangle}{Triangle}{Circle}{Trapezoid}
\answerExplain{C}{A cylinder has circular bases. Any horizontal cut parallel to the base creates a circle with the same radius.}

\circleAnswer{Slicing a rectangular pyramid parallel to its base always gives a:}{Triangle}{Smaller rectangle}{Circle}{Pentagon}
\answerExplain{B}{A horizontal cut parallel to the rectangular base of a pyramid produces a smaller rectangle that is similar to the base.}
\end{practiceBox}

% ── Practice D: True or False ────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{True or False?}

\trueOrFalse{Every cross-section of a cube is a square.}
\answerExplain{False}{Horizontal cuts parallel to a face give squares, but diagonal or angled cuts can give rectangles, triangles, or even hexagons.}

\trueOrFalse{A cross-section of a sphere is always a circle.}
\answerExplain{True}{No matter where or at what angle you cut a sphere, the cross-section is always a circle.}

\trueOrFalse{If you slice a rectangular prism parallel to its base, the cross-section has the same shape and size as the base.}
\answerExplain{True}{A horizontal cut parallel to the base of a prism always produces a shape identical to the base — same dimensions, same shape.}

\trueOrFalse{A vertical cut through a cylinder always produces a circle.}
\answerExplain{False}{A vertical cut through a cylinder (perpendicular to the base) produces a rectangle, not a circle. The rectangle's height equals the cylinder's height and its width equals the diameter.}
\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob A cube is sliced by a plane that passes through the midpoints of three edges that all meet at one corner. What shape is the cross-section? How does the cross-section's area compare to the face area?
\answerExplain{Equilateral triangle; area is $\frac{\sqrt{3}}{4} \cdot \frac{s^2}{2}$, which is about $43\%$ of a face}{The plane cuts through three midpoints of edges meeting at one vertex, creating an equilateral triangle with side length $\frac{s\sqrt{2}}{2}$. Its area is $\frac{\sqrt{3}}{4}\left(\frac{s\sqrt{2}}{2}\right)^2 = \frac{s^2\sqrt{3}}{8}$. A face has area $s^2$, so the cross-section is about $\frac{\sqrt{3}}{8} \approx 21.7\%$ of a face.}

\prob Name a 3D figure where every possible cross-section is a circle.
\answerBlank[3cm]
\answerExplain{Sphere}{A sphere is the only common 3D figure where every possible plane cross-section is a circle, regardless of where or at what angle you cut.}
\end{challengeBox}
