\section{Simple Interest: Earning and Paying Interest}
\topicTitle{Simple Interest: Earning and Paying Interest}

\begin{quickReview}{Simple Interest}
\textbf{Simple interest} is money earned on savings or owed on loans, calculated with:
$$I = Prt$$

\begin{itemize}[leftmargin=*]
    \item $I$ = interest (dollars earned or owed)
    \item $P$ = principal (starting amount)
    \item $r$ = annual interest rate (as a decimal)
    \item $t$ = time (in years)
\end{itemize}

\textbf{Total amount} $= P + I$

\medskip
\textbf{Example:} You deposit $\$600$ at $5\%$ for $3$ years. $I = 600 \times 0.05 \times 3 = \$90$. Total $= 600 + 90 = \$690$.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Convert Rates and Identify Variables}

\begin{multicols}{2}
\prob Convert $7\%$ to a decimal. \answerBlank[3cm]
\answerExplain{$0.07$}{Divide by $100$: $7\% = 7 \div 100 = 0.07$. Always convert percent to a decimal before using the formula.}

\prob Convert $3.5\%$ to a decimal. \answerBlank[3cm]
\answerExplain{$0.035$}{Divide by $100$: $3.5\% = 3.5 \div 100 = 0.035$.}

\prob $P = \$500$, $r = 4\%$, $t = 1$ year. Find $I$. \answerBlank[3cm]
\answerExplain{$\$20$}{$I = Prt = 500 \times 0.04 \times 1 = \$20$. In one year, $4\%$ of $\$500$ is $\$20$.}

\prob $P = \$1{,}000$, $r = 10\%$, $t = 2$ years. Find $I$. \answerBlank[3cm]
\answerExplain{$\$200$}{$I = Prt = 1{,}000 \times 0.10 \times 2 = \$200$.}

\prob $P = \$200$, $r = 6\%$, $t = 5$ years. Find the total amount. \answerBlank[3cm]
\answerExplain{$\$260$}{$I = 200 \times 0.06 \times 5 = \$60$. Total $= 200 + 60 = \$260$.}

\prob $18$ months equals how many years? \answerBlank[3cm]
\answerExplain{$1.5$ years}{Time must be in years for the formula. Divide months by $12$: $18 \div 12 = 1.5$ years.}
\end{multicols}
\end{practiceBox}

% ── Practice A: Finding Interest ─────────────────────────────────────────────
\begin{practiceBox}{Calculating Simple Interest}

\practiceHeader{Use $I = Prt$ to find the interest.}

\prob $P = \$800$, $r = 5\%$, $t = 3$ years. \answerBlank[3cm]
\answerExplain{$\$120$}{$I = 800 \times 0.05 \times 3 = \$120$.}

\prob $P = \$1{,}500$, $r = 3\%$, $t = 4$ years. \answerBlank[3cm]
\answerExplain{$\$180$}{$I = 1{,}500 \times 0.03 \times 4 = \$180$.}

\prob $P = \$2{,}400$, $r = 7\%$, $t = 2$ years. \answerBlank[3cm]
\answerExplain{$\$336$}{$I = 2{,}400 \times 0.07 \times 2 = \$336$.}

\prob $P = \$950$, $r = 4.5\%$, $t = 6$ years. \answerBlank[3cm]
\answerExplain{$\$256.50$}{$I = 950 \times 0.045 \times 6 = \$256.50$.}

\prob $P = \$3{,}000$, $r = 2\%$, $t = 18$ months. \answerBlank[3cm]
\answerExplain{$\$90$}{Convert time: $18$ months $= 1.5$ years. $I = 3{,}000 \times 0.02 \times 1.5 = \$90$.}

\end{practiceBox}

% ── Practice B: Finding Total Amount ─────────────────────────────────────────
\begin{practiceBox}[funTeal]{Finding the Total Amount}

\practiceHeader[funTeal]{Find the interest and the total amount ($P + I$).}

\prob You deposit $\$1{,}200$ at $4\%$ for $5$ years. What is the total? \answerBlank[3cm]
\answerExplain{$\$1{,}440$}{$I = 1{,}200 \times 0.04 \times 5 = \$240$. Total $= 1{,}200 + 240 = \$1{,}440$.}

\prob You borrow $\$3{,}500$ at $8\%$ for $3$ years. How much do you pay back in total? \answerBlank[3cm]
\answerExplain{$\$4{,}340$}{$I = 3{,}500 \times 0.08 \times 3 = \$840$. Total $= 3{,}500 + 840 = \$4{,}340$.}

\prob A savings account has $\$750$ at $2.5\%$ for $4$ years. Total amount? \answerBlank[3cm]
\answerExplain{$\$825$}{$I = 750 \times 0.025 \times 4 = \$75$. Total $= 750 + 75 = \$825$.}

\prob You invest $\$5{,}000$ at $6\%$ for $2$ years. Total amount? \answerBlank[3cm]
\answerExplain{$\$5{,}600$}{$I = 5{,}000 \times 0.06 \times 2 = \$600$. Total $= 5{,}000 + 600 = \$5{,}600$.}

\end{practiceBox}

% ── Practice C: Solving for Other Variables ──────────────────────────────────
\begin{practiceBox}[funPurple]{Working Backward with $I = Prt$}

\practiceHeader[funPurple]{Rearrange the formula to find the missing variable.}

\prob A loan of $\$2{,}000$ accumulated $\$360$ in interest over $3$ years. What is the rate? \answerBlank[3cm]
\answerExplain{$6\%$}{Solve for $r$: $360 = 2{,}000 \times r \times 3$. So $360 = 6{,}000r$. Divide: $r = 360 \div 6{,}000 = 0.06 = 6\%$.}

\prob You earned $\$150$ interest on $\$2{,}500$ at $3\%$. For how many years? \answerBlank[3cm]
\answerExplain{$2$ years}{Solve for $t$: $150 = 2{,}500 \times 0.03 \times t$. So $150 = 75t$. Divide: $t = 150 \div 75 = 2$ years.}

\prob After $4$ years at $5\%$, you earned $\$100$ in interest. What was the principal? \answerBlank[3cm]
\answerExplain{$\$500$}{Solve for $P$: $100 = P \times 0.05 \times 4$. So $100 = 0.20P$. Divide: $P = 100 \div 0.20 = \$500$.}

\prob A bank account earned $\$180$ on $\$1{,}200$ over some time at $6\%$. How long? \answerBlank[3cm]
\answerExplain{$2.5$ years}{Solve for $t$: $180 = 1{,}200 \times 0.06 \times t$. So $180 = 72t$. Divide: $t = 180 \div 72 = 2.5$ years.}

\end{practiceBox}

% ── Practice D: Word Problems ────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{Maria deposits $\$1{,}800$ in a savings account that earns $3.5\%$ simple interest per year. How much money will she have after $4$ years?}{dollars}
\answerExplain{$\$2{,}052$}{$I = 1{,}800 \times 0.035 \times 4 = \$252$. Total $= 1{,}800 + 252 = \$2{,}052$.}

\wordProblem{Jake borrows $\$4{,}000$ for a used car at $7\%$ simple interest for $5$ years. How much interest will he pay? What is the total he pays back?}{dollars}
\answerExplain{Interest: $\$1{,}400$; Total: $\$5{,}400$}{$I = 4{,}000 \times 0.07 \times 5 = \$1{,}400$. Total $= 4{,}000 + 1{,}400 = \$5{,}400$.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob Bank A offers $4\%$ simple interest. Bank B offers $3\%$ simple interest. You deposit $\$2{,}000$ in Bank A and $\$3{,}000$ in Bank B. After how many years will you have earned the same total interest from both banks? \answerBlank[3cm]
\answerExplain{Never (Bank B always earns more)}{Bank A interest per year: $2{,}000 \times 0.04 = \$80$. Bank B interest per year: $3{,}000 \times 0.03 = \$90$. Bank B earns $\$10$ more per year forever. They never produce the same total interest; Bank B always earns more.}

\prob Two investments: $\$1{,}000$ at $8\%$ and $\$2{,}000$ at $5\%$. How much total interest do you earn if both are invested for $3$ years? \answerBlank[3cm]
\answerExplain{$\$540$}{Investment 1: $I = 1{,}000 \times 0.08 \times 3 = \$240$. Investment 2: $I = 2{,}000 \times 0.05 \times 3 = \$300$. Total interest: $240 + 300 = \$540$.}
\end{challengeBox}
