\section{Solving Real-World Problems with Rational Numbers}
\topicTitle{Solving Real-World Problems with Rational Numbers}

\begin{quickReview}{Putting It All Together}
Real-world problems often mix \textbf{all four operations} with rational numbers — integers, fractions, and decimals.

\begin{itemize}[leftmargin=*]
    \item \textbf{Read carefully:} Identify which operation each part needs.
    \item \textbf{Watch the signs:} Gains, deposits, rises $\to$ positive. Losses, withdrawals, drops $\to$ negative.
    \item \textbf{Order of operations:} Multiply and divide before adding and subtracting.
    \item \textbf{Check your answer:} Does the sign make sense? Is the size reasonable?
\end{itemize}

\medskip
\textbf{Example:} Balance $= \$125.50 - \$40.75 + \$18.20 - \$3.50 = \$99.45$.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Choose the correct operation and solve.}

\prob A hiker climbs $350$ m, descends $125$ m, then climbs $80$ m. What is the net elevation change? \answerBlank[3cm]
\answerExplain{$305$ m}{Add all changes: $350 + (-125) + 80 = 350 - 125 + 80 = 305$ meters gained.}

\prob You have $\$50$ and spend $\$12.75$. How much remains? \answerBlank[3cm]
\answerExplain{$\$37.25$}{Subtract: $50 - 12.75 = 37.25$.}

\prob A baker uses $\frac{2}{3}$ cup of sugar for each batch and makes $6$ batches. How much sugar total? \answerBlank[3cm]
\answerExplain{$4$ cups}{Multiply: $\frac{2}{3} \times 6 = \frac{12}{3} = 4$ cups.}

\prob Split $\$27.60$ equally among $4$ friends. How much does each get? \answerBlank[3cm]
\answerExplain{$\$6.90$}{Divide: $27.60 \div 4 = 6.90$. Each friend gets $\$6.90$.}

\prob The temperature is $-5.5°$C and rises $8.3°$C. What is the new temperature? \answerBlank[3cm]
\answerExplain{$2.8°$C}{Add: $-5.5 + 8.3 = 2.8°$C. Different signs, subtract: $8.3 - 5.5 = 2.8$. Positive because $8.3$ is larger.}

\end{practiceBox}

% ── Practice A: Money Problems ───────────────────────────────────────────────
\begin{practiceBox}{Money and Finance}

\wordProblem{Your bank account has $\$230.50$. You withdraw $\$85.25$, deposit $\$42.00$, and pay a $\$4.75$ fee. What is your new balance?}{\$}
\answerExplain{$\$182.50$}{$230.50 - 85.25 + 42.00 - 4.75 = 145.25 + 42.00 - 4.75 = 187.25 - 4.75 = 182.50$.}

\wordProblem{A store marks up a product by $\frac{1}{4}$ of the original $\$48$ price, then offers a discount of $\$5.50$. What is the final price?}{\$}
\answerExplain{$\$54.50$}{Markup: $\frac{1}{4} \times 48 = 12$. After markup: $48 + 12 = 60$. After discount: $60 - 5.50 = 54.50$.}

\wordProblem{Over $5$ months, a company's stock changed by $+\$3.20$, $-\$1.50$, $-\$4.80$, $+\$2.10$, and $+\$0.60$. What was the overall change?}{\$}
\answerExplain{$-\$0.40$}{Sum all changes: $3.20 + (-1.50) + (-4.80) + 2.10 + 0.60$. Positives: $3.20 + 2.10 + 0.60 = 5.90$. Negatives: $(-1.50) + (-4.80) = -6.30$. Total: $5.90 + (-6.30) = -0.40$.}

\end{practiceBox}

% ── Practice B: Temperature and Science ──────────────────────────────────────
\begin{practiceBox}[funTeal]{Temperature and Science}

\wordProblem{At midnight the temperature is $-8.4°$C. It drops $3.6°$C by 4 AM, then rises $12.5°$C by noon. What is the noon temperature?}{°C}
\answerExplain{$0.5°$C}{$-8.4 + (-3.6) = -12$. Then $-12 + 12.5 = 0.5°$C.}

\wordProblem{A weather balloon descends at $-2\frac{1}{2}$ meters per second. How far has it descended after $6$ seconds?}{meters}
\answerExplain{$-15$ meters}{Multiply: $(-2\frac{1}{2}) \times 6 = (-\frac{5}{2}) \times 6 = -\frac{30}{2} = -15$ meters.}

\wordProblem{A canyon floor is $-185.5$ m below the rim. A hiker has descended $\frac{2}{5}$ of the way. What is the hiker's position relative to the rim?}{m}
\answerExplain{$-74.2$ m}{$\frac{2}{5} \times (-185.5) = \frac{-371}{5} = -74.2$. The hiker is $74.2$ m below the rim.}

\end{practiceBox}

% ── Practice C: Multi-Step Expressions ───────────────────────────────────────
\begin{practiceBox}[funOrange]{Multi-Step Expressions}

\practiceHeader[funOrange]{Evaluate each expression. Follow the order of operations.}

\prob $(-3) \times 4 + 10 \div (-2)$ \answerBlank[3cm]
\answerExplain{$-17$}{Multiply first: $(-3) \times 4 = -12$. Divide: $10 \div (-2) = -5$. Add: $-12 + (-5) = -17$.}

\prob $-\dfrac{1}{2} \times 8 + \dfrac{3}{4} \div \left(-\dfrac{1}{2}\right)$ \answerBlank[3cm]
\answerExplain{$-5\dfrac{1}{2}$}{Multiply: $-\frac{1}{2} \times 8 = -4$. Divide: $\frac{3}{4} \times (-2) = -\frac{3}{2}$. Add: $-4 + (-\frac{3}{2}) = -\frac{8}{2} + (-\frac{3}{2}) = -\frac{11}{2} = -5\frac{1}{2}$.}

\prob $(-2.4) \times 5 - (-3.6) \div 0.9$ \answerBlank[3cm]
\answerExplain{$-8$}{Multiply: $(-2.4) \times 5 = -12$. Divide: $(-3.6) \div 0.9 = -4$. Subtract: $-12 - (-4) = -12 + 4 = -8$.}

\prob $\dfrac{(-6) + 10}{(-2)}$ \answerBlank[2cm]
\answerExplain{$-2$}{Numerator: $(-6) + 10 = 4$. Divide: $4 \div (-2) = -2$.}

\end{practiceBox}

% ── Practice D: Mixed Fractions and Decimals ─────────────────────────────────
\begin{practiceBox}[funPink]{Mixing Fractions and Decimals}

\prob A recipe calls for $\frac{3}{4}$ cup of oil and $1.5$ cups of water. How much liquid in total? \answerBlank[3cm]
\answerExplain{$2\frac{1}{4}$ or $2.25$ cups}{Convert: $\frac{3}{4} = 0.75$. Add: $0.75 + 1.5 = 2.25 = 2\frac{1}{4}$ cups.}

\prob You walk $1\frac{1}{3}$ miles, then ride $2.75$ miles. How far did you travel? \answerBlank[3cm]
\answerExplain{$4\frac{1}{12}$ miles (or approximately $4.08$ mi)}{Convert: $1\frac{1}{3} = \frac{4}{3}$ and $2.75 = \frac{11}{4}$. Common denominator $12$: $\frac{16}{12} + \frac{33}{12} = \frac{49}{12} = 4\frac{1}{12}$.}

\prob Evaluate: $-0.5 + \dfrac{3}{8} - 1.25$ \answerBlank[3cm]
\answerExplain{$-1\dfrac{3}{8}$ or $-1.375$}{Convert all to decimals: $-0.5 + 0.375 - 1.25 = -0.125 - 1.25 = -1.375 = -1\frac{3}{8}$.}

\end{practiceBox}

% ── Practice E: True or False? ───────────────────────────────────────────────
\begin{practiceBox}[funPurple]{True or False?}

\trueOrFalse{When a problem says ``the temperature dropped $5°$,'' you should use $+5$.}
\answerExplain{False}{A drop means a decrease, which is negative. You should use $-5$. ``Dropped,'' ``fell,'' and ``lost'' all indicate negative values.}

\trueOrFalse{To check if an answer is reasonable, you can estimate with rounded numbers.}
\answerExplain{True}{Estimation is a great way to check. For example, $-8.4 + 12.5 \approx -8 + 13 = 5$, which is close to the exact answer of $4.1$.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}

\prob A scuba diver starts at sea level ($0$ m). She descends $\frac{3}{4}$ m every second for $8$ seconds, rests, then ascends $0.25$ m per second for $12$ seconds. What is her final position? \answerBlank[4cm]
\answerExplain{$-3$ m}{Descent: $-\frac{3}{4} \times 8 = -6$ m. Ascent: $0.25 \times 12 = 3$ m. Final: $-6 + 3 = -3$ m (still $3$ m below sea level).}

\prob A football team starts at the $20$-yard line. On three plays, they gain $4\frac{1}{2}$ yards, lose $2.75$ yards, and gain $8\frac{1}{4}$ yards. Where are they now? \answerBlank[4cm]
\answerExplain{$30$-yard line}{Change: $4.5 + (-2.75) + 8.25 = 10$ yards gained. New position: $20 + 10 = 30$-yard line.}

\end{challengeBox}
