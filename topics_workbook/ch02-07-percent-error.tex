\section{Percent Error: How Close Are Your Estimates?}
\topicTitle{Percent Error: How Close Are Your Estimates?}

\begin{quickReview}{Percent Error}
\textbf{Percent error} measures how far off an estimate or measurement is from the actual value.
$$\text{percent error} = \frac{|\text{estimated} - \text{actual}|}{\text{actual}} \times 100$$

\begin{itemize}[leftmargin=*]
    \item The absolute value $| \; |$ makes the answer always positive --- it does not matter if you guessed too high or too low.
    \item A \textbf{smaller} percent error means a \textbf{better} estimate.
    \item A percent error of $0\%$ means the estimate was exactly correct.
\end{itemize}

\medskip
\textbf{Example:} You estimated $50$ marbles; the actual count is $40$. Percent error $= \frac{|50 - 40|}{40} \times 100 = \frac{10}{40} \times 100 = 25\%$.
\end{quickReview}

\begin{multicols}{2}
\prob Estimated: $30$, Actual: $25$ \answerBlank[3cm]

\prob Estimated: $18$, Actual: $20$ \answerBlank[3cm]

\prob Estimated: $100$, Actual: $100$ \answerBlank[3cm]
\answerExplain{$0$}{$|100 - 100| = 0$. The estimate was exactly correct.}

\prob Estimated: $45$, Actual: $50$ \answerBlank[3cm]
\answerExplain{$5$}{$|45 - 50| = |-5| = 5$. The absolute value removes the negative sign.}

\prob Estimated: $7.5$, Actual: $6$ \answerBlank[3cm]
\answerExplain{$1.5$}{$|7.5 - 6| = |1.5| = 1.5$.}

\prob Estimated: $200$, Actual: $250$ \answerBlank[3cm]
\answerExplain{$50$}{$|200 - 250| = |-50| = 50$.}
\end{multicols}

% ── Practice A: Computing Percent Error ──────────────────────────────────────
\begin{practiceBox}{Computing Percent Error}

\practiceHeader{Use the formula to find the percent error. Round to the nearest tenth if needed.}

\prob Estimated: $30$, Actual: $25$. \answerBlank[3cm]
\answerExplain{$20\%$}{Difference: $|30 - 25| = 5$. Divide by actual: $5 \div 25 = 0.20$. Multiply by $100$: $20\%$.}

\prob Estimated: $60$, Actual: $50$. \answerBlank[3cm]
\answerExplain{$20\%$}{Difference: $|60 - 50| = 10$. Divide by actual: $10 \div 50 = 0.20$. Multiply by $100$: $20\%$.}

\prob Estimated: $15$, Actual: $18$. \answerBlank[3cm]
\answerExplain{$\approx 16.7\%$}{Difference: $|15 - 18| = 3$. Divide by actual: $3 \div 18 = 0.1\overline{6}$. Multiply by $100$: about $16.7\%$.}

\prob Estimated: $92$, Actual: $80$. \answerBlank[3cm]
\answerExplain{$15\%$}{Difference: $|92 - 80| = 12$. Divide by actual: $12 \div 80 = 0.15$. Multiply by $100$: $15\%$.}

\prob Estimated: $3.5$ kg, Actual: $4$ kg. \answerBlank[3cm]
\answerExplain{$12.5\%$}{Difference: $|3.5 - 4| = 0.5$. Divide by actual: $0.5 \div 4 = 0.125$. Multiply by $100$: $12.5\%$.}

\prob Estimated: $250$, Actual: $240$. \answerBlank[3cm]
\answerExplain{$\approx 4.2\%$}{Difference: $|250 - 240| = 10$. Divide by actual: $10 \div 240 = 0.041\overline{6}$. Multiply by $100$: about $4.2\%$.}

\end{practiceBox}

% ── Practice B: True or False ────────────────────────────────────────────────
\begin{practiceBox}[funPurple]{True or False?}

\trueOrFalse{Percent error can be a negative number.}
\answerExplain{False}{The absolute value in the formula ensures percent error is always zero or positive. It does not matter whether you overestimated or underestimated.}

\trueOrFalse{If your estimate is exactly correct, the percent error is $0\%$.}
\answerExplain{True}{If estimated $=$ actual, then $|\text{estimated} - \text{actual}| = 0$, and $0 \div \text{actual} \times 100 = 0\%$.}

\trueOrFalse{An estimate of $55$ for an actual value of $50$ has the same percent error as an estimate of $45$ for an actual value of $50$.}
\answerExplain{True}{Both have $|\text{difference}| = 5$. Percent error $= 5 \div 50 \times 100 = 10\%$ in both cases. Overestimating and underestimating by the same amount give the same percent error.}

\trueOrFalse{Percent error is always less than $100\%$.}
\answerExplain{False}{Percent error can exceed $100\%$. For example, estimating $30$ when the actual is $10$: $|30 - 10| \div 10 \times 100 = 200\%$.}

\end{practiceBox}

% ── Practice C: Word Problems ────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{A student estimated a classroom is $35$ feet long. The actual length is $30$ feet. What is the percent error?}{}
\answerExplain{$\approx 16.7\%$}{Difference: $|35 - 30| = 5$. Divide by actual: $5 \div 30 \approx 0.167$. Multiply by $100$: about $16.7\%$.}

\wordProblem{A recipe calls for $2$ cups of sugar. You accidentally used $2.3$ cups. What is the percent error?}{}
\answerExplain{$15\%$}{Difference: $|2.3 - 2| = 0.3$. Divide by actual: $0.3 \div 2 = 0.15$. Multiply by $100$: $15\%$ error.}

\wordProblem{A scientist predicted an experiment would produce $120$ grams of a substance. The actual yield was $108$ grams. Find the percent error.}{}
\answerExplain{$\approx 11.1\%$}{Difference: $|120 - 108| = 12$. Divide by actual: $12 \div 108 \approx 0.111$. Multiply by $100$: about $11.1\%$.}

\wordProblem{A builder estimated a project would cost $\$8{,}000$. The actual cost was $\$7{,}500$. Find the percent error.}{}
\answerExplain{$\approx 6.7\%$}{Difference: $|8{,}000 - 7{,}500| = 500$. Divide by actual: $500 \div 7{,}500 \approx 0.0667$. Multiply by $100$: about $6.7\%$.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob Two students estimated the number of seats in an auditorium. Student A guessed $450$ and Student B guessed $520$. The actual number is $500$. Who had the smaller percent error? \answerBlank[3cm]
\answerExplain{Student B ($4\%$ vs.\ $10\%$)}{Student A: $|450 - 500| \div 500 \times 100 = 50 \div 500 \times 100 = 10\%$. Student B: $|520 - 500| \div 500 \times 100 = 20 \div 500 \times 100 = 4\%$. Student B had the smaller percent error.}

\prob A measurement has a percent error of $8\%$. The actual value is $75$ cm. What are the two possible estimated values (one too high, one too low)? \answerBlank[4cm]
\answerExplain{$81$ cm or $69$ cm}{$8\%$ of $75 = 0.08 \times 75 = 6$. Too high: $75 + 6 = 81$ cm. Too low: $75 - 6 = 69$ cm. Both give $|6| \div 75 \times 100 = 8\%$.}
\end{challengeBox}
