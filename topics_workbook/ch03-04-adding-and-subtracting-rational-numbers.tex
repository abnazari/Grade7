\section{Adding and Subtracting Rational Numbers}
\topicTitle{Adding and Subtracting Rational Numbers}

\begin{quickReview}{Adding and Subtracting Rational Numbers}
The same sign rules from integers apply to \textbf{all rational numbers} — fractions and decimals.

\begin{itemize}[leftmargin=*]
    \item \textbf{Same signs:} Add absolute values, keep the sign.
    \item \textbf{Different signs:} Subtract absolute values, keep the sign of the larger.
    \item For \textbf{fractions}, find a common denominator first, then apply the sign rules.
    \item Subtraction $=$ adding the opposite: $a - b = a + (-b)$.
\end{itemize}

\medskip
\textbf{Example:} $-\dfrac{2}{3} + \dfrac{1}{4} = -\dfrac{8}{12} + \dfrac{3}{12} = -\dfrac{5}{12}$ \quad (different signs, larger absolute value is negative).
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Add or subtract these decimals.}

\begin{multicols}{2}
\prob $3.5 + (-1.2)$ \answerBlank[2cm]
\answerExplain{$2.3$}{Different signs, subtract: $3.5 - 1.2 = 2.3$. Positive because $3.5$ has the larger absolute value.}

\prob $(-4.7) + (-2.3)$ \answerBlank[2cm]
\answerExplain{$-7$}{Same signs (both negative), add: $4.7 + 2.3 = 7$. Keep negative: $-7$.}

\prob $(-6.1) + 6.1$ \answerBlank[2cm]
\answerExplain{$0$}{These are opposites, so their sum is $0$.}

\prob $5.4 - 8.9$ \answerBlank[2cm]
\answerExplain{$-3.5$}{Rewrite: $5.4 + (-8.9)$. Different signs, subtract: $8.9 - 5.4 = 3.5$. Negative because $8.9$ is larger: $-3.5$.}

\prob $(-2.6) - (-4.1)$ \answerBlank[2cm]
\answerExplain{$1.5$}{Rewrite: $(-2.6) + 4.1$. Different signs, subtract: $4.1 - 2.6 = 1.5$. Positive because $4.1$ is larger.}

\prob $1.8 + (-3.3)$ \answerBlank[2cm]
\answerExplain{$-1.5$}{Different signs, subtract: $3.3 - 1.8 = 1.5$. Negative because $3.3$ is larger: $-1.5$.}
\end{multicols}
\end{practiceBox}

% ── Practice A: Signed Fractions ─────────────────────────────────────────────
\begin{practiceBox}{Adding and Subtracting Signed Fractions}

\practiceHeader{Find each sum or difference. Simplify your answer.}

\prob $-\dfrac{1}{4} + \dfrac{3}{4}$ \answerBlank[2cm]
\answerExplain{$\dfrac{1}{2}$}{Same denominator. Different signs, subtract numerators: $3 - 1 = 2$. Positive because $3 > 1$. Result: $\frac{2}{4} = \frac{1}{2}$.}

\prob $-\dfrac{5}{6} + \left(-\dfrac{1}{6}\right)$ \answerBlank[2cm]
\answerExplain{$-1$}{Same denominator, same signs. Add numerators: $5 + 1 = 6$. Keep negative: $-\frac{6}{6} = -1$.}

\prob $\dfrac{2}{3} - \dfrac{5}{6}$ \answerBlank[2cm]
\answerExplain{$-\dfrac{1}{6}$}{Rewrite: $\frac{2}{3} + (-\frac{5}{6})$. Common denominator $6$: $\frac{4}{6} + (-\frac{5}{6})$. Different signs, subtract: $5 - 4 = 1$. Negative because $5 > 4$: $-\frac{1}{6}$.}

\prob $-\dfrac{3}{5} + \dfrac{7}{10}$ \answerBlank[2cm]
\answerExplain{$\dfrac{1}{10}$}{Common denominator $10$: $-\frac{6}{10} + \frac{7}{10}$. Different signs, subtract: $7 - 6 = 1$. Positive because $7 > 6$: $\frac{1}{10}$.}

\prob $-\dfrac{1}{2} - \dfrac{3}{8}$ \answerBlank[2cm]
\answerExplain{$-\dfrac{7}{8}$}{Rewrite: $-\frac{1}{2} + (-\frac{3}{8})$. Common denominator $8$: $-\frac{4}{8} + (-\frac{3}{8})$. Same signs, add: $4 + 3 = 7$. Keep negative: $-\frac{7}{8}$.}

\prob $\dfrac{5}{8} - \left(-\dfrac{1}{4}\right)$ \answerBlank[2cm]
\answerExplain{$\dfrac{7}{8}$}{Rewrite: $\frac{5}{8} + \frac{1}{4}$. Common denominator $8$: $\frac{5}{8} + \frac{2}{8} = \frac{7}{8}$.}

\end{practiceBox}

% ── Practice B: Mixed Decimals and Fractions ─────────────────────────────────
\begin{practiceBox}[funTeal]{Mixed Practice — Decimals and Fractions}

\prob $-7.25 + 3.5$ \answerBlank[2cm]
\answerExplain{$-3.75$}{Different signs, subtract: $7.25 - 3.5 = 3.75$. Negative because $7.25$ is larger: $-3.75$.}

\prob $4.6 - (-2.8)$ \answerBlank[2cm]
\answerExplain{$7.4$}{Rewrite: $4.6 + 2.8 = 7.4$. Subtracting a negative adds the positive.}

\prob $-\dfrac{2}{3} - \dfrac{4}{9}$ \answerBlank[2cm]
\answerExplain{$-1\dfrac{1}{9}$}{Rewrite: $-\frac{2}{3} + (-\frac{4}{9})$. Common denominator $9$: $-\frac{6}{9} + (-\frac{4}{9})$. Same signs, add: $6 + 4 = 10$. Keep negative: $-\frac{10}{9} = -1\frac{1}{9}$.}

\prob $-1.5 + 0.75 + (-2.25)$ \answerBlank[2cm]
\answerExplain{$-3$}{Step by step: $-1.5 + 0.75 = -0.75$. Then $-0.75 + (-2.25) = -3$. Both negative, add: $0.75 + 2.25 = 3$.}

\end{practiceBox}

% ── Practice C: True or False? ───────────────────────────────────────────────
\begin{practiceBox}[funPurple]{True or False?}

\trueOrFalse{$-0.5 + 0.5 = 1$}
\answerExplain{False}{$-0.5$ and $0.5$ are opposites, so their sum is $0$, not $1$.}

\trueOrFalse{When adding two fractions with different signs, you always need a common denominator first.}
\answerExplain{True}{You must have a common denominator to compare and combine the numerators, just like with positive fractions. Then apply the sign rules.}

\trueOrFalse{$\dfrac{3}{4} - 1 = -\dfrac{1}{4}$}
\answerExplain{True}{$\frac{3}{4} - 1 = \frac{3}{4} - \frac{4}{4} = -\frac{1}{4}$. Since $1 > \frac{3}{4}$, the result is negative.}

\end{practiceBox}

% ── Practice D: Word Problems ────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{A diver is at $-12.5$ m below sea level. She swims up $4.8$ m, then back down $3.2$ m. What is her final depth?}{m}
\answerExplain{$-10.9$ m}{Start at $-12.5$. Swim up: $-12.5 + 4.8 = -7.7$. Swim down: $-7.7 + (-3.2) = -10.9$ m.}

\wordProblem{A recipe needs $2\frac{1}{3}$ cups of liquid. You already added $\frac{3}{4}$ cup. How much more liquid do you need?}{cups}
\answerExplain{$1\frac{7}{12}$ cups}{Subtract: $2\frac{1}{3} - \frac{3}{4} = \frac{7}{3} - \frac{3}{4}$. Common denominator $12$: $\frac{28}{12} - \frac{9}{12} = \frac{19}{12} = 1\frac{7}{12}$ cups.}

\wordProblem{The stock price of a company changed by $-\$2.75$ on Monday and $+\$1.50$ on Tuesday. What was the overall change?}{\$}
\answerExplain{$-\$1.25$}{Add the changes: $(-2.75) + 1.50 = -1.25$. Different signs, subtract: $2.75 - 1.50 = 1.25$. Negative because $2.75$ is larger.}

\end{practiceBox}

\begin{challengeBox}

\prob Evaluate: $-\dfrac{3}{4} + \dfrac{5}{6} - \dfrac{1}{3} + \dfrac{1}{12}$ \answerBlank[3cm]
\answerExplain{$-\dfrac{1}{6}$}{Common denominator $12$: $-\frac{9}{12} + \frac{10}{12} - \frac{4}{12} + \frac{1}{12}$. Combine: $(-9 + 10 - 4 + 1)/12 = -2/12 = -\frac{1}{6}$.}

\prob The temperature at midnight is $-4.8°$C. It drops $2\frac{1}{2}°$C by 3 AM, then rises $7.3°$C by noon. What is the noon temperature? \answerBlank[3cm]
\answerExplain{$0°$C}{Midnight: $-4.8$. Drop: $-4.8 + (-2.5) = -7.3$. Rise: $-7.3 + 7.3 = 0°$C.}

\end{challengeBox}
