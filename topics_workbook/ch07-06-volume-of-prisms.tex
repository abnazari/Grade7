\section{Volume of Prisms}
\topicTitle{Volume of Prisms}

\begin{quickReview}{Volume of Prisms}
\textbf{Volume} measures the space inside a 3D object, in \textbf{cubic units} (cm$^3$, m$^3$, in$^3$, etc.).

\begin{itemize}[leftmargin=*]
    \item \textbf{General prism:} $V = Bh$, where $B$ = area of the base and $h$ = height of the prism.
    \item \textbf{Rectangular prism:} $V = lwh$.
    \item \textbf{Cube:} $V = s^3$.
    \item \textbf{Triangular prism:} $B = \frac{1}{2} \times \text{base} \times \text{height of triangle}$, then $V = Bh$.
\end{itemize}

\medskip
\textbf{Example:} A box $6 \times 4 \times 3$ cm: $V = 6 \times 4 \times 3 = 72$ cm$^3$.
\end{quickReview}

% ── Practice C: Working Backwards ────────────────────────────────────────────
\begin{practiceBox}[funOrange]{Find the Missing Dimension}

\practiceHeader[funOrange]{Use $V = lwh$ to find the unknown.}

\prob A rectangular prism has $V = 180$ cm$^3$, $l = 9$ cm, $w = 4$ cm. Find $h$. \answerBlank[3cm]
\answerExplain{$5$ cm}{$h = V \div (l \times w) = 180 \div (9 \times 4) = 180 \div 36 = 5$ cm.}

\prob A rectangular prism has $V = 360$ in$^3$, $l = 12$ in, $h = 6$ in. Find $w$. \answerBlank[3cm]
\answerExplain{$5$ in}{$w = V \div (l \times h) = 360 \div (12 \times 6) = 360 \div 72 = 5$ in.}

\prob A cube has volume $343$ m$^3$. What is the side length? \answerBlank[3cm]
\answerExplain{$7$ m}{$s^3 = 343$. Since $7 \times 7 \times 7 = 343$, the side length is $7$ m.}

\end{practiceBox}

% ── True or False ────────────────────────────────────────────────────────────
\begin{practiceBox}[funPurple]{True or False?}

\trueOrFalse{Volume is measured in square units.}
\answerExplain{False}{Volume measures the space inside a 3D object, so it uses \emph{cubic} units (cm$^3$, m$^3$, etc.). Square units are for area.}

\trueOrFalse{If you double the height of a rectangular prism but keep $l$ and $w$ the same, the volume doubles.}
\answerExplain{True}{$V = lwh$. If $h$ becomes $2h$, then $V = lw(2h) = 2lwh$, which is double the original volume.}

\trueOrFalse{A cube with side $4$ cm has the same volume as a rectangular prism $8 \times 4 \times 2$ cm.}
\answerExplain{True}{Cube: $4^3 = 64$ cm$^3$. Prism: $8 \times 4 \times 2 = 64$ cm$^3$. Both have the same volume.}

\end{practiceBox}

% ── Word Problems ────────────────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{A fish tank is $60$ cm long, $30$ cm wide, and $40$ cm tall. How many cubic centimeters of water can it hold?}{cm$^3$}
\answerExplain{$72{,}000$ cm$^3$}{$V = lwh = 60 \times 30 \times 40 = 72{,}000$ cm$^3$.}

\wordProblem{A sandbox shaped like a triangular prism has a triangular base with base $3$ m and height $2$ m. The sandbox is $4$ m long. How much sand does it hold?}{m$^3$}
\answerExplain{$12$ m$^3$}{$B = \frac{1}{2} \times 3 \times 2 = 3$ m$^2$. $V = 3 \times 4 = 12$ m$^3$.}

\wordProblem{A shipping container is $12$ ft long, $8$ ft wide, and $8$ ft tall. Each small box is a cube with side $2$ ft. How many small boxes fit inside?}{boxes}
\answerExplain{$96$ boxes}{Container: $V = 12 \times 8 \times 8 = 768$ ft$^3$. Box: $V = 2^3 = 8$ ft$^3$. Number of boxes: $768 \div 8 = 96$.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob A rectangular pool is $10$ m long, $6$ m wide, and $1.5$ m deep. Water fills $\frac{3}{4}$ of the pool. How many cubic meters of water are in the pool? \answerBlank[3cm]
\answerExplain{$67.5$ m$^3$}{Full volume: $10 \times 6 \times 1.5 = 90$ m$^3$. Three-quarters: $90 \times \frac{3}{4} = 67.5$ m$^3$.}

\prob Two rectangular prisms are placed side by side. The first is $5 \times 3 \times 4$ cm and the second is $5 \times 3 \times 6$ cm. What is their combined volume? \answerBlank[3cm]
\answerExplain{$150$ cm$^3$}{First: $5 \times 3 \times 4 = 60$ cm$^3$. Second: $5 \times 3 \times 6 = 90$ cm$^3$. Combined: $60 + 90 = 150$ cm$^3$.}
\end{challengeBox}
