\section{Area of Composite Shapes}
\topicTitle{Area of Composite Shapes}

\begin{quickReview}{Area of Composite Shapes}
A \textbf{composite shape} is made by combining (or removing) simpler shapes. To find its area:

\begin{itemize}[leftmargin=*]
    \item \textbf{Break} the figure into familiar shapes (rectangles, triangles, circles, semicircles, etc.).
    \item \textbf{Find} the area of each piece using the correct formula.
    \item \textbf{Add} the areas if pieces are joined. \textbf{Subtract} if a piece is cut out.
\end{itemize}

\medskip
\textbf{Key formulas:} Rectangle: $A = lw$. \quad Triangle: $A = \frac{1}{2}bh$. \quad Circle: $A = \pi r^2$.

\smallskip
\textbf{Example:} A rectangle $8 \times 4$ cm with a semicircle (radius $4$ cm) on top:
Rectangle $= 32$ cm$^2$. Semicircle $= \frac{1}{2}(3.14)(16) = 25.12$ cm$^2$. Total $= 57.12$ cm$^2$.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Find Each Area (review of basic formulas)}

\begin{multicols}{2}
\prob Rectangle: $l = 10$ cm, $w = 4$ cm \answerBlank[3cm]
\answerExplain{$40$ cm$^2$}{Area of a rectangle: $A = lw = 10 \times 4 = 40$ cm$^2$.}

\prob Triangle: $b = 6$ m, $h = 8$ m \answerBlank[3cm]
\answerExplain{$24$ m$^2$}{Area of a triangle: $A = \frac{1}{2}bh = \frac{1}{2} \times 6 \times 8 = 24$ m$^2$.}

\prob Square: side $= 5$ in \answerBlank[3cm]
\answerExplain{$25$ in$^2$}{Area of a square: $A = s^2 = 5^2 = 25$ in$^2$.}

\prob Circle: $r = 3$ ft (use $\pi \approx 3.14$) \answerBlank[3cm]
\answerExplain{$28.26$ ft$^2$}{Area of a circle: $A = \pi r^2 = 3.14 \times 9 = 28.26$ ft$^2$.}

\prob Semicircle: $r = 4$ cm (use $\pi \approx 3.14$) \answerBlank[3cm]
\answerExplain{$25.12$ cm$^2$}{Semicircle is half a circle: $A = \frac{1}{2}\pi r^2 = \frac{1}{2} \times 3.14 \times 16 = 25.12$ cm$^2$.}

\prob Parallelogram: $b = 9$ m, $h = 3$ m \answerBlank[3cm]
\answerExplain{$27$ m$^2$}{Area of a parallelogram: $A = bh = 9 \times 3 = 27$ m$^2$.}
\end{multicols}
\end{practiceBox}

% ── Practice A: Adding Areas ─────────────────────────────────────────────────
\begin{practiceBox}{Composite Shapes — Add the Areas}

\practiceHeader{Break each figure into simpler shapes, find each area, then add. Use $\pi \approx 3.14$.}

\prob A figure is made of a rectangle $12 \times 5$ cm with a triangle (base $12$ cm, height $4$ cm) on top. Find the total area. \answerBlank[3cm]
\answerExplain{$84$ cm$^2$}{Rectangle: $12 \times 5 = 60$ cm$^2$. Triangle: $\frac{1}{2} \times 12 \times 4 = 24$ cm$^2$. Total: $60 + 24 = 84$ cm$^2$.}

\prob A figure is a rectangle $10 \times 6$ m with a semicircle (diameter $6$ m) attached to one shorter side. Find the total area. \answerBlank[3cm]
\answerExplain{$74.13$ m$^2$}{Rectangle: $10 \times 6 = 60$ m$^2$. Semicircle: $r = 3$ m, $A = \frac{1}{2} \times 3.14 \times 9 = 14.13$ m$^2$. Total: $60 + 14.13 = 74.13$ m$^2$.}

\prob An L-shaped figure is made of two rectangles: one $8 \times 3$ in and one $4 \times 5$ in. Find the total area. \answerBlank[3cm]
\answerExplain{$44$ in$^2$}{First rectangle: $8 \times 3 = 24$ in$^2$. Second rectangle: $4 \times 5 = 20$ in$^2$. Total: $24 + 20 = 44$ in$^2$.}

\prob A figure is made of a square with side $6$ cm and an equilateral triangle with base $6$ cm and height $5.2$ cm on top. Find the total area. \answerBlank[3cm]
\answerExplain{$51.6$ cm$^2$}{Square: $6^2 = 36$ cm$^2$. Triangle: $\frac{1}{2} \times 6 \times 5.2 = 15.6$ cm$^2$. Total: $36 + 15.6 = 51.6$ cm$^2$.}

\prob A T-shaped figure is made of two rectangles: $10 \times 2$ m across the top and $3 \times 6$ m going down. Find the total area. \answerBlank[3cm]
\answerExplain{$38$ m$^2$}{Top rectangle: $10 \times 2 = 20$ m$^2$. Bottom rectangle: $3 \times 6 = 18$ m$^2$. Total: $20 + 18 = 38$ m$^2$.}

\end{practiceBox}

% ── Practice B: Subtracting Areas ────────────────────────────────────────────
\begin{practiceBox}[funTeal]{Composite Shapes — Subtract the Areas}

\practiceHeader[funTeal]{Find the shaded area by subtracting the cut-out shape. Use $\pi \approx 3.14$.}

\prob A square with side $10$ cm has a circle with radius $3$ cm cut out of the center. Find the remaining area. \answerBlank[3cm]
\answerExplain{$71.74$ cm$^2$}{Square: $10^2 = 100$ cm$^2$. Circle: $3.14 \times 9 = 28.26$ cm$^2$. Remaining: $100 - 28.26 = 71.74$ cm$^2$.}

\prob A rectangle $8 \times 6$ in has a triangle with base $4$ in and height $3$ in cut from one corner. Find the remaining area. \answerBlank[3cm]
\answerExplain{$42$ in$^2$}{Rectangle: $8 \times 6 = 48$ in$^2$. Triangle: $\frac{1}{2} \times 4 \times 3 = 6$ in$^2$. Remaining: $48 - 6 = 42$ in$^2$.}

\prob A circle with radius $5$ m has a square with side $4$ m cut from the center. Find the remaining area. \answerBlank[3cm]
\answerExplain{$62.5$ m$^2$}{Circle: $3.14 \times 25 = 78.5$ m$^2$. Square: $16$ m$^2$. Remaining: $78.5 - 16 = 62.5$ m$^2$.}

\prob A rectangle $14 \times 8$ ft has two semicircles with diameter $8$ ft cut from the shorter sides. Find the remaining area. \answerBlank[3cm]
\answerExplain{$61.76$ ft$^2$}{Rectangle: $14 \times 8 = 112$ ft$^2$. Two semicircles make one full circle with $r = 4$ ft: $3.14 \times 16 = 50.24$ ft$^2$. Remaining: $112 - 50.24 = 61.76$ ft$^2$.}

\end{practiceBox}

% ── Practice C: True or False ────────────────────────────────────────────────
\begin{practiceBox}[funPurple]{True or False?}

\trueOrFalse{To find the area of a composite shape, you always add the areas of each piece.}
\answerExplain{False}{Sometimes you subtract. If a piece is removed (cut out) from a larger shape, you subtract its area from the larger shape's area.}

\trueOrFalse{A quarter circle has an area equal to $\frac{1}{4}\pi r^2$.}
\answerExplain{True}{A quarter circle is one-fourth of a full circle, so its area is $\frac{1}{4}$ of $\pi r^2$.}

\trueOrFalse{Two rectangles $5 \times 3$ and $4 \times 2$ joined together have a combined area of $22$ square units.}
\answerExplain{False}{First rectangle: $5 \times 3 = 15$. Second: $4 \times 2 = 8$. Total: $15 + 8 = 23$ square units, not $22$.}

\end{practiceBox}

% ── Word Problems ────────────────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{A room is shaped like a rectangle $12 \times 8$ ft with a semicircular alcove (diameter $8$ ft) on one side. What is the total floor area? Use $\pi \approx 3.14$.}{ft$^2$}
\answerExplain{$121.12$ ft$^2$}{Rectangle: $12 \times 8 = 96$ ft$^2$. Semicircle: $r = 4$, $A = \frac{1}{2} \times 3.14 \times 16 = 25.12$ ft$^2$. Total: $96 + 25.12 = 121.12$ ft$^2$.}

\wordProblem{A rectangular yard $20 \times 15$ m has a circular pool with radius $3$ m in the middle. How much grass (area) is around the pool? Use $\pi \approx 3.14$.}{m$^2$}
\answerExplain{$271.74$ m$^2$}{Yard: $20 \times 15 = 300$ m$^2$. Pool: $3.14 \times 9 = 28.26$ m$^2$. Grass: $300 - 28.26 = 271.74$ m$^2$.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob A figure consists of a rectangle $10 \times 6$ cm with a semicircle on each shorter side (diameter $6$ cm). What is the total area? Use $\pi \approx 3.14$. \answerBlank[3cm]
\answerExplain{$88.26$ cm$^2$}{Rectangle: $10 \times 6 = 60$ cm$^2$. Two semicircles make one full circle with $r = 3$: $3.14 \times 9 = 28.26$ cm$^2$. Total: $60 + 28.26 = 88.26$ cm$^2$.}

\prob A circular piece of cardboard has a radius of $8$ in. A rectangle $6 \times 4$ in is cut from the center. What area of cardboard remains? Use $\pi \approx 3.14$. \answerBlank[3cm]
\answerExplain{$176.96$ in$^2$}{Circle: $3.14 \times 64 = 200.96$ in$^2$. Rectangle: $6 \times 4 = 24$ in$^2$. Remaining: $200.96 - 24 = 176.96$ in$^2$.}
\end{challengeBox}
