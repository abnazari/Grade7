\section{Populations and Samples}
\topicTitle{Populations and Samples}

\begin{quickReview}{Populations and Samples}
\textbf{Population}: The \emph{entire} group you want to learn about.
\textbf{Sample}: A \emph{smaller} group chosen from the population.

\begin{itemize}[leftmargin=*]
    \item A sample must be \textbf{representative} — it should reflect the characteristics of the whole population.
    \item \textbf{Random sampling} means every member of the population has an equal chance of being selected. This helps avoid \textbf{bias}.
    \item A \textbf{biased sample} favors certain outcomes and can lead to incorrect conclusions.
\end{itemize}

\medskip
\textbf{Example:} A school has $800$ students. To learn about favorite sports, you randomly survey $60$ students.
Population $=$ all $800$ students. Sample $=$ the $60$ surveyed.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Identify the Population and Sample}

\prob A city has $50{,}000$ residents. A researcher surveys $500$ residents about transportation. What is the population? \answerBlank[4cm]
\answerExplain{All $50{,}000$ residents}{The population is the entire group the researcher wants to learn about — all residents of the city.}

\prob In Problem 1, what is the sample? \answerBlank[4cm]
\answerExplain{The $500$ surveyed residents}{The sample is the smaller group actually surveyed.}

\prob A factory makes $10{,}000$ light bulbs per day. An inspector tests $200$ bulbs. What is the population? \answerBlank[4cm]
\answerExplain{All $10{,}000$ light bulbs}{The population is the entire group of interest — all bulbs produced that day.}

\prob A teacher wants to know how much time her $120$ students spend on homework. She asks $25$ students. What is the sample? \answerBlank[4cm]
\answerExplain{The $25$ students asked}{The sample is the smaller group chosen from the population.}

\prob A veterinarian wants to study the weight of dogs in a town with $3{,}000$ dogs. She weighs $150$ dogs at random. What is the population? \answerBlank[4cm]
\answerExplain{All $3{,}000$ dogs in the town}{The population is the entire group the vet wants to learn about.}

\end{practiceBox}

% ── Practice A: Random vs. Biased Samples ────────────────────────────────────
\begin{practiceBox}{Random vs.\ Biased Samples}

\practiceHeader{Is the sampling method random or biased? Explain.}

\prob A principal wants to know students' opinions about school lunch. She surveys every student whose last name starts with the letter M.  \answerBlank[3cm]
\answerExplain{Biased}{Selecting by last name does not give every student an equal chance. Students with certain last names may share cultural backgrounds with similar food preferences.}

\prob A store manager puts all customer receipts into a box and draws $50$ receipts without looking. \answerBlank[3cm]
\answerExplain{Random}{Every receipt has an equal chance of being drawn, making this a random sample.}

\prob A coach surveys only the basketball team to find out students' favorite sport. \answerBlank[3cm]
\answerExplain{Biased}{Basketball players are more likely to choose basketball. The sample does not represent all students.}

\prob A researcher assigns every household a number and uses a computer to randomly select $200$ households. \answerBlank[3cm]
\answerExplain{Random}{Every household has an equal chance of being selected. This is a proper random sample.}

\prob A news website surveys only visitors to its sports page about the most important news topic. \answerBlank[3cm]
\answerExplain{Biased}{Sports-page visitors are likely to say sports, so the sample does not represent all website visitors.}

\prob A librarian picks every $10$th person who enters the library to ask about reading habits. \answerBlank[3cm]
\answerExplain{Random (systematic)}{Systematic sampling gives each visitor an approximately equal chance of being selected, so it is considered a type of random sampling.}

\end{practiceBox}

% ── Practice B: True or False ────────────────────────────────────────────────
\begin{practiceBox}[funPurple]{True or False?}

\trueOrFalse{A sample is always smaller than the population.}
\answerExplain{True}{A sample is a subset of the population, so it contains fewer members than the entire population.}

\trueOrFalse{If you survey only your friends, your sample is random.}
\answerExplain{False}{Your friends are not randomly chosen from the population. This is a convenience sample, which is biased because your friends likely share similar opinions.}

\trueOrFalse{A larger random sample generally gives a more reliable estimate than a smaller one.}
\answerExplain{True}{Larger samples are more likely to represent the population accurately. They reduce the effect of random variation.}

\trueOrFalse{A biased sample can still give accurate results about a population.}
\answerExplain{False}{While a biased sample might occasionally match the population, it systematically favors certain outcomes, making its conclusions unreliable.}

\end{practiceBox}

% ── Practice C: Design a Sampling Method ─────────────────────────────────────
\begin{practiceBox}[funTeal]{Design a Sampling Method}

\practiceHeader[funTeal]{Describe a good random sampling method for each situation.}

\prob A school with $600$ students wants to know how many students would join an after-school coding club.
\answerExplain{Assign each student a number $1$--$600$ and use a random number generator to select $60$ students to survey}{This ensures every student has an equal chance of being picked. Surveying about $10\%$ of the school gives a good estimate.}

\prob A farmer has $2{,}000$ apple trees and wants to estimate the average number of apples per tree.
\answerExplain{Number all trees and randomly select $100$ trees to count}{Random selection avoids bias from choosing only the healthiest-looking or most accessible trees.}

\prob A city planner wants to know residents' opinions about a new park. The city has $5$ neighborhoods.
\answerExplain{Randomly select the same number of residents from each neighborhood}{This stratified random sampling ensures all neighborhoods are represented equally in the sample.}

\end{practiceBox}

% ── Word Problems ────────────────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{A company has $4{,}000$ employees. They randomly survey $200$ employees and find that $60\%$ prefer working from home. Estimate how many of all $4{,}000$ employees prefer working from home.}{employees}
\answerExplain{$2{,}400$ employees}{If $60\%$ of the sample prefers working from home, we estimate $60\%$ of the population does too: $0.60 \times 4{,}000 = 2{,}400$.}

\wordProblem{A pet store owner surveys $50$ random customers and finds that $18$ own a cat. If the store has $800$ customers, about how many own a cat?}{customers}
\answerExplain{$288$ customers}{The fraction of cat owners in the sample is $\frac{18}{50} = 0.36 = 36\%$. Estimate for the population: $0.36 \times 800 = 288$.}

\wordProblem{A school surveys $80$ randomly selected students and finds that $35$ walk to school. If the school has $640$ students, estimate the number who walk.}{students}
\answerExplain{$280$ students}{The fraction who walk is $\frac{35}{80} = 0.4375$. Estimate: $0.4375 \times 640 = 280$ students.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob A town has $12{,}000$ voters. Pollster A surveys $100$ people at a shopping mall and finds $55\%$ support a new law. Pollster B randomly calls $100$ voters and finds $42\%$ support it. Which result is more trustworthy, and why? \answerBlank[3cm]
\answerExplain{Pollster B's result}{Pollster B used random sampling, which gives every voter an equal chance of being selected. Pollster A's mall sample is biased — it only includes people who go to that mall, who may not represent all voters.}

\prob Explain why surveying $500$ randomly chosen people is generally more reliable than surveying $500$ volunteers who sign up online.
\answerExplain{Random selection avoids self-selection bias}{Volunteers choose to participate, which means the sample may over-represent people with strong opinions. Random selection ensures every person has an equal chance regardless of their views.}
\end{challengeBox}
