\section{Subtracting Integers}
\topicTitle{Subtracting Integers}

\begin{quickReview}{Subtracting Integers}
To subtract an integer, \textbf{add its opposite} (Keep--Change--Flip):
$$p - q = p + (-q)$$

\begin{itemize}[leftmargin=*]
    \item \textbf{Keep} the first number. \textbf{Change} subtraction to addition. \textbf{Flip} the sign of the second number.
    \item $8 - 13 = 8 + (-13) = -5$
    \item $(-6) - (-4) = (-6) + 4 = -2$
    \item The \textbf{distance} between two numbers is $|a - b|$, always $\geq 0$.
\end{itemize}

\medskip
\textbf{Example:} $(-3) - (-10) = (-3) + 10 = 7$.
\end{quickReview}

\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Rewrite as addition, then solve.}

\begin{multicols}{2}
\prob $9 - 4 = $ \answerBlank[2cm]
\answerExplain{$5$}{Rewrite: $9 + (-4)$. Different signs, subtract: $9 - 4 = 5$. Positive because $9$ has the larger absolute value.}

\prob $3 - 7 = $ \answerBlank[2cm]
\answerExplain{$-4$}{Rewrite: $3 + (-7)$. Different signs, subtract: $7 - 3 = 4$. Negative because $7$ has the larger absolute value.}

\prob $(-5) - 2 = $ \answerBlank[2cm]
\answerExplain{$-7$}{Rewrite: $(-5) + (-2)$. Same signs (both negative), add: $5 + 2 = 7$. Keep negative: $-7$.}

\prob $(-8) - (-3) = $ \answerBlank[2cm]
\answerExplain{$-5$}{Rewrite: $(-8) + 3$. Different signs, subtract: $8 - 3 = 5$. Negative because $8$ has the larger absolute value: $-5$.}

\prob $0 - (-6) = $ \answerBlank[2cm]
\answerExplain{$6$}{Rewrite: $0 + 6 = 6$. Subtracting a negative is the same as adding its opposite.}

\prob $(-10) - (-10) = $ \answerBlank[2cm]
\answerExplain{$0$}{Rewrite: $(-10) + 10 = 0$. A number minus itself always equals $0$.}
\end{multicols}
\end{practiceBox}

% ── Practice B: Distance Between Integers ────────────────────────────────────
\begin{practiceBox}[funTeal]{Distance Between Two Integers}

\practiceHeader[funTeal]{Find the distance between each pair of integers.}

\prob $-8$ and $5$ \answerBlank[2cm]
\answerExplain{$13$ units}{Distance $= |{-8} - 5| = |{-13}| = 13$. From $-8$ to $0$ is $8$ units, and $0$ to $5$ is $5$ units: $8 + 5 = 13$.}

\prob $-3$ and $-11$ \answerBlank[2cm]
\answerExplain{$8$ units}{Distance $= |{-3} - (-11)| = |{-3} + 11| = |8| = 8$ units.}

\prob $14$ and $-6$ \answerBlank[2cm]
\answerExplain{$20$ units}{Distance $= |14 - (-6)| = |14 + 6| = |20| = 20$ units.}

\prob $0$ and $-17$ \answerBlank[2cm]
\answerExplain{$17$ units}{Distance $= |0 - (-17)| = |17| = 17$ units. The number $-17$ is $17$ units from $0$.}

\end{practiceBox}

% ── Practice C: Multi-Step ───────────────────────────────────────────────────
\begin{practiceBox}[funOrange]{Multi-Step Subtraction}

\practiceHeader[funOrange]{Simplify each expression.}

\prob $5 - 8 - (-3)$ \answerBlank[2cm]
\answerExplain{$0$}{Rewrite: $5 + (-8) + 3$. Add: $5 + 3 = 8$, then $8 + (-8) = 0$.}

\prob $(-4) - 7 - (-12)$ \answerBlank[2cm]
\answerExplain{$1$}{Rewrite: $(-4) + (-7) + 12$. Add negatives: $(-4) + (-7) = -11$. Then $-11 + 12 = 1$.}

\prob $(-2) - (-9) - 5 - (-1)$ \answerBlank[2cm]
\answerExplain{$3$}{Rewrite: $(-2) + 9 + (-5) + 1$. Positives: $9 + 1 = 10$. Negatives: $(-2) + (-5) = -7$. Combine: $10 + (-7) = 3$.}

\end{practiceBox}

% ── Practice D: True or False? ───────────────────────────────────────────────
\begin{practiceBox}[funPurple]{True or False?}

\trueOrFalse{$a - b$ always equals $b - a$.}
\answerExplain{False}{Subtraction is not commutative. For example, $5 - 3 = 2$ but $3 - 5 = -2$. In general, $a - b = -(b - a)$.}

\trueOrFalse{Subtracting a negative number is the same as adding a positive number.}
\answerExplain{True}{$a - (-b) = a + b$. Flipping the sign of a negative makes it positive, so you add.}

\trueOrFalse{The distance between $-4$ and $4$ is $0$.}
\answerExplain{False}{Distance $= |{-4} - 4| = |{-8}| = 8$ units. The numbers are opposites, each $4$ units from $0$, so they are $8$ units apart.}

\end{practiceBox}

% ── Practice E: Word Problems ────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{The highest point in Death Valley is $1{,}669$ m above sea level. The lowest point is $86$ m below sea level. What is the difference in elevation?}{meters}
\answerExplain{$1{,}755$ meters}{Below sea level is $-86$ m. Difference: $1{,}669 - (-86) = 1{,}669 + 86 = 1{,}755$ meters.}

\wordProblem{At 6 AM the temperature was $-3°$C. By 2 PM it was $11°$C. How much did the temperature change?}{°C}
\answerExplain{$14°$C increase}{Change $= 11 - (-3) = 11 + 3 = 14°$C. The temperature rose $14$ degrees.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}

\prob Find a value of $x$ such that $x - (-x) = 16$. \answerBlank[2cm]
\answerExplain{$x = 8$}{Simplify: $x - (-x) = x + x = 2x$. Set $2x = 16$, so $x = 8$. Check: $8 - (-8) = 8 + 8 = 16$. \checkmark}

\prob The temperature at the top of a mountain is $-18°$C. At the base it is $7°$C. A hiker is halfway up, where the temperature is exactly the average. What is the temperature at the halfway point? \answerBlank[3cm]
\answerExplain{$-5.5°$C}{Average $= \frac{-18 + 7}{2} = \frac{-11}{2} = -5.5°$C. The halfway temperature is $-5.5°$C.}

\end{challengeBox}
