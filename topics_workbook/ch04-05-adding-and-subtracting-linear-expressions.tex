\section{Adding and Subtracting Linear Expressions}
\topicTitle{Adding and Subtracting Linear Expressions}

\begin{quickReview}{Adding and Subtracting Expressions}
\textbf{Adding expressions:} Remove parentheses and combine like terms.
$$(3x + 5) + (2x - 4) = 5x + 1$$

\textbf{Subtracting expressions:} Distribute the negative sign to \textbf{every term} in the second expression, then combine.
$$(3x + 5) - (2x - 4) = 3x + 5 - 2x + 4 = x + 9$$

\medskip
\textbf{Key idea:} Subtracting flips every sign in the second expression.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Add these expressions.}

\begin{multicols}{2}
\prob $(3x + 2) + (5x + 4)$ \answerBlank[3cm]
\answerExplain{$8x + 6$}{Remove parentheses: $3x + 2 + 5x + 4$. Combine like terms: $(3x + 5x) + (2 + 4) = 8x + 6$.}

\prob $(7a - 1) + (2a + 6)$ \answerBlank[3cm]
\answerExplain{$9a + 5$}{Remove parentheses: $7a - 1 + 2a + 6$. Combine: $(7a + 2a) + (-1 + 6) = 9a + 5$.}

\prob $(4n + 3) + (n - 8)$ \answerBlank[3cm]
\answerExplain{$5n - 5$}{Remove parentheses: $4n + 3 + n - 8$. Combine: $(4n + n) + (3 - 8) = 5n - 5$.}

\prob $(-2y + 7) + (6y + 1)$ \answerBlank[3cm]
\answerExplain{$4y + 8$}{Remove parentheses: $-2y + 7 + 6y + 1$. Combine: $(-2y + 6y) + (7 + 1) = 4y + 8$.}

\prob $(x - 5) + (x + 5)$ \answerBlank[3cm]
\answerExplain{$2x$}{Remove parentheses: $x - 5 + x + 5$. Combine: $(x + x) + (-5 + 5) = 2x + 0 = 2x$.}
\end{multicols}
\end{practiceBox}

% ── Practice A: Subtracting Expressions ──────────────────────────────────────
\begin{practiceBox}{Subtracting Linear Expressions}

\practiceHeader{Distribute the negative sign, then combine like terms.}

\begin{multicols}{2}
\prob $(6x + 5) - (2x + 3)$ \answerBlank[3cm]
\answerExplain{$4x + 2$}{Distribute the negative: $6x + 5 - 2x - 3$. Combine: $(6x - 2x) + (5 - 3) = 4x + 2$.}

\prob $(9a - 4) - (5a - 1)$ \answerBlank[3cm]
\answerExplain{$4a - 3$}{Distribute the negative: $9a - 4 - 5a + 1$. Combine: $(9a - 5a) + (-4 + 1) = 4a - 3$.}

\prob $(3n + 8) - (7n + 2)$ \answerBlank[3cm]
\answerExplain{$-4n + 6$}{Distribute the negative: $3n + 8 - 7n - 2$. Combine: $(3n - 7n) + (8 - 2) = -4n + 6$.}

\prob $(5y - 3) - (5y + 4)$ \answerBlank[3cm]
\answerExplain{$-7$}{Distribute the negative: $5y - 3 - 5y - 4$. Combine: $(5y - 5y) + (-3 - 4) = 0 - 7 = -7$.}

\prob $(-2b + 9) - (-6b + 3)$ \answerBlank[3cm]
\answerExplain{$4b + 6$}{Distribute the negative: $-2b + 9 + 6b - 3$. Combine: $(-2b + 6b) + (9 - 3) = 4b + 6$.}

\prob $(4x - 1) - (-3x - 5)$ \answerBlank[3cm]
\answerExplain{$7x + 4$}{Distribute the negative: $4x - 1 + 3x + 5$. Combine: $(4x + 3x) + (-1 + 5) = 7x + 4$.}
\end{multicols}
\end{practiceBox}

% ── Practice B: Mixed Adding and Subtracting ─────────────────────────────────
\begin{practiceBox}[funTeal]{Add or Subtract — Watch the Operation!}

\prob $(2x + 5) + (4x - 3) - (x + 1)$ \answerBlank[3cm]
\answerExplain{$5x + 1$}{Remove the first set of parentheses and distribute the negative on the third group: $2x + 5 + 4x - 3 - x - 1$. Combine: $(2x + 4x - x) + (5 - 3 - 1) = 5x + 1$.}

\prob $(8a - 2) - (3a + 4) + (a - 6)$ \answerBlank[3cm]
\answerExplain{$6a - 12$}{Distribute the negative on the second group: $8a - 2 - 3a - 4 + a - 6$. Combine: $(8a - 3a + a) + (-2 - 4 - 6) = 6a - 12$.}

\prob $(\frac{1}{2}n + 3) + (\frac{3}{4}n - 5)$ \answerBlank[3cm]
\answerExplain{$\frac{5}{4}n - 2$}{Combine the $n$-terms: $\frac{1}{2}n + \frac{3}{4}n = \frac{2}{4}n + \frac{3}{4}n = \frac{5}{4}n$. Combine constants: $3 - 5 = -2$. Result: $\frac{5}{4}n - 2$.}

\prob $(0.6y - 1.2) - (0.4y + 0.8)$ \answerBlank[3cm]
\answerExplain{$0.2y - 2$}{Distribute the negative: $0.6y - 1.2 - 0.4y - 0.8$. Combine: $(0.6y - 0.4y) + (-1.2 - 0.8) = 0.2y - 2$.}

\end{practiceBox}

% ── Practice C: Real-World Perimeters ────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Geometry and Word Problems}

\prob Find the perimeter of a triangle with sides $(3x + 1)$, $(2x - 4)$, and $(x + 7)$. \answerBlank[4cm]
\answerExplain{$6x + 4$}{Add all three sides: $(3x + 1) + (2x - 4) + (x + 7) = (3x + 2x + x) + (1 - 4 + 7) = 6x + 4$.}

\prob A rectangle has length $(5a + 2)$ and width $(3a - 1)$. Write a simplified expression for the perimeter.
\answerBlank[4cm]
\answerExplain{$16a + 2$}{Perimeter $= 2(\text{length}) + 2(\text{width}) = 2(5a + 2) + 2(3a - 1) = 10a + 4 + 6a - 2 = 16a + 2$.}

\prob Jen earned $(4x + 10)$ dollars this week and $(3x - 5)$ dollars last week. How much more did she earn this week?
\answerBlank[4cm]
\answerExplain{$x + 15$ dollars}{Subtract last week from this week: $(4x + 10) - (3x - 5) = 4x + 10 - 3x + 5 = x + 15$.}

\end{practiceBox}

% ── Practice D: Error Analysis ───────────────────────────────────────────────
\begin{errorBox}{Spot the Mistake}

\prob Alex simplified $(5x + 2) - (3x - 1)$ and wrote $2x + 1$. What went wrong?
\answerBlank[6cm]
\answerExplain{Alex did not flip the sign of $-1$}{When subtracting, the negative sign must flip \emph{both} signs: $-3x$ and $+1$. The correct work is $5x + 2 - 3x + 1 = 2x + 3$, not $2x + 1$.}

\end{errorBox}

% ── Practice E: True or False ────────────────────────────────────────────────
\begin{practiceBox}[funOrange]{True or False?}

\trueOrFalse{$(4x + 3) + (2x - 3) = 6x$.}
\answerExplain{True}{Combine: $4x + 3 + 2x - 3 = (4x + 2x) + (3 - 3) = 6x + 0 = 6x$.}

\trueOrFalse{$(a + 5) - (a + 5) = 2a + 10$.}
\answerExplain{False}{Distribute the negative: $a + 5 - a - 5 = 0$. Subtracting an expression from itself always gives $0$, not $2a + 10$.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob Simplify $(7x - 2y + 3) - (4x + 3y - 5) + (2x - y + 1)$. \answerBlank[4cm]
\answerExplain{$5x - 6y + 9$}{Distribute the negative on the second group: $7x - 2y + 3 - 4x - 3y + 5 + 2x - y + 1$. Combine $x$: $7x - 4x + 2x = 5x$. Combine $y$: $-2y - 3y - y = -6y$. Constants: $3 + 5 + 1 = 9$. Result: $5x - 6y + 9$.}

\prob The sides of a quadrilateral are $(2x + 3)$, $(x - 1)$, $(3x + 4)$, and $(x + 2)$. If the perimeter is $50$, find $x$.
\answerBlank[3cm]
\answerExplain{$x = 6$}{Perimeter: $(2x + 3) + (x - 1) + (3x + 4) + (x + 2) = 7x + 8$. Set equal to $50$: $7x + 8 = 50$. Subtract $8$: $7x = 42$. Divide: $x = 6$.}
\end{challengeBox}
