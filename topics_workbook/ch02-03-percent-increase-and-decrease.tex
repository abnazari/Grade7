\section{Percent Increase and Decrease}
\topicTitle{Percent Increase and Decrease}

\begin{quickReview}{Percent Increase and Decrease}
\textbf{Percent change} tells you how much a value changed compared to the original.
$$\text{percent change} = \frac{|\text{new} - \text{original}|}{\text{original}} \times 100$$

\textbf{Multiplier shortcut:}
\begin{itemize}[leftmargin=*]
    \item \textbf{Increase} by $p\%$: multiply by $1 + \frac{p}{100}$ \quad (e.g., increase by $15\%$ $\to$ multiply by $1.15$)
    \item \textbf{Decrease} by $p\%$: multiply by $1 - \frac{p}{100}$ \quad (e.g., decrease by $20\%$ $\to$ multiply by $0.80$)
\end{itemize}

\medskip
\textbf{Example:} A shirt was $\$40$ and now costs $\$46$. Percent increase $= \frac{|46 - 40|}{40} \times 100 = \frac{6}{40} \times 100 = 15\%$.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Write the Multiplier}
Write the single multiplier for each percent change.

\begin{multicols}{2}
\prob Increase by $10\%$ \answerBlank[3cm]
\answerExplain{$1.10$}{An increase of $10\%$ means you keep $100\%$ and add $10\%$: $1 + 0.10 = 1.10$.}

\prob Decrease by $25\%$ \answerBlank[3cm]
\answerExplain{$0.75$}{A decrease of $25\%$ means you keep $100\% - 25\% = 75\%$: $1 - 0.25 = 0.75$.}

\prob Increase by $50\%$ \answerBlank[3cm]
\answerExplain{$1.50$}{An increase of $50\%$: $1 + 0.50 = 1.50$. The new value is $150\%$ of the original.}

\prob Decrease by $40\%$ \answerBlank[3cm]
\answerExplain{$0.60$}{A decrease of $40\%$: $1 - 0.40 = 0.60$. You keep $60\%$ of the original.}

\prob Increase by $5\%$ \answerBlank[3cm]
\answerExplain{$1.05$}{An increase of $5\%$: $1 + 0.05 = 1.05$.}

\prob Decrease by $75\%$ \answerBlank[3cm]
\answerExplain{$0.25$}{A decrease of $75\%$: $1 - 0.75 = 0.25$. Only $25\%$ of the original remains.}
\end{multicols}
\end{practiceBox}

% ── Practice A: Using Multipliers ────────────────────────────────────────────
\begin{practiceBox}{Using Multipliers to Find New Values}

\practiceHeader{Use the multiplier to find the new amount.}

\prob A $\$60$ jacket increases in price by $15\%$. What is the new price? \answerBlank[3cm]
\answerExplain{$\$69$}{Multiplier for $15\%$ increase: $1.15$. New price: $60 \times 1.15 = \$69$.}

\prob A $\$200$ TV is on sale for $30\%$ off. What is the sale price? \answerBlank[3cm]
\answerExplain{$\$140$}{Multiplier for $30\%$ decrease: $0.70$. Sale price: $200 \times 0.70 = \$140$.}

\prob A town's population of $5{,}000$ grows by $8\%$. What is the new population? \answerBlank[3cm]
\answerExplain{$5{,}400$}{Multiplier for $8\%$ increase: $1.08$. New population: $5{,}000 \times 1.08 = 5{,}400$.}

\prob A $\$45$ backpack is marked down by $20\%$. What is the new price? \answerBlank[3cm]
\answerExplain{$\$36$}{Multiplier for $20\%$ decrease: $0.80$. New price: $45 \times 0.80 = \$36$.}

\prob A gym has $240$ members and membership increases by $35\%$. How many members now? \answerBlank[3cm]
\answerExplain{$324$ members}{Multiplier for $35\%$ increase: $1.35$. New count: $240 \times 1.35 = 324$.}

\end{practiceBox}

% ── Practice B: Finding Percent Change ───────────────────────────────────────
\begin{practiceBox}[funTeal]{Finding the Percent Change}

\practiceHeader[funTeal]{Find the percent increase or decrease.}

\prob A plant grew from $10$ cm to $13$ cm. \answerBlank[3cm]
\answerExplain{$30\%$ increase}{Change: $13 - 10 = 3$. Divide by original: $3 \div 10 = 0.30$. Multiply by $100$: $30\%$ increase.}

\prob A stock dropped from $\$80$ to $\$68$. \answerBlank[3cm]
\answerExplain{$15\%$ decrease}{Change: $80 - 68 = 12$. Divide by original: $12 \div 80 = 0.15$. Multiply by $100$: $15\%$ decrease.}

\prob School enrollment went from $600$ to $750$ students. \answerBlank[3cm]
\answerExplain{$25\%$ increase}{Change: $750 - 600 = 150$. Divide by original: $150 \div 600 = 0.25$. Multiply by $100$: $25\%$ increase.}

\prob A river's water level dropped from $8$ feet to $6$ feet. \answerBlank[3cm]
\answerExplain{$25\%$ decrease}{Change: $8 - 6 = 2$. Divide by original: $2 \div 8 = 0.25$. Multiply by $100$: $25\%$ decrease.}

\prob The price of gas went from $\$3.00$ to $\$3.60$ per gallon. \answerBlank[3cm]
\answerExplain{$20\%$ increase}{Change: $3.60 - 3.00 = 0.60$. Divide by original: $0.60 \div 3.00 = 0.20$. Multiply by $100$: $20\%$ increase.}

\end{practiceBox}

% ── Practice C: True or False ────────────────────────────────────────────────
\begin{practiceBox}[funPurple]{True or False?}

\trueOrFalse{A $50\%$ increase followed by a $50\%$ decrease gives you the original amount.}
\answerExplain{False}{Start with $100$. Increase $50\%$: $100 \times 1.50 = 150$. Decrease $50\%$: $150 \times 0.50 = 75$. That is less than $100$, not equal. The decrease applies to a larger number.}

\trueOrFalse{The multiplier for a $100\%$ increase is $2.00$.}
\answerExplain{True}{A $100\%$ increase means the value doubles. Multiplier: $1 + 1.00 = 2.00$, so the new value is twice the original.}

\trueOrFalse{If a price decreases by $20\%$, you multiply by $0.20$ to find the new price.}
\answerExplain{False}{Multiplying by $0.20$ gives the discount amount, not the new price. To find the new price, multiply by $1 - 0.20 = 0.80$.}

\trueOrFalse{A $\$50$ item that increases by $10\%$ and then increases by $10\%$ again costs $\$60.50$.}
\answerExplain{True}{First increase: $50 \times 1.10 = 55$. Second increase: $55 \times 1.10 = 60.50$. Two $10\%$ increases do not equal one $20\%$ increase ($50 \times 1.20 = 60$).}

\end{practiceBox}

\answerExplain{$\$204$}{Multiplier for $15\%$ decrease: $0.85$. Sale price: $240 \times 0.85 = \$204$.}

\wordProblem{A town had $12{,}000$ people last year. The population increased by $6\%$ this year. How many people live there now?}{people}
\answerExplain{$12{,}720$ people}{Multiplier for $6\%$ increase: $1.06$. New population: $12{,}000 \times 1.06 = 12{,}720$.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob A store buys an item for $\$80$. It increases the price by $50\%$, then offers a $25\%$ discount. What is the final selling price? Is it more or less than the original $\$80$? \answerBlank[3cm]
\answerExplain{$\$90$; more than $\$80$}{Markup: $80 \times 1.50 = \$120$. Discount: $120 \times 0.75 = \$90$. The final price is $\$90$, which is $\$10$ more than the original cost.}

\prob A population of $2{,}000$ decreased by $10\%$ one year and then increased by $20\%$ the next year. What is the population after two years? \answerBlank[3cm]
\answerExplain{$2{,}160$}{After year 1: $2{,}000 \times 0.90 = 1{,}800$. After year 2: $1{,}800 \times 1.20 = 2{,}160$. The $20\%$ increase applies to the reduced value, not the original.}
\end{challengeBox}
