\section{Expanding with the Distributive Property}
\topicTitle{Expanding with the Distributive Property}

\begin{quickReview}{The Distributive Property}
Multiply the number \textbf{outside} the parentheses by \textbf{every term inside}:
$$a(b + c) = ab + ac$$

\begin{itemize}[leftmargin=*]
    \item $3(2x + 5) = 6x + 15$
    \item $-2(4x - 3) = -8x + 6$ \quad (a negative times a negative is positive!)
\end{itemize}

\medskip
\textbf{Example:} Expand $-4(3x - 7)$: \quad $-4 \times 3x = -12x$, \quad $-4 \times (-7) = +28$. \quad Result: $-12x + 28$.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Expand each expression.}

\begin{multicols}{2}
\prob $2(x + 3)$ \answerBlank[3cm]
\answerExplain{$2x + 6$}{Multiply $2$ by each term: $2 \times x = 2x$ and $2 \times 3 = 6$. Result: $2x + 6$.}

\prob $5(y - 4)$ \answerBlank[3cm]
\answerExplain{$5y - 20$}{Multiply $5$ by each term: $5 \times y = 5y$ and $5 \times (-4) = -20$. Result: $5y - 20$.}

\prob $3(2a + 1)$ \answerBlank[3cm]
\answerExplain{$6a + 3$}{Multiply $3$ by each term: $3 \times 2a = 6a$ and $3 \times 1 = 3$. Result: $6a + 3$.}

\prob $4(n - 5)$ \answerBlank[3cm]
\answerExplain{$4n - 20$}{Multiply $4$ by each term: $4 \times n = 4n$ and $4 \times (-5) = -20$. Result: $4n - 20$.}

\prob $6(3 + 2b)$ \answerBlank[3cm]
\answerExplain{$18 + 12b$}{Multiply $6$ by each term: $6 \times 3 = 18$ and $6 \times 2b = 12b$. Result: $18 + 12b$.}

\prob $7(x + 1)$ \answerBlank[3cm]
\answerExplain{$7x + 7$}{Multiply $7$ by each term: $7 \times x = 7x$ and $7 \times 1 = 7$. Result: $7x + 7$.}
\end{multicols}
\end{practiceBox}

% ── Practice A: Expanding with Negatives ─────────────────────────────────────
\begin{practiceBox}{Expanding with Negative Factors}

\practiceHeader{Watch the signs carefully!}

\begin{multicols}{2}
\prob $-3(x + 5)$ \answerBlank[3cm]
\answerExplain{$-3x - 15$}{Multiply $-3$ by each term: $-3 \times x = -3x$ and $-3 \times 5 = -15$. Result: $-3x - 15$.}

\prob $-2(4a - 7)$ \answerBlank[3cm]
\answerExplain{$-8a + 14$}{Multiply $-2$ by each term: $-2 \times 4a = -8a$ and $-2 \times (-7) = +14$. A negative times a negative is positive.}

\prob $-5(2n + 3)$ \answerBlank[3cm]
\answerExplain{$-10n - 15$}{Multiply $-5$ by each term: $-5 \times 2n = -10n$ and $-5 \times 3 = -15$. Result: $-10n - 15$.}

\prob $-(6x - 1)$ \answerBlank[3cm]
\answerExplain{$-6x + 1$}{The negative sign in front means $-1$ times each term: $-1 \times 6x = -6x$ and $-1 \times (-1) = +1$. Result: $-6x + 1$.}

\prob $-4(-2y + 3)$ \answerBlank[3cm]
\answerExplain{$8y - 12$}{Multiply $-4$ by each term: $-4 \times (-2y) = 8y$ and $-4 \times 3 = -12$. Result: $8y - 12$.}

\prob $-1(3m - 8)$ \answerBlank[3cm]
\answerExplain{$-3m + 8$}{Multiply $-1$ by each term: $-1 \times 3m = -3m$ and $-1 \times (-8) = +8$. Multiplying by $-1$ flips every sign.}
\end{multicols}
\end{practiceBox}

% ── Practice B: Fractional Coefficients ──────────────────────────────────────
\begin{practiceBox}[funTeal]{Expanding with Fractions and Decimals}

\prob $\frac{1}{2}(6x - 10)$ \answerBlank[3cm]
\answerExplain{$3x - 5$}{Multiply $\frac{1}{2}$ by each term: $\frac{1}{2} \times 6x = 3x$ and $\frac{1}{2} \times (-10) = -5$. Result: $3x - 5$.}

\prob $\frac{1}{3}(9a + 12)$ \answerBlank[3cm]
\answerExplain{$3a + 4$}{Multiply $\frac{1}{3}$ by each term: $\frac{1}{3} \times 9a = 3a$ and $\frac{1}{3} \times 12 = 4$. Result: $3a + 4$.}

\prob $0.5(4y - 8)$ \answerBlank[3cm]
\answerExplain{$2y - 4$}{Multiply $0.5$ by each term: $0.5 \times 4y = 2y$ and $0.5 \times (-8) = -4$. Result: $2y - 4$.}

\prob $-\frac{2}{3}(6n - 9)$ \answerBlank[3cm]
\answerExplain{$-4n + 6$}{Multiply $-\frac{2}{3}$ by each term: $-\frac{2}{3} \times 6n = -4n$ and $-\frac{2}{3} \times (-9) = +6$. Result: $-4n + 6$.}

\end{practiceBox}

% ── Practice C: Expand and Simplify ──────────────────────────────────────────
\begin{practiceBox}[funPurple]{Expand and Simplify}

\practiceHeader[funPurple]{First expand, then combine like terms.}

\prob $3(x + 2) + 5x$ \answerBlank[3cm]
\answerExplain{$8x + 6$}{Expand: $3x + 6 + 5x$. Combine like terms: $(3x + 5x) + 6 = 8x + 6$.}

\prob $2(4a - 1) - 3a$ \answerBlank[3cm]
\answerExplain{$5a - 2$}{Expand: $8a - 2 - 3a$. Combine like terms: $(8a - 3a) - 2 = 5a - 2$.}

\prob $-2(x - 5) + 3(2x + 1)$ \answerBlank[3cm]
\answerExplain{$4x + 13$}{Expand each part: $-2x + 10 + 6x + 3$. Combine: $(-2x + 6x) + (10 + 3) = 4x + 13$.}

\prob $4(y + 3) - 2(y - 1)$ \answerBlank[3cm]
\answerExplain{$2y + 14$}{Expand: $4y + 12 - 2y + 2$. Combine: $(4y - 2y) + (12 + 2) = 2y + 14$.}

\end{practiceBox}

% ── Practice D: True or False ────────────────────────────────────────────────
\begin{practiceBox}[funOrange]{True or False?}

\trueOrFalse{$3(x + 4) = 3x + 4$.}
\answerExplain{False}{The distributive property says you must multiply $3$ by \emph{every} term inside: $3(x + 4) = 3x + 12$, not $3x + 4$.}

\trueOrFalse{$-2(a - 5) = -2a + 10$.}
\answerExplain{True}{Multiply $-2$ by each term: $-2 \times a = -2a$ and $-2 \times (-5) = +10$. Result: $-2a + 10$.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob Expand and simplify $5(2x - 3) - 3(4x - 1) + 2$. \answerBlank[4cm]
\answerExplain{$-2x - 10$}{Expand: $10x - 15 - 12x + 3 + 2$. Combine $x$-terms: $10x - 12x = -2x$. Combine constants: $-15 + 3 + 2 = -10$. Result: $-2x - 10$.}

\prob A rectangular garden has length $(3x + 2)$~feet and width $4$~feet. Write and simplify an expression for the perimeter.
\answerBlank[4cm]
\answerExplain{$6x + 12$ feet}{Perimeter $= 2 \times \text{length} + 2 \times \text{width} = 2(3x + 2) + 2(4) = 6x + 4 + 8 = 6x + 12$ feet.}
\end{challengeBox}
