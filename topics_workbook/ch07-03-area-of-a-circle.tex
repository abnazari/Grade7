\section{Area of a Circle}
\topicTitle{Area of a Circle}

\begin{quickReview}{Area of a Circle}
The \textbf{area} of a circle measures the space inside. Use the formula:

\[A = \pi r^2\]

\begin{itemize}[leftmargin=*]
    \item $r$ is the \textbf{radius}. If given the diameter, first find $r = d \div 2$.
    \item Square the radius \emph{first}, then multiply by $\pi$.
    \item Area is measured in \textbf{square units} (cm$^2$, m$^2$, in$^2$, etc.).
\end{itemize}

\medskip
\textbf{Example:} $r = 4$ cm. \quad $A = 3.14 \times 4^2 = 3.14 \times 16 = 50.24$ cm$^2$.
\end{quickReview}

% ── Practice B: Working Backwards ────────────────────────────────────────────
\begin{practiceBox}[funTeal]{Find the Radius from the Area}

\practiceHeader[funTeal]{Use $r^2 = A \div \pi$ to find the radius. Use $\pi \approx 3.14$.}

\prob A circle has an area of $78.5$ cm$^2$. What is its radius? \answerBlank[3cm]
\answerExplain{$5$ cm}{$r^2 = A \div \pi = 78.5 \div 3.14 = 25$. Take the square root: $r = 5$ cm.}

\prob A circle has an area of $200.96$ in$^2$. What is its radius? \answerBlank[3cm]
\answerExplain{$8$ in}{$r^2 = 200.96 \div 3.14 = 64$. Take the square root: $r = 8$ in.}

\prob A circle has an area of $28.26$ m$^2$. What is its diameter? \answerBlank[3cm]
\answerExplain{$6$ m}{$r^2 = 28.26 \div 3.14 = 9$. So $r = 3$ m and $d = 2 \times 3 = 6$ m.}

\prob A circle has an area of $314$ ft$^2$. What is its diameter? \answerBlank[3cm]
\answerExplain{$20$ ft}{$r^2 = 314 \div 3.14 = 100$. So $r = 10$ ft and $d = 2 \times 10 = 20$ ft.}

\end{practiceBox}

% ── Practice C: True or False ────────────────────────────────────────────────
\begin{practiceBox}[funPurple]{True or False?}

\trueOrFalse{If you double the radius, the area doubles.}
\answerExplain{False}{If you double the radius, the area quadruples. For example, $r = 3$: $A = 9\pi$. Double to $r = 6$: $A = 36\pi$. Since $36\pi = 4 \times 9\pi$, the area is $4$ times as large, not $2$.}

\trueOrFalse{To find the area, you need the radius, not the diameter.}
\answerExplain{False}{You can use either. If given the diameter, divide by $2$ to get the radius, then apply $A = \pi r^2$. Both measurements work.}

\trueOrFalse{A circle with radius $7$ cm has an area of $153.86$ cm$^2$ (using $\pi \approx 3.14$).}
\answerExplain{True}{$A = 3.14 \times 7^2 = 3.14 \times 49 = 153.86$ cm$^2$. This is correct.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob A semicircular window has a diameter of $10$ ft. What is the area of the window? Use $\pi \approx 3.14$. \answerBlank[3cm]
\answerExplain{$39.25$ ft$^2$}{Radius: $10 \div 2 = 5$ ft. Full circle area: $3.14 \times 25 = 78.5$ ft$^2$. Semicircle: $78.5 \div 2 = 39.25$ ft$^2$.}

\prob A large circular garden has a radius of $10$ m. Inside it, a circular fountain has a radius of $3$ m. What is the area of the garden that is \emph{not} taken up by the fountain? Use $\pi \approx 3.14$. \answerBlank[3cm]
\answerExplain{$285.74$ m$^2$}{Garden area: $3.14 \times 10^2 = 314$ m$^2$. Fountain area: $3.14 \times 3^2 = 28.26$ m$^2$. Difference: $314 - 28.26 = 285.74$ m$^2$.}
\end{challengeBox}
