\topicTitle{Multiplying Integers and Rational Numbers}

\begin{quickReview}{Multiplying Signed Numbers}
\textbf{Sign rules for multiplication:}

\begin{itemize}[leftmargin=*]
    \item \textbf{Same signs $\to$ positive:} $(+)(+) = +$ \quad and \quad $(-)(-)= +$
    \item \textbf{Different signs $\to$ negative:} $(+)(-) = -$ \quad and \quad $(-)(+) = -$
\end{itemize}

\medskip
\textbf{Quick trick:} Count the negative factors. \textbf{Even} count $\to$ positive. \textbf{Odd} count $\to$ negative.

\smallskip
\textbf{Examples:} $(-4)(3) = -12$ \quad $(-2)(-5) = 10$ \quad $(-1)(-2)(-3) = -6$ (three negatives $\to$ negative)
\end{quickReview}

% ── Practice B: Signed Fractions and Decimals ────────────────────────────────
\begin{practiceBox}[funTeal]{Multiplying Rational Numbers}

\practiceHeader[funTeal]{Multiply. Simplify fractions.}

\prob $\left(-\dfrac{2}{3}\right) \times \dfrac{3}{5}$ \answerBlank[2cm]
\answerExplain{$-\dfrac{2}{5}$}{Multiply numerators: $2 \times 3 = 6$. Multiply denominators: $3 \times 5 = 15$. Different signs $\to$ negative: $-\frac{6}{15} = -\frac{2}{5}$.}

\prob $\left(-\dfrac{3}{4}\right) \times \left(-\dfrac{2}{5}\right)$ \answerBlank[2cm]
\answerExplain{$\dfrac{3}{10}$}{Multiply: $\frac{3 \times 2}{4 \times 5} = \frac{6}{20} = \frac{3}{10}$. Same signs $\to$ positive.}

\prob $\left(-\dfrac{5}{8}\right) \times \dfrac{4}{3}$ \answerBlank[2cm]
\answerExplain{$-\dfrac{5}{6}$}{Multiply: $\frac{5 \times 4}{8 \times 3} = \frac{20}{24} = \frac{5}{6}$. Different signs $\to$ negative: $-\frac{5}{6}$.}

\prob $(-1.5) \times 0.4$ \answerBlank[2cm]
\answerExplain{$-0.6$}{Multiply absolute values: $1.5 \times 0.4 = 0.6$. Different signs $\to$ negative: $-0.6$.}

\prob $(-0.3) \times (-0.7)$ \answerBlank[2cm]
\answerExplain{$0.21$}{Multiply absolute values: $0.3 \times 0.7 = 0.21$. Same signs $\to$ positive: $0.21$.}

\prob $(-2.5) \times (-4)$ \answerBlank[2cm]
\answerExplain{$10$}{Multiply absolute values: $2.5 \times 4 = 10$. Same signs $\to$ positive: $10$.}

\end{practiceBox}

% ── Practice C: True or False? ───────────────────────────────────────────────
\begin{practiceBox}[funPurple]{True or False?}

\trueOrFalse{The product of two negative numbers is always positive.}
\answerExplain{True}{Same signs rule: negative times negative gives a positive result. This is always true.}

\trueOrFalse{$(-1) \times (-1) \times (-1) \times (-1) = -1$}
\answerExplain{False}{Four negative factors (even count) $\to$ positive. $(-1)^4 = 1$, not $-1$.}

\trueOrFalse{If a product is negative, then exactly one of the two factors must be negative.}
\answerExplain{True}{For two factors, a negative product means the factors have different signs, so exactly one is negative and the other is positive.}

\end{practiceBox}

% ── Practice D: Word Problems ────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{A stock drops $\$1.25$ per hour for $6$ hours. What is the total change in the stock price?}{\$}
\answerExplain{$-\$7.50$}{A drop of $\$1.25$ per hour is $-1.25$. Multiply: $(-1.25) \times 6 = -7.50$. The stock fell a total of $\$7.50$.}

\wordProblem{A submarine descends at a rate of $-15$ meters per minute. Where is it after $\frac{2}{3}$ of a minute?}{meters}
\answerExplain{$-10$ meters}{Multiply: $(-15) \times \frac{2}{3} = \frac{-30}{3} = -10$. The submarine is $10$ meters below its starting point.}

\wordProblem{Each day for $5$ days, a town's water level drops by $0.8$ cm. What is the total change?}{cm}
\answerExplain{$-4$ cm}{Each day: $-0.8$ cm. For $5$ days: $(-0.8) \times 5 = -4$ cm. The water level fell $4$ cm total.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}

\prob Without computing, decide: Is $(-3) \times (-4) \times (-5) \times (-6)$ positive or negative? Explain. \answerBlank[3cm]
\answerExplain{Positive}{Count the negative factors: there are $4$ (an even number). An even number of negative factors gives a positive product. ($= 360$)}

\prob Find the value of $\left(-\dfrac{1}{2}\right)^3$. \answerBlank[2cm]
\answerExplain{$-\dfrac{1}{8}$}{$\left(-\frac{1}{2}\right)^3 = \left(-\frac{1}{2}\right)\left(-\frac{1}{2}\right)\left(-\frac{1}{2}\right)$. First two: $\frac{1}{4}$ (positive). Then $\frac{1}{4} \times (-\frac{1}{2}) = -\frac{1}{8}$. Three negatives (odd) $\to$ negative.}

\end{challengeBox}
