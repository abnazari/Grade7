\section{Drawing Geometric Figures with Given Conditions}
\topicTitle{Drawing Geometric Figures with Given Conditions}

\begin{quickReview}{When Can You Draw a Shape?}
Given conditions about sides and angles may produce:

\begin{itemize}[leftmargin=*]
    \item \textbf{Exactly one shape} — the conditions fully determine it (e.g., three specific sides of a triangle).
    \item \textbf{More than one shape} — the conditions allow flexibility (e.g., a quadrilateral with four given sides but no angles).
    \item \textbf{No shape} — the conditions are impossible (e.g., a triangle with sides $1$, $2$, $10$).
\end{itemize}

\medskip
\textbf{Triangle Inequality:} A triangle is possible only if the sum of any two sides is \textbf{greater than} the third side.

\smallskip
\textbf{Example:} Sides $3$, $5$, $7$: \quad $3 + 5 = 8 > 7$ \checkmark \quad $3 + 7 = 10 > 5$ \checkmark \quad $5 + 7 = 12 > 3$ \checkmark \quad Triangle is possible!
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Can a Triangle Be Formed?}
Check the triangle inequality. Write Yes or No.

\begin{multicols}{2}
\prob Sides: $3$, $4$, $5$ \answerBlank[2cm]
\answerExplain{Yes}{$3 + 4 = 7 > 5$ \checkmark, $3 + 5 = 8 > 4$ \checkmark, $4 + 5 = 9 > 3$ \checkmark. All three checks pass, so a triangle can be formed.}

\prob Sides: $1$, $2$, $8$ \answerBlank[2cm]
\answerExplain{No}{$1 + 2 = 3$, but $3 < 8$. The two shorter sides cannot reach each other, so no triangle is possible.}

\prob Sides: $6$, $6$, $6$ \answerBlank[2cm]
\answerExplain{Yes}{$6 + 6 = 12 > 6$ \checkmark. All three checks pass (the triangle is equilateral).}

\prob Sides: $2$, $3$, $5$ \answerBlank[2cm]
\answerExplain{No}{$2 + 3 = 5$, which equals the third side. The sum must be \emph{greater than}, not equal to, the third side. The ``triangle'' would be a flat line segment.}

\prob Sides: $7$, $10$, $12$ \answerBlank[2cm]
\answerExplain{Yes}{$7 + 10 = 17 > 12$ \checkmark, $7 + 12 = 19 > 10$ \checkmark, $10 + 12 = 22 > 7$ \checkmark. All pass.}

\prob Sides: $4$, $4$, $9$ \answerBlank[2cm]
\answerExplain{No}{$4 + 4 = 8$, but $8 < 9$. The two shorter sides cannot connect around the longest side.}
\end{multicols}
\end{practiceBox}

% ── Practice A: One, Many, or None ───────────────────────────────────────────
\begin{practiceBox}{One, Many, or No Shapes?}

\practiceHeader{Determine: does each set of conditions produce \textbf{exactly one shape}, \textbf{more than one shape}, or \textbf{no shape}?}

\prob A triangle with sides $5$ cm, $7$ cm, and $9$ cm. \answerBlank[3cm]
\answerExplain{Exactly one}{Three specific side lengths (SSS) that satisfy the triangle inequality determine exactly one triangle.}

\prob A rectangle with a perimeter of $24$ cm. \answerBlank[3cm]
\answerExplain{More than one}{The sides must add to $12$ (half the perimeter). Many pairs work: $1 \times 11$, $2 \times 10$, $3 \times 9$, $4 \times 8$, $5 \times 7$, $6 \times 6$, etc.}

\prob A triangle with sides $1$ cm, $3$ cm, and $7$ cm. \answerBlank[3cm]
\answerExplain{No shape}{$1 + 3 = 4 < 7$. The triangle inequality fails, so no triangle is possible.}

\prob A quadrilateral with sides $3$, $4$, $5$, and $6$ cm. \answerBlank[3cm]
\answerExplain{More than one}{Four side lengths do not determine a unique quadrilateral. You can change the angles and get many different shapes with the same sides.}

\prob A triangle with angles $60°$, $60°$, and $60°$. \answerBlank[3cm]
\answerExplain{More than one}{Three angles (AAA) do not fix the size. You can draw equilateral triangles of any size — they all have the same shape but different dimensions.}

\prob A triangle with sides $5$ cm and $8$ cm and an included angle of $45°$. \answerBlank[3cm]
\answerExplain{Exactly one}{Two sides and the included angle (SAS) fully determine a unique triangle.}
\end{practiceBox}

% ── Practice B: True or False ────────────────────────────────────────────────
\begin{practiceBox}[funPurple]{True or False?}

\trueOrFalse{If you know all three angles of a triangle, you can draw exactly one triangle.}
\answerExplain{False}{Knowing all three angles (AAA) tells you the shape but not the size. You can draw infinitely many similar triangles with those angles.}

\trueOrFalse{A triangle with sides $5$, $12$, and $13$ is possible.}
\answerExplain{True}{$5 + 12 = 17 > 13$ \checkmark, $5 + 13 = 18 > 12$ \checkmark, $12 + 13 = 25 > 5$ \checkmark. This is actually a right triangle ($5^2 + 12^2 = 13^2$).}

\trueOrFalse{A quadrilateral with all four sides known always has exactly one shape.}
\answerExplain{False}{Four sides do not determine all the angles. A quadrilateral can flex at its corners, creating many different shapes with the same four side lengths.}

\trueOrFalse{If two sides of a triangle are $10$ and $15$, the third side could be $25$.}
\answerExplain{False}{$10 + 15 = 25$, which is not greater than $25$. The sum must be strictly greater, so $25$ is not a valid third side.}
\end{practiceBox}

% ── Practice C: Find the Range ───────────────────────────────────────────────
\begin{practiceBox}[funTeal]{Possible Third Sides}

\practiceHeader[funTeal]{Two sides of a triangle are given. Find the range of possible lengths for the third side.}

\prob Sides: $5$ and $9$. Third side must be between \answerBlank[1.5cm] and \answerBlank[1.5cm].
\answerExplain{Between $4$ and $14$ (exclusive)}{The third side must be greater than $|9 - 5| = 4$ and less than $9 + 5 = 14$. So the third side is between $4$ and $14$, not including $4$ or $14$.}

\prob Sides: $7$ and $7$. Third side must be between \answerBlank[1.5cm] and \answerBlank[1.5cm].
\answerExplain{Between $0$ and $14$ (exclusive)}{The third side must be greater than $|7 - 7| = 0$ and less than $7 + 7 = 14$. Any positive length less than $14$ works.}

\prob Sides: $3$ and $10$. Third side must be between \answerBlank[1.5cm] and \answerBlank[1.5cm].
\answerExplain{Between $7$ and $13$ (exclusive)}{The third side must be greater than $|10 - 3| = 7$ and less than $10 + 3 = 13$.}

\prob Sides: $8$ and $15$. Third side must be between \answerBlank[1.5cm] and \answerBlank[1.5cm].
\answerExplain{Between $7$ and $23$ (exclusive)}{The third side must be greater than $|15 - 8| = 7$ and less than $15 + 8 = 23$.}
\end{practiceBox}

% ── Word Problems ────────────────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{A landscape designer wants to build a triangular garden with sides $4$ m, $6$ m, and $11$ m. Is this possible? Explain.}{m}
\answerExplain{Not possible}{Check the triangle inequality: $4 + 6 = 10$, but $10 < 11$. The two shorter sides are not long enough to connect around the longest side.}

\wordProblem{Marcus wants to build a rectangular picture frame with a perimeter of $36$ inches. He wants the length to be twice the width. What are the dimensions?}{inches}
\answerExplain{$6$ inches by $12$ inches}{Let width $= w$, length $= 2w$. Perimeter: $2(w + 2w) = 6w = 36$. So $w = 6$ and length $= 12$ inches.}

\wordProblem{Two sides of a triangular fence are $20$ ft and $30$ ft. The third side must be a whole number. What are the possible lengths?}{ft}
\answerExplain{$11, 12, 13, \ldots, 49$ ft}{The third side must be greater than $|30 - 20| = 10$ and less than $30 + 20 = 50$. Whole-number values range from $11$ to $49$.}
\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob You have three sticks. The longest is $15$ cm. The shortest is $4$ cm. What are all the whole-number lengths the middle stick could be so that the three sticks form a triangle?
\answerExplain{$12, 13, 14$ cm}{The middle stick $m$ must satisfy both: $4 + m > 15$ (so $m > 11$) and $4 + 15 > m$ (so $m < 19$). Also $m \leq 15$ since $15$ is the longest. Whole numbers: $12, 13, 14$.}
\end{challengeBox}
