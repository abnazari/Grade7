\small
\renewcommand{\arraystretch}{1.6}
\begin{tabular}{@{}c@{\hspace{4mm}}l@{\hspace{3mm}}p{10cm}@{}}
    \textcolor{funOrange}{\fontsize{20}{20}\selectfont\bfseries S} &
    \textbf{Study the question} &
    Read it fully. Identify what is given and what is being asked. \\
    \textcolor{funOrange}{\fontsize{20}{20}\selectfont\bfseries C} &
    \textbf{Choose a strategy} &
    Proportion? Equation? Estimation? Pick your tool before computing. \\
    \textcolor{funOrange}{\fontsize{20}{20}\selectfont\bfseries A} &
    \textbf{Apply and solve} &
    Show every step. Label units. Write the equation before plugging in. \\
    \textcolor{funOrange}{\fontsize{20}{20}\selectfont\bfseries N} &
    \textbf{Navigate back} &
    Re-read the question. Does your answer match what was asked? \\

% --- Green-Light Moves (full width) ---
\begin{tcolorbox}[
    enhanced,
    colback=funGreen!6,
    colframe=funGreen!40,
    arc=8pt, boxrule=1pt,
    left=4mm, right=4mm, top=3mm, bottom=3mm,
    title={\textcolor{white}{\faCheckCircle~Green-Light Moves}},
    colbacktitle=funGreen!65,
    fonttitle=\bfseries\sffamily,
]
    \small
    \begin{itemize}[leftmargin=*, itemsep=3pt]
        \item Read the \textit{entire} question before glancing at choices
        \item Circle key numbers; underline what is asked
        \item Convert fractions, decimals, and percents when it helps
        \item Use estimation to check reasonableness
        \item Skip tough problems and return with fresh eyes
    \end{itemize}
\end{tcolorbox}

\vspace{2mm}

% --- Red-Alert Traps (full width) ---
\begin{tcolorbox}[
    enhanced,
    colback=funRed!6,
    colframe=funRed!40,
    arc=8pt, boxrule=1pt,
    left=4mm, right=4mm, top=3mm, bottom=3mm,
    title={\textcolor{white}{\faTimesCircle~Red-Alert Traps}},
    colbacktitle=funRed!65,
    fonttitle=\bfseries\sffamily,
]
    \small
    \begin{itemize}[leftmargin=*, itemsep=3pt]
        \item Rushing through word problems
        \item Forgetting negative signs (rational numbers!)
        \item Confusing ``percent of'' with ``percent increase''
        \item Choosing an answer that solves only \textit{part} of the problem
        \item Leaving constructed-response blank instead of attempting it
    \end{itemize}
\end{tcolorbox}
