    \textcolor{musGold!90!black}{\fontsize{18}{18}\selectfont\bfseries C} &
    \textbf{Calculate} &
    Extract all given values, units, and relationships from the score. \\
    \textcolor{musGold!90!black}{\fontsize{18}{18}\selectfont\bfseries O} &
    \textbf{Operate} &
    Choose the right strategy. Set up the equation, proportion, or model. \\
    \textcolor{musGold!90!black}{\fontsize{18}{18}\selectfont\bfseries R} &
    \textbf{Resolve} &
    Execute step by step. Show every calculation and label units. \\
    \textcolor{musGold!90!black}{\fontsize{18}{18}\selectfont\bfseries E} &
    \textbf{Encore} &
    Review your answer. Does it make sense? Would you play it again the same way? \\

\newcommand{\stageDir}[3]{%
    \noindent
    \begin{minipage}[t]{0.06\linewidth}
        \centering
        \textcolor{#2}{\fontsize{16}{16}\selectfont\bfseries #1.}
    \end{minipage}%
    \begin{minipage}[t]{0.92\linewidth}
        \small #3
    \end{minipage}
    \vspace{2mm}
}

\stageDir{1}{musGold!90!black}{%
    \textbf{Sight-read twice.} The first pass gives you the tempo; the second reveals the melody.
}
\stageDir{2}{musBurgundy}{%
    \textbf{Annotate the score.} Circle given values. Underline the unknown. Mark rests (distractors).
}
\stageDir{3}{musBlack}{%
    \textbf{Hum before you play.} A quick mental estimate catches wrong notes before they hit the page.
}
\stageDir{4}{musGold!90!black}{%
    \textbf{Set up in rehearsal.} Write the equation or proportion \textit{before} performing calculations.
}
\stageDir{5}{musBurgundy}{%
    \textbf{Watch your sharps and flats.} Positive/negative sign errors are the \#1 wrong note in Grade~7.
}
\stageDir{6}{musBlack}{%
    \textbf{Eliminate wrong notes.} On multiple choice, cross out clearly wrong answers before selecting.
}
\stageDir{7}{musCrimson}{%
    \textbf{Never leave the stage silent.} Even a partial answer earns credit. Always play something.
}
