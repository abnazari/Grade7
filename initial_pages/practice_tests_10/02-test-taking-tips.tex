% --- Seven Precision Rules (numbered, clean) ---
\noindent\textcolor{funBlueDark}{\large\bfseries\faRulerCombined~Seven Precision Rules}

\vspace{2mm}

\newcommand{\precisionRule}[3]{%
    \noindent
    \begin{minipage}[t]{0.06\linewidth}
        \centering
        \textcolor{#2}{\fontsize{16}{16}\selectfont\bfseries #1.}
    \end{minipage}%
    \begin{minipage}[t]{0.92\linewidth}
        \small #3
    \end{minipage}
    \vspace{2mm}
}

\precisionRule{1}{funBlue}{%
    \textbf{Read twice, solve once.} The first read gives context; the second reveals what to compute.
}
\precisionRule{2}{funTeal}{%
    \textbf{Annotate the problem.} Circle given values. Underline the unknown. Cross out distractors.
}
\precisionRule{3}{funPurple}{%
    \textbf{Estimate before calculating.} A quick mental approximation catches major errors.
}
\precisionRule{4}{funGreen}{%
    \textbf{Set up before you solve.} Write the equation or proportion \textit{before} plugging in numbers.
}
\precisionRule{5}{funOrange}{%
    \textbf{Track your signs.} Rational number operations are the \#1 error source in Grade 7.
}
\precisionRule{6}{funPink}{%
    \textbf{Eliminate, then choose.} On multiple choice, cross out clearly wrong options first.
}
\precisionRule{7}{funRed}{%
    \textbf{Never submit blanks.} On open response, even a partial setup can earn credit.
}
