\newcommand{\seaRule}[3]{%
    \noindent
    \begin{minipage}[t]{0.06\linewidth}
        \centering
        \textcolor{#2}{\fontsize{16}{16}\selectfont\bfseries #1.}
    \end{minipage}%
    \begin{minipage}[t]{0.92\linewidth}
        \small #3
    \end{minipage}
    \vspace{2mm}
}

\seaRule{1}{navTeal}{%
    \textbf{Read the horizon twice.} The first read gives you context; the second reveals the destination.
}
\seaRule{2}{navGold!80!black}{%
    \textbf{Mark your chart.} Circle known values. Underline the unknown. Cross out distractors like debris.
}
\seaRule{3}{navDeep}{%
    \textbf{Estimate your position.} A quick mental approximation catches you before you drift off course.
}
\seaRule{4}{navTeal}{%
    \textbf{Plot before you sail.} Write the equation or proportion \textit{before} plugging in numbers.
}
\seaRule{5}{navGold!80!black}{%
    \textbf{Watch for crosscurrents.} Rational number signs are the \#1 error --- positive and negative mix-ups.
}
\seaRule{6}{navDeep}{%
    \textbf{Never abandon ship.} On open response, even a partial setup can earn credit. Always write something.
}
