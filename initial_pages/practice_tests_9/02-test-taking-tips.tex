    \textcolor{alpCrimson}{\fontsize{18}{18}\selectfont\bfseries C} &
    \textbf{Collect} &
    Read the problem carefully. Extract all given information: numbers, units, relationships. \\
    \textcolor{alpCrimson}{\fontsize{18}{18}\selectfont\bfseries L} &
    \textbf{Layout} &
    Organize your data. Draw a diagram, table, or number line if it helps. \\
    \textcolor{alpCrimson}{\fontsize{18}{18}\selectfont\bfseries I} &
    \textbf{Identify} &
    What operation or strategy fits? Choose the right formula or approach. \\
    \textcolor{alpCrimson}{\fontsize{18}{18}\selectfont\bfseries M} &
    \textbf{Math} &
    Solve step by step. Show every calculation. Label every unit. \\
    \textcolor{alpCrimson}{\fontsize{18}{18}\selectfont\bfseries B} &
    \textbf{Bridge} &
    Does the answer make sense? Bridge back to the question and verify. \\
\end{tabular}
\end{tcolorbox}

% --- Seven Alpine Rules ---
\noindent\textcolor{alpSlate}{\large\bfseries\faFlag~Seven Alpine Rules}

\vspace{2mm}

\newcommand{\alpineRule}[3]{%
    \noindent
    \begin{minipage}[t]{0.06\linewidth}
        \centering
        \textcolor{#2}{\fontsize{16}{16}\selectfont\bfseries #1.}
    \end{minipage}%
    \begin{minipage}[t]{0.92\linewidth}
        \small #3
    \end{minipage}
    \vspace{2mm}
}

\alpineRule{1}{alpPine}{%
    \textbf{Scout the trail twice.} First read for context, second read for the actual question.
}
\alpineRule{2}{alpSunrise!90!black}{%
    \textbf{Mark your footholds.} Circle given values. Underline the unknown. Cross out loose rocks.
}
\alpineRule{3}{alpCrimson}{%
    \textbf{Check your altitude.} A quick estimate catches you before you fall off-route.
}
\alpineRule{4}{alpPine}{%
    \textbf{Secure your rope.} Write the equation or proportion \textit{before} climbing into calculations.
}
\alpineRule{5}{alpSunrise!90!black}{%
    \textbf{Watch for ice patches.} Positives and negatives are the \#1 slip point in Grade 7.
}
\alpineRule{6}{alpCrimson}{%
    \textbf{Eliminate dead ends.} On multiple choice, cross out clearly wrong paths first.
}
\alpineRule{7}{alpSlate}{%
    \textbf{Never turn back empty-handed.} Even a partial setup earns credit. Always attempt every problem.
}
