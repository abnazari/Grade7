\wbSectionHead{How to Use This Workbook}{Four steps to get the most from every section.}

% ── Step Badges ───────────────────────────────────────────────────────────
\begin{center}
\begin{minipage}{0.84\textwidth}

\wbStepBadge{1}{funBlue}{Preview the Concept}{%
    Each section opens with a short reminder of the key idea or formula.
    If something feels unfamiliar, pair this workbook with the study
    guide for a deeper explanation before jumping into problems.}

\wbStepBadge{2}{funTeal}{Start Solving}{%
    Problems are arranged from basic to challenging. Begin with the
    easier ones to warm up your thinking, then work your way through
    the tougher questions. Don't skip around --- the order matters.}

\wbStepBadge{3}{funOrange}{Track Your Steps}{%
    Write every step on scratch paper. If you get a wrong answer,
    your written work helps you pinpoint exactly where the mistake
    happened --- so you can fix it and avoid it next time.}

\wbStepBadge{4}{funPurple}{Review and Correct}{%
    After finishing a section, flip to the Answer Key and check every
    problem. For each mistake, figure out \emph{why} it happened.
    Then try a similar problem again. That's how real learning works.}

\end{minipage}
\end{center}

    \vspace{3mm}

    \begin{itemize}[leftmargin=1.2em, itemsep=2pt]
        \item[\textcolor{funBlueDark}{\faChevronRight}]
            \textbf{Weekdays:} Complete one section per day
            (about 20--30 minutes).
        \item[\textcolor{funBlueDark}{\faChevronRight}]
            \textbf{End of week:} Re-do problems you missed during
            the week.
        \item[\textcolor{funBlueDark}{\faChevronRight}]
            \textbf{Weekend:} Rest, or try a bonus challenge for fun.
    \end{itemize}
