% ── Study Space Tips ──────────────────────────────────────────────────────
\begin{wbDashBox}{funOrange}
    {\bfseries\color{funOrangeDark}\faLightbulb\hspace{2mm}Study Space Tips}

    \vspace{3mm}

    \begin{itemize}[leftmargin=*, itemsep=3pt]
        \item Find a quiet spot with good light.
        \item Put your phone out of reach.
        \item Set a timer for 20--30 minutes, then take a short break.
        \item Keep a water bottle nearby.
    \end{itemize}
\end{wbDashBox}

\vspace{5mm}

% ── When to Use a Calculator ─────────────────────────────────────────────
\begin{wbDashBox}{funPurple}
    {\bfseries\color{funPurpleDark}\faCalculator\hspace{2mm}When to Use a Calculator}

    \vspace{3mm}

    \begin{itemize}[leftmargin=*, itemsep=3pt]
        \item Always try problems by hand first.
        \item Use a calculator to \emph{check} your work, not to replace it.
        \item For multi-step problems, a calculator helps verify tricky
              intermediate steps.
    \end{itemize}
\end{wbDashBox}
