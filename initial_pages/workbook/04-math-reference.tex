% --- Key Formulas ---
\begin{wbCallout}{funGreen}
    {\color{funGreen}\faCalculator}\;\;{\bfseries\color{black!80}Key Formulas}
    \par\vspace{3pt}
    \small
    \begin{multicols}{2}
    \begin{itemize}[leftmargin=*, itemsep=3pt]
        \item \textbf{Proportional relationship:}\\
            $y = kx$ \quad where $k = \dfrac{y}{x}$
        \item \textbf{Percent of a number:}\\
            $\text{part} = \text{percent} \times \text{whole}$
        \item \textbf{Percent change:}\\
            $\dfrac{|\text{new} - \text{original}|}{|\text{original}|} \times 100\%$
        \item \textbf{Simple interest:}\\
            $I = Prt$
        \item \textbf{Area of a circle:}\\
            $A = \pi r^{2}$
        \item \textbf{Circumference of a circle:}\\
            $C = 2\pi r = \pi d$
        \item \textbf{Area of a triangle:}\\
            $A = \dfrac{1}{2}\,b\,h$
        \item \textbf{Volume of a prism:}\\
            $V = B \times h$ \; ($B$ = area of base)
        \item \textbf{Probability:}\\
            $P(\text{event}) = \dfrac{\text{favorable outcomes}}{\text{total outcomes}}$
        \item \textbf{Scale factor:}\\
            $\dfrac{\text{drawing length}}{\text{actual length}}$
    \end{itemize}
    \end{multicols}
\end{wbCallout}

\vspace{3mm}

% --- Rational Number Rules ---
\begin{wbCallout}{funPurple}
    {\color{funPurple}\faBalanceScale}\;\;{\bfseries\color{black!80}Rational Number Rules}
    \par\vspace{3pt}
    \small
    \begin{multicols}{2}
    \begin{itemize}[leftmargin=*, itemsep=2pt]
        \item \textbf{Same signs} $\to$ add, keep the sign\\
            $(-3)+(-5) = -8$
        \item \textbf{Different signs} $\to$ subtract, keep the sign of the larger absolute value\\
            $(-9)+4 = -5$
        \item \textbf{Subtracting} = adding the opposite\\
            $6-(-2)=6+2=8$
        \item \textbf{Multiply / Divide:}\\
            Same signs $\to$ positive result\\
            Different signs $\to$ negative result
    \end{itemize}
    \end{multicols}
\end{wbCallout}

% --- Key Math Vocabulary ---
\begin{wbCallout}{funTeal}
    {\color{funTeal}\faBook}\;\;{\bfseries\color{black!80}Key Math Vocabulary}
    \par\vspace{3pt}
    \small
    \begin{multicols}{2}
    \begin{itemize}[leftmargin=*, itemsep=2pt]
        \item \textbf{Unit rate} — rate per \emph{one} unit
        \item \textbf{Proportion} — an equation stating two ratios are equal
        \item \textbf{Rational number} — any number that can be written as $\frac{a}{b}$ ($b \neq 0$)
        \item \textbf{Opposite} — same distance from $0$, other side
        \item \textbf{Like terms} — terms with the same variable and exponent
        \item \textbf{Coefficient} — the number multiplied by a variable
        \item \textbf{Inequality} — a statement using $<, >, \leq, \geq$
        \item \textbf{Complementary angles} — two angles summing to~$90\degree$
        \item \textbf{Supplementary angles} — two angles summing to~$180\degree$
        \item \textbf{Cross-section} — the face you see when slicing a 3D shape
        \item \textbf{Sample space} — all possible outcomes
        \item \textbf{Compound event} — two or more simple events combined
    \end{itemize}
    \end{multicols}
\end{wbCallout}

\vspace{3mm}
