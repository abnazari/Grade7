Ratio / Fraction &

Inequality &
$\geq$ means \emph{greater than or equal to};\; $\leq$ is the reverse. \\

{\fontsize{18}{20}\selectfont\textcolor{funRed}{$2x\!-\!4\!=\!10$}} &
Equation &
A statement that two expressions are \emph{equal}. \\

{\fontsize{18}{20}\selectfont\textcolor{funPink}{$P(\text{event})$}} &

A number from $0$ to $1$ measuring how likely an event is. \\

% ── Key Formulas ─────────────────────────────────────────────────────────
\begin{wbDashBox}{funGreen}
    {\color{funGreen}\faCalculator}\;\;{\bfseries\color{black!80}Key Formulas}
    \par\vspace{3pt}
    \small
    \begin{multicols}{2}
    \begin{itemize}[leftmargin=*, itemsep=3pt]
        \item \textbf{Proportional relationship:}\\
            $y = kx$ \quad where $k = \dfrac{y}{x}$
        \item \textbf{Percent of a number:}\\
            $\text{part} = \text{percent} \times \text{whole}$
        \item \textbf{Percent change:}\\
            $\dfrac{|\text{new} - \text{original}|}{|\text{original}|} \times 100\%$
        \item \textbf{Simple interest:}\\
            $I = Prt$
        \item \textbf{Area of a circle:}\\
            $A = \pi r^{2}$
        \item \textbf{Circumference:}\\
            $C = 2\pi r = \pi d$
        \item \textbf{Area of a triangle:}\\
            $A = \dfrac{1}{2}\,b\,h$
        \item \textbf{Volume of a prism:}\\
            $V = B \times h$ \; ($B$ = area of base)
        \item \textbf{Probability:}\\
            $P = \dfrac{\text{favorable outcomes}}{\text{total outcomes}}$
        \item \textbf{Scale factor:}\\
            $\dfrac{\text{drawing length}}{\text{actual length}}$
    \end{itemize}
    \end{multicols}
\end{wbDashBox}

\vspace{3mm}

% ── Rational Number Rules ────────────────────────────────────────────────
\begin{wbDashBox}{funPurple}
    {\color{funPurple}\faBalanceScale}\;\;{\bfseries\color{black!80}Rational Number Rules}
    \par\vspace{3pt}
    \small
    \begin{multicols}{2}
    \begin{itemize}[leftmargin=*, itemsep=2pt]
        \item \textbf{Same signs} $\to$ add, keep the sign\\
            $(-3)+(-5) = -8$
        \item \textbf{Different signs} $\to$ subtract, keep the sign
            of the larger absolute value\\
            $(-9)+4 = -5$
        \item \textbf{Subtracting} = adding the opposite\\
            $6-(-2)=6+2=8$
        \item \textbf{Multiply / Divide:}\\
            Same signs $\to$ positive\\
            Different signs $\to$ negative
    \end{itemize}
    \end{multicols}
\end{wbDashBox}
