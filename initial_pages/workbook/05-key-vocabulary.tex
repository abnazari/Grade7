% --- Proportional Relationships ---
{\normalsize\bfseries\color{funBlueDark}Proportional Relationships\par}
\vspace{1.5mm}
\wbVocabWord{Ratio}{A comparison of two quantities.}
\wbVocabWord{Proportion}{An equation stating two ratios are equal.}
\wbVocabWord{Unit rate}{A rate with a denominator of 1.}
\wbVocabWord{Constant of proportionality}{The value $k$ in $y = kx$.}

\vspace{3mm}

% --- Percents ---
{\normalsize\bfseries\color{funBlueDark}Percents\par}
\vspace{1.5mm}
\wbVocabWord{Percent}{A ratio that compares a number to 100.}
\wbVocabWord{Markup}{The extra amount a store adds to cost.}
\wbVocabWord{Discount}{A reduction from the original price.}
\wbVocabWord{Simple interest}{Interest calculated on the principal only: $I = Prt$.}
\wbVocabWord{Percent error}{How far an estimate is from the actual value, as a percent.}

\vspace{3mm}

% --- Rational Numbers ---
{\normalsize\bfseries\color{funBlueDark}Rational Numbers\par}
\vspace{1.5mm}
\wbVocabWord{Integer}{A whole number or its opposite (\ldots $-2, -1, 0, 1, 2$ \ldots).}
\wbVocabWord{Rational number}{Any number that can be written as $\frac{a}{b}$ ($b \neq 0$).}
\wbVocabWord{Absolute value}{The distance of a number from zero on a number line.}
\wbVocabWord{Opposite}{Two numbers the same distance from zero on opposite sides.}
\wbVocabWord{Additive inverse}{A number and its opposite sum to $0$.}
\wbVocabWord{Terminating decimal}{A decimal that ends: $0.75$.}
\wbVocabWord{Repeating decimal}{A decimal with a repeating pattern: $0.\overline{3}$.}

\vspace{3mm}

% --- Expressions & Equations ---
{\normalsize\bfseries\color{funBlueDark}Expressions \& Equations\par}
\vspace{1.5mm}
\wbVocabWord{Variable}{A letter that represents an unknown number.}
\wbVocabWord{Coefficient}{The number multiplied by a variable ($6$ in $6x$).}
\wbVocabWord{Constant}{A fixed number with no variable ($+3$ in $2x+3$).}
\wbVocabWord{Like terms}{Terms with the same variable and exponent.}
\wbVocabWord{Expression}{Numbers, variables, and operations (no $=$ sign).}
\wbVocabWord{Equation}{A statement that two expressions are equal.}
\wbVocabWord{Inequality}{A statement that uses $<, >, \leq$, or $\geq$.}
\wbVocabWord{Distributive property}{$a(b+c) = ab + ac$.}

\columnbreak

% --- Geometry ---
{\normalsize\bfseries\color{funBlueDark}Geometry \& Measurement\par}
\vspace{1.5mm}
\wbVocabWord{Scale drawing}{A proportional representation of an object.}
\wbVocabWord{Scale factor}{The ratio of drawing length to actual length.}
\wbVocabWord{Radius}{The distance from the center to the edge of a circle.}
\wbVocabWord{Diameter}{The distance across a circle through its center ($d = 2r$).}
\wbVocabWord{Circumference}{The distance around a circle: $C = \pi d$.}
\wbVocabWord{Area}{The space inside a flat shape, in square units.}
\wbVocabWord{Surface area}{The total area of all faces of a 3-D shape.}
\wbVocabWord{Volume}{The space inside a 3-D shape, in cubic units.}
\wbVocabWord{Cross-section}{The 2-D shape you see when slicing a 3-D figure.}
\wbVocabWord{Complementary angles}{Two angles that sum to $90^\circ$.}
\wbVocabWord{Supplementary angles}{Two angles that sum to $180^\circ$.}
\wbVocabWord{Vertical angles}{Opposite angles formed by intersecting lines; always equal.}

\vspace{3mm}

% --- Statistics ---
{\normalsize\bfseries\color{funBlueDark}Statistics\par}
\vspace{1.5mm}
\wbVocabWord{Population}{The entire group you want information about.}
\wbVocabWord{Sample}{A smaller group chosen from the population.}
\wbVocabWord{Random sample}{A sample where every member has an equal chance of being chosen.}
\wbVocabWord{Mean}{The average: sum of values divided by the count.}
\wbVocabWord{Median}{The middle value when data is sorted.}
\wbVocabWord{MAD}{Mean Absolute Deviation --- average distance from the mean.}

\vspace{3mm}

% --- Probability ---
{\normalsize\bfseries\color{funBlueDark}Probability\par}
\vspace{1.5mm}
\wbVocabWord{Probability}{A number from $0$ to $1$ measuring how likely an event is.}
\wbVocabWord{Theoretical probability}{Probability based on reasoning, not experiments.}
\wbVocabWord{Experimental probability}{Probability based on observed data.}
\wbVocabWord{Sample space}{The set of all possible outcomes.}
\wbVocabWord{Compound event}{An event made of two or more simple events.}
\wbVocabWord{Simulation}{An experiment that models a real-world situation.}

\end{multicols}
