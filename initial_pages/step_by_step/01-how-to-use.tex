% ── Arrow Steps ───────────────────────────────────────────────────────────
\begin{center}
\begin{minipage}{0.84\textwidth}

\stArrowStep{1}{funOrange}{Read the Steps}{%
    Each topic opens with a numbered list of steps that tells you
    \emph{exactly} what to do. Read through the steps first so you
    have the full picture before diving into problems.}

\stArrowStep{2}{funBlue}{Follow the Examples}{%
    Worked examples show every step applied to a real problem.
    Follow along with pencil and paper --- try covering the solution
    and solving it yourself before looking.}

\stArrowStep{3}{funTeal}{Try the Guided Practice}{%
    ``Your Turn'' problems give you blanks to fill in as you
    apply each step. This bridges the gap between watching
    and doing.}

\stArrowStep{4}{funPurple}{Practise on Your Own}{%
    Each section ends with independent practice problems. Write out
    all your steps, then check answers in the back. When you find a
    mistake, figure out \emph{where} your reasoning went wrong and
    try again.}

\end{minipage}
\end{center}
