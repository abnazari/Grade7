% --- Step-by-Step Guide ---
\begin{tcolorbox}[
    enhanced,
    colback=funBlueLight,
    colframe=funBlue!50,
    arc=8pt, boxrule=1pt,
    left=5mm, right=5mm, top=3mm, bottom=3mm,
    title={\textcolor{white}{\faClipboardCheck~How to Take a Practice Test}},
    colbacktitle=funBlue!70,
    fonttitle=\bfseries\sffamily,
]
    \small
    \begin{enumerate}[leftmargin=*, itemsep=3pt]
        \item \textbf{Find a quiet spot} with no distractions. Clear your desk.
        \item \textbf{Gather your supplies}: pencil, eraser, and scratch paper.
        \item \textbf{Set a timer} if you want to practice with a time limit (ask a grown-up
            to help!).
        \item \textbf{Read each question carefully} --- don't rush!
        \item \textbf{Try your best} on every question. Skip hard ones and come back later.
        \item \textbf{When you're done}, use the Answer Key to check your work.
        \item \textbf{Read the explanations} for any questions you missed.
    \end{enumerate}
\end{tcolorbox}
