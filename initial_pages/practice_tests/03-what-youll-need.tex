% --- Tips for parents/teachers ---
\begin{tcolorbox}[
    enhanced,
    colback=funCream,
    colframe=funYellow!60,
    arc=10pt,
    outer arc=10pt,
    boxrule=1.5pt,
    left=6mm, right=6mm, top=4mm, bottom=4mm,
    title={\textcolor{funYellowDark}{\faUsers}~\textbf{A Note for Parents \& Teachers}},
    colbacktitle=funYellow!30,
    coltitle=funYellowDark,
    fonttitle=\sffamily\bfseries,
]
    \sffamily\small
    \begin{itemize}[leftmargin=*, itemsep=3pt]
        \item Simulate real test conditions: quiet room, timed sessions, no help.
        \item \textbf{Don't do all tests at once.} Space them out over several weeks.
        \item After each test, review the Answer Key and explanations together.
        \item Focus on \textbf{understanding}, not just scores.
        \item If your child struggles with a topic, use our \textbf{Grade 7 Math Study Guide}
            to review that concept before the next practice test.
        \item Celebrate effort and improvement, not just perfect scores!
    \end{itemize}
\end{tcolorbox}
