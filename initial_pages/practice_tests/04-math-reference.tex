% --- Key Math Vocabulary ---
\begin{tcolorbox}[
    enhanced,
    colback=funTealLight,
    colframe=funTeal!50,
    arc=10pt,
    boxrule=1.5pt,
    left=5mm, right=5mm, top=4mm, bottom=4mm,
    title={\textcolor{white}{\faBook~Key Math Vocabulary}},
    colbacktitle=funTeal!70,
    fonttitle=\bfseries\sffamily,
]
    \small
    \begin{multicols}{2}
    \begin{itemize}[leftmargin=*, itemsep=2pt]
        \item \textbf{Sum} --- the result of addition
        \item \textbf{Difference} --- the result of subtraction
        \item \textbf{Product} --- the result of multiplication
        \item \textbf{Quotient} --- the result of division
        \item \textbf{Remainder} --- what's left over after dividing
        \item \textbf{Factor} --- a number you multiply
        \item \textbf{Expression} --- numbers and operations without $=$
        \item \textbf{Equation} --- a math sentence with $=$
        \item \textbf{Numerator} --- the top number of a fraction
        \item \textbf{Denominator} --- the bottom number of a fraction
        \item \textbf{Mixed number} --- a whole number + a fraction
        \item \textbf{Equivalent fractions} --- fractions with equal value
        \item \textbf{Decimal} --- a number written with a decimal point
        \item \textbf{Volume} --- the space inside a 3D shape
        \item \textbf{Coordinate plane} --- a grid with $x$ and $y$ axes
        \item \textbf{Ordered pair} --- $(x, y)$ location on the plane
    \end{itemize}
    \end{multicols}
\end{tcolorbox}
