% Row macro: one test entry with colored number circle + write lines
\newcount\testrow
\def\makeScoreRow{%
    % Pick colors before entering tikz to avoid alignment shifts
    \ifodd\testrow
        \colorlet{rowFill}{funGreen!20}%
        \colorlet{rowFrame}{funGreen!50}%
        \colorlet{rowText}{funGreenDark}%
    \else
        \colorlet{rowFill}{funBlue!15}%
        \colorlet{rowFrame}{funBlue!40}%
        \colorlet{rowText}{funBlueDark}%
    \fi
    \par\vspace{4mm}\noindent
    \hspace{4mm}%
    \makebox[3cm][c]{%
        \tikz[baseline=(n.base)]{%
            \node[circle, fill=rowFill, draw=rowFrame, inner sep=0pt, minimum size=8mm] (n)
                {\bfseries\textcolor{rowText}{\the\testrow}};
        }%
    }%
    \hfill
    \makebox[4cm][c]{\textcolor{funGray!60}{\rule[-2pt]{3.2cm}{0.4pt}}}%
    \hfill
    \makebox[3.5cm][c]{\textcolor{funGray!60}{\rule[-2pt]{1.2cm}{0.4pt}~/~\rule[-2pt]{1.2cm}{0.4pt}}}%
    \hfill
    \makebox[3cm][c]{\textcolor{funGray!30}{\Large\faSmile}}%
    \hspace{4mm}%
    \par
    \vspace{3mm}
    \noindent\textcolor{funBlue!15}{\rule{\linewidth}{0.4pt}}%
    \global\advance\testrow by 1\relax
}

\testrow=1
\loop
    \makeScoreRow
\ifnum\testrow<\numexpr\NumPracticeTests+1\relax
\repeat
