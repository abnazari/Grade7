% --- During the Test ---
\begin{tcolorbox}[
    enhanced,
    colback=funBlueLight,
    colframe=funBlue!50,
    arc=10pt,
    outer arc=10pt,
    boxrule=1.5pt,
    left=5mm, right=5mm, top=4mm, bottom=4mm,
    title={\textcolor{white}{\faPen~During the Test}},
    colbacktitle=funBlue!70,
    fonttitle=\bfseries\sffamily,
]
    \small
    \renewcommand{\labelenumi}{\textcolor{funBlueDark}{\textbf{\arabic{enumi}.}}}
    \begin{enumerate}[leftmargin=*, itemsep=4pt]
        \item \textbf{Read each question twice.}
            Make sure you understand what it's asking before you answer.
        \item \textbf{Look for key words.}
            Words like \textit{``how many more,''} \textit{``in all,''} and
            \textit{``each''} tell you what to do.
        \item \textbf{Show your work.}
            Write out your steps --- it helps you think clearly and catch mistakes.
        \item \textbf{Use scratch paper.}
            Don't try to do everything in your head!
        \item \textbf{Skip and come back.}
            If a question is hard, put a small mark next to it and move on.
            Come back to it later.
        \item \textbf{Check your answers.}
            If you finish early, go back and double-check your work.
    \end{enumerate}
\end{tcolorbox}

\begin{minipage}[t]{0.48\textwidth}
\begin{tcolorbox}[

    arc=8pt, boxrule=1pt,

    title={\textcolor{white}{\faCircle~Multiple Choice Tricks}},
    colbacktitle=funOrange!70,
    fonttitle=\bfseries\sffamily,
    height=5.5cm,
    valign=center,
]
    \small
    \begin{itemize}[leftmargin=*, itemsep=3pt]
        \item Read \textbf{all} four choices before picking one
        \item \textbf{Cross out} answers you know are wrong
        \item \textbf{Plug in} your answer to check if it works
        \item If two answers look close, re-read the question
        \item \textbf{Never leave a question blank!} Make your best guess
    \end{itemize}
\end{tcolorbox}
\end{minipage}%
