\sgFormulaCard{funTeal}{Proportional Relationships}{%
    $y = kx$, where $k = \dfrac{y}{x}$\\[3pt]
    The constant of proportionality $k$ is the unit rate.
    The graph is a straight line through the origin.}

\sgFormulaCard{funBlue}{Percent Problems}{%
    $\text{part} = \text{percent} \times \text{whole}$\\[3pt]
    \textbf{Percent change:}
    $\dfrac{|\text{new} - \text{original}|}{|\text{original}|} \times 100\%$\\[3pt]
    \textbf{Simple interest:} $I = Prt$}

\sgFormulaCard{funGreen}{Rational Number Rules}{%
    \textbf{Same signs} $\to$ add, keep the sign\\
    \textbf{Different signs} $\to$ subtract, keep larger sign\\
    \textbf{Subtract} = add the opposite\\
    \textbf{Multiply/Divide:} same signs $\to$ $+$, different $\to$ $-$}

\sgFormulaCard{funOrange}{Expressions \& Equations}{%
    \textbf{Distributive Property:}
    $a(b + c) = ab + ac$\\[3pt]
    \textbf{Two-step equation:}
    $ax + b = c$\\
    Solve: subtract $b$, then divide by $a$.}

\sgFormulaCard{funPurple}{Inequalities}{%
    Solve like equations, but \textbf{flip the sign}
    when multiplying or dividing by a negative.\\[3pt]
    $x > 3$ \;\;$\to$\;\; open circle at $3$\\
    $x \geq 3$ \;\;$\to$\;\; closed circle at $3$}

\sgFormulaCard{funRed}{Geometry \& Scale}{%
    \textbf{Scale factor:}
    $\dfrac{\text{drawing length}}{\text{actual length}}$\\[3pt]
    \textbf{Supplementary:} sum $= 180°$\\
    \textbf{Complementary:} sum $= 90°$\\
    \textbf{Vertical angles} are equal.}

\sgFormulaCard{funPink}{Circles, Area \& Volume}{%
    \textbf{Circumference:}
    $C = 2\pi r = \pi d$\\[3pt]
    \textbf{Area of circle:} $A = \pi r^{2}$\\[3pt]
    \textbf{Area of triangle:} $A = \tfrac{1}{2}\,b\,h$\\[3pt]
    \textbf{Volume of prism:} $V = B \times h$}

\sgFormulaCard{funBlue}{Probability \& Statistics}{%
    $P(\text{event}) = \dfrac{\text{favorable outcomes}}{\text{total outcomes}}$\\[4pt]
    $0 \leq P \leq 1$\\[3pt]
    \textbf{Compound events:} use tree diagrams, tables,
    or organised lists to find the sample space.}

\end{multicols}
