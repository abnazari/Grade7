\begin{practiceQuestion}{m-6-6-q01}{mc}
\begin{questionText}
Two supplementary angles measure $(4x + 12)¬į$ and $(2x + 48)¬į$. What is the value of $x$?
\end{questionText}
\choiceA{$15$}
\choiceB{$20$}
\choiceC{$24$}
\choiceD{$30$}
\correctAnswer{B}
\explanation{Supplementary angles sum to $180¬į$: $(4x + 12) + (2x + 48) = 180$. Combine: $6x + 60 = 180$. Subtract $60$: $6x = 120$. Divide: $x = 20$.}
\end{practiceQuestion}

\end{questionText}
\choiceA{$14¬į$}
\choiceB{$34¬į$}
\choiceC{$56¬į$}

\explanation{Complementary angles sum to $90¬į$: $(3x - 7) + (x + 13) = 90$. Combine: $4x + 6 = 90$. Solve: $4x = 84$, $x = 21$. Angles: $3(21) - 7 = 56¬į$ and $21 + 13 = 34¬į$. The smaller angle is $34¬į$.}
\end{practiceQuestion}

\correctAnswer{$x = 15$; $\angle A = 50¬į$, $\angle C = 45¬į$, $\angle B = 85¬į$, exterior angle $= 95¬į$}

Exterior angle $= 7(15) - 10 = 95¬į$. Verify: $50 + 45 = 95$ \checkmark.}
\end{practiceQuestion}
