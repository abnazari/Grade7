\begin{practiceQuestion}{2-5-q01}{mc}
\begin{questionText}
A family's dinner bill is $\$65$. They leave a $20\%$ tip. How much is the tip?
\end{questionText}
\choiceA{$\$6.50$}
\choiceB{$\$10.00$}
\choiceC{$\$13.00$}
\choiceD{$\$15.00$}
\correctAnswer{C}
\explanation{Convert: $20\% = 0.20$. Tip: $65 \times 0.20 = \$13.00$.}
\end{practiceQuestion}

\begin{practiceQuestion}{2-5-q13}{mc}
\begin{questionText}
A waiter received tips of $\$8$, $\$12$, $\$15$, and $\$5$ on four tables whose bills were $\$40$, $\$60$, $\$100$, and $\$25$ respectively. What was the waiter's overall tip percentage?
\end{questionText}
\choiceA{$15\%$}
