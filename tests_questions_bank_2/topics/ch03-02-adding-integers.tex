\begin{practiceQuestion}{3-2-q01}{mc}
\begin{questionText}
The temperature at dawn was $-7°$C. By afternoon it had risen $12°$C. What was the afternoon temperature?
\end{questionText}
\choiceA{$-19°$C}
\choiceB{$-5°$C}
\choiceC{$5°$C}
\choiceD{$19°$C}
\correctAnswer{C}
\explanation{Different signs: subtract $12 - 7 = 5$. The larger absolute value ($12$) is positive, so $(-7) + 12 = 5°$C.}
\end{practiceQuestion}
