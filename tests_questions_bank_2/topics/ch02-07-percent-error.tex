\begin{practiceQuestion}{2-7-q01}{mc}
\begin{questionText}
A student estimates a rope is $8$ feet long. The actual length is $10$ feet. What is the percent error?
\end{questionText}
\choiceA{$2\%$}
\choiceB{$10\%$}
\choiceC{$20\%$}
\choiceD{$25\%$}
\correctAnswer{C}
\explanation{$\text{Percent error} = \dfrac{|8 - 10|}{10} \times 100 = \dfrac{2}{10} \times 100 = 20\%$.}
\end{practiceQuestion}

\end{questionText}

\choiceB{The actual value}
\choiceC{The difference between the values}
