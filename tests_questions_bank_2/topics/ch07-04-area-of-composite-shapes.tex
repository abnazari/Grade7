\begin{practiceQuestion}{7-4-q01}{mc}
\begin{questionText}
A composite shape is made of a rectangle that is $12$ cm by $5$ cm and a right triangle with legs $5$ cm and $6$ cm attached to one end. What is the total area?
\end{questionText}
\choiceA{$60$ cm$^2$}
\choiceB{$75$ cm$^2$}
\choiceC{$90$ cm$^2$}
\choiceD{$120$ cm$^2$}
\correctAnswer{B}
\explanation{Rectangle: $12 \times 5 = 60$ cm$^2$. Triangle: $\frac{1}{2} \times 5 \times 6 = 15$ cm$^2$. Total: $60 + 15 = 75$ square centimeters.}
\end{practiceQuestion}
