\begin{practiceQuestion}{3-8-q01}{mc}
\begin{questionText}
A baker has $5\dfrac{1}{2}$ pounds of flour. She uses $2\dfrac{3}{4}$ pounds for bread and $1\dfrac{1}{4}$ pounds for muffins. How much flour remains?
\end{questionText}
\choiceA{$1\dfrac{1}{2}$ pounds}
\choiceB{$1\dfrac{1}{4}$ pounds}
\choiceC{$2$ pounds}
\choiceD{$\dfrac{3}{4}$ pound}
\correctAnswer{A}
\explanation{Used: $2\dfrac{3}{4} + 1\dfrac{1}{4} = 4$ pounds. Remaining: $5\dfrac{1}{2} - 4 = 1\dfrac{1}{2}$ pounds.}
\end{practiceQuestion}

\begin{practiceQuestion}{3-8-q02}{mc}
\begin{questionText}
The temperature at $6$ a.m. was $-8.5°$F. By noon it rose $15.3°$F, then dropped $4.2°$F by $6$ p.m. What was the temperature at $6$ p.m.?
\end{questionText}
\choiceA{$2.6°$F}
\choiceB{$-2.6°$F}
\choiceC{$11.1°$F}
\choiceD{$28.0°$F}
\correctAnswer{A}
\explanation{$-8.5 + 15.3 = 6.8$. Then $6.8 - 4.2 = 2.6°$F.}
\end{practiceQuestion}

\choiceA{$60$ feet}

\begin{practiceQuestion}{3-8-q07}{mc}

A recipe serves $4$ people and calls for $\dfrac{2}{3}$ cup of sugar. How much sugar is needed to serve $10$ people?

\choiceB{$1\dfrac{2}{3}$ cups}

\begin{practiceQuestion}{3-8-q14}{mc}
\begin{questionText}
The water level in a reservoir changes by $-2.4$ inches per week for $5$ weeks, then rises $3.6$ inches per week for $3$ weeks. What is the overall change?
\end{questionText}
\choiceA{$-1.2$ inches}
\choiceB{$1.2$ inches}
\choiceC{$-22.8$ inches}
\choiceD{$22.8$ inches}

\begin{practiceQuestion}{3-8-q15}{mc}
\begin{questionText}
A pizza weighing $1\dfrac{1}{2}$ pounds is cut into $6$ equal slices. How much does each slice weigh?
\end{questionText}
\choiceA{$\dfrac{1}{6}$ pound}
\choiceB{$\dfrac{1}{4}$ pound}
\choiceC{$\dfrac{1}{3}$ pound}
\choiceD{$\dfrac{3}{4}$ pound}
\correctAnswer{B}
\explanation{$1\dfrac{1}{2} \div 6 = \dfrac{3}{2} \div 6 = \dfrac{3}{2} \times \dfrac{1}{6} = \dfrac{3}{12} = \dfrac{1}{4}$ pound.}
\end{practiceQuestion}
