\begin{practiceQuestion}{2-4-q01}{mc}
\begin{questionText}
A store buys sunglasses for $\$15$ and marks them up $60\%$. What is the selling price?
\end{questionText}
\choiceA{$\$9$}
\choiceB{$\$21$}
\choiceC{$\$24$}
\choiceD{$\$75$}
\correctAnswer{C}
\explanation{A $60\%$ markup means the selling price is $1.60$ times the cost: $15 \times 1.60 = \$24$.}
\end{practiceQuestion}
