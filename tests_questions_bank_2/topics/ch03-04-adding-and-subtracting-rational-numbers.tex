\begin{practiceQuestion}{3-4-q01}{mc}
\begin{questionText}
What is $\dfrac{3}{8} + \left(-\dfrac{7}{8}\right)$?
\end{questionText}
\choiceA{$\dfrac{10}{8}$}
\choiceB{$-\dfrac{10}{8}$}
\choiceC{$\dfrac{4}{8}$}
\choiceD{$-\dfrac{4}{8}$}
\correctAnswer{D}
\explanation{Same denominator: $\dfrac{3 + (-7)}{8} = \dfrac{-4}{8} = -\dfrac{1}{2}$. Different signs: $7 - 3 = 4$, keep the sign of the larger absolute value (negative).}
\end{practiceQuestion}

\begin{practiceQuestion}{3-4-q27}{gsa}

\begin{center}
