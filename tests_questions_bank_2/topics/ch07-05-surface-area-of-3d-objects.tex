\begin{practiceQuestion}{7-5-q01}{mc}
\begin{questionText}
What is the surface area of a cube with side length $7$ cm?
\end{questionText}
\choiceA{$49$ cm$^2$}
\choiceB{$147$ cm$^2$}
\choiceC{$294$ cm$^2$}
\choiceD{$343$ cm$^2$}
\correctAnswer{C}
\explanation{A cube has $6$ identical square faces. $SA = 6s^2 = 6 \times 7^2 = 6 \times 49 = 294$ square centimeters.}
\end{practiceQuestion}

\begin{practiceQuestion}{7-5-q02}{mc}

\begin{practiceQuestion}{7-5-q10}{mc}
\begin{questionText}
Two identical cubes, each with side $4$ cm, are glued together along one face. What is the surface area of the new solid?
\end{questionText}
\choiceA{$96$ cm$^2$}
\choiceB{$128$ cm$^2$}
\choiceC{$160$ cm$^2$}
\choiceD{$192$ cm$^2$}
\correctAnswer{C}
\explanation{Each cube has $SA = 6 \times 16 = 96$ cm$^2$, so two cubes have $192$ cm$^2$. The two glued faces ($2 \times 16 = 32$ cm$^2$) are hidden. New $SA = 192 - 32 = 160$ square centimeters.}
\end{practiceQuestion}

\begin{practiceQuestion}{7-5-q11}{mc}
\begin{questionText}
A triangular prism has two right-triangle bases with legs $3$ cm and $4$ cm (hypotenuse $5$ cm). The prism is $10$ cm long. What is the total surface area?
\end{questionText}
\choiceA{$108$ cm$^2$}
\choiceB{$120$ cm$^2$}
\choiceC{$132$ cm$^2$}
\choiceD{$144$ cm$^2$}
\correctAnswer{C}
\explanation{Two triangles: $2 \times \frac{1}{2}(3)(4) = 12$ cm$^2$. Three rectangles: $3 \times 10 = 30$, $4 \times 10 = 40$, $5 \times 10 = 50$. Total: $12 + 30 + 40 + 50 = 132$ square centimeters.}
\end{practiceQuestion}

\begin{practiceQuestion}{7-5-q15}{mc}
\begin{questionText}
If a cube's edge length is doubled, how does its surface area change?
\end{questionText}
\choiceA{It doubles.}
\choiceB{It triples.}
\choiceC{It quadruples.}
\choiceD{It increases by $8$ times.}
\correctAnswer{C}
\explanation{Original: $SA = 6s^2$. New: $SA = 6(2s)^2 = 6 \times 4s^2 = 24s^2$. The ratio is $24s^2 \div 6s^2 = 4$. The surface area quadruples.}
\end{practiceQuestion}
