\begin{practiceQuestion}{7-6-q01}{mc}
\begin{questionText}
What is the volume of a rectangular prism with length $8$ cm, width $5$ cm, and height $4$ cm?
\end{questionText}
\choiceA{$17$ cm$^3$}
\choiceB{$40$ cm$^3$}
\choiceC{$80$ cm$^3$}
\choiceD{$160$ cm$^3$}
\correctAnswer{D}
\explanation{The volume of a rectangular prism is $V = lwh = 8 \times 5 \times 4 = 160$ cubic centimeters.}
\end{practiceQuestion}

\begin{practiceQuestion}{7-6-q05}{mc}
\begin{questionText}
A rectangular box has a volume of $240$ in$^3$. Its length is $8$ in and width is $5$ in. What is its height?
\end{questionText}
\choiceA{$3$ in}
\choiceB{$6$ in}
\choiceC{$8$ in}
\choiceD{$12$ in}
\correctAnswer{B}
\explanation{Use $V = lwh$. Solve for $h$: $h = V \div (l \times w) = 240 \div (8 \times 5) = 240 \div 40 = 6$ inches.}
\end{practiceQuestion}

\begin{practiceQuestion}{7-6-q12}{mc}
\begin{questionText}
If the length of a rectangular prism is tripled while the width and height stay the same, the volume is:
\end{questionText}
\choiceA{tripled}
\choiceB{doubled}
\choiceC{the same}
\choiceD{nine times as large}
\correctAnswer{A}
\explanation{$V = lwh$. If $l$ becomes $3l$, then $V_{new} = 3l \times w \times h = 3(lwh) = 3V$. The volume triples.}
\end{practiceQuestion}
