  \draw[thick, fill=funBlue!10] (0,0) rectangle (4,4);
  \node at (2,2) {Area $= n^8$};
  \node[below] at (2,0) {$s = ?$};

\choiceC{$n^{16}$}
\choiceD{$n^6$}
