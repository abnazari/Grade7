\begin{practiceQuestion}{6-7-q01}{mc}
\begin{questionText}
Point $P(4, -3)$ is translated left $6$ and up $5$. What are the coordinates of $P'$?
\end{questionText}
\choiceA{$(10, 2)$}
\choiceB{$(-2, 2)$}
\choiceC{$(-2, -8)$}
\choiceD{$(10, -8)$}
\correctAnswer{B}
\explanation{Translate left $6$: $4 - 6 = -2$. Translate up $5$: $-3 + 5 = 2$. So $P'(-2, 2)$.}
\end{practiceQuestion}
