\begin{practiceQuestion}{3-10-q01}{mc}
\begin{questionText}
A recipe calls for $\dfrac{3}{4}$ cup of oil. If you triple the recipe but then pour out $\dfrac{1}{2}$ cup, how much oil remains?
\end{questionText}
\choiceA{$1\dfrac{1}{2}$ cups}
\choiceB{$1\dfrac{3}{4}$ cups}
\choiceC{$2$ cups}
\choiceD{$2\dfrac{1}{4}$ cups}
\correctAnswer{B}
\explanation{Triple: $\dfrac{3}{4} \times 3 = \dfrac{9}{4} = 2\dfrac{1}{4}$. Pour out: $2\dfrac{1}{4} - \dfrac{1}{2} = 2\dfrac{1}{4} - \dfrac{2}{4} = 1\dfrac{3}{4}$ cups.}
\end{practiceQuestion}

\begin{practiceQuestion}{3-10-q02}{mc}
\begin{questionText}
A store discounts an item by $25\%$, then adds $8\%$ sales tax on the discounted price. If the original price is $\$80$, what is the final cost?
\end{questionText}
\choiceA{$\$60.00$}
\choiceB{$\$64.80$}
\choiceC{$\$66.40$}
\choiceD{$\$68.00$}
\correctAnswer{B}
\explanation{Discount: $80 \times 0.25 = 20$. Sale price: $80 - 20 = 60$. Tax: $60 \times 0.08 = 4.80$. Final: $60 + 4.80 = \$64.80$.}
\end{practiceQuestion}

\begin{practiceQuestion}{3-10-q03}{mc}
\begin{questionText}
The temperature at midnight was $-12.5°$C. It rose $4.8°$C by $6$ a.m., dropped $7.3°$C by noon, and rose $9.5°$C by $6$ p.m. What was the temperature at $6$ p.m.?
\end{questionText}
\choiceA{$-5.5°$C}
\choiceB{$5.5°$C}
\choiceC{$-14.5°$C}
\choiceD{$-34.1°$C}
\correctAnswer{A}
\explanation{$-12.5 + 4.8 = -7.7$. Then $-7.7 - 7.3 = -15.0$. Then $-15.0 + 9.5 = -5.5°$C.}
\end{practiceQuestion}

\begin{practiceQuestion}{3-10-q10}{mc}

\begin{center}

  \draw[thick,->] (0,0) -- (0,8);
  \draw[thick,->] (0,0) -- (10,0);
  \foreach \y in {1,...,7} \draw (-0.2,\y) -- (0.2,\y) node[left=3pt,font=\footnotesize] {$\y$k};
