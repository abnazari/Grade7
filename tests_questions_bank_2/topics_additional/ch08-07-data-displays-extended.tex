\begin{practiceQuestion}{8-7-q01}{mc}
\begin{questionText}
A frequency table shows that the interval $10$--$14$ has a frequency of $7$ and the interval $15$--$19$ has a frequency of $12$. How many data values fall between $10$ and $19$?
\end{questionText}
\choiceA{$5$}
\choiceB{$7$}
\choiceC{$12$}
\choiceD{$19$}
\correctAnswer{D}
\explanation{Add the frequencies of both intervals: $7 + 12 = 19$ data values.}
\end{practiceQuestion}

\choiceA{Tuesday}

\begin{practiceQuestion}{8-7-q18}{sa}
\begin{questionText}
A relative frequency table shows three categories with relative frequencies $\frac{3}{10}$, $\frac{1}{4}$, and $\frac{1}{5}$. If there are $100$ data values, how many data values are in the remaining (fourth) category?
\end{questionText}
\correctAnswer{$25$}
\explanation{Convert to common denominator: $\frac{3}{10} = 0.30$, $\frac{1}{4} = 0.25$, $\frac{1}{5} = 0.20$. Total: $0.30 + 0.25 + 0.20 = 0.75$. Remaining: $1 - 0.75 = 0.25$. Values: $0.25 \times 100 = 25$.}
\end{practiceQuestion}

\begin{practiceQuestion}{8-7-q27}{gsa}

The frequency table below shows how many minutes $25$ students studied for a quiz.

\begin{center}

\hline
\textbf{Minutes} & \textbf{Frequency} \\
\hline

$30$--$39$ & $4$ \\

Create a relative frequency table by finding the relative frequency (as a fraction and decimal) for each interval.
