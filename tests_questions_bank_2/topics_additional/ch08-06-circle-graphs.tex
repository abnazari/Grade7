\begin{practiceQuestion}{8-6-q01}{mc}
\begin{questionText}
A circle graph shows that $25\%$ of students prefer math. What fraction of the circle does this section represent?
\end{questionText}
\choiceA{$\frac{1}{2}$}
\choiceB{$\frac{1}{3}$}
\choiceC{$\frac{1}{4}$}
\choiceD{$\frac{1}{5}$}
\correctAnswer{C}
\explanation{$25\% = \frac{25}{100} = \frac{1}{4}$. The math section takes up one-quarter of the circle.}
\end{practiceQuestion}
