\section{Data Displays Extended}

\begin{stepGoal}
\begin{itemize}
    \item Read and create frequency tables, double bar graphs, and line graphs
    \item Solve problems using information from multiple data displays
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Frequency Table}{A table listing categories and how \textbf{often} each occurs.}
    \vocabItem{Double Bar Graph}{A bar graph that compares \textbf{two} data sets side by side.}
    \vocabItem{Line Graph}{A graph that connects data points to show \textbf{change over time}.}
\end{stepVocabBox}

\begin{stepsCard}[How to Read and Use Data Displays]
    \stepItem{Identify the \textbf{type} of display (frequency table, bar graph, or line graph).}
    \stepItem{Read the \textbf{labels} — what do the axes, columns, or rows represent?}
    \stepItem{Pull out \textbf{specific values} by reading the graph or table carefully.}
    \stepItem{\textbf{Calculate} or \textbf{compare}: find totals, differences, trends, or averages.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{A frequency table shows grades: A: $6$, B: $10$, C: $8$, D: $4$, F: $2$. What percent earned an A or B?}
    \stepShow{1} This is a frequency table — each row gives a category and its count.

    \stepShow{2} Labels: categories are letter grades; values are number of students.

    \stepShow{3} A $= 6$, B $= 10$. Total students $= 6 + 10 + 8 + 4 + 2 = 30$.

    \stepShow{4} A or B $= 6 + 10 = 16$. Percent $= \frac{16}{30} \approx 0.533 = 53.3\%$.
    \stepResult{About $53.3\%$ of students earned an A or B.}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{A double bar graph shows sales: Store X sold $40, 55, 60$ items over three months. Store Y sold $50, 45, 70$. Which store improved more from Month 1 to Month 3?}
    \stepShow{1} Double bar graph — two data sets on the same axes.

    \stepShow{2} Labels: months on the horizontal axis, items sold on the vertical.

    \stepShow{3} Store X: Month 1 $= 40$, Month 3 $= 60$. Store Y: Month 1 $= 50$, Month 3 $= 70$.

    \stepShow{4} Store X increase: $60 - 40 = 20$. Store Y increase: $70 - 50 = 20$. Both improved the same.
    \stepResult{Both stores improved by $20$ items from Month 1 to Month 3.}
\end{stepExample}

\watchOut{When comparing bar graphs, look at the \textbf{change} (difference), not just the height of the bars!}

\begin{stepPractice}{Data Displays Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Frequency Tables}
    \prob A table shows: Cats: $12$, Dogs: $18$, Fish: $5$, Birds: $5$. How many pets total? \answerBlank[2.5cm]
    \answer{$12 + 18 + 5 + 5 = 40$}

    \prob What fraction of pets are dogs? \answerBlank[2.5cm]
    \answer{$\frac{18}{40} = \frac{9}{20}$}

    \stepPracticeHeader[step2Color]{Double Bar Graphs}
    \prob Team A scored $30, 45, 50$ over three games. Team B scored $35, 40, 55$. Find each team's average. \answerBlank[3cm]
    \answer{Team A: $\frac{125}{3} \approx 41.7$. Team B: $\frac{130}{3} \approx 43.3$.}

    \stepPracticeHeader[step3Color]{Line Graphs}
    \prob Temperatures were $58°, 62°, 59°, 65°$ on Mon--Thu. Between which two days was the biggest change? \answerBlank[3cm]
    \answer{Wed to Thu: $|65 - 59| = 6°$}

    \wordProblem{A frequency table shows that $15$ out of $60$ students walk to school, $35$ take the bus, and $10$ get a ride. What percent walks? What percent takes the bus?}{~}
    \answer{Walk: $\frac{15}{60} = 25\%$. Bus: $\frac{35}{60} \approx 58.3\%$.}
\end{stepPractice}
