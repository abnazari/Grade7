\section{Rational Number Operations in Extended Contexts}

\begin{stepGoal}
\begin{itemize}
    \item Apply all four operations with rational numbers in real-world situations
    \item Interpret positive and negative answers in context
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Elevation}{Height above or below sea level — negative means below.}
    \vocabItem{Net Change}{The overall result after all increases and decreases are combined.}
\end{stepVocabBox}

\begin{stepsCard}[How to Solve Extended-Context Rational Number Problems]
    \stepItem{\textbf{Assign signs}: increases, gains, and above are positive; decreases, losses, and below are negative.}
    \stepItem{\textbf{Choose the operation}: combine $\to$ add/subtract; equal groups $\to$ multiply/divide.}
    \stepItem{\textbf{Solve and interpret}: compute the result and explain what the sign means in context.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{At $6$ AM the temperature is $-8.5°$C. It rises $3.2°$ each hour for $4$ hours, then drops $5.6°$. Final temperature?}
    \stepShow{1} Rise is positive: $+3.2$ per hour for $4$ hours. Drop is negative: $-5.6$.

    \stepShow{2} Multiply for the rise, then add/subtract: $-8.5 + (3.2 \times 4) - 5.6$.

    \stepShow{3} Rise: $3.2 \times 4 = 12.8$. After rise: $-8.5 + 12.8 = 4.3$. After drop: $4.3 - 5.6 = -1.3°$C. Negative means below $0$.
    \stepResult{$-1.3°$C}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{A submarine is at $-250$ ft. It rises $80$ ft, then dives $120$ ft. What is its depth?}
    \stepShow{1} Starting depth: $-250$. Rise: $+80$. Dive: $-120$.

    \stepShow{2} Add each change: $-250 + 80 + (-120)$.

    \stepShow{3} $-250 + 80 = -170$. Then $-170 + (-120) = -290$ ft. Negative means below sea level.
    \stepResult{$-290$ ft}
\end{stepExample}

\watchOut{Always re-read the problem after solving. Does your sign match the context? A negative temperature means below zero — make sure that's what the problem describes!}

\newpage
\begin{stepPractice}{Extended Contexts Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Assign Signs}
    \prob A stock opens at $\$34.50$ and changes by $-\$1.25$, $+\$0.80$, and $-\$2.10$. Write the expression. \answerBlank[4cm]
    \answer{$34.50 - 1.25 + 0.80 - 2.10$}

    \stepPracticeHeader[step2Color]{Solve}
    \prob Find the stock's closing price from the problem above. \answerBlank[2cm]
    \answer{$\$31.95$}

    \prob The temperature is $-4\frac{1}{2}°$F. It drops another $6\frac{3}{4}°$. New temperature? \answerBlank[2.5cm]
    \answer{$-11\frac{1}{4}°$F}

    \stepPracticeHeader[step3Color]{Solve and Interpret}
    \prob You owe $\$85.50$. You pay $\frac{2}{3}$ of it. How much do you still owe? \answerBlank[2.5cm]
    \answer{$\$28.50$}

    \wordProblem{Death Valley is at $-282$ ft. Mt.\ Whitney is at $14{,}505$ ft. What is the difference in elevation?}{ft}
    \answer{$14{,}787$ ft}
\end{stepPractice}
