\section{Introduction to Square Roots}

\begin{stepGoal}
\begin{itemize}
    \item Find square roots of perfect squares
    \item Estimate square roots of non-perfect squares
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Perfect Square}{A number you get by multiplying an integer by itself (e.g., $25 = 5 \times 5$).}
    \vocabItem{Square Root}{The number that, when multiplied by itself, gives the original: $\sqrt{25} = 5$.}
\end{stepVocabBox}

\begin{stepsCard}[How to Find or Estimate a Square Root]
    \stepItem{Check if the number is a \textbf{perfect square}. If so, find the integer whose square equals it.}
    \stepItem{If not a perfect square, find the two \textbf{consecutive perfect squares} it sits between.}
    \stepItem{\textbf{Estimate} — decide which perfect square the number is closer to and refine.}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{Find $\sqrt{144}$.}
    \stepShow{1} Is $144$ a perfect square? Yes — $12 \times 12 = 144$.

    \stepShow{2} (Not needed — it is a perfect square.)

    \stepShow{3} (No estimating needed.)
    \stepResult{$\sqrt{144} = 12$}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{Estimate $\sqrt{40}$.}
    \stepShow{1} Is $40$ a perfect square? No.

    \stepShow{2} $36 < 40 < 49$, so $6 < \sqrt{40} < 7$.

    \stepShow{3} $40$ is closer to $36$ than to $49$, so $\sqrt{40} \approx 6.3$.
    \stepResult{$\sqrt{40} \approx 6.3$}
\end{stepExample}

\mascotStepTip{Memorize perfect squares up to $15^2 = 225$. It makes estimating much faster!}

\begin{stepPractice}{Square Roots Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Perfect Squares}
    \begin{multicols}{2}
    \prob $\sqrt{81} =$ \answerBlank[2cm]
    \answer{$9$}
    \prob $\sqrt{196} =$ \answerBlank[2cm]
    \answer{$14$}
    \end{multicols}

    \stepPracticeHeader[step2Color]{Between Which Two Whole Numbers?}
    \begin{multicols}{2}
    \prob $\sqrt{50}$ is between \answerBlank[1.5cm] and \answerBlank[1.5cm]
    \answer{$7$ and $8$}
    \prob $\sqrt{70}$ is between \answerBlank[1.5cm] and \answerBlank[1.5cm]
    \answer{$8$ and $9$}
    \end{multicols}

    \stepPracticeHeader[step3Color]{Estimate to One Decimal Place}
    \prob Estimate $\sqrt{20} \approx$ \answerBlank[2cm]
    \answer{$\approx 4.5$}

    \wordProblem{A square garden has an area of $225$ ft$^2$. Find the side length.}{ft}
    \answer{$15$ ft}
\end{stepPractice}
