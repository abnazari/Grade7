\section{Proportional Reasoning with Scale Models}

\begin{stepGoal}
\begin{itemize}
    \item Use scale factors to find actual dimensions from a model
    \item Use scale factors to find model dimensions from actual sizes
\end{itemize}
\end{stepGoal}

\begin{stepVocabBox}
    \vocabItem{Scale Model}{A smaller (or larger) copy of a real object built to a fixed ratio.}
    \vocabItem{Scale Factor}{The ratio of model size to actual size, like $1 : 50$.}
\end{stepVocabBox}

\begin{stepsCard}[How to Solve Scale Model Problems]
    \stepItem{Identify the \textbf{scale factor} (e.g., $1 : 25$ means $1$ unit on the model $= 25$ units in real life).}
    \stepItem{To find \textbf{actual size}: multiply the model measurement by the scale factor.}
    \stepItem{To find \textbf{model size}: divide the actual measurement by the scale factor.}
    \stepItem{Convert units if needed (cm $\to$ m: divide by $100$).}
\end{stepsCard}

% --- Worked Example 1 ---
\begin{stepExample}{A model car is $8$ cm long. The scale is $1 : 25$. What is the actual length?}
    \stepShow{1} Scale factor is $25$.

    \stepShow{2} Actual $= 8 \times 25 = 200$ cm.

    \stepShow{3} Not finding model size — skip this step.

    \stepShow{4} Convert: $200 \div 100 = 2$ m.
    \stepResult{The real car is $2$ m long.}
\end{stepExample}

% --- Worked Example 2 ---
\begin{stepExample}{A bridge is $450$ m long. The scale is $1 : 500$. How long is the model?}
    \stepShow{1} Scale factor is $500$.

    \stepShow{2} Not finding actual size — skip this step.

    \stepShow{3} Model $= 450 \div 500 = 0.9$ m.

    \stepShow{4} Convert: $0.9$ m $= 90$ cm.
    \stepResult{The model bridge is $90$ cm long.}
\end{stepExample}

\mascotStepTip{Scale factor works like a unit rate — multiply to go BIG, divide to go SMALL!}

\newpage
\begin{stepPractice}{Scale Models Practice}
    \resetProblems

    \stepPracticeHeader[step1Color]{Identify the Scale Factor}
    \prob A model airplane wing is $15$ cm. The real wing is $12$ m. What is the scale factor? \answerBlank[2.5cm]
    \answer{$1 : 80$}

    \stepPracticeHeader[step2Color]{Find the Actual Size}
    \prob A model building is $12$ cm tall. Scale: $1 : 200$. Find the actual height. \answerBlank[2.5cm]
    \answer{$24$ m}

    \prob A $1 : 100$ model house is $9$ cm tall and $14$ cm wide. Find the actual dimensions. \answerBlank[3cm]
    \answer{$9$ m tall, $14$ m wide}

    \stepPracticeHeader[step3Color]{Find the Model Size}
    \prob A room is $6$ m by $4$ m. Scale: $1 : 50$. Find the model's dimensions. \answerBlank[3cm]
    \answer{$12$ cm by $8$ cm}

    \wordProblem{A statue is $15$ m tall. An architect builds a $1 : 30$ scale model. How tall is the model in centimeters?}{cm}
    \answer{$15 \div 30 = 0.5$ m $= 50$ cm}
\end{stepPractice}
