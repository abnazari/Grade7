\section{Solving Percent Problems}
\topicTitle{Solving Percent Problems}

\begin{quickReview}{Percent Problems in Everyday Money}
You use percents every time you shop, tip, or calculate tax:

\begin{itemize}[leftmargin=*]
    \item \textbf{Discount:} Sale price $=$ original $-$ (discount~\% $\times$ original)
    \item \textbf{Sales tax:} Total $=$ price $+$ (tax~\% $\times$ price)
    \item \textbf{Tip:} Tip amount $=$ tip~\% $\times$ bill
    \item \textbf{Markup:} Retail price $=$ cost $+$ (markup~\% $\times$ cost)
\end{itemize}

\medskip
\textbf{Example:} A jacket costs $\$60$ and is $25\%$ off. What is the sale price?

Discount $= 0.25 \times 60 = \$15$. Sale price $= 60 - 15 = \$45$.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Convert and Calculate}

\prob What is $15\%$ of $\$80$? \answerBlank[3cm]
\answerExplain{$\$12$}{$0.15 \times 80 = 12$.}

\prob What is $7\%$ of $\$45$? \answerBlank[3cm]
\answerExplain{$\$3.15$}{$0.07 \times 45 = 3.15$.}

\prob A shirt is $\$30$ and the discount is $20\%$. What is the discount amount? \answerBlank[3cm]
\answerExplain{$\$6$}{$0.20 \times 30 = 6$.}

\prob A meal costs $\$24$. You leave a $15\%$ tip. What is the tip? \answerBlank[3cm]
\answerExplain{$\$3.60$}{$0.15 \times 24 = 3.60$.}

\prob A store buys a hat for $\$10$ and marks it up $50\%$. What is the selling price? \answerBlank[3cm]
\answerExplain{$\$15$}{Markup $= 0.50 \times 10 = \$5$. Selling price $= 10 + 5 = \$15$.}

\prob A backpack is $\$40$ plus $8\%$ tax. What is the total? \answerBlank[3cm]
\answerExplain{$\$43.20$}{Tax $= 0.08 \times 40 = \$3.20$. Total $= 40 + 3.20 = \$43.20$.}
\end{practiceBox}

% ── Practice A: Discounts and Sales ──────────────────────────────────────────
\begin{practiceBox}{Discounts and Sales}

\practiceHeader{Find the sale price after the discount.}

\prob A pair of shoes is $\$85$. The store offers $30\%$ off. What is the sale price?
\answerBlank[3cm]
\answerExplain{$\$59.50$}{Discount $= 0.30 \times 85 = \$25.50$. Sale price $= 85 - 25.50 = \$59.50$.}

\prob A laptop bag is $\$42$ with a $15\%$ discount. What do you pay?
\answerBlank[3cm]
\answerExplain{$\$35.70$}{Discount $= 0.15 \times 42 = \$6.30$. Sale price $= 42 - 6.30 = \$35.70$.}

\prob A bookstore has a ``Buy one, get the second $40\%$ off'' deal. Both books are $\$20$. What is your total?
\answerBlank[3cm]
\answerExplain{$\$32$}{First book: $\$20$. Second book: $20 - (0.40 \times 20) = 20 - 8 = \$12$. Total: $20 + 12 = \$32$.}

\prob A $\$65$ jacket is marked $25\%$ off, then an extra $10\%$ off the sale price. Final price?
\answerBlank[3cm]
\answerExplain{$\$43.88$}{First discount: $65 \times 0.75 = \$48.75$. Second discount: $48.75 \times 0.90 = \$43.875 \approx \$43.88$.}

\end{practiceBox}

% ── Practice B: Tax and Tips ─────────────────────────────────────────────────
\begin{practiceBox}[funTeal]{Tax and Tips}

\practiceHeader[funTeal]{Find the total after tax or tip.}

\prob Dinner costs $\$36$. Sales tax is $6.5\%$. What is the total?
\answerBlank[3cm]
\answerExplain{$\$38.34$}{Tax $= 0.065 \times 36 = \$2.34$. Total $= 36 + 2.34 = \$38.34$.}

\prob A family's restaurant bill is $\$72$. They leave a $20\%$ tip. What is the tip and the total paid?
\answerBlank[3cm]
\answerExplain{Tip $= \$14.40$; Total $= \$86.40$}{Tip $= 0.20 \times 72 = \$14.40$. Total $= 72 + 14.40 = \$86.40$.}

\prob You buy a phone case for $\$18.50$ plus $7\%$ tax. What do you pay?
\answerBlank[3cm]
\answerExplain{$\$19.80$}{Tax $= 0.07 \times 18.50 = \$1.295 \approx \$1.30$. Total $= 18.50 + 1.30 = \$19.80$.}

\prob A pizza costs $\$14$. After $8\%$ tax and a $15\%$ tip (on the pre-tax price), what is the grand total?
\answerBlank[3cm]
\answerExplain{$\$17.22$}{Tax $= 0.08 \times 14 = \$1.12$. Tip $= 0.15 \times 14 = \$2.10$. Total $= 14 + 1.12 + 2.10 = \$17.22$.}

\end{practiceBox}

% ── Practice C: Markup and Profit ────────────────────────────────────────────
\begin{practiceBox}[funOrange]{Markup and Profit}

\practiceHeader[funOrange]{Find the retail price after markup.}

\prob A store buys T-shirts for $\$8$ each and marks them up $75\%$. What is the selling price?
\answerBlank[3cm]
\answerExplain{$\$14$}{Markup $= 0.75 \times 8 = \$6$. Selling price $= 8 + 6 = \$14$.}

\prob A shop buys jewelry for $\$25$ and sells it for $\$40$. What is the markup percent?
\answerBlank[3cm]
\answerExplain{$60\%$}{Markup amount $= 40 - 25 = \$15$. Percent $= \frac{15}{25} = 0.60 = 60\%$.}

\prob A furniture store marks up a desk by $120\%$ over its $\$150$ cost. What is the retail price?
\answerBlank[3cm]
\answerExplain{$\$330$}{Markup $= 1.20 \times 150 = \$180$. Retail $= 150 + 180 = \$330$.}

\end{practiceBox}

% ── Practice D: True or False ────────────────────────────────────────────────
\begin{practiceBox}[funPurple]{True or False?}

\trueOrFalse{A $20\%$ discount followed by a $10\%$ discount is the same as a $30\%$ discount.}
\answerExplain{False}{A $\$100$ item: $20\%$ off $= \$80$, then $10\%$ off $= \$72$. A straight $30\%$ off $= \$70$. The successive discounts leave a higher price because the second discount applies to the already-reduced price.}

\trueOrFalse{If sales tax is $8\%$, a $\$50$ item costs $\$54$ after tax.}
\answerExplain{True}{Tax $= 0.08 \times 50 = \$4$. Total $= 50 + 4 = \$54$.}

\trueOrFalse{A $100\%$ markup means the selling price is double the cost.}
\answerExplain{True}{$100\%$ markup $=$ adding the full cost again. Selling price $= \text{cost} + \text{cost} = 2 \times \text{cost}$.}

\trueOrFalse{A $15\%$ tip on a $\$40$ meal is $\$8$.}
\answerExplain{False}{$0.15 \times 40 = \$6$, not $\$8$.}

\end{practiceBox}

% ── Word Problems ────────────────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{A bike is on sale for $35\%$ off the original price of $\$240$. If sales tax is $7\%$ on the sale price, what is the total you pay?}{dollars}
\answerExplain{$\$166.92$}{Sale price $= 240 \times 0.65 = \$156$. Tax $= 0.07 \times 156 = \$10.92$. Total $= 156 + 10.92 = \$166.92$.}

\wordProblem{A restaurant bill is $\$55$. You add an $18\%$ tip and then pay $6\%$ tax on the food only. How much do you pay in total?}{dollars}
\answerExplain{$\$68.20$}{Tip $= 0.18 \times 55 = \$9.90$. Tax $= 0.06 \times 55 = \$3.30$. Total $= 55 + 9.90 + 3.30 = \$68.20$.}

\wordProblem{A store buys headphones for $\$35$ and sells them at a $60\%$ markup. A customer uses a $25\%$ off coupon. What does the customer pay?}{dollars}
\answerExplain{$\$42$}{Retail $= 35 + 0.60 \times 35 = 35 + 21 = \$56$. After coupon: $56 \times 0.75 = \$42$.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob During a clearance sale, prices drop $40\%$. On the last day, there is an extra $50\%$ off the sale price. Is this the same as $90\%$ off the original? If not, what single discount is it equivalent to? \answerBlank[5cm]
\answerExplain{No; it equals $70\%$ off}{First: pay $60\%$ of original. Then pay $50\%$ of that: $0.60 \times 0.50 = 0.30$. You pay $30\%$, so the total discount is $70\%$, not $90\%$.}

\prob A store has a ``price match plus $10\%$'' policy: if a competitor sells an item cheaper, the store beats that price by $10\%$ of the \emph{difference}. A blender is $\$80$ at the store and $\$65$ at a competitor. What do you pay? \answerBlank[5cm]
\answerExplain{$\$63.50$}{Difference $= 80 - 65 = \$15$. Extra saving $= 0.10 \times 15 = \$1.50$. You pay$= 65 - 1.50 = \$63.50$.}
\end{challengeBox}
