\section{Applying Proportional Reasoning}
\topicTitle{Applying Proportional Reasoning to Real-World Problems}

\begin{quickReview}{Estimate First, Then Solve}
Before solving any proportion problem, \textbf{estimate the answer} using rounded numbers. After solving, check that your exact answer is close to your estimate.

\begin{enumerate}[leftmargin=*]
    \item \textbf{Round} the numbers to make the math easy.
    \item \textbf{Estimate} the answer mentally.
    \item \textbf{Solve} using unit rate, proportion, equation, or scale factor.
    \item \textbf{Check} --- is your answer close to the estimate? Does it make sense?
\end{enumerate}

\medskip
\textbf{Example:} $7$ shirts cost $\$48.93$. How much for $12$?

\textit{Estimate:} $\$48.93 \approx \$49$. Per shirt $\approx \$7$. For $12$: about $\$84$.

\textit{Solve:} Unit rate $= 48.93 \div 7 = \$6.99$. For $12$: $6.99 \times 12 = \$83.88$.

\textit{Check:} $\$83.88 \approx \$84$ \checkmark
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Estimate first, then solve.}

\prob $6$ notebooks cost $\$8.94$. About how much for $1$ notebook? Find the exact unit price.
\answerBlank[3cm]
\answerExplain{Estimate: about $\$1.50$; Exact: $\$1.49$}{Estimate: $\$8.94 \approx \$9$, and $9 \div 6 = \$1.50$. Exact: $8.94 \div 6 = \$1.49$. Close to the estimate \checkmark.}

\prob A car travels $195$ miles on $6$ gallons. About how far on $1$ gallon? Find the exact mileage.
\answerBlank[3cm]
\answerExplain{Estimate: about $33$ mi; Exact: $32.5$ mi}{Estimate: $195 \approx 198$, and $198 \div 6 = 33$. Exact: $195 \div 6 = 32.5$ miles per gallon.}

\prob A recipe for $3$ servings uses $5$ eggs. About how many eggs for $10$ servings? Find the exact answer.
\answerBlank[3cm]
\answerExplain{Estimate: about $17$; Exact: $16\frac{2}{3}$ eggs}{Estimate: $10 \div 3 \approx 3.3$ batches, and $3.3 \times 5 \approx 17$. Exact: $\frac{10}{3} \times 5 = \frac{50}{3} = 16\frac{2}{3}$.}

\prob A map scale says $1$~cm $= 4.8$~km. Two cities are $7$~cm apart. Estimate, then find the actual distance.
\answerBlank[3cm]
\answerExplain{Estimate: about $35$~km; Exact: $33.6$~km}{Estimate: $4.8 \approx 5$, and $5 \times 7 = 35$~km. Exact: $4.8 \times 7 = 33.6$~km.}

\prob $15$ muffins cost $\$11.25$. Estimate the cost per muffin, then find the exact price.
\answerBlank[3cm]
\answerExplain{Estimate: about $\$0.75$; Exact: $\$0.75$}{Estimate: $\$11.25 \approx \$12$, and $12 \div 15 = \$0.80$. Exact: $11.25 \div 15 = \$0.75$.}
\end{practiceBox}

% ── Practice A: Estimate-Then-Solve Proportions ──────────────────────────────
\begin{practiceBox}{Estimate-Then-Solve Proportions}

\practiceHeader{For each problem, write an estimate \emph{before} solving.}

\prob A factory produces $372$ bottles in $8$ hours. Estimate, then find how many in $15$ hours.
\answerBlank[3cm]
\answerExplain{Estimate: about $700$; Exact: $697.5$}{Estimate: $372 \approx 400$, rate $\approx 50$/hr, in $15$~hr $\approx 750$. Exact: $372 \div 8 = 46.5$/hr, and $46.5 \times 15 = 697.5$.}

\prob A cyclist rides $43.4$~km in $3$ hours. Estimate, then find the distance in $5$ hours.
\answerBlank[3cm]
\answerExplain{Estimate: about $72$~km; Exact: $72.\overline{3}$~km}{Estimate: $43.4 \approx 42$, rate $\approx 14$~km/hr, in $5$~hr $\approx 70$. Exact: $43.4 \div 3 \approx 14.47$, and $14.47 \times 5 \approx 72.3$~km.}

\prob $6$ pounds of apples cost $\$8.34$. Estimate, then find the cost for $10$ pounds.
\answerBlank[3cm]
\answerExplain{Estimate: about $\$14$; Exact: $\$13.90$}{Estimate: $\$8.34 \approx \$8.40$, rate $= \$1.40$/lb, for $10$~lb $= \$14$. Exact: $8.34 \div 6 = \$1.39$, and $1.39 \times 10 = \$13.90$.}

\prob A printer prints $88$ pages in $6$ minutes. Estimate, then find the time for $300$ pages.
\answerBlank[3cm]
\answerExplain{Estimate: about $20$~min; Exact: $\approx 20.5$~min}{Estimate: $88 \approx 90$, rate $= 15$~pg/min, time $= 300 \div 15 = 20$~min. Exact: $88 \div 6 \approx 14.67$~pg/min, time $= 300 \div 14.67 \approx 20.5$~min.}

\end{practiceBox}

% ── Practice B: Estimate to Check Reasonableness ─────────────────────────────
\begin{practiceBox}[funTeal]{Is the Answer Reasonable?}

\practiceHeader[funTeal]{A student solved each problem. Use estimation to decide if the answer is reasonable.}

\prob A car gets $28.5$ miles per gallon. The student says a $400$-mile trip needs $\mathbf{40}$ gallons. Reasonable?
\answerBlank[3cm]
\answerExplain{Not reasonable; about $14$ gallons}{Estimate: $400 \div 30 \approx 13.3$ gallons. The answer of $40$ is way too high. Exact: $400 \div 28.5 \approx 14.0$ gallons.}

\prob $8$ tickets cost $\$53.60$. The student says $12$ tickets cost $\$80.40$. Reasonable?
\answerBlank[3cm]
\answerExplain{Reasonable}{Estimate: per ticket $\approx \$54 \div 8 \approx \$6.75$. For $12$: $6.75 \times 12 = \$81$. The answer $\$80.40$ is close. Exact: $53.60 \div 8 = \$6.70$, and $6.70 \times 12 = \$80.40$ \checkmark.}

\prob A shadow problem: a $6$-ft person casts a $4$-ft shadow. The student says a $20$-ft shadow belongs to a $30$-ft tree. Reasonable?
\answerBlank[3cm]
\answerExplain{Reasonable}{Estimate: ratio is $6:4 = 1.5$. For a $20$-ft shadow: $20 \times 1.5 = 30$~ft. The answer $30$~ft checks out.}

\prob A recipe for $4$ servings needs $\frac{3}{2}$ cups of rice. The student says $14$ servings need $\mathbf{3}$ cups. Reasonable?
\answerBlank[3cm]
\answerExplain{Not reasonable; about $5\frac{1}{4}$ cups}{Estimate: $14$ servings is $3.5$ times $4$ servings. $1.5 \times 3.5 = 5.25$ cups. The answer of $3$ cups is too low.}

\end{practiceBox}

% ── Practice C: Which Is the Better Deal? ────────────────────────────────────
\begin{practiceBox}[funOrange]{Which Is the Better Deal?}

\practiceHeader[funOrange]{Estimate unit prices first, then find exact values.}

\prob Brand A: $5$ bars of soap for $\$4.45$. Brand B: $8$ bars for $\$6.80$. \answerBlank[4cm]
\answerExplain{Brand B (about $\$0.85$/bar vs.\ $\$0.89$/bar)}{Estimates: A $\approx 4.50 \div 5 = \$0.90$. B $\approx 6.80 \div 8 = \$0.85$. Exact: A $= 4.45 \div 5 = \$0.89$, B $= 6.80 \div 8 = \$0.85$. Brand B is cheaper.}

\prob Small juice: $\frac{3}{4}$~L for $\$1.79$. Large juice: $1\frac{1}{2}$~L for $\$3.29$. \answerBlank[4cm]
\answerExplain{Large (about $\$2.19$/L vs.\ $\$2.39$/L)}{Estimates: Small $\approx 1.80 \div 0.75 = \$2.40$/L. Large $\approx 3.30 \div 1.5 = \$2.20$/L. Exact: Small $= \$2.39$/L, Large $= \$2.19$/L. Large is cheaper.}

\prob Pack A: $15$ pencils for $\$3.75$. Pack B: $24$ pencils for $\$5.52$. \answerBlank[4cm]
\answerExplain{Pack B ($\$0.23$/pencil vs.\ $\$0.25$/pencil)}{Estimate: A $\approx 3.75 \div 15 = \$0.25$. B $\approx 5.50 \div 25 = \$0.22$. Exact: A $= \$0.25$, B $= \$0.23$. Pack B is the better value.}

\end{practiceBox}

% ── Practice D: Multi-Step Word Problems ─────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Multi-Step Word Problems}

\wordProblem{A car travels $295$ miles on $8$ gallons. Gas costs $\$3.49$ per gallon. Estimate, then find the exact fuel cost for a $700$-mile trip.}{dollars}
\answerExplain{Estimate: about $\$66$; Exact: $\$66.30$}{Estimate: mpg $\approx 300 \div 8 \approx 37.5$, gallons $\approx 700 \div 37.5 \approx 19$, cost $\approx 19 \times 3.50 = \$66.50$. Exact: mpg $= 36.875$, gallons $= 18.98$, cost $= 18.98 \times 3.49 \approx \$66.24$.}

\wordProblem{On a map, $2$~cm $= 15$ miles. A trail measures $9.4$~cm. A cyclist goes $12$~mph. Estimate, then find the exact ride time.}{hours}
\answerExplain{Estimate: about $6$~hr; Exact: $\approx 5.88$~hr}{Estimate: distance $\approx 9 \times 7.5 = 67.5$~mi, time $\approx 67.5 \div 12 \approx 5.6$~hr. Exact: distance $= 9.4 \times 7.5 = 70.5$~mi, time $= 70.5 \div 12 = 5.875$~hr.}

\wordProblem{A recipe for $6$ servings calls for $\frac{5}{4}$ cups of sugar and $3$ cups of flour. You need $15$ servings. Estimate, then find the exact amounts.}{cups of sugar and cups of flour}
\answerExplain{Estimate: about $3$ cups sugar, $7.5$ cups flour; Exact: $3\frac{1}{8}$ cups sugar, $7.5$ cups flour}{Scale factor: $15 \div 6 = 2.5$. Sugar: $1.25 \times 2.5 = 3.125 = 3\frac{1}{8}$ cups. Flour: $3 \times 2.5 = 7.5$ cups. Estimates close \checkmark.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob A store sells paint: Small $\frac{3}{4}$ gallon for $\$17.89$, Large $2\frac{1}{2}$ gallons for $\$54.75$. You need $5$ gallons. Estimate the cheapest option, then find the exact cost. \answerBlank[5cm]
\answerExplain{Estimate: $2$ large $\approx \$110$; Exact: $2 \times \$54.75 = \$109.50$}{Estimate unit prices: Small $\approx \$18 \div 0.75 = \$24$/gal. Large $\approx \$55 \div 2.5 = \$22$/gal. Large is cheaper. Two large cans $= 5$ gal exactly. Exact cost: $2 \times 54.75 = \$109.50$.}

\prob A student solves a proportion and gets $x = 450$. The original problem says $3$ workers complete $28$ tasks in $5$ days; how many tasks do $8$ workers complete in $10$ days? Without solving, explain whether $450$ is reasonable.
\answerExplain{Not reasonable; estimate about $150$ tasks}{Estimate: $8$ workers is about $2.7\times$ more workers, and $10$ days is $2\times$ more time. So roughly $28 \times 2.7 \times 2 \approx 150$ tasks. The answer $450$ is far too high.}
\end{challengeBox}
