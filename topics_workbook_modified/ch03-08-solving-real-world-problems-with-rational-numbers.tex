\section{Solving Real-World Problems with Rational Numbers}
\topicTitle{Solving Real-World Problems with Rational Numbers}

\begin{quickReview}{Putting It All Together --- Alaska Style}
Real-world problems often mix addition, subtraction, multiplication, and division with fractions \textbf{and} decimals.

\begin{enumerate}[leftmargin=*]
    \item \textbf{Read carefully.} Decide which operation each part needs.
    \item \textbf{Watch the signs.} Gains/increases are positive; losses/decreases are negative.
    \item \textbf{Estimate first.} Use rounded numbers to predict a reasonable answer.
    \item \textbf{Check your answer.} Does the sign make sense? Is the size reasonable?
\end{enumerate}

\medskip
\textbf{Example:} At midnight in Fairbanks, it is $-28°$F. By noon it rises $\frac{3}{4}$ of the way toward $0°$F.

Rise $= \frac{3}{4} \times 28 = 21$. Noon temp $= -28 + 21 = -7°$F.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Compute. Show your work.}

\prob $-15 + 8.5 =$ \answerBlank[2cm]
\answerExplain{$-6.5$}{Adding a positive to a negative: $-15 + 8.5 = -(15 - 8.5) = -6.5$.}

\prob $-\dfrac{3}{4} \times 12 =$ \answerBlank[2cm]
\answerExplain{$-9$}{$\frac{3}{4} \times 12 = 9$. Negative times positive is negative: $-9$.}

\prob $-6.2 - (-3.8) =$ \answerBlank[2cm]
\answerExplain{$-2.4$}{Subtracting a negative means adding: $-6.2 + 3.8 = -2.4$.}

\prob $\dfrac{2}{5} \div \left(-\dfrac{1}{3}\right) =$ \answerBlank[2cm]
\answerExplain{$-\frac{6}{5}$}{Multiply by the reciprocal: $\frac{2}{5} \times (-3) = -\frac{6}{5}$.}

\prob $-20 + 7.5 + (-3.25) =$ \answerBlank[2cm]
\answerExplain{$-15.75$}{$-20 + 7.5 = -12.5$. Then $-12.5 + (-3.25) = -15.75$.}

\end{practiceBox}

% ── Practice A: Alaska Temperature Problems ──────────────────────────────────
\begin{practiceBox}{Alaska Temperature Problems}

\practiceHeader{Solve each temperature problem.}

\prob The temperature in Barrow is $-18°$F at 6~AM. It drops $5.5°$F by midnight, then rises $12.25°$F by noon the next day. What is the noon temperature?
\answerExplain{$-11.25°$F}{$-18 + (-5.5) = -23.5$. Then $-23.5 + 12.25 = -11.25°$F.}

\prob Anchorage records $-4.5°$C on Monday morning. By Thursday, the temperature has changed by $+8.2°$C, then $-3.7°$C, then $+1.5°$C. What is the temperature on Thursday?
\answerExplain{$1.5°$C}{$-4.5 + 8.2 = 3.7$. Then $3.7 - 3.7 = 0$. Then $0 + 1.5 = 1.5°$C.}

\prob The average January temperature in Fairbanks is $-15.8°$F and in Juneau is $28.4°$F. What is the difference?
\answerExplain{$44.2°$F}{$28.4 - (-15.8) = 28.4 + 15.8 = 44.2°$F.}

\prob The high temperature was $-2°$F and the low was $-19°$F. What is the range?
\answerExplain{$17°$F}{Range $= -2 - (-19) = -2 + 19 = 17°$F.}

\end{practiceBox}

% ── Practice B: Fishing and Wildlife ─────────────────────────────────────────
\begin{practiceBox}[funTeal]{Fishing and Wildlife}

\practiceHeader[funTeal]{Solve each problem.}

\prob A fishing boat catches $345.5$~lb of salmon on Monday and $278.75$~lb on Tuesday. The crew keeps $\frac{2}{3}$ of the total and sells the rest at $\$4.50$/lb. How much do they earn?
\answerExplain{$\$937.88$}{Total: $345.5 + 278.75 = 624.25$~lb. Sold: $\frac{1}{3} \times 624.25 = 208.08\overline{3}$~lb. Revenue: $208.08\overline{3} \times 4.50 \approx \$937.88$.}

\prob A research team tracks a whale pod. The whales dive to $-85.5$~m, rise $42.25$~m, then dive another $33.75$~m. What is the final depth?
\answerExplain{$-77$~m}{$-85.5 + 42.25 = -43.25$. Then $-43.25 + (-33.75) = -77$~m.}

\prob A village catches $1{,}240$ fish over $5$ days. On the first $3$ days they catch $\frac{3}{5}$ of the total. How many fish are left for the last $2$ days?
\answerExplain{$496$ fish}{First $3$ days: $\frac{3}{5} \times 1{,}240 = 744$. Remaining: $1{,}240 - 744 = 496$ fish.}

\prob A guide charges $\$185$ per person for a $4$-hour fishing trip. If $\frac{1}{4}$ of the fee covers boat fuel, how much fuel money comes from a group of $6$ people?
\answerExplain{$\$277.50$}{Total fees: $6 \times 185 = \$1{,}110$. Fuel: $\frac{1}{4} \times 1{,}110 = \$277.50$.}

\end{practiceBox}

% ── Practice C: Elevation and Distance ───────────────────────────────────────
\begin{practiceBox}[funOrange]{Elevation and Distance}

\practiceHeader[funOrange]{Solve.}

\prob A hiker descends $1{,}250$~ft from a summit at $4{,}380$~ft, rests, then climbs $\frac{1}{2}$ of the descent back up. What is the final elevation?
\answerExplain{$3{,}755$~ft}{After descent: $4{,}380 - 1{,}250 = 3{,}130$. Climbs back: $\frac{1}{2} \times 1{,}250 = 625$. Final: $3{,}130 + 625 = 3{,}755$~ft.}

\prob A float plane takes off from a lake at $820$~ft elevation, climbs $3{,}450$~ft, then descends $1{,}275.5$~ft. What is the current altitude?
\answerExplain{$2{,}994.5$~ft}{$820 + 3{,}450 = 4{,}270$. Then $4{,}270 - 1{,}275.5 = 2{,}994.5$~ft.}

\prob Two villages are $48.6$ miles apart by road. A dogsled team covers $\frac{2}{3}$ of the distance the first day and $\frac{1}{4}$ of the remaining distance the next morning. How far is left?
\answerExplain{$12.15$ miles}{Day 1: $\frac{2}{3} \times 48.6 = 32.4$~mi. Remaining: $48.6 - 32.4 = 16.2$~mi. Day 2: $\frac{1}{4} \times 16.2 = 4.05$~mi. Left: $16.2 - 4.05 = 12.15$~mi.}

\end{practiceBox}

% ── Practice D: Budgets and Money ────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Budgets and Money}

\wordProblem{A family budgets $\$1{,}200$ per month. They spend $\frac{2}{5}$ on housing, $\frac{1}{4}$ on food, and $\$180$ on fuel. How much is left?}{dollars}
\answerExplain{$\$240$}{Housing: $\frac{2}{5} \times 1{,}200 = \$480$. Food: $\frac{1}{4} \times 1{,}200 = \$300$. Total spent: $480 + 300 + 180 = \$960$. Left: $1{,}200 - 960 = \$240$.}

\wordProblem{A store sells hand-carved crafts. A carver is paid $\frac{3}{4}$ of each sale. If $8$ items sell for a total of $\$342$, how much does the carver earn?}{dollars}
\answerExplain{$\$256.50$}{$\frac{3}{4} \times 342 = \$256.50$.}

\wordProblem{A community center has a budget of $\$5{,}000$ for a winter festival. They spend $38.5\%$ on food, $\frac{1}{5}$ on decorations, and the rest on prizes. How much goes to prizes?}{dollars}
\answerExplain{$\$2{,}075$}{Food: $0.385 \times 5{,}000 = \$1{,}925$. Decor: $\frac{1}{5} \times 5{,}000 = \$1{,}000$. Prizes: $5{,}000 - 1{,}925 - 1{,}000 = \$2{,}075$.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob A sled dog team runs $4\frac{1}{2}$ hours the first day, covering $7.2$ miles per hour. The second day, the trail conditions slow them to $\frac{2}{3}$ of their first-day speed, and they run $6$ hours. What is the total distance covered? \answerBlank[5cm]
\answerExplain{$61.2$ miles}{Day 1: $4.5 \times 7.2 = 32.4$~mi. Day 2 speed: $\frac{2}{3} \times 7.2 = 4.8$~mph. Day 2: $6 \times 4.8 = 28.8$~mi. Total: $32.4 + 28.8 = 61.2$~mi.}

\prob Evaluate: $-\dfrac{1}{2} \times 8 + \dfrac{3}{4} \div \left(-\dfrac{1}{2}\right)$ \answerBlank[3cm]
\answerExplain{$-5\frac{1}{2}$}{Multiply: $-\frac{1}{2} \times 8 = -4$. Divide: $\frac{3}{4} \div (-\frac{1}{2}) = \frac{3}{4} \times (-2) = -\frac{3}{2}$. Add: $-4 + (-\frac{3}{2}) = -\frac{11}{2} = -5\frac{1}{2}$.}
\end{challengeBox}
