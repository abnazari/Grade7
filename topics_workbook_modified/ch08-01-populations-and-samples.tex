\section{Populations and Samples}
\topicTitle{Populations and Samples}

\begin{quickReview}{Populations, Samples, and Experimental Design}
\textbf{Population}: The entire group you want to study.
\textbf{Sample}: A smaller group chosen from the population.

A \textbf{random sample} gives every member an equal chance of being selected, avoiding \textbf{bias}.

\textbf{Five steps to design an experiment:}
\begin{enumerate}[leftmargin=*]
    \item \textbf{Ask a question} --- What do you want to find out?
    \item \textbf{Identify the population} --- Who or what are you studying?
    \item \textbf{Choose a sampling method} --- How will you pick your sample?
    \item \textbf{Collect data} --- Gather the information.
    \item \textbf{Analyze and conclude} --- What does the data tell you?
\end{enumerate}

\medskip
\textbf{Example:} A school wants to know the favorite sport of all $600$ students. Use a random number generator to pick $60$ student IDs and survey them.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Identify the Population and Sample}

\prob A factory tests $50$ bulbs from a batch of $5{,}000$. Population? Sample?
\answerBlank[4cm]
\answerExplain{Population: $5{,}000$ bulbs; Sample: $50$ bulbs}{The factory uses the sample to estimate the quality of the whole batch.}

\prob A city surveys $500$ of its $50{,}000$ residents about transportation. Population? Sample?
\answerBlank[4cm]
\answerExplain{Population: $50{,}000$ residents; Sample: $500$ surveyed}{The sample represents the whole city.}

\prob A teacher asks $25$ of her $120$ students about homework time. Population? Sample?
\answerBlank[4cm]
\answerExplain{Population: $120$ students; Sample: $25$ students}{The $25$ students are chosen from the full class.}

\prob A vet weighs $150$ of $3{,}000$ dogs in town. Population? Sample?
\answerBlank[4cm]
\answerExplain{Population: $3{,}000$ dogs; Sample: $150$ dogs}{The sample gives an estimate for all dogs' weights.}

\prob A farm tests $200$ of $10{,}000$ eggs. Population? Sample?
\answerBlank[4cm]
\answerExplain{Population: $10{,}000$ eggs; Sample: $200$ eggs}{Testing all $10{,}000$ would be impractical; the sample provides a reliable estimate.}
\end{practiceBox}

% ── Practice A: Random vs. Biased ────────────────────────────────────────────
\begin{practiceBox}{Random vs.\ Biased Samples}

\practiceHeader{Is the sampling method random or biased? Explain.}

\prob A principal surveys every student whose last name starts with M. \answerBlank[3cm]
\answerExplain{Biased}{Not every student has an equal chance. Students with M names may share cultural backgrounds.}

\prob A manager draws $50$ receipts from a box without looking. \answerBlank[3cm]
\answerExplain{Random}{Every receipt has an equal chance of being drawn.}

\prob A coach surveys only the basketball team about favorite sports. \answerBlank[3cm]
\answerExplain{Biased}{Basketball players will likely choose basketball. The sample doesn't represent all students.}

\prob A researcher uses a computer to randomly select $200$ household numbers. \answerBlank[3cm]
\answerExplain{Random}{Every household has an equal chance.}

\prob A teacher asks the first $10$ students who arrive each morning. \answerBlank[3cm]
\answerExplain{Biased}{Early arrivals may have different habits than late arrivals. Not everyone has an equal chance.}

\end{practiceBox}

% ── Practice B: Design an Experiment ─────────────────────────────────────────
\begin{practiceBox}[funTeal]{Design an Experiment}

\practiceHeader[funTeal]{Use the 5-step process for each situation.}

\prob Design an experiment to find out which school lunch is most popular.
\answerExplain{1) Question: Which lunch is most popular? 2) Population: all students. 3) Method: randomly select $50$. 4) Collect: survey their choice. 5) Analyze: tally and find most common.}{Following the $5$-step process ensures a valid, unbiased study.}

\prob Design an experiment to estimate how many hours per week students in your school spend on homework.
\answerExplain{1) Question: Average weekly homework hours? 2) Population: all students. 3) Method: randomly select $60$ from student IDs. 4) Collect: each records hours for one week. 5) Analyze: find the mean and compare by grade.}{Random selection avoids bias from surveying only hard-working or struggling students.}

\prob Design an experiment to determine whether a new fertilizer increases plant growth.
\answerExplain{1) Question: Does the fertilizer help? 2) Population: plants of the same type. 3) Method: randomly assign $20$ plants to fertilizer group and $20$ to no-fertilizer (control). 4) Collect: measure growth over $4$ weeks. 5) Analyze: compare average growth of both groups.}{Using a control group and random assignment ensures valid comparison.}

\end{practiceBox}

% ── Practice C: True or False ────────────────────────────────────────────────
\begin{practiceBox}[funPurple]{True or False?}

\trueOrFalse{A larger random sample always gives a more reliable estimate.}
\answerExplain{True}{Larger random samples reduce the effect of chance and better represent the population.}

\trueOrFalse{If you survey only your friends, your sample is random.}
\answerExplain{False}{Friends share similar opinions. This convenience sample is biased.}

\trueOrFalse{A biased sample can still give accurate results about a population.}
\answerExplain{False}{A biased sample systematically favors certain outcomes, making conclusions unreliable.}

\trueOrFalse{In an experiment, a control group helps you measure the effect of the variable you're testing.}
\answerExplain{True}{Without a control group, you cannot tell whether changes are due to your variable or other factors.}

\end{practiceBox}

% ── Word Problems ────────────────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{A company randomly surveys $200$ of $4{,}000$ employees and finds $60\%$ prefer working from home. Estimate the number of all employees who prefer remote work.}{employees}
\answerExplain{$2{,}400$}{$0.60 \times 4{,}000 = 2{,}400$ employees.}

\wordProblem{A pet store surveys $50$ random customers; $18$ own a cat. If the store has $800$ total customers, estimate how many own cats.}{customers}
\answerExplain{$288$}{$\frac{18}{50} = 0.36$. Estimate: $0.36 \times 800 = 288$.}

\wordProblem{A school surveys $80$ random students; $35$ walk to school. If the school has $640$ students, estimate the total who walk.}{students}
\answerExplain{$280$}{$\frac{35}{80} = 0.4375$. Estimate: $0.4375 \times 640 = 280$.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob Pollster~A surveys $100$ people at a mall: $55\%$ support a new law. Pollster~B randomly calls $100$ voters: $42\%$ support it. Which result is more trustworthy? Design a better study. \answerBlank[5cm]
\answerExplain{Pollster B is more trustworthy; design: randomly call $500$ voters from the phone registry}{Mall surveys only reach mall-goers, introducing bias. Random phone calls give every voter an equal chance. Increasing to $500$ improves reliability.}

\prob A student claims: ``I surveyed $10$ friends about the school dress code, so I know what the whole school thinks.'' Identify \emph{two} problems with this claim.
\answerExplain{1) Sample is too small; 2) Friends are biased (not random)}{A sample of $10$ is small and may not represent $500$+ students. Friends share similar views, so the sample is biased by self-selection.}
\end{challengeBox}
