\section{Rational Number Operations in Extended Contexts}
\topicTitle{Rational Number Operations in Extended Contexts}

\begin{quickReview}{Rational Numbers in the Real World}
Rational numbers (positive and negative fractions, decimals, integers) appear in many real-world contexts:

\begin{itemize}[leftmargin=*]
    \item \textbf{Temperature:} $-15°$F means $15$ degrees below zero.
    \item \textbf{Elevation:} $-200$ ft means $200$ ft below sea level.
    \item \textbf{Money:} $-\$50$ means $\$50$ in debt.
    \item \textbf{Stocks:} $+2.75$ means the price went up $\$2.75$.
\end{itemize}

\medskip
\textbf{Key idea:} Choose the right operation, watch the signs, and check that your answer makes sense in context.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Basic Operations with Signed Numbers}

\begin{multicols}{2}
\prob $-8 + 5 =$ \answerBlank[2cm]
\answerExplain{$-3$}{Adding a positive to a negative: $|-8| - |5| = 3$, keep the negative sign: $-3$.}

\prob $-12 - 7 =$ \answerBlank[2cm]
\answerExplain{$-19$}{Subtracting from a negative makes it more negative: $-12 + (-7) = -19$.}

\prob $6 \times (-4) =$ \answerBlank[2cm]
\answerExplain{$-24$}{A positive times a negative gives a negative: $6 \times 4 = 24$, so the answer is $-24$.}

\prob $-15 \div 3 =$ \answerBlank[2cm]
\answerExplain{$-5$}{A negative divided by a positive gives a negative: $15 \div 3 = 5$, so the answer is $-5$.}

\prob $-3.5 + 8.2 =$ \answerBlank[2cm]
\answerExplain{$4.7$}{$8.2 - 3.5 = 4.7$. The positive number is larger, so the result is positive.}

\prob $-\dfrac{1}{2} - \dfrac{3}{4} =$ \answerBlank[2cm]
\answerExplain{$-1\frac{1}{4}$}{$-\frac{2}{4} - \frac{3}{4} = -\frac{5}{4} = -1\frac{1}{4}$.}
\end{multicols}
\end{practiceBox}

% ── Practice A: Temperature Problems ─────────────────────────────────────────
\begin{practiceBox}{Temperature Changes}

\prob At $6$ AM the temperature is $-14°$F. By noon it rises $22°$. What is the temperature at noon? \answerBlank[3cm]
\answerExplain{$8°$F}{$-14 + 22 = 8°$F. The temperature rose above zero.}

\prob The temperature is $5°$C. It drops $12°$ overnight. What is the new temperature? \answerBlank[3cm]
\answerExplain{$-7°$C}{$5 - 12 = -7°$C.}

\prob At dawn the temperature is $-9.5°$C. It rises $3.2°$ each hour for $5$ hours. What is the temperature? \answerBlank[3cm]
\answerExplain{$6.5°$C}{Rise: $3.2 \times 5 = 16°$. New temp: $-9.5 + 16 = 6.5°$C.}

\prob The high is $4°$F and the low is $-18°$F. What is the difference? \answerBlank[3cm]
\answerExplain{$22°$F}{Difference $= 4 - (-18) = 4 + 18 = 22°$F.}

\prob The temperature at $8$ AM is $-6.5°$C. It rises $4.8°$ by $10$ AM, then drops $7.3°$ by $4$ PM. What is the final temperature? \answerBlank[3cm]
\answerExplain{$-9°$C}{$-6.5 + 4.8 = -1.7$. Then $-1.7 - 7.3 = -9°$C.}
\end{practiceBox}

% ── Practice B: Elevation Problems ───────────────────────────────────────────
\begin{practiceBox}[funTeal]{Elevation Changes}

\prob A submarine is at $-250$ ft. It rises $80$ ft, then dives $120$ ft. What is its depth? \answerBlank[3cm]
\answerExplain{$-290$ ft}{$-250 + 80 = -170$. Then $-170 - 120 = -290$ ft.}

\prob Death Valley is at $-282$ ft. The top of a nearby hill is $1{,}218$ ft. What is the elevation change? \answerBlank[3cm]
\answerExplain{$1{,}500$ ft}{$1{,}218 - (-282) = 1{,}218 + 282 = 1{,}500$ ft.}

\prob A scuba diver starts at the surface ($0$ ft), descends $45.5$ ft, then rises $18.3$ ft. Current depth? \answerBlank[3cm]
\answerExplain{$-27.2$ ft}{$0 - 45.5 + 18.3 = -27.2$ ft (still $27.2$ ft below the surface).}

\prob A hikers starts at $-50$ ft (below sea level) and climbs $320$ ft. What is the new elevation? \answerBlank[3cm]
\answerExplain{$270$ ft}{$-50 + 320 = 270$ ft above sea level.}
\end{practiceBox}

% ── Practice C: Money and Finance ────────────────────────────────────────────
\begin{practiceBox}[funOrange]{Money Transactions}

\prob You have $\$125.50$ in your account. You spend $\$89.75$ and then deposit $\$40$. What is your balance? \answerBlank[3cm]
\answerExplain{$\$75.75$}{$125.50 - 89.75 + 40 = \$75.75$.}

\prob A stock opens at $\$34.50$. It changes by $-\$1.25$, $+\$0.80$, and $-\$2.10$. What is the closing price? \answerBlank[3cm]
\answerExplain{$\$31.95$}{$34.50 - 1.25 + 0.80 - 2.10 = \$31.95$.}

\prob You owe $\$85.50$. You pay $\frac{2}{3}$ of it. How much do you still owe? \answerBlank[3cm]
\answerExplain{$\$28.50$}{Paid: $\frac{2}{3} \times 85.50 = \$57$. Remaining: $85.50 - 57 = \$28.50$.}

\prob A business loses $\$250$/day for $5$ days, then gains $\$400$/day for $3$ days. What is the net change? \answerBlank[3cm]
\answerExplain{$-\$50$}{Loss: $-250 \times 5 = -\$1{,}250$. Gain: $400 \times 3 = \$1{,}200$. Net: $-1{,}250 + 1{,}200 = -\$50$.}
\end{practiceBox}

% ── Practice D: Mixed Contexts ───────────────────────────────────────────────
\begin{practiceBox}[funPurple]{Mixed Real-World Problems}

\prob The temperature was $-4\frac{1}{2}°$F and dropped another $6\frac{3}{4}°$. What is the new temperature? \answerBlank[3cm]
\answerExplain{$-11\frac{1}{4}°$F}{$-\frac{9}{2} - \frac{27}{4} = -\frac{18}{4} - \frac{27}{4} = -\frac{45}{4} = -11\frac{1}{4}°$F.}

\prob Death Valley is $-282$ ft. Mount Whitney is $14{,}505$ ft. What is the difference in elevation? \answerBlank[3cm]
\answerExplain{$14{,}787$ ft}{$14{,}505 - (-282) = 14{,}505 + 282 = 14{,}787$ ft.}

\prob A fish swims at $-12.6$ m. It goes up $4.8$ m, then back down $7.2$ m. Where is it now? \answerBlank[3cm]
\answerExplain{$-15$ m}{$-12.6 + 4.8 - 7.2 = -15$ m.}
\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob Over five days, the temperature changes were $+3.5°$, $-7.2°$, $+1.8°$, $-4.6°$, $+2.1°$. If the starting temperature was $-2.0°$C, what was the final temperature? \answerBlank[3cm]
\answerExplain{$-6.4°$C}{Sum of changes: $3.5 - 7.2 + 1.8 - 4.6 + 2.1 = -4.4°$. Final: $-2.0 + (-4.4) = -6.4°$C.}

\prob An elevator starts at floor $-2$ (basement). It goes up $8$ floors, down $3$, up $5$, and down $12$. What floor is it on? \answerBlank[3cm]
\answerExplain{Floor $-4$}{$-2 + 8 - 3 + 5 - 12 = -4$. The elevator is on floor $-4$ (the $4$th basement level).}
\end{challengeBox}
