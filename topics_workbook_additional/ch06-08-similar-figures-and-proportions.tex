\section{Similar Figures and Proportions}
\topicTitle{Similar Figures and Proportions}

\begin{quickReview}{Similar Figures}
Two figures are \textbf{similar} ($\sim$) if they have the same shape but not necessarily the same size.

\begin{itemize}[leftmargin=*]
    \item Corresponding angles are \textbf{equal}.
    \item Corresponding sides are \textbf{proportional} (same scale factor).
    \item Scale factor $= \dfrac{\text{side of larger figure}}{\text{corresponding side of smaller figure}}$
\end{itemize}

\medskip
\textbf{Example:} $\triangle ABC \sim \triangle DEF$. $AB = 6$, $DE = 9$, $BC = 8$. Find $EF$.

Scale factor $= \frac{9}{6} = 1.5$. So $EF = 8 \times 1.5 = 12$.
\end{quickReview}

% ── Warm-Up ──────────────────────────────────────────────────────────────────
\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Find the Scale Factor}

\begin{multicols}{2}
\prob Sides $3$ and $9$. Scale factor $=$ \answerBlank[2cm]
\answerExplain{$3$}{$9 \div 3 = 3$. The larger figure is $3$ times the smaller.}

\prob Sides $5$ and $15$. Scale factor $=$ \answerBlank[2cm]
\answerExplain{$3$}{$15 \div 5 = 3$.}

\prob Sides $4$ and $10$. Scale factor $=$ \answerBlank[2cm]
\answerExplain{$2.5$}{$10 \div 4 = 2.5$.}

\prob Sides $8$ and $12$. Scale factor $=$ \answerBlank[2cm]
\answerExplain{$1.5$}{$12 \div 8 = 1.5$.}

\prob Sides $6$ and $6$. Scale factor $=$ \answerBlank[2cm]
\answerExplain{$1$}{$6 \div 6 = 1$. The figures are the same size (congruent).}

\prob Sides $10$ and $4$. Scale factor (small to large) $=$ \answerBlank[2cm]
\answerExplain{$2.5$}{$10 \div 4 = 2.5$. The scale factor from the smaller ($4$) to the larger ($10$) is $2.5$.}
\end{multicols}
\end{practiceBox}

% ── Practice A: Finding Missing Sides ────────────────────────────────────────
\begin{practiceBox}{Find the Missing Side}

\practiceHeader{The figures are similar. Use the scale factor to find the missing side.}

\prob $\triangle ABC \sim \triangle DEF$. $AB = 4$, $DE = 12$, $BC = 5$. Find $EF$. \answerBlank[3cm]
\answerExplain{$15$}{Scale factor: $12 \div 4 = 3$. $EF = 5 \times 3 = 15$.}

\prob $\triangle PQR \sim \triangle STU$. $PQ = 10$, $ST = 4$, $QR = 15$. Find $TU$. \answerBlank[3cm]
\answerExplain{$6$}{Scale factor: $4 \div 10 = 0.4$. $TU = 15 \times 0.4 = 6$.}

\prob Rectangle $ABCD \sim$ Rectangle $EFGH$. $AB = 6$, $EF = 15$, $BC = 4$. Find $FG$. \answerBlank[3cm]
\answerExplain{$10$}{Scale factor: $15 \div 6 = 2.5$. $FG = 4 \times 2.5 = 10$.}

\prob $\triangle LMN \sim \triangle XYZ$. $LM = 8$, $XY = 20$, $MN = 6$, $LN = 10$. Find $YZ$ and $XZ$. \answerBlank[3cm]
\answerExplain{$YZ = 15$, $XZ = 25$}{Scale: $20 \div 8 = 2.5$. $YZ = 6 \times 2.5 = 15$. $XZ = 10 \times 2.5 = 25$.}

\prob Two similar pentagons have corresponding sides of $7$ cm and $21$ cm. Another side of the small pentagon is $5$ cm. Find the corresponding side of the large pentagon. \answerBlank[3cm]
\answerExplain{$15$ cm}{Scale: $21 \div 7 = 3$. Missing side: $5 \times 3 = 15$ cm.}
\end{practiceBox}

% ── Practice B: Setting Up Proportions ───────────────────────────────────────
\begin{practiceBox}[funTeal]{Solve Using a Proportion}

\practiceHeader[funTeal]{Set up $\dfrac{\text{side}_1}{\text{side}_2} = \dfrac{\text{side}_3}{\text{side}_4}$ and cross-multiply.}

\prob $\triangle ABC \sim \triangle DEF$. $AB = 3$, $DE = x$, $BC = 7$, $EF = 21$. Find $x$. \answerBlank[3cm]
\answerExplain{$9$}{$\frac{3}{x} = \frac{7}{21}$. Cross-multiply: $3 \times 21 = 7x$, so $63 = 7x$ and $x = 9$.}

\prob Two similar rectangles: width $5$ cm matches width $x$ cm; length $12$ cm matches length $30$ cm. Find $x$. \answerBlank[3cm]
\answerExplain{$12.5$ cm}{$\frac{5}{x} = \frac{12}{30}$. Cross-multiply: $150 = 12x$, so $x = 12.5$ cm.}

\prob Two similar triangles have sides $6$ and $x$, and $8$ and $20$. Find $x$. \answerBlank[3cm]
\answerExplain{$15$}{$\frac{6}{x} = \frac{8}{20}$. Cross-multiply: $120 = 8x$, so $x = 15$.}

\prob $\triangle RST \sim \triangle UVW$. $RS = 9$, $UV = 6$, $ST = x$, $VW = 10$. Find $x$. \answerBlank[3cm]
\answerExplain{$15$}{$\frac{9}{6} = \frac{x}{10}$. Cross-multiply: $90 = 6x$, so $x = 15$.}
\end{practiceBox}

% ── Practice C: Indirect Measurement ─────────────────────────────────────────
\begin{practiceBox}[funOrange]{Indirect Measurement (Shadows)}

\prob A tree casts a $12$-ft shadow. A $5$-ft person casts a $3$-ft shadow at the same time. How tall is the tree? \answerBlank[3cm]
\answerExplain{$20$ ft}{$\frac{5}{3} = \frac{h}{12}$. Cross-multiply: $60 = 3h$, so $h = 20$ ft.}

\prob A streetlight casts a $15$-ft shadow. A $4$-ft mailbox casts a $3$-ft shadow. How tall is the streetlight? \answerBlank[3cm]
\answerExplain{$20$ ft}{$\frac{4}{3} = \frac{h}{15}$. Cross-multiply: $60 = 3h$, so $h = 20$ ft.}

\prob A flagpole casts a $24$-ft shadow. A $6$-ft teacher casts an $8$-ft shadow. How tall is the flagpole? \answerBlank[3cm]
\answerExplain{$18$ ft}{$\frac{6}{8} = \frac{h}{24}$. Cross-multiply: $144 = 8h$, so $h = 18$ ft.}
\end{practiceBox}

% ── True or False ────────────────────────────────────────────────────────────
\begin{practiceBox}[funPurple]{True or False?}

\trueOrFalse{All squares are similar to each other.}
\answerExplain{True}{All squares have $90°$ angles and all sides equal, so any two squares have equal angles and proportional sides.}

\trueOrFalse{All rectangles are similar to each other.}
\answerExplain{False}{Rectangles have equal angles ($90°$), but their side ratios can be different. A $2 \times 4$ rectangle is not similar to a $2 \times 6$ rectangle.}
\end{practiceBox}

% ── Word Problems ────────────────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{A photo is $4$ in.\ by $6$ in. It is enlarged so the short side is $10$ in. What is the long side?}{in.}
\answerExplain{$15$ in.}{Scale: $10 \div 4 = 2.5$. Long side: $6 \times 2.5 = 15$ in.}

\wordProblem{A map uses a scale where $2$ cm represents $50$ km. Two cities are $7$ cm apart on the map. What is the actual distance?}{km}
\answerExplain{$175$ km}{$\frac{2}{50} = \frac{7}{d}$. Cross-multiply: $2d = 350$, so $d = 175$ km.}
\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob Two similar triangles have perimeters of $30$ cm and $45$ cm. The shortest side of the smaller triangle is $8$ cm. What is the shortest side of the larger triangle? \answerBlank[3cm]
\answerExplain{$12$ cm}{Scale factor $= 45 \div 30 = 1.5$. Shortest side: $8 \times 1.5 = 12$ cm. (Perimeters of similar figures scale by the same factor as their sides.)}
\end{challengeBox}
