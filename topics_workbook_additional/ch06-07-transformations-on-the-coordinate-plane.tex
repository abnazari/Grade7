\section{Transformations on the Coordinate Plane}
\topicTitle{Transformations on the Coordinate Plane}

\begin{quickReview}{Three Types of Transformations}
All three transformations keep the \textbf{size and shape} the same (rigid motions).

\begin{itemize}[leftmargin=*]
    \item \textbf{Translation} (slide): $(x, y) \to (x + a,\; y + b)$
    \item \textbf{Reflection} (flip):
        Over the $x$-axis: $(x, y) \to (x, -y)$. \quad
        Over the $y$-axis: $(x, y) \to (-x, y)$.
    \item \textbf{Rotation} (turn):
        $90В°$ clockwise about origin: $(x, y) \to (y, -x)$.
        $180В°$ about origin: $(x, y) \to (-x, -y)$.
\end{itemize}

\medskip
\textbf{Example:} Translate $A(2, 5)$ right $3$, down $4$: $(2+3,\; 5-4) = A'(5, 1)$.
\end{quickReview}

\begin{practiceBox}[funGreen]{\faSun~Warm-Up}

\practiceHeader[funGreen]{Translations}

\begin{multicols}{2}
\prob Translate $(3, 2)$ right $4$, up $1$. \answerBlank[2cm]
\answerExplain{$(7, 3)$}{Add $4$ to $x$ and $1$ to $y$: $(3+4, 2+1) = (7, 3)$.}

\prob Translate $(-1, 5)$ left $3$, down $2$. \answerBlank[2cm]
\answerExplain{$(-4, 3)$}{Left means subtract from $x$: $(-1-3, 5-2) = (-4, 3)$.}

\prob Translate $(0, -4)$ right $6$, up $7$. \answerBlank[2cm]
\answerExplain{$(6, 3)$}{$(0+6, -4+7) = (6, 3)$.}

\prob Translate $(5, 1)$ left $5$, down $1$. \answerBlank[2cm]
\answerExplain{$(0, 0)$}{$(5-5, 1-1) = (0, 0)$. The point lands on the origin.}

\prob Translate $(-2, -3)$ right $8$, up $5$. \answerBlank[2cm]
\answerExplain{$(6, 2)$}{$(-2+8, -3+5) = (6, 2)$.}

\prob Translate $(4, -6)$ left $2$, down $3$. \answerBlank[2cm]
\answerExplain{$(2, -9)$}{$(4-2, -6-3) = (2, -9)$.}
\end{multicols}
\end{practiceBox}

\begin{practiceBox}{Reflections}

\practiceHeader{Reflect each point over the given axis.}

\prob Reflect $(4, -2)$ over the $x$-axis. \answerBlank[3cm]
\answerExplain{$(4, 2)$}{Over the $x$-axis: keep $x$, flip the sign of $y$. $(4, -2) \to (4, 2)$.}

\prob Reflect $(-5, 3)$ over the $y$-axis. \answerBlank[3cm]
\answerExplain{$(5, 3)$}{Over the $y$-axis: flip the sign of $x$, keep $y$. $(-5, 3) \to (5, 3)$.}

\prob Reflect $(2, 7)$ over the $x$-axis. \answerBlank[3cm]
\answerExplain{$(2, -7)$}{$(2, 7) \to (2, -7)$. The $y$-coordinate changes sign.}

\prob Reflect $(-3, -4)$ over the $y$-axis. \answerBlank[3cm]
\answerExplain{$(3, -4)$}{$(-3, -4) \to (3, -4)$. The $x$-coordinate changes sign.}

\prob Reflect $(0, 6)$ over the $x$-axis. \answerBlank[3cm]
\answerExplain{$(0, -6)$}{$(0, 6) \to (0, -6)$. Points on the $y$-axis stay on the $y$-axis.}

\prob Reflect $(-1, -8)$ over the $x$-axis. \answerBlank[3cm]
\answerExplain{$(-1, 8)$}{$(-1, -8) \to (-1, 8)$. Only the $y$-coordinate flips.}
\end{practiceBox}

% в”Ђв”Ђ Practice B: Rotations в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ
\begin{practiceBox}[funTeal]{Rotations About the Origin}

\practiceHeader[funTeal]{Apply the given rotation.}

\prob Rotate $(3, 1)$ by $90В°$ clockwise. \answerBlank[3cm]
\answerExplain{$(1, -3)$}{$90В°$ clockwise: $(x, y) \to (y, -x)$. $(3, 1) \to (1, -3)$.}

\prob Rotate $(2, -5)$ by $90В°$ clockwise. \answerBlank[3cm]
\answerExplain{$(-5, -2)$}{$(2, -5) \to (-5, -2)$.}

\prob Rotate $(-4, 2)$ by $180В°$. \answerBlank[3cm]
\answerExplain{$(4, -2)$}{$180В°$: $(x, y) \to (-x, -y)$. $(-4, 2) \to (4, -2)$.}

\prob Rotate $(6, 0)$ by $180В°$. \answerBlank[3cm]
\answerExplain{$(-6, 0)$}{$(6, 0) \to (-6, 0)$.}

\prob Rotate $(-1, -3)$ by $90В°$ clockwise. \answerBlank[3cm]
\answerExplain{$(-3, 1)$}{$(-1, -3) \to (-3, -(-1)) = (-3, 1)$.}
\end{practiceBox}

% в”Ђв”Ђ Practice C: Identifying the Transformation в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ
\begin{practiceBox}[funOrange]{Name That Transformation}

\practiceHeader[funOrange]{Tell whether each is a translation, reflection, or rotation.}

\prob $(2, 3) \to (2, -3)$ \answerBlank[3cm]
\answerExplain{Reflection over the $x$-axis}{The $x$-coordinate stayed the same and the $y$-coordinate changed sign, which is a reflection over the $x$-axis.}

\prob $(2, 3) \to (5, 7)$ \answerBlank[3cm]
\answerExplain{Translation (right $3$, up $4$)}{Both coordinates changed by a constant amount: $x$ increased by $3$, $y$ increased by $4$. This is a translation.}

\prob $(4, -1) \to (-4, 1)$ \answerBlank[3cm]
\answerExplain{Rotation $180В°$ about the origin}{Both coordinates changed sign: $(x, y) \to (-x, -y)$. This is a $180В°$ rotation.}

\prob $(-3, 5) \to (3, 5)$ \answerBlank[3cm]
\answerExplain{Reflection over the $y$-axis}{The $y$-coordinate stayed the same and the $x$-coordinate changed sign, which is a reflection over the $y$-axis.}
\end{practiceBox}

% в”Ђв”Ђ Word Problem в”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђв”Ђ
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{A triangle has vertices $P(1, 2)$, $Q(4, 2)$, $R(4, 5)$. Translate it right $2$ and down $3$. List the new vertices.}{}
\answerExplain{$P'(3, -1)$, $Q'(6, -1)$, $R'(6, 2)$}{Add $2$ to each $x$, subtract $3$ from each $y$: $(1+2, 2-3)=(3,-1)$; $(4+2, 2-3)=(6,-1)$; $(4+2, 5-3)=(6,2)$.}

\wordProblem{A rectangle has vertices $A(1, 1)$, $B(4, 1)$, $C(4, 3)$, $D(1, 3)$. Reflect it over the $x$-axis. List the new vertices.}{}
\answerExplain{$A'(1, -1)$, $B'(4, -1)$, $C'(4, -3)$, $D'(1, -3)$}{Flip each $y$-coordinate: $(1,-1)$, $(4,-1)$, $(4,-3)$, $(1,-3)$.}
\end{practiceBox}

\answerExplain{$(4, 3)$}{$90В°$ clockwise: $(3, 4) \to (4, -3)$. Reflect over $x$-axis: $(4, -3) \to (4, 3)$.}
\end{challengeBox}
