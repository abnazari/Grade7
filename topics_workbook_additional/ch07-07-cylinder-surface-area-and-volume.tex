\topicTitle{Cylinder Surface Area and Volume}

\begin{quickReview}{Cylinder Surface Area and Volume}
A \textbf{cylinder} has two circular bases and a curved rectangular side.

\begin{itemize}[leftmargin=*]
    \item \textbf{Volume:} $V = \pi r^2 h$ \quad (area of circular base $\times$ height)
    \item \textbf{Surface Area:} $SA = 2\pi r^2 + 2\pi r h$ \quad (two circles $+$ curved side)
    \item The curved side, when unrolled, is a rectangle with width $= h$ and length $= 2\pi r$ (the circumference).
\end{itemize}

\medskip
\textbf{Example:} $r = 3$ cm, $h = 10$ cm, $\pi \approx 3.14$:
$V = 3.14 \times 9 \times 10 = 282.6$ cm$^3$. \quad $SA = 2(3.14)(9) + 2(3.14)(3)(10) = 56.52 + 188.4 = 244.92$ cm$^2$.
\end{quickReview}

% ── Practice B: Diameter Given ───────────────────────────────────────────────
\begin{practiceBox}[funTeal]{Given Diameter — Find Volume and Surface Area}

\practiceHeader[funTeal]{First find the radius, then compute. Use $\pi \approx 3.14$.}

\prob A can has diameter $6$ cm and height $10$ cm. Find the volume. \answerBlank[3cm]
\answerExplain{$282.6$ cm$^3$}{$r = 3$ cm. $V = 3.14 \times 9 \times 10 = 282.6$ cm$^3$.}

\prob A pipe has diameter $4$ in and length $20$ in. Find the volume. \answerBlank[3cm]
\answerExplain{$251.2$ in$^3$}{$r = 2$ in. $V = 3.14 \times 4 \times 20 = 251.2$ in$^3$.}

\prob A drum has diameter $12$ ft and height $4$ ft. Find the surface area. \answerBlank[3cm]
\answerExplain{$376.8$ ft$^2$}{$r = 6$ ft. $SA = 2(3.14)(36) + 2(3.14)(6)(4) = 226.08 + 150.72 = 376.8$ ft$^2$.}

\prob A cylinder has diameter $10$ m and height $7$ m. Find both the volume and surface area. \answerBlank[3cm]
\answerExplain{$V = 549.5$ m$^3$; $SA = 376.8$ m$^2$}{$r = 5$ m. $V = 3.14 \times 25 \times 7 = 549.5$ m$^3$. $SA = 2(3.14)(25) + 2(3.14)(5)(7) = 157 + 219.8 = 376.8$ m$^2$.}

\end{practiceBox}

% ── Practice C: Working Backwards ────────────────────────────────────────────
\begin{practiceBox}[funOrange]{Find the Missing Dimension}

\practiceHeader[funOrange]{Use $V = \pi r^2 h$ to find the unknown. Use $\pi \approx 3.14$.}

\prob A cylinder has volume $628$ cm$^3$ and radius $5$ cm. Find the height. \answerBlank[3cm]
\answerExplain{$8$ cm}{$h = V \div (\pi r^2) = 628 \div (3.14 \times 25) = 628 \div 78.5 = 8$ cm.}

\prob A cylinder has volume $471$ in$^3$ and height $6$ in. Find the radius. \answerBlank[3cm]
\answerExplain{$5$ in}{$r^2 = V \div (\pi h) = 471 \div (3.14 \times 6) = 471 \div 18.84 = 25$. So $r = 5$ in.}

\prob A cylinder has volume $1{,}256$ m$^3$ and radius $10$ m. Find the height. \answerBlank[3cm]
\answerExplain{$4$ m}{$h = 1{,}256 \div (3.14 \times 100) = 1{,}256 \div 314 = 4$ m.}

\end{practiceBox}

% ── Word Problems ────────────────────────────────────────────────────────────
\begin{practiceBox}[funBlue]{\faGlobe~Word Problems}

\wordProblem{A water bottle is shaped like a cylinder with radius $3$ cm and height $20$ cm. How many cubic centimeters of water can it hold? Use $\pi \approx 3.14$.}{cm$^3$}
\answerExplain{$565.2$ cm$^3$}{$V = \pi r^2 h = 3.14 \times 9 \times 20 = 565.2$ cm$^3$.}

\wordProblem{A soup can has radius $4$ cm and height $12$ cm. The label covers the curved side only. What is the area of the label? Use $\pi \approx 3.14$.}{cm$^2$}
\answerExplain{$301.44$ cm$^2$}{The curved side area (lateral area): $2\pi rh = 2 \times 3.14 \times 4 \times 12 = 301.44$ cm$^2$.}

\wordProblem{A cylindrical storage tank has diameter $8$ m and height $5$ m. The outside will be painted. What is the total surface area to paint? Use $\pi \approx 3.14$.}{m$^2$}
\answerExplain{$226.08$ m$^2$}{$r = 4$ m. $SA = 2(3.14)(16) + 2(3.14)(4)(5) = 100.48 + 125.6 = 226.08$ m$^2$.}

\end{practiceBox}

% ── Challenge ────────────────────────────────────────────────────────────────
\begin{challengeBox}
\prob A cylindrical tank with radius $5$ m and height $8$ m is half-full of water. How many cubic meters of water are in the tank? Use $\pi \approx 3.14$. \answerBlank[3cm]
\answerExplain{$314$ m$^3$}{Full volume: $3.14 \times 25 \times 8 = 628$ m$^3$. Half-full: $628 \div 2 = 314$ m$^3$.}

\prob A company makes cylindrical cans with radius $3$ cm and height $10$ cm. Each can is made from a flat sheet of metal. What is the minimum area of metal needed for one can (top, bottom, and side)? Use $\pi \approx 3.14$. \answerBlank[3cm]
\answerExplain{$244.92$ cm$^2$}{$SA = 2\pi r^2 + 2\pi rh = 2(3.14)(9) + 2(3.14)(3)(10) = 56.52 + 188.4 = 244.92$ cm$^2$.}
\end{challengeBox}
